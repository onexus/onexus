053public class BoxesPanel extends Panel {
054
055    @PaxWicketBean(name = "collectionManager")
056    private ICollectionManager collectionManager;
057
058    private boolean quickList;
059
060    public BoxesPanel(String id, SearchPageStatus status, ORI baseUri, FilterConfig filterConfig) {
061        super(id);
062        setMarkupId("boxes");
063        setOutputMarkupId(true);
064
065        add(new AttributeModifier("class", "accordion"));
066
067        RepeatingView boxes = new RepeatingView("boxes");
068
069        SearchType type = status.getType();
070
071        if (Strings.isEmpty(status.getSearch())) {
072
073            // Nothing selected
074            add(new EmptyPanel("disambiguation").setVisible(false));
075            List<SearchLink> links = type.getFixLinks();
076            if (links != null && !links.isEmpty()) {
077                boxes.add(new MainLinksBox(boxes.newChildId(), links));
078            }
079
080            if (type.getFigures() != null) {
081                for (FigureConfig figure : type.getFigures()) {
082                    if (!Strings.isEmpty(figure.getVisible()) && "NONE".equalsIgnoreCase(figure.getVisible())) {
083                        boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, null));
084                    }
085                }
086            }
087
088        } else {
089            ORI collectionUri = type.getCollection().toAbsolute(baseUri);
090            if (filterConfig == null && status.getSearch().indexOf(',') == -1) {
091
092                // Single entity selection
093                IEntityTable table = getSingleEntityTable(collectionManager, type, collectionUri, baseUri, status.getSearch(), true);
094
095                boolean found;
096                if (table.next()) {
097                    found = true;
098                } else {
099
100                    // If we don't found an exact match, look for a similar one
101                    table.close();
102                    table = getSingleEntityTable(collectionManager, type, collectionUri, baseUri, status.getSearch(), false);
103                    found = table.next();
104                }
105
106                if (found) {
107
108                    IEntity entity = table.getEntity(collectionUri);
109
110                    boxes.add(new LinksBox(boxes.newChildId(), status, entity));
111
112                    if (type.getFigures() != null) {
113                        for (FigureConfig figure : type.getFigures()) {
114                            if (Strings.isEmpty(figure.getVisible()) || "SINGLE".equalsIgnoreCase(figure.getVisible())) {
115                                boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, new SingleEntitySelection(entity)));
116                            }
117                        }
118                    }
119
120                    if (table.next()) {
121                        add(new DisambiguationPanel("disambiguation", table, collectionUri) {
122
123                            @Override
124                            protected void onSelection(AjaxRequestTarget target, String newSearch) {
125                                onDisambiguation(target, newSearch);
126                            }
127                        });
128                    } else {
129                        add(new EmptyPanel("disambiguation").setVisible(false));
130                    }
131
132                } else {
133                    add(new EmptyPanel("disambiguation").setVisible(false));
134                    boxes.add(new Label(boxes.newChildId(), "No results found").add(new AttributeModifier("class", "alert")));
135                }
136                table.close();
137
138            } else {
139
140                // Multiple entities selection
141
142                add(new EmptyPanel("disambiguation").setVisible(false));
143
144                quickList = false;
145                if (filterConfig == null) {
146                    quickList = true;
147                    filterConfig = new FilterConfig(status.getSearch());
148
149                    filterConfig.setCollection(collectionUri);
150                    filterConfig.setDefine("fc='" + collectionUri + "'");
151                    String mainKey = type.getKeysList().get(0);
152                    In where = new In("fc", mainKey);
153                    String[] values = status.getSearch().split(",");
154                    for (String value : values) {
155                        where.addValue(value.trim());
156                    }
157                    filterConfig.setWhere(where.toString());
158                }
159
160                IEntityTable table = getMultipleEntityTable(collectionManager, type, collectionUri, baseUri, filterConfig);
161                boxes.add(new LinksBox(boxes.newChildId(), status, collectionUri, filterConfig, new EntityIterator(table, collectionUri)) {
162                    @Override
163                    protected void onNotFound(Set<String> valuesNotFound) {
164                        if (valuesNotFound.isEmpty() || !quickList) {
165                            BoxesPanel.this.addOrReplace(new EmptyPanel("disambiguation").setVisible(false));
166                        } else {
167                            BoxesPanel.this.addOrReplace(new DisambiguationPanel("disambiguation", valuesNotFound) {
168                                @Override
169                                protected void onSelection(AjaxRequestTarget target, String newSearch) {
170                                    onDisambiguation(target, newSearch);
171                                }
172                            });
173                        }
174                    }
175                });
176
177                if (type.getFigures() != null) {
178                    for (FigureConfig figure : type.getFigures()) {
179                        if (Strings.isEmpty(figure.getVisible()) || "LIST".equalsIgnoreCase(figure.getVisible())) {
180                            boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, new MultipleEntitySelection(filterConfig)));
181                        }
182                    }
183                }
184            }
185        }
186
187        add(boxes);
188
189    }
190
191    protected void onDisambiguation(AjaxRequestTarget target, String query) {
192    }
193
194    private static IEntityTable getMultipleEntityTable(ICollectionManager collectionManager, SearchType type, ORI collectionUri, ORI baseOri, FilterConfig filter) {
195
196        Query query = new Query();
197        query.setOn(baseOri);
198
199        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
200        query.setFrom(collectionAlias);
201
202
203        query.addSelect(collectionAlias, null);
204
205        IEntitySelection selection = new MultipleEntitySelection(filter);
206        query.setWhere(selection.buildFilter(query));
207
208        List<String> fieldList = type.getFieldsList();
209        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
210
211        return collectionManager.load(query);
212    }
213
214
215    public static IEntityTable getSingleEntityTable(ICollectionManager collectionManager, SearchType type, ORI collectionUri, ORI baseUri, String search, boolean equal) {
216
217        Query query = new Query();
218        query.setOn(baseUri);
219
220        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
221        query.setFrom(collectionAlias);
222
223        List<String> fieldList = type.getFieldsList();
224        query.addSelect(collectionAlias, null);
225
226        for (String field : fieldList) {
227            if (equal) {
228                QueryUtils.or(query, new Equal(collectionAlias, field, search));
229            } else {
230                QueryUtils.or(query, new Contains(collectionAlias, field, search));
231            }
232        }
233
234        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
235        query.setCount(20);
236
237        return collectionManager.load(query);
238    }
