067public class DownloadWidget extends Widget<DownloadWidgetConfig, DownloadWidgetStatus> {
068
069    public static final ResourceReference CSS = new CssResourceReference(DownloadWidget.class, "prettify/prettify.css");
070    public static final ResourceReference JS = new JavaScriptResourceReference(DownloadWidget.class, "prettify/prettify.js");
071    private static final Pattern COMMA_PATTERN = Pattern.compile(",");
072
073    @PaxWicketBean(name = "collectionManager")
074    private ICollectionManager collectionManager;
075
076    public static final Map<String, IQueryScript> SCRIPTS_MAP = new LinkedHashMap<String, IQueryScript>();
077
078    static {
079        addScript(new RScript());
080        addScript(new Python2Script());
081        addScript(new Python3Script());
082        addScript(new PerlScript());
083        addScript(new BashScript());
084    }
085
086    private static void addScript(IQueryScript queryScript) {
087        SCRIPTS_MAP.put(queryScript.getLabel().toLowerCase(), queryScript);
088    }
089
090    private static final Map<String, IDownloadFormat> FORMATS_MAP = new LinkedHashMap<String, IDownloadFormat>();
091
092    static {
093        addFormat(new TsvFormat());
094        addFormat(new ExcelFormat());
095    }
096
097    private static void addFormat(IDownloadFormat format) {
098        FORMATS_MAP.put(format.getLabel().toLowerCase(), format);
099    }
100
101
102    private String webserviceUrl;
103    private IDownloadFormat format;
104
105    public DownloadWidget(String componentId, IModel<DownloadWidgetStatus> statusModel) {
106        super(componentId, statusModel);
107        onEventFireUpdate(EventQueryUpdate.class);
108
109        // Get query
110        final StringBuilder oql = new StringBuilder();
111        Query query = getQuery();
112        query.toString(oql, true);
113
114        // Webservice URL
115        String serviceMount = collectionManager.getMount();
116        webserviceUrl = WebsiteApplication.toAbsolutePath('/' + serviceMount);
117
118        // Download form
119        final Form<String> downloadForm = new Form<String>("form");
120        downloadForm.setOutputMarkupId(true);
121
122        List<IDownloadFormat> formats = getFormats();
123        if (formats.isEmpty()) {
124            throw new UnsupportedOperationException("Download widget: At least one format is needed.");
125        }
126        setFormat(formats.get(0));
127        downloadForm.add(new DropDownChoice<IDownloadFormat>("format", new PropertyModel<IDownloadFormat>(this, "format"), formats));
128        add(downloadForm);
129
130        final AjaxDownloadBehavior ajaxDownloadBehavior = new AjaxDownloadBehavior() {
131            @Override
132            protected String getFileName() {
133                return DownloadWidget.this.getFileName();
134            }
135
136            @Override
137            protected IResource getResource() {
138                return new DownloadResource();
139            }
140        };
141        downloadForm.add(ajaxDownloadBehavior);
142
143        AjaxButton link = new AjaxButton("download") {
144
145            @Override
146            protected void onSubmit(AjaxRequestTarget target, Form<?> form) {
147                ajaxDownloadBehavior.initiate(target);
148                target.add(form);
149            }
150        };
151        downloadForm.add(link);
152
153        List<IQueryScript> scripts = getScripts();
154        // Add scripts
155        ListView scriptsView = new ListView<IQueryScript>("scripts", scripts) {
156            @Override
157            protected void populateItem(ListItem<IQueryScript> item) {
158
159                IQueryScript script = item.getModelObject();
160
161                // Code body
162                WebMarkupContainer body = new WebMarkupContainer("body");
163                body.setMarkupId(item.getMarkupId() + "-body");
164                item.add(body);
165                body.add(new Label("code", script.getContent(oql.toString(), webserviceUrl)).setEscapeModelStrings(false));
166
167                // Code toggle
168                Label toggle = new Label("toggle", "Use in " + script.getLabel() + " script");
169                toggle.add(new AttributeModifier("href", "#" + body.getMarkupId()));
170                item.add(toggle);
171
172            }
173        };
174
175        if (scripts.isEmpty()) {
176            scriptsView.setVisible(false);
177        }
178        add(scriptsView);
179
180    }
181
182    private String getFileName() {
183        BrowserPageStatus status = DownloadWidget.this.findParentStatus(BrowserPageStatus.class);
184
185        if (status == null) {
186            return getFormat().getFileName("datafile");
187        }
188
189        return getFormat().getFileName(status.getCurrentTabId());
190    }
191
192    private List<IDownloadFormat> getFormats() {
193        List<IDownloadFormat> formats = new ArrayList<IDownloadFormat>();
194
195        String formatsStr = getConfig().getFormats();
196
197        if (formatsStr == null) {
198            formats.addAll(FORMATS_MAP.values());
199        } else {
200            String[] values = COMMA_PATTERN.split(formatsStr);
201            for (String value : values) {
202                IDownloadFormat format = FORMATS_MAP.get(value.trim().toLowerCase());
203                if (format != null) {
204                    formats.add(format);
205                }
206            }
207        }
208
209        return formats;
210    }
211
212
213    private List<IQueryScript> getScripts() {
214        List<IQueryScript> scripts = new ArrayList<IQueryScript>();
215
216        String scriptsStr = getConfig().getScripts();
217
218        if (scriptsStr == null) {
219            scripts.addAll(SCRIPTS_MAP.values());
220        } else {
221            String[] values = COMMA_PATTERN.split(scriptsStr);
222            for (String value : values) {
223                IQueryScript script = SCRIPTS_MAP.get(value.trim().toLowerCase());
224                if (script != null) {
225                    scripts.add(script);
226                }
227            }
228        }
229
230        return scripts;
231    }

239        }
240
241    }

249    public IDownloadFormat getFormat() {
250        return format;

254        this.format = format;

257    private ICollectionManager getCollectionManager() {
258
259        if (collectionManager == null) {
260            WebsiteApplication.inject(this);
261        }
262
263        return collectionManager;
264    }
265
266    private class DownloadResource extends AbstractResource {
267
268        @Override
269        protected ResourceResponse newResourceResponse(Attributes attributes) {
270
271            ResourceResponse resourceResponse = new ResourceResponse();
272
273            resourceResponse.setContentDisposition(ContentDisposition.ATTACHMENT);
274            resourceResponse.setContentType(getFormat().getContentType());
275            resourceResponse.setFileName(getFileName());
276            resourceResponse.setWriteCallback(new WriteCallback() {
277                @Override
278                public void writeData(Attributes attributes) throws IOException {
279
280                    Query query = getQuery();
281                    IDownloadFormat format = getFormat();
282
283                    if (format.getMaxRowsLimit() != null && (query.getCount() == null || format.getMaxRowsLimit() < query.getCount())) {
284                        query.setCount(format.getMaxRowsLimit());
285                    }
286
287                    IEntityTable table = getCollectionManager().load(query);
288                    OutputStream out = attributes.getResponse().getOutputStream();
289                    getFormat().write(table, out);
290                }
291            });
292
293            return resourceResponse;
294        }
295    }
