022import org.apache.wicket.markup.head.HeaderItem;

024import org.apache.wicket.markup.head.JavaScriptHeaderItem;
025import org.apache.wicket.markup.head.OnLoadHeaderItem;
026import org.apache.wicket.markup.html.image.Image;
027import org.apache.wicket.model.IModel;
028import org.apache.wicket.request.resource.CssResourceReference;
029import org.apache.wicket.request.resource.PackageResourceReference;
030import org.apache.wicket.request.resource.ResourceReference;
031import org.apache.wicket.resource.JQueryPluginResourceReference;
032import org.onexus.collection.api.ICollectionManager;
033import org.onexus.collection.api.IEntityTable;
034import org.onexus.collection.api.query.Query;
035import org.onexus.resource.api.ORI;
036import org.onexus.website.api.WebsiteApplication;
037import org.onexus.website.api.widgets.Widget;
038import org.onexus.website.api.widgets.tableviewer.columns.ColumnConfig;
039import org.ops4j.pax.wicket.api.PaxWicketBean;
040
041import java.net.URLEncoder;
042import java.util.ArrayList;
043import java.util.List;
044
045public class HeatmapViewer extends Widget<HeatmapViewerConfig, HeatmapViewerStatus> {
046
047    private static final HeaderItem CSS = CssHeaderItem.forReference(new CssResourceReference(HeatmapViewer.class, "css/jheatmap-1.0.0-alpha.css"));
048    private static final HeaderItem JS_JHEATMAP = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(HeatmapViewer.class, "js/jheatmap-1.0.0-alpha.js"));
049
050    private static final ResourceReference LOADING_IMG = new PackageResourceReference(HeatmapViewer.class, "images/loading.gif");
051
052
053    @PaxWicketBean(name = "collectionManager")
054    private ICollectionManager collectionManager;
055
056    public HeatmapViewer(String componentId, IModel<HeatmapViewerStatus> status) {
057        super(componentId, status);
058
059        add(new Image("loading", LOADING_IMG));
060    }
061
062    @Override
063    public void renderHead(IHeaderResponse response) {
064        response.render(CSS);
065        response.render(JS_JHEATMAP);
066        response.render(OnLoadHeaderItem.forScript(newJavaScriptHeatmap()));
067    }
068
069    private String newJavaScriptHeatmap() {
070        StringBuilder code = new StringBuilder();
071
072        HeatmapViewerConfig config = getConfig();
073
074        List<IColumn<IEntityTable, String>> columns = new ArrayList<IColumn<IEntityTable, String>>();
075
076        ORI parentUri = getPageBaseOri();
077        for (ColumnConfig column : getConfig().getColumns()) {
078            column.addColumns(columns, parentUri, false);
079        }
080        int lastColumnFields = columns.size();
081        columns.clear();
082
083        for (ColumnConfig row : getConfig().getRows()) {
084            row.addColumns(columns, parentUri, false);
085        }
086        int lastRowFields = lastColumnFields + columns.size();
087        columns.clear();
088
089        // Webservice URL
090        code.append(
091                "$('#heatmap').heatmap({\n" +
092                        "       data : {\n" +
093                        "                 type : \"tdm\",\n" +
094                        "                 values : '" + getQueryUrl() + "', \n");
095
096        code.append("\t\t\t\tcols_annotations : [");
097        if (config.getColAnnotations() == null) {
098            for (int i = 0; i < lastColumnFields; i++) {
099                if (i != 0) {
100                    code.append(",");
101                }
102                code.append(i);
103            }
104        } else {
105            code.append(config.getColAnnotations());
106        }
107        code.append("], \n");
108        code.append("\t\t\t\trows_annotations : [");
109        if (config.getRowAnnotations() == null) {
110            for (int i = lastColumnFields; i < lastRowFields; i++) {
111                if (i != lastColumnFields) {
112                    code.append(",");
113                }
114                code.append(i);
115            }
116        } else {
117            code.append(config.getRowAnnotations());
118        }
119        code.append("]\n" +
120                "              },\n" +
121                "       init : function(heatmap) {\n");
122        code.append(config.getInit());
123        code.append("}});");
124
125        return code.toString();
126    }
127
128    private String getQueryUrl() {
129        String serviceMount = collectionManager.getMount();
130        String webserviceUrl = WebsiteApplication.toAbsolutePath('/' + serviceMount);
131        Query query = getQuery();
132
133        String strQuery = query.toString();
134        String fileName = "file-" + Integer.toHexString(strQuery.hashCode()) + ".tsv";
135
136        return webserviceUrl + "?query=" + URLEncoder.encode(strQuery) + "&filename=" + fileName;
137    }
138}
