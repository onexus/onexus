018package org.onexus.data.loader.file.internal;
019
020import org.apache.commons.io.FileUtils;
021import org.apache.commons.io.FilenameUtils;
022import org.apache.commons.io.filefilter.IOFileFilter;
023import org.apache.commons.io.filefilter.TrueFileFilter;
024import org.apache.commons.io.filefilter.WildcardFileFilter;
025import org.onexus.data.api.Data;
026import org.onexus.data.api.IDataStreams;
027import org.onexus.data.api.utils.UrlDataStreams;
028import org.onexus.resource.api.Plugin;
029import org.onexus.resource.api.Progress;
030
031import java.io.File;
032import java.net.MalformedURLException;
033import java.net.URL;
034import java.util.ArrayList;
035import java.util.Collection;
036import java.util.List;
037import java.util.concurrent.Callable;
038
039public class FileCallable implements Callable<IDataStreams> {
040
041    private Progress progress;
042    private Plugin plugin;
043    private Data data;
044
045    private boolean compressed;
046
047    public FileCallable(Progress progress, Plugin plugin, Data data) {
048        this.progress = progress;
049        this.plugin = plugin;
050        this.data = data;
051
052        String format = plugin.getParameter("format");
053
054        if (format != null && format.equals("gz")) {
055            compressed = true;
056        } else {
057            compressed = false;
058        }
059    }
060
061    @Override
062    public IDataStreams call() throws Exception {
063        progress.done();
064        return new UrlDataStreams(progress, getUrls(plugin, data), compressed);
065    }
066
067    private List<URL> getUrls(Plugin plugin, Data data) {
068
069        String location = plugin.getParameter("location");
070        String mirror = plugin.getParameter("mirror");
071        String basePath = data.getLoader().getParameter("base-path");
072
073        if (basePath == null) {
074            basePath = "";
075        }
076
077        if (!basePath.isEmpty() && !basePath.endsWith(File.separator)) {
078            basePath = basePath + File.separator;
079        }
080
081
082        List<String> paths = data.getLoader().getParameterList("path");
083
084        List<URL> urls = new ArrayList<URL>();
085
086        for (String templatePath : paths) {
087            String path = templatePath;
088            String fileName = FilenameUtils.getName(path);
089
090            // Check if it is a wildcard filter
091            if (path.contains("*") || path.contains("?")) {
092
093                // Is recursive?
094                IOFileFilter dirFilter = null;
095                if (path.contains("**/")) {
096                    dirFilter = TrueFileFilter.INSTANCE;
097                    path = path.replace("**/", "");
098                }
099
100                String sourceContainer = location + File.separator + basePath + FilenameUtils.getFullPathNoEndSeparator(path);
101                File sourceFile = new File(sourceContainer);
102
103                for (File file : (Collection<File>) FileUtils.listFiles(sourceFile, new WildcardFileFilter(fileName), dirFilter)) {
104                    try {
105                        urls.add(file.toURI().toURL());
106                    } catch (MalformedURLException e) {
107                        throw new RuntimeException(e);
108                    }
109                }
110
111            } else {
112                String sourcePath = location + File.separator + basePath + path;
113                File sourceFile = new File(sourcePath);
114
115                if (sourceFile.exists()) {
116                    try {
117                        urls.add(sourceFile.toURI().toURL());
118                    } catch (MalformedURLException e) {
119                        throw new RuntimeException(e);
120                    }
121                } else {
122
123                    // Try mirror
124                    if (mirror != null) {
125
126                        String remoteFile = mirror + '/' + path;
127
128                        try {
129                            URL url = new URL(remoteFile);
130                            urls.add(url);
131                        } catch (MalformedURLException e) {
132                            throw new RuntimeException(e);
133                        }
134                    }
135                }
136            }
137        }
138
139        return urls;
140    }
141}
