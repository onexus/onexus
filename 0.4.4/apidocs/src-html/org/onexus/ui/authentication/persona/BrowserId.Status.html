018package org.onexus.ui.authentication.persona;
019
020
021import org.apache.wicket.util.string.Strings;
022
023import java.io.Serializable;
024import java.util.Date;
025
026/**
027 * An object representing the response from https://browserid.org
028 */
029public class BrowserId implements Serializable {
030
031    private static final long serialVersionUID = 1L;
032
033    public enum Status {
034        OK, FAIL;
035
036        public static Status parse(final String t) {
037            if ("okay".equals(t)) {
038                return OK;
039            } else if ("failure".equals(t)) {
040                return FAIL;
041            } else {
042                throw new IllegalArgumentException("Unrecognized status: " + t);
043            }
044        }
045    }
046
047    ;
048
049    private Status status;
050
051    private String email;
052
053    private String audience;
054
055    private Date validUntil;
056
057    private String issuer;
058
059    private String reason;
060
061    public String getReason() {
062        return reason;
063    }
064
065    public void setReason(String reason) {
066        this.reason = reason;
067    }
068
069    public Status getStatus() {
070        return status;
071    }
072
073    public void setStatus(Status status) {
074        this.status = status;
075    }
076
077    public String getEmail() {
078        return email;
079    }
080
081    public void setEmail(String email) {
082        this.email = email;
083    }
084
085    public String getAudience() {
086        return audience;
087    }
088
089    public void setAudience(String audience) {
090        this.audience = audience;
091    }
092
093    public Date getValidUntil() {
094        return validUntil;
095    }
096
097    public void setValidUntil(Date validUntil) {
098        this.validUntil = validUntil;
099    }
100
101    public String getIssuer() {
102        return issuer;
103    }
104
105    public void setIssuer(String issuer) {
106        this.issuer = issuer;
107    }
108
109    /**
110     * Create {@link BrowserId} object from success or failure JSON response
111     *
112     * @param json the response returned by https://browserid.org
113     * @return a {@link BrowserId} instance with status OK if the authentication passed, or with
114     *         status FAIL if the authentication failed, or {@code null} if the passed JSON response
115     *         is invalid
116     */
117    public static BrowserId of(String json) {
118        BrowserId browserId = null;
119
120        if (json != null && json.startsWith("{") && json.endsWith("}")) {
121            json = json.substring(1);
122            json = json.substring(0, json.length() - 1);
123
124            String[] settings = Strings.split(json, ',');
125            if (settings.length > 0) {
126                browserId = new BrowserId();
127                for (String setting : settings) {
128                    String[] pair = Strings.split(setting, ':');
129                    pair[0] = Strings.replaceAll(pair[0], "\"", "").toString();
130                    pair[1] = Strings.replaceAll(pair[1], "\"", "").toString();
131
132                    if ("status".equals(pair[0])) {
133                        browserId.status = Status.parse(pair[1]);
134                    } else if ("email".equals(pair[0])) {
135                        browserId.email = pair[1];
136                    } else if ("audience".equals(pair[0])) {
137                        browserId.audience = pair[1];
138                    } else if ("issuer".equals(pair[0])) {
139                        browserId.issuer = pair[1];
140                    } else if ("expires".equals(pair[0])) {
141                        Long millis = Long.valueOf(pair[1]);
142                        browserId.validUntil = new Date(millis);
143                    } else if ("reason".equals(pair[0])) {
144                        browserId.reason = pair[1];
145                    } else {
146                        throw new IllegalArgumentException("Unknown setting: " + pair[0]);
147                    }
148                }
149            }
150        }
151
152        return browserId;
153    }
154}
