037import org.apache.wicket.RestartResponseException;
038import org.apache.wicket.authentication.IAuthenticationStrategy;
039import org.apache.wicket.markup.html.WebMarkupContainer;
040import org.apache.wicket.markup.html.form.CheckBox;
041import org.apache.wicket.markup.html.form.PasswordTextField;
042import org.apache.wicket.markup.html.form.StatelessForm;
043import org.apache.wicket.markup.html.form.TextField;
044import org.apache.wicket.markup.html.panel.FeedbackPanel;
045import org.apache.wicket.markup.html.panel.Panel;
046import org.apache.wicket.model.CompoundPropertyModel;
047
048
049/**
050 * Reusable user sign in panel with username and password as well as support for persistence of the
051 * both. When the SignInPanel's form is submitted, the method signIn(String, String) is called,
052 * passing the username and password submitted. The signIn() method should authenticate the user's
053 * session.
054 *
055 * @author Jonathan Locke
056 * @author Juergen Donnerstag
057 * @author Eelco Hillenius
058 * @see {@link org.apache.wicket.authentication.IAuthenticationStrategy}
059 * @see {@link org.apache.wicket.settings.ISecuritySettings#getAuthenticationStrategy()}
060 * @see {@link org.apache.wicket.authentication.strategy.DefaultAuthenticationStrategy}
061 */
062public class SignInPanel extends Panel {
063
064    private static final long serialVersionUID = 1L;
065    private static final String SIGN_IN_FORM = "signInForm";
066
067    /**
068     * True if the panel should display a remember-me checkbox
069     */
070    private boolean includeRememberMe = true;
071
072    /**
073     * True if the user should be remembered via form persistence (cookies)
074     */
075    private boolean rememberMe = true;
076
077    /**
078     * password.
079     */
080    private String password;
081
082    /**
083     * user name.
084     */
085    private String username;
086
087    /**
088     * @see org.apache.wicket.Component#Component(String)
089     */
090    public SignInPanel(final String id) {
091        this(id, true);
092    }
093
094    /**
095     * @param id                See Component constructor
096     * @param includeRememberMe True if form should include a remember-me checkbox
097     * @see org.apache.wicket.Component#Component(String)
098     */
099    public SignInPanel(final String id, final boolean includeRememberMe) {
100        super(id);
101
102        this.includeRememberMe = includeRememberMe;
103
104        // Create feedback panel and add to page
105        add(new FeedbackPanel("feedback"));
106
107        // Add sign-in form to page, passing feedback panel as
108        // validation error handler
109        add(new SignInForm(SIGN_IN_FORM));
110    }
111
112    /**
113     * @return signin form
114     */
115    protected SignInForm getForm() {
116        return (SignInForm) get(SIGN_IN_FORM);
117    }
118
119    /**
120     * @see org.apache.wicket.Component#onBeforeRender()
121     */
122    @Override
123    protected void onBeforeRender() {
124        // logged in already?
125        if (!isSignedIn()) {
126            IAuthenticationStrategy authenticationStrategy = getApplication().getSecuritySettings()
127                    .getAuthenticationStrategy();
128            // get username and password from persistence store
129            String[] data = authenticationStrategy.load();
130
131            if (data!=null && data.length>1) {
132                // try to sign in the user
133                if (signIn(data[0], data[1])) {
134                    username = data[0];
135                    password = data[1];
136
137                    // logon successful. Continue to the original destination
138                    continueToOriginalDestination();
139                    // Ups, no original destination. Go to the home page
140                    throw new RestartResponseException(getSession().getPageFactory().newPage(
141                            getApplication().getHomePage()));
142                } else {
143                    // the loaded credentials are wrong. erase them.
144                    authenticationStrategy.remove();
145                }
146            }
147        }
148
149        // don't forget
150        super.onBeforeRender();
151    }
152
153    /**
154     * Convenience method to access the password.
155     *
156     * @return The password
157     */
158    public String getPassword() {
159        return password;
160    }
161
162    /**
163     * Set the password
164     *
165     * @param password
166     */
167    public void setPassword(final String password) {
168        this.password = password;
169    }
170
171    /**
172     * Convenience method to access the username.
173     *
174     * @return The user name
175     */
176    public String getUsername() {
177        return username;
178    }
179
180    /**
181     * Set the username
182     *
183     * @param username
184     */
185    public void setUsername(final String username) {
186        this.username = username;
187    }
188
189    /**
190     * Get model object of the rememberMe checkbox
191     *
192     * @return True if user should be remembered in the future
193     */
194    public boolean getRememberMe() {
195        return rememberMe;
196    }
197
198    /**
199     * @param rememberMe If true, rememberMe will be enabled (username and password will be persisted
200     *                   somewhere)
201     */
202    public void setRememberMe(final boolean rememberMe) {
203        this.rememberMe = rememberMe;
204    }
205
206    /**
207     * Sign in user if possible.
208     *
209     * @param username The username
210     * @param password The password
211     * @return True if signin was successful
212     */
213    private boolean signIn(String username, String password) {
214        return JaasSignInPage.getAuthenticatedSession().authenticate(username, password);
215    }
216
217    /**
218     * @return true, if signed in
219     */
220    private boolean isSignedIn() {
221        return JaasSignInPage.getAuthenticatedSession().isSignedIn();
222    }
223
224    /**
225     * Called when sign in failed
226     */
227    protected void onSignInFailed() {
228        // Try the component based localizer first. If not found try the
229        // application localizer. Else use the default
230        error(getLocalizer().getString("signInFailed", this, "Sign in failed"));
231    }
232
233    /**
234     * Called when sign in was successful
235     */
236    protected void onSignInSucceeded() {
237        // If login has been called because the user was not yet logged in, than continue to the
238        // original destination, otherwise to the Home page
239        continueToOriginalDestination();
240        setResponsePage(getApplication().getHomePage());
241    }
242
243    /**
244     * Sign in form.
245     */
246    public final class SignInForm extends StatelessForm<SignInPanel> {
247        private static final long serialVersionUID = 1L;
248
249        /**
250         * Constructor.
251         *
252         * @param id id of the form component
253         */
254        public SignInForm(final String id) {
255            super(id);
256
257            setModel(new CompoundPropertyModel<SignInPanel>(SignInPanel.this));
258
259            // Attach textfields for username and password
260            add(new TextField<String>("username"));
261            add(new PasswordTextField("password"));
262
263            // MarkupContainer row for remember me checkbox
264            WebMarkupContainer rememberMeRow = new WebMarkupContainer("rememberMeRow");
265            add(rememberMeRow);
266
267            // Add rememberMe checkbox
268            rememberMeRow.add(new CheckBox("rememberMe"));
269
270            // Show remember me checkbox?
271            rememberMeRow.setVisible(includeRememberMe);
272        }
273
274        /**
275         * @see org.apache.wicket.markup.html.form.Form#onSubmit()
276         */
277        @Override
278        public void onSubmit() {
279            IAuthenticationStrategy strategy = getApplication().getSecuritySettings()
280                    .getAuthenticationStrategy();
281
282            if (signIn(getUsername(), getPassword())) {
283                if (rememberMe) {
284                    strategy.save(username, password);
285                } else {
286                    strategy.remove();
287                }
288
289                onSignInSucceeded();
290            } else {
291                onSignInFailed();
292                strategy.remove();
293            }
294        }
295    }
296}
