018package org.onexus.ui.website.internal.wizards;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.extensions.wizard.FinishButton;
022import org.apache.wicket.extensions.wizard.IDefaultButtonProvider;
023import org.apache.wicket.extensions.wizard.IWizard;
024import org.apache.wicket.extensions.wizard.IWizardStep;
025import org.apache.wicket.extensions.wizard.Wizard;
026import org.apache.wicket.extensions.wizard.WizardButtonBar;
027import org.apache.wicket.markup.html.form.Form;
028import org.apache.wicket.model.CompoundPropertyModel;
029
030public abstract class AbstractWizard extends Wizard {
031
032    public AbstractWizard(String id) {
033        super(id);
034    }
035
036    @Override
037    public void onCancel() {
038        setVisible(false);
039    }
040
041    @Override
042    public void onFinish() {
043        setVisible(false);
044    }
045
046
047    @Override
048    protected Component newButtonBar(String id) {
049        return new MyWizardButtonBar(id, this);
050    }
051
052
053    @Override
054    protected Component newFeedbackPanel(String id) {
055        return super.newFeedbackPanel(id).setOutputMarkupId(true);
056    }
057
058    @Override
059    @SuppressWarnings("unchecked")
060    protected Form<AbstractWizard> newForm(String id) {
061
062        Form<AbstractWizard> form = new Form<AbstractWizard>(id, new CompoundPropertyModel<AbstractWizard>(this));
063        form.setOutputMarkupId(true);
064
065        return form;
066
067    }
068
069
070    /**
071     * This is a hack, because we expect that the finish button is enabled only
072     * if the last step is completed. By default, finish button is always
073     * enabled on the last step.
074     *
075     * @author Jordi Deu-Pons
076     */
077    public static final class MyWizardButtonBar extends WizardButtonBar implements IDefaultButtonProvider {
078
079        public MyWizardButtonBar(String id, IWizard wizard) {
080            super(id, wizard);
081            replace(new FinishButton("finish", wizard) {
082
083                @Override
084                public boolean isEnabled() {
085                    IWizardStep activeStep = getWizardModel().getActiveStep();
086                    return activeStep!=null && getWizardModel().isLastStep(activeStep) && activeStep.isComplete();
087                }
088
089            });
090        }
091
092    }
093
094}
