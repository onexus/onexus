018package org.onexus.ui.api.progressbar;
019
020import org.apache.wicket.MetaDataKey;
021import org.apache.wicket.Session;
022import org.onexus.resource.api.Progress;
023
024import java.io.Serializable;
025import java.util.Collection;
026import java.util.HashSet;
027import java.util.Iterator;
028import java.util.Set;
029
030public class ProgressExpansion implements Set<Progress>, Serializable {
031    private static final long serialVersionUID = 1L;
032
033    private static final MetaDataKey<ProgressExpansion> KEY = new MetaDataKey<ProgressExpansion>() {
034        private static final long serialVersionUID = 1L;
035    };
036
037    private Set<String> ids = new HashSet<String>();
038
039    private boolean inverse = false;
040
041    public void expandAll() {
042        ids.clear();
043
044        inverse = true;
045    }
046
047    public void collapseAll() {
048        ids.clear();
049
050        inverse = false;
051    }
052
053    @Override
054    public boolean add(Progress foo) {
055        if (inverse) {
056            return ids.remove(foo.getId());
057        } else {
058            return ids.add(foo.getId());
059        }
060    }
061
062    @Override
063    public boolean remove(Object o) {
064        Progress foo = (Progress) o;
065
066        if (inverse) {
067            return ids.add(foo.getId());
068        } else {
069            return ids.remove(foo.getId());
070        }
071    }
072
073    @Override
074    public boolean contains(Object o) {
075        Progress foo = (Progress) o;
076
077        if (inverse) {
078            return !ids.contains(foo.getId());
079        } else {
080            return ids.contains(foo.getId());
081        }
082    }
083
084    @Override
085    public void clear() {
086        throw new UnsupportedOperationException();
087    }
088
089    @Override
090    public int size() {
091        throw new UnsupportedOperationException();
092    }
093
094    @Override
095    public boolean isEmpty() {
096        throw new UnsupportedOperationException();
097    }
098
099    @Override
100    public <A> A[] toArray(A[] a) {
101        throw new UnsupportedOperationException();
102    }
103
104    @Override
105    public Iterator<Progress> iterator() {
106        throw new UnsupportedOperationException();
107    }
108
109    @Override
110    public Object[] toArray() {
111        throw new UnsupportedOperationException();
112    }
113
114    @Override
115    public boolean containsAll(Collection<?> c) {
116        throw new UnsupportedOperationException();
117    }
118
119    @Override
120    public boolean addAll(Collection<? extends Progress> c) {
121        throw new UnsupportedOperationException();
122    }
123
124    @Override
125    public boolean retainAll(Collection<?> c) {
126        throw new UnsupportedOperationException();
127    }
128
129    @Override
130    public boolean removeAll(Collection<?> c) {
131        throw new UnsupportedOperationException();
132    }
133
134    /**
135     * Get the expansion for the session.
136     *
137     * @return expansion
138     */
139    public static ProgressExpansion get() {
140        ProgressExpansion expansion = Session.get().getMetaData(KEY);
141        if (expansion == null) {
142            expansion = new ProgressExpansion();
143
144            Session.get().setMetaData(KEY, expansion);
145        }
146        return expansion;
147    }
148
149}
