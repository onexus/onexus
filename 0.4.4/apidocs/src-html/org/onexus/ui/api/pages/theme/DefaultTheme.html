018package org.onexus.ui.api.pages.theme;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.behavior.Behavior;
022import org.apache.wicket.markup.head.CssHeaderItem;
023import org.apache.wicket.markup.head.IHeaderResponse;
024import org.apache.wicket.markup.head.JavaScriptHeaderItem;
025import org.apache.wicket.request.resource.CssResourceReference;
026import org.apache.wicket.resource.JQueryPluginResourceReference;
027
028public class DefaultTheme extends Behavior {
029
030    public static final CssResourceReference BOOTSTRAP_CSS = new CssResourceReference(DefaultTheme.class, "css/bootstrap.min.css");
031    public static final JQueryPluginResourceReference BOOTSTRAP_JS = new JQueryPluginResourceReference(DefaultTheme.class, "js/bootstrap.min.js");
032    public static final CssResourceReference STYLE_CSS = new CssResourceReference(DefaultTheme.class, "css/style.css");
033
034    @Override
035    public void renderHead(Component component, IHeaderResponse response) {
036
037        response.render(CssHeaderItem.forReference(BOOTSTRAP_CSS));
038        response.render(CssHeaderItem.forReference(STYLE_CSS));
039
040        response.render(JavaScriptHeaderItem.forReference(BOOTSTRAP_JS));
041        response.render(JavaScriptHeaderItem.forScript("     $(document).ready(function () {\n" +
042                "        $(\"[rel=tooltip]\").tooltip({ placement: 'bottom'});\n" +
043                "        $(\"div.modal\").modal({ show: false }); \n" +
044                "        });\n" +
045                "   ", "bootstrap-init"));
046
047    }
048}
