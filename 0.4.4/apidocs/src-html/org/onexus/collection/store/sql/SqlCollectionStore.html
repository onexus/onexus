042public abstract class SqlCollectionStore implements ICollectionStore {
043    private static final Logger LOGGER = LoggerFactory
044            .getLogger(SqlCollectionStore.class);
045
046    private static final int INSERT_BATCH_SIZE = 400;
047    private static final int INSERT_BUFFER_SIZE = INSERT_BATCH_SIZE * 400;
048
049    private DataSource dataSource;
050
051    private Map<ORI, SqlCollectionDDL> ddls;
052
053    private SqlDialect sqlDialect;
054
055    private Map<String, String> propertiesMap;
056
057    public SqlCollectionStore() {
058        this(new SqlDialect());
059    }
060
061    public SqlCollectionStore(SqlDialect sqlDialect) {
062        super();
063
064        this.sqlDialect = sqlDialect;
065    }
066
067    public void init() {
068
069        dataSource = newDataSource();
070
071        // Create the store properties table if it's necessary
072        try {
073            sqlDialect.createSystemPropertiesTable(dataSource.getConnection());
074        } catch (SQLException e) {
075            throw new RuntimeException(e);
076        }
077
078        // Load all properties
079        try {
080            propertiesMap = sqlDialect.loadPropertyMap(dataSource.getConnection());
081        } catch (Exception e) {
082            throw new RuntimeException(e);
083        }
084
085        this.ddls = new HashMap<ORI, SqlCollectionDDL>();
086    }
087
088    protected abstract DataSource newDataSource();
089
090    @Override
091    public boolean isRegistered(ORI collectionURI) {
092
093        String tableName = getProperty(collectionURI.toString());
094        return tableName != null;
095
096    }
097
098    private String getProperty(String propertyKey) {
099        synchronized (propertiesMap) {
100            return propertiesMap.get(propertyKey);
101        }
102    }
103
104
105    @Override
106    public void register(ORI collectionURI) {
107        LOGGER.debug("Registering collection {}", collectionURI);
108
109        // Rebuild always the DDL before registering
110        ddls.put(collectionURI, new SqlCollectionDDL(sqlDialect, getCollection(collectionURI), getProperty(collectionURI.toString())));
111
112        SqlCollectionDDL ddl = getDDL(collectionURI);
113
114        Connection conn = null;
115        try {
116            synchronized (propertiesMap) {
117                conn = dataSource.getConnection();
118
119                String dropTable = ddl.getDropTable();
120                LOGGER.debug(dropTable);
121                try {
122                    sqlDialect.execute(conn, dropTable);
123                } catch (Exception e) {
124                    LOGGER.debug("Error droping table '" + dropTable + "' at register()", e);
125                }
126
127                List<String> dropIndex = ddl.getDropIndex();
128                for (String indexSQL : dropIndex) {
129                    LOGGER.debug(indexSQL);
130                    try {
131                        sqlDialect.execute(conn, indexSQL);
132                    } catch (Exception e) {
133                        LOGGER.debug("Error creating index for table '" + dropTable + "' at register()", e);
134                    }
135                }
136
137
138                String createTable = ddl.getCreateTable();
139                LOGGER.debug(createTable);
140                sqlDialect.execute(conn, createTable);
141
142                List<String> createIndex = ddl.getCreateIndex();
143                for (String indexSQL : createIndex) {
144                    LOGGER.debug(indexSQL);
145                    sqlDialect.execute(conn, indexSQL);
146                }
147
148                sqlDialect.createSystemPropertiesTable(conn, false);
149                sqlDialect.saveProperty(conn, collectionURI.toString(), ddl.getTableName());
150                propertiesMap.put(collectionURI.toString(), ddl.getTableName());
151            }
152
153
154        } catch (Exception e) {
155            String msg = String.format(
156                    "Registering collection '%s' with SQL create table '%s'",
157                    collectionURI, ddl.getCreateTable());
158            LOGGER.error(msg, e);
159            throw new RuntimeException(msg, e);
160        } finally {
161            if (conn != null) {
162                try {
163                    conn.close();
164                } catch (SQLException e) {
165                    LOGGER.error("Closing connection", e);
166                }
167            }
168        }
169
170    }
171
172    @Override
173    public void deregister(ORI collectionURI) {
174        LOGGER.debug("Unregistering collection {}", collectionURI);
175
176        Connection conn = null;
177        String tableName = null;
178        try {
179
180            synchronized (propertiesMap) {
181                conn = dataSource.getConnection();
182
183                tableName = sqlDialect.loadProperty(conn, collectionURI.toString());
184                sqlDialect.removeProperty(conn, collectionURI.toString());
185
186                sqlDialect.execute(conn, "DROP TABLE `" + tableName + "`");
187                propertiesMap.remove(collectionURI.toString());
188            }
189
190        } catch (Exception e) {
191            String msg = String.format(
192                    "Unregistering collection '%s' with table '%s'",
193                    collectionURI, tableName);
194            LOGGER.error(msg, e);
195            throw new RuntimeException(msg, e);
196        } finally {
197            if (conn != null) {
198                try {
199                    conn.close();
200                } catch (SQLException e) {
201                    LOGGER.error("Closing connection", e);
202                }
203            }
204        }
205
206    }

209    @Override
210    public List<String> getRegistered() {
211        Connection conn = null;
212        try {
213            conn = dataSource.getConnection();
214            return sqlDialect.loadPropertyKeys(conn);
215        } catch (Exception e) {
216            LOGGER.error("Error getRegistered()", e);
217            return Collections.EMPTY_LIST;
218        } finally {
219            if (conn != null) {
220                try {
221                    conn.close();
222                } catch (SQLException e) {
223                    LOGGER.error("Closing connection", e);
224                }
225            }
226        }

230    @Override
231    public IEntityTable load(Query query) {
232        LOGGER.debug("Loading query\n --------------------------------------------------------\n {} \n --------------------------------------------------------\n", query);
233
234        try {
235
236            SqlEntityTable entitySet = new SqlEntityTable(this, query,
237                    dataSource.getConnection());
238            return entitySet;
239
240        } catch (SQLException e) {
241            String msg = "Error loading query '" + query + "'";
242            LOGGER.error(msg, e);
243            throw new RuntimeException(msg, e);
244        }
245    }

247    @Override
248    public void insert(IEntitySet entitySet) {
249
250        if (!isRegistered(entitySet.getCollection().getORI())) {
251            throw new RuntimeException("The collection '"
252                    + entitySet.getCollection().getORI()
253                    + "' is NOT registered in this collection store.");
254        }
255
256        SqlCollectionDDL ddl = getDDL(entitySet.getCollection().getORI());
257
258        StringBuilder sql = new StringBuilder(INSERT_BUFFER_SIZE);
259
260        sqlDialect.openInsert(sql, ddl);
261
262        Connection conn = null;
263        try {
264            int batch = 0;
265            conn = dataSource.getConnection();
266            while (entitySet.next()) {
267
268                // Skip if the primary key is null
269                boolean allPrimaryKeysNotNull = true;
270                for (SqlCollectionDDL.ColumnInfo columnInfo : ddl.getColumnInfos()) {
271                    Field field = columnInfo.getField();
272                    if (field.isPrimaryKey() != null
273                            && field.isPrimaryKey()
274                            && entitySet.get(field.getId()) == null) {
275                        allPrimaryKeysNotNull = false;
276                        break;
277                    }
278                }
279
280                if (!allPrimaryKeysNotNull) {
281                    LOGGER.warn("Primary key value is NULL. Skipping row " + entitySet.toString());
282                    continue;
283                }
284
285                if (batch > 0) {
286                    sql.append(",");
287                }
288                sqlDialect.addValues(sql, ddl, entitySet);
289                if (batch < INSERT_BATCH_SIZE) {
290                    batch++;
291                } else {
292                    sqlDialect.execute(conn, sql.toString());
293                    sql = new StringBuilder(INSERT_BUFFER_SIZE);
294                    sqlDialect.openInsert(sql, ddl);
295                    batch = 0;
296                }
297            }
298
299            if (batch > 0) {
300                sqlDialect.execute(conn, sql.toString());
301                sql = null;
302            }
303        } catch (SQLException e) {
304            throw new RuntimeException(e);
305        } finally {
306            if (conn != null) {
307                try {
308                    conn.close();
309                } catch (SQLException e) {
310                    LOGGER.error("Closing connection", e);
311                }
312            }
313        }
314
315    }
316
317    @Override
318    public void insert(IEntity entity) {
319
320        if (!isRegistered(entity.getCollection().getORI())) {
321            throw new RuntimeException("The collection '"
322                    + entity.getCollection().getORI()
323                    + "' is NOT registered in this collection store.");
324        }
325
326        StringBuilder sql = new StringBuilder();
327
328        SqlCollectionDDL ddl = getDDL(entity.getCollection().getORI());
329
330        // Open insert
331        sqlDialect.openInsert(sql, ddl);
332        sqlDialect.addValues(sql, ddl, entity);
333
334        Connection conn = null;
335        try {
336            conn = dataSource.getConnection();
337            sqlDialect.execute(conn, sql.toString());
338        } catch (SQLException e) {
339            String msg = "Error inserting entity: '" + entity + "'";
340            LOGGER.error(msg, e);
341            throw new RuntimeException(msg, e);
342        } finally {
343            if (conn != null) {
344                try {
345                    conn.close();
346                } catch (SQLException e) {
347                    LOGGER.error("Closing connection", e);
348                }
349            }
350        }
351    }
352
353    public Collection getCollection(ORI collectionURI) {
354        return getResourceManager().load(Collection.class, collectionURI);
355    }
356
357    public SqlCollectionDDL getDDL(ORI collectionURI) {
358        if (!ddls.containsKey(collectionURI)) {
359            ddls.put(collectionURI, new SqlCollectionDDL(sqlDialect,
360                    getCollection(collectionURI), getProperty(collectionURI.toString())));
361        }
362
363        return ddls.get(collectionURI);
364    }
365
366    public abstract IResourceManager getResourceManager();
367
368    protected DataSource getDataSource() {
369        return this.dataSource;
370    }
371
372    public SqlDialect getSqlDialect() {
373        return sqlDialect;
374    }
375
376    public Statement createReadStatement(Connection dataConn) throws SQLException {
377        return dataConn.createStatement(ResultSet.TYPE_FORWARD_ONLY,
378                ResultSet.CONCUR_READ_ONLY);
379    }
380
381    public SqlQuery newSqlQuery(Query query) {
382        return new SqlQuery(this, query);
383    }
