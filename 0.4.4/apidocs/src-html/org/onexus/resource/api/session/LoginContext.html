018package org.onexus.resource.api.session;
019
020import java.io.Serializable;
021import java.util.HashMap;
022import java.util.HashSet;
023import java.util.Map;
024import java.util.Set;
025
026public class LoginContext implements Serializable {
027
028    private static Map<String, LoginContext> contexts = new HashMap<String, LoginContext>();
029
030    public static LoginContext ANONYMOUS_CONTEXT = new LoginContext();
031    public static LoginContext SERVICE_CONTEXT = new LoginContext("service");
032
033    private static ThreadLocal<LoginContext> loginContext = new ThreadLocal<LoginContext>() {
034        @Override
035        protected LoginContext initialValue() {
036            return ANONYMOUS_CONTEXT;
037        }
038    };
039
040    public static LoginContext get() {
041        return loginContext.get();
042    }
043
044    public static LoginContext get(String sessionId) {
045        return contexts.get(sessionId);
046    }
047
048    public static void set(LoginContext ctx, String sessionId) {
049        loginContext.set(ctx);
050
051        if (sessionId != null) {
052            contexts.put(sessionId, ctx);
053        }
054    }
055
056    private String userName;
057    private Set<String> roles = new HashSet<String>();
058
059    public LoginContext() {
060        this.userName = null;
061    }
062
063    public LoginContext(String userName) {
064        this.userName = userName;
065    }
066
067    public void logout() {
068        this.userName = null;
069    }
070
071    public boolean isAnonymous() {
072        return userName==null;
073    }
074
075    public String getUserName() {
076        return userName;
077    }
078
079    public Set<String> getRoles() {
080        return roles;
081    }
082
083    public void addRole(String role) {
084        roles.add(role);
085    }
086
087    /**
088     * Whether this context object containes the provided role.
089     *
090     * @param role the role to check
091     * @return true if it contains the role, false otherwise
092     */
093    public boolean hasRole(final String role) {
094        if (userName != null && role != null) {
095            return roles.contains(role);
096        }
097        return false;
098    }
099
100
101}
