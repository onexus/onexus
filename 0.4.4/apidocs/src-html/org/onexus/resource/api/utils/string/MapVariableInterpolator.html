018package org.onexus.resource.api.utils.string;
019
020import java.util.Map;
021
022/**
023 * Original code extract from Apache Wicket core *
024 */
025public class MapVariableInterpolator extends VariableInterpolator {
026    /**
027     * Map of variables
028     */
029    private Map<?, ?> variables;
030
031    /**
032     * Constructor.
033     *
034     * @param string    a <code>String</code> to interpolate into
035     * @param variables the variables to substitute
036     */
037    public MapVariableInterpolator(final String string, final Map<?, ?> variables) {
038        super(string);
039        this.variables = variables;
040    }
041
042    /**
043     * Constructor.
044     *
045     * @param string                  a <code>String</code> to interpolate into
046     * @param variables               the variables to substitute
047     * @param exceptionOnNullVarValue if <code>true</code> an {@link IllegalStateException} will be thrown if
048     *                                {@link #getValue(String)} returns <code>null</code>, otherwise the
049     *                                <code>${varname}</code> string will be left in the <code>String</code> so that
050     *                                multiple interpolators can be chained
051     */
052    public MapVariableInterpolator(final String string, final Map<?, ?> variables,
053                                   final boolean exceptionOnNullVarValue) {
054        super(string, exceptionOnNullVarValue);
055        this.variables = variables;
056    }
057
058    /**
059     * Sets the <code>Map</code> of variables.
060     *
061     * @param variables the <code>Map</code> of variables
062     */
063    public final void setVariables(final Map<?, ?> variables) {
064        this.variables = variables;
065    }
066
067    /**
068     * Retrieves a value for a variable name during interpolation.
069     *
070     * @param variableName the variable name
071     * @return the value
072     */
073    @Override
074    protected String getValue(final String variableName) {
075        Object value = variables.get(variableName);
076
077        if (value == null) {
078            return null;
079        }
080
081        return value.toString();
082    }
083
084    /**
085     * Interpolates a <code>String</code> with the arguments defined in the given <code>Map</code>.
086     *
087     * @param string    a <code>String</code> to interpolate into
088     * @param variables the variables to substitute
089     * @return the interpolated <code>String</code>
090     */
091    public static String interpolate(final String string, final Map<?, ?> variables) {
092        return new MapVariableInterpolator(string, variables).toString();
093    }
094
095}
