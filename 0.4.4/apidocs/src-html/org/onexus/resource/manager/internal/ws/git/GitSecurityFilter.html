020import org.eclipse.jgit.util.Base64;
021import org.eclipse.jgit.util.StringUtils;
022import org.onexus.resource.api.IAuthorizationManager;
023import org.onexus.resource.api.IProfileManager;
024import org.onexus.resource.api.IResourceManager;

031import javax.servlet.Filter;
032import javax.servlet.FilterChain;
033import javax.servlet.FilterConfig;
034import javax.servlet.ServletException;
035import javax.servlet.ServletRequest;
036import javax.servlet.ServletResponse;
037import javax.servlet.http.HttpServletRequest;

043
044public class GitSecurityFilter implements Filter {
045
046    private static final Logger LOGGER = LoggerFactory.getLogger(GitSecurityFilter.class);
047
048    private static final String BASIC = "Basic";
049    private static final String CHALLENGE = BASIC + " realm=\"karaf\"";
050
051    private static final String gitReceivePack = "/git-receive-pack";
052    private static final String gitUploadPack = "/git-upload-pack";
053    private static final String[] suffixes = {gitReceivePack, gitUploadPack, "/info/refs", "/HEAD", "/objects"};
054
055    private IAuthorizationManager authorizationManager;
056    private IResourceManager resourceManager;
057    private IProfileManager profileManager;
058
059
060    @Override
061    public void init(FilterConfig filterConfig) throws ServletException {
062        LOGGER.debug("FilterConfig: " + filterConfig);
063    }
064
065    @Override
066    public void doFilter(final ServletRequest request, final ServletResponse response,
067                         final FilterChain chain) throws IOException, ServletException {
068
069        HttpServletRequest httpRequest = (HttpServletRequest) request;
070        HttpServletResponse httpResponse = (HttpServletResponse) response;
071
072        String fullUrl = getFullUrl(httpRequest);
073        String repository = extractRepositoryName(fullUrl);
074        String fullSuffix = fullUrl.substring(repository.length());
075        String requestPrivilege = getUrlRequestAction(fullSuffix);
076        String user = getUser(httpRequest);
077
078        if (IAuthorizationManager.WRITE.equals(requestPrivilege) && user == null) {
079            httpResponse.setHeader("WWW-Authenticate", CHALLENGE);
080            httpResponse.sendError(HttpServletResponse.SC_UNAUTHORIZED);
081            return;
082        }
083
084        // Find a project with this name
085        ORI resource = null;
086        for (Project project : resourceManager.getProjects()) {
087            if (repository.equals(project.getName())) {
088                resource = project.getORI();
089                break;
090            }
091        }
092
093        if (resource == null) {
094            httpResponse.sendError(HttpServletResponse.SC_NOT_FOUND);
095            return;
096        }
097
098        if (!authorizationManager.check(requestPrivilege, resource) && user == null) {
099            httpResponse.setHeader("WWW-Authenticate", CHALLENGE);
100            httpResponse.sendError(HttpServletResponse.SC_UNAUTHORIZED);
101            return;
102        }
103
104        Project project = null;
105        try {
106            project = resourceManager.getProject(resource.getProjectUrl());
107        } catch (Exception e) {
108            LOGGER.error("At git serving '" + resource + "'", e);
109        }
110
111        //TODO automatic project creation on push
112        // if (project == null && requestPrivilege.equals(IAuthorizationManager.WRITE)) {
113        //
114        // }
115
116        if (project == null) {
117            httpResponse.sendError(HttpServletResponse.SC_NOT_FOUND);
118            return;
119        }
120
121        chain.doFilter(new AuthenticatedRequest(user, httpRequest), httpResponse);
122
123    }
124
125    protected String getUser(HttpServletRequest httpRequest) {
126
127        final String authorization = httpRequest.getHeader("Authorization");
128
129        if (authorization != null && authorization.startsWith(BASIC)) {
130
131            // Authorization: Basic base64credentials
132            String base64Credentials = authorization.substring(BASIC.length()).trim();
133            String credentials = new String(Base64.decode(base64Credentials), Charset.forName("UTF-8"));
134
135            // credentials = username:password
136            final String[] values = credentials.split(":", 2);
137
138            if (values.length == 2) {
139                String username = values[0];
140                String password = values[1];
141
142                LoginContext.set(new LoginContext(username), httpRequest.getRequestedSessionId());
143
144                boolean valid = false;
145                String[] tokens = profileManager.getValueArray("tokens");
146                if (tokens != null) {
147                    for (String token : tokens) {
148                        if (password.equals(token)) {
149                            valid = true;
150                            break;
151                        }
152                    }
153                }
154
155                if (valid) {
156                    return username;
157                }
158
159                LoginContext.get().logout();
160            }
161            LOGGER.info(MessageFormat.format("AUTH: invalid credentials ({0})", credentials));
162        }
163        return null;
164    }
165
166    @Override
167    public void destroy() {
168    }
169
170    private String getFullUrl(HttpServletRequest httpRequest) {
171        String servletUrl = httpRequest.getContextPath() + httpRequest.getServletPath();
172        String url = httpRequest.getRequestURI().substring(servletUrl.length());
173        String params = httpRequest.getQueryString();
174        if (url.length() > 0 && url.charAt(0) == '/') {
175            url = url.substring(1);
176        }
177
178        return url + (StringUtils.isEmptyOrNull(params) ? "" : "?" + params);
179    }
180
181    private String extractRepositoryName(String url) {
182        String repository = url;
183        // get the repository name from the url by finding a known url suffix
184        for (String urlSuffix : suffixes) {
185            if (repository.indexOf(urlSuffix) > -1) {
186                repository = repository.substring(0, repository.indexOf(urlSuffix));
187            }
188        }
189        return repository;
190    }
191
192    private String getUrlRequestAction(String suffix) {
193        if (!StringUtils.isEmptyOrNull(suffix)) {
194            if (suffix.startsWith(gitReceivePack)) {
195                return IAuthorizationManager.WRITE;
196            } else if (suffix.startsWith(gitUploadPack)) {
197                return IAuthorizationManager.READ;
198            } else if (suffix.contains("?service=git-receive-pack")) {
199                return IAuthorizationManager.WRITE;
200            } else if (suffix.contains("?service=git-upload-pack")) {
201                return IAuthorizationManager.READ;
202            } else {
203                return IAuthorizationManager.READ;
204            }
205        }
206        return null;
207    }
208

210        return authorizationManager;
211    }

214        this.authorizationManager = authorizationManager;
215    }

218        return resourceManager;
219    }
220

222        this.resourceManager = resourceManager;
223    }
224
225    public IProfileManager getProfileManager() {
226        return profileManager;
227    }
228
229    public void setProfileManager(IProfileManager profileManager) {
230        this.profileManager = profileManager;
231    }

233    /**
234     * Wraps a standard HttpServletRequest and overrides user principal methods.
235     */
236    public static class AuthenticatedRequest extends ServletRequestWrapper {
237
238        private String user;
239
240        public AuthenticatedRequest(String user, HttpServletRequest req) {
241            super(req);
242            this.user = user;
243        }
244
245        @Override
246        public String getRemoteUser() {
247            return user;
248        }
249
250        @Override
251        public Principal getUserPrincipal() {
252            return new Principal() {
253                @Override
254                public String getName() {
255                    return user;
256                }
257            };
258        }
259    }
