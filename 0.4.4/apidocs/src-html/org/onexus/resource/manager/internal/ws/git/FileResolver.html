001package org.onexus.resource.manager.internal.ws.git;
002
003import org.eclipse.jgit.errors.RepositoryNotFoundException;
004import org.eclipse.jgit.lib.Constants;
005import org.eclipse.jgit.lib.Repository;
006import org.eclipse.jgit.lib.RepositoryCache;
007import org.eclipse.jgit.lib.RepositoryCache.FileKey;
008import org.eclipse.jgit.transport.resolver.RepositoryResolver;
009import org.eclipse.jgit.transport.resolver.ServiceNotEnabledException;
010import org.eclipse.jgit.util.FS;
011import org.onexus.resource.api.IResourceManager;
012import org.onexus.resource.api.ORI;
013import org.onexus.resource.api.Project;
014import org.onexus.resource.manager.internal.ResourceManager;
015import org.onexus.resource.manager.internal.providers.AbstractProjectProvider;
016import org.onexus.resource.manager.internal.providers.GitProjectProvider;
017
018import java.io.File;
019import java.io.IOException;
020import java.util.Map;
021import java.util.concurrent.ConcurrentHashMap;
022
023/**
024 * Default resolver serving from the local filesystem.
025 *
026 * @param <C>
027 *            type of connection
028 */
029public class FileResolver<C> implements RepositoryResolver<C> {
030
031    private ResourceManager resourceManager;
032    private final Map<String, Repository> exports;
033
034    /** Initialize an empty file based resolver. */
035    public FileResolver(ResourceManager resourceManager) {
036        this.resourceManager = resourceManager;
037        this.exports = new ConcurrentHashMap<String, Repository>();
038    }
039
040    public Repository open(final C req, final String name)
041            throws RepositoryNotFoundException, ServiceNotEnabledException {
042        if (isUnreasonableName(name))
043            throw new RepositoryNotFoundException(name);
044
045        Repository db = exports.get(nameWithDotGit(name));
046        if (db != null) {
047            db.incrementOpen();
048            return db;
049        }
050
051        ORI projectORI = null;
052        for (Project project : resourceManager.getProjects()) {
053            if (project.getName().equals(name)) {
054                projectORI = project.getORI();
055                break;
056            }
057        }
058
059        if (projectORI == null) {
060            throw new RepositoryNotFoundException(name);
061        }
062
063        AbstractProjectProvider projectProvider = resourceManager.getProjectsContainer().getProjectProvider(projectORI.toString());
064
065        if (!(projectProvider instanceof GitProjectProvider)) {
066            throw new RepositoryNotFoundException(name);
067        }
068
069        File projectFolder = projectProvider.getProjectFolder();
070        File dir = FileKey.resolve(projectFolder, FS.DETECTED);
071        if (dir == null) {
072            throw new RepositoryNotFoundException(name);
073        }
074
075
076        try {
077            FileKey key = FileKey.exact(dir, FS.DETECTED);
078            db = RepositoryCache.open(key, true);
079        } catch (IOException e) {
080            throw new RepositoryNotFoundException(name, e);
081        }
082
083        return db;
084
085    }
086
087    /**
088     * Add a single repository to the set that is exported by this daemon.
089     * <p>
090     * The existence (or lack-thereof) of <code>git-daemon-export-ok</code> is
091     * ignored by this method. The repository is always published.
092     *
093     * @param name
094     *            name the repository will be published under.
095     * @param db
096     *            the repository instance.
097     */
098    public void exportRepository(String name, Repository db) {
099        exports.put(nameWithDotGit(name), db);
100    }
101
102    private static String nameWithDotGit(String name) {
103        if (name.endsWith(Constants.DOT_GIT_EXT))
104            return name;
105        return name + Constants.DOT_GIT_EXT;
106    }
107
108    private static boolean isUnreasonableName(final String name) {
109        if (name.length() == 0)
110            return true; // no empty paths
111
112        if (name.indexOf('\\') >= 0)
113            return true; // no windows/dos style paths
114        if (new File(name).isAbsolute())
115            return true; // no absolute paths
116
117        if (name.startsWith("../"))
118            return true; // no "l../etc/passwd"
119        if (name.contains("/../"))
120            return true; // no "foo/../etc/passwd"
121        if (name.contains("/./"))
122            return true; // "foo/./foo" is insane to ask
123        if (name.contains("//"))
124            return true; // double slashes is sloppy, don't use it
125
126        return false; // is a reasonable name
127    }
128
129}
