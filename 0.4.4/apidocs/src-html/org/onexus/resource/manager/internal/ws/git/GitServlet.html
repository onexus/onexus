018package org.onexus.resource.manager.internal.ws.git;
019
020
021import org.onexus.resource.api.IResourceManager;
022import org.onexus.resource.manager.internal.ProjectsContainer;
023import org.onexus.resource.manager.internal.ResourceManager;
024
025import javax.servlet.ServletConfig;
026import javax.servlet.ServletContext;
027import javax.servlet.ServletException;
028import javax.servlet.http.HttpServletRequest;
029import java.util.Enumeration;
030
031public class GitServlet extends org.eclipse.jgit.http.server.GitServlet implements ServletConfig {
032
033    private ServletConfig config;
034    private IResourceManager resourceManager;
035
036    @Override
037    public void init(final ServletConfig config) throws ServletException {
038        this.config = config;
039
040        setRepositoryResolver(new FileResolver<HttpServletRequest>((ResourceManager) resourceManager));
041        super.init(this);
042    }
043
044    @Override
045    public String getServletName() {
046        return GitServlet.class.getName();
047    }
048
049    public ServletContext getServletContext() {
050        return config.getServletContext();
051    }
052
053    public String getInitParameter(String name) {
054        if (name.equals("base-path")) {
055            return ProjectsContainer.ONEXUS_PROJECTS_FOLDER;
056        }
057        if (name.equals("export-all")) {
058            return "1";
059        }
060        return config.getInitParameter(name);
061    }
062
063    public Enumeration getInitParameterNames() {
064        return config.getInitParameterNames();
065    }
066
067    public IResourceManager getResourceManager() {
068        return resourceManager;
069    }
070
071    public void setResourceManager(IResourceManager resourceManager) {
072        this.resourceManager = resourceManager;
073    }
074}
