018package org.onexus.resource.serializer.xstream.internal;
019
020import com.thoughtworks.xstream.XStream;
021import com.thoughtworks.xstream.converters.ConversionException;
022import com.thoughtworks.xstream.mapper.MapperWrapper;
023import org.onexus.resource.api.Folder;
024import org.onexus.resource.api.IResourceSerializer;
025import org.onexus.resource.api.Loader;
026import org.onexus.resource.api.Parameter;
027import org.onexus.resource.api.Plugin;
028import org.onexus.resource.api.Project;
029import org.onexus.resource.api.Property;
030import org.onexus.resource.api.Resource;
031import org.onexus.resource.api.annotations.ResourceAlias;
032import org.onexus.resource.api.annotations.ResourceImplicitList;
033import org.onexus.resource.api.annotations.ResourceRegister;
034import org.onexus.resource.api.exceptions.UnserializeException;
035import org.onexus.resource.api.utils.AbstractMetadata;
036import org.slf4j.Logger;
037import org.slf4j.LoggerFactory;
038
039import java.io.InputStream;
040import java.io.OutputStream;
041import java.lang.reflect.Field;
042import java.lang.reflect.ParameterizedType;
043import java.lang.reflect.Type;
044import java.util.HashSet;
045import java.util.Set;
046
047public class ResourceSerializer implements IResourceSerializer {
048
049    private static final Logger LOGGER = LoggerFactory.getLogger(ResourceSerializer.class);
050
051    private Set<ClassLoader> registeredLoaders = new HashSet<ClassLoader>();
052
053    private XStream xstream;
054
055    public ResourceSerializer() {
056        super();
057        this.xstream = new XStream() {
058            @Override
059            protected MapperWrapper wrapMapper(MapperWrapper next) {
060                return new MapperWrapper(next) {
061                    @Override
062                    public boolean shouldSerializeMember(Class definedIn,
063                                                         String fieldName) {
064                        if (definedIn == Object.class) {
065                            LOGGER.warn("Tag '" + fieldName + "' not defined.");
066                            return false;
067                        }
068                        return super.shouldSerializeMember(definedIn, fieldName);
069                    }
070                };
071            }
072        };
073
074        this.xstream.setClassLoader(new RegisteredClassLoader());
075
076        // Resource
077        xstream.addImplicitCollection(Resource.class, "properties", "property", Property.class);
078        xstream.addImplicitCollection(AbstractMetadata.class, "properties", "property", Property.class);
079
080        // Project
081        alias("project", Project.class);
082        xstream.addImplicitCollection(Plugin.class, "parameters", "parameter", Parameter.class);
083        xstream.omitField(Project.class, "name");
084
085        // Plugin
086        alias("plugin", Plugin.class);
087
088        // Folder
089        alias("folder", Folder.class);
090
091        // Loader
092        alias("parameter", Parameter.class);
093        xstream.addImplicitCollection(Loader.class, "parameters", "parameter", Parameter.class);
094
095        alias("property", Property.class);
096
097        xstream.registerConverter(new ClassConverter());
098        xstream.registerConverter(new ORIConverter());
099
100
101    }
102
103    @Override
104    public String getMediaType() {
105        return "text/xml";
106    }
107
108    @SuppressWarnings("unchecked")
109    @Override
110    public <T> T unserialize(Class<T> resourceType, InputStream input) throws UnserializeException {
111        try {
112            return (T) xstream.fromXML(input);
113        } catch (ConversionException e) {
114
115            String path = e.get("path");
116            String line = e.get("line number");
117
118            throw new UnserializeException(path, line, e);
119        }
120    }
121
122    @Override
123    public void serialize(Object resource, OutputStream output) {
124        xstream.toXML(resource, output);
125    }
126
127    private void alias(String alias, Class<?> resourceType) {
128        xstream.alias(alias, resourceType);
129        register(resourceType);
130    }
131
132    @Override
133    public void register(Class<?> resourceType) {
134
135        processAnnotations(resourceType);
136        registeredLoaders.add(resourceType.getClassLoader());
137
138        // Register dependent types
139        ResourceRegister resourceRegister = resourceType.getAnnotation(ResourceRegister.class);
140        if (resourceRegister != null) {
141            for (Class type : resourceRegister.value()) {
142                processAnnotations(type);
143            }
144        }
145    }
146
147    private void processAnnotations(Class<?> resourceType) {
148
149        ResourceAlias resourceAlias = resourceType.getAnnotation(ResourceAlias.class);
150        if (resourceAlias != null) {
151           xstream.alias(resourceAlias.value(), resourceType);
152        }
153
154        for (Field field : resourceType.getDeclaredFields()) {
155
156            ResourceAlias resourceFieldAlias = field.getAnnotation(ResourceAlias.class);
157            if (resourceFieldAlias != null) {
158                xstream.aliasField(resourceFieldAlias.value(), resourceType, field.getName());
159            }
160
161            ResourceImplicitList implicitList = field.getAnnotation(ResourceImplicitList.class);
162            if (implicitList != null) {
163                xstream.addImplicitCollection(resourceType, field.getName(), implicitList.value(), getCollectionItemClass(field));
164            }
165        }
166
167    }
168
169    private Class<?> getCollectionItemClass(Field field) {
170
171        Class<?> type = null;
172        final Type genericType = field.getGenericType();
173        if (genericType instanceof ParameterizedType) {
174            final Type typeArgument = ((ParameterizedType)genericType).getActualTypeArguments()[0];
175            if (typeArgument instanceof ParameterizedType) {
176                type = (Class<?>)((ParameterizedType)typeArgument).getRawType();
177            } else if (typeArgument instanceof Class) {
178                type = (Class<?>)typeArgument;
179            }
180        }
181
182        return type;
183    }
184
185    private class RegisteredClassLoader extends ClassLoader {
186
187        @SuppressWarnings({"rawtypes", "unchecked"})
188        public Class loadClass(String name) throws ClassNotFoundException {
189
190            for (ClassLoader loader : registeredLoaders) {
191                try {
192                    return loader.loadClass(name);
193                } catch (ClassNotFoundException e) {
194                    // Continue
195                }
196            }
197
198            throw new ClassNotFoundException();
199        }
200
201    }
202
203}
