018package org.onexus.website.api.utils.error;
019
020import org.apache.wicket.core.request.mapper.StalePageException;
021import org.apache.wicket.markup.html.basic.MultiLineLabel;
022import org.apache.wicket.markup.html.pages.AbstractErrorPage;
023import org.apache.wicket.request.http.WebResponse;
024import org.apache.wicket.util.lang.Generics;
025import org.onexus.resource.api.exceptions.OnexusException;
026import org.onexus.website.api.Website;
027import org.onexus.website.api.theme.DefaultTheme;
028
029import javax.servlet.http.HttpServletResponse;
030import java.util.List;
031
032
033public class ExceptionErrorPage extends AbstractErrorPage {
034    private static final long serialVersionUID = 1L;
035
036    /**
037     * Keep a reference to the root cause. WicketTester will use it
038     */
039    private final transient Throwable throwable;
040
041    /**
042     * Constructor.
043     *
044     * @param throwable The exception to show
045     */
046    public ExceptionErrorPage(final Throwable throwable) {
047        this.throwable = throwable;
048
049        add(new DefaultTheme());
050
051        // Add exception label
052        add(new MultiLineLabel("exception", getErrorMessage(throwable)));
053
054        add(new MultiLineLabel("stacktrace", getStackTrace(throwable)));
055
056    }
057
058    /**
059     * Converts a Throwable to a string.
060     *
061     * @param throwable The throwable
062     * @return The string
063     */
064    public String getErrorMessage(final Throwable throwable) {
065        if (throwable != null) {
066            StringBuilder sb = new StringBuilder(256);
067
068            // first print the last cause or the first onexus exception
069            Throwable cause = null;
070            List<Throwable> al = convertToList(throwable);
071            for (Throwable e : al) {
072                if (OnexusException.class.isAssignableFrom(e.getClass())) {
073                    cause = e;
074                    break;
075                }
076            }
077
078            if (cause == null) {
079                int length = al.size() - 1;
080                cause = al.get(length);
081            }
082
083            sb.append(cause.getMessage());
084            return sb.toString();
085        } else {
086            return "[Unknown]";
087        }
088    }
089
090    /**
091     * Converts a Throwable to a string.
092     *
093     * @param throwable The throwable
094     * @return The string
095     */
096    public String getStackTrace(final Throwable throwable) {
097        if (throwable != null) {
098            List<Throwable> al = convertToList(throwable);
099
100            StringBuilder sb = new StringBuilder(256);
101
102            // first print the last cause
103            int length = al.size() - 1;
104            Throwable cause = al.get(length);
105
106            outputThrowable(cause, sb, false);
107
108            if (length > 0) {
109                sb.append("\n\nComplete stack:\n\n");
110                for (int i = 0; i < length; i++) {
111                    outputThrowable(al.get(i), sb, true);
112                    sb.append("\n");
113                }
114            }
115            return sb.toString();
116        } else {
117            return "<Null Throwable>";
118        }
119    }
120
121    /**
122     * @param throwable
123     * @return xxx
124     */
125    private List<Throwable> convertToList(final Throwable throwable) {
126        List<Throwable> al = Generics.newArrayList();
127        Throwable cause = throwable;
128        al.add(cause);
129        while (cause.getCause()!= null && cause!= cause.getCause()) {
130            cause = cause.getCause();
131            al.add(cause);
132        }
133        return al;
134    }
135
136    /**
137     * Outputs the throwable and its stacktrace to the stringbuffer. If stopAtWicketSerlvet is true
138     * then the output will stop when the org.apache.wicket servlet is reached. sun.reflect.
139     * packages are filtered out.
140     *
141     * @param cause
142     * @param sb
143     * @param stopAtWicketServlet
144     */
145    private void outputThrowable(Throwable cause, StringBuilder sb, boolean stopAtWicketServlet) {
146        sb.append(cause);
147        sb.append("\n");
148        StackTraceElement[] trace = cause.getStackTrace();
149        for (int i = 0; i < trace.length; i++) {
150            String traceString = trace[i].toString();
151            if (!(traceString.startsWith("sun.reflect.") && i > 1)) {
152                sb.append("     at ");
153                sb.append(traceString);
154                sb.append("\n");
155                if (stopAtWicketServlet &&
156                        (traceString.startsWith("org.apache.wicket.protocol.http.WicketServlet") || traceString.startsWith("org.apache.wicket.protocol.http.WicketFilter"))) {
157                    return;
158                }
159            }
160        }
161    }
162
163    @Override
164    protected void setHeaders(final WebResponse response) {
165        response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
166    }
167
168    /**
169     * Get access to the exception
170     *
171     * @return The exception
172     */
173    public Throwable getThrowable() {
174        return throwable;
175    }
176
177    @Override
178    protected void onBeforeRender() {
179
180        if (getThrowable() instanceof StalePageException) {
181            setResponsePage(Website.class);
182        }
183
184        super.onBeforeRender();
185    }
186}
