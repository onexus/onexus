018package org.onexus.website.api.utils;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.behavior.Behavior;
022import org.apache.wicket.markup.head.CssHeaderItem;
023import org.apache.wicket.markup.head.IHeaderResponse;
024import org.apache.wicket.request.resource.CssResourceReference;
025import org.apache.wicket.util.string.Strings;
026import org.onexus.data.api.IDataManager;
027import org.onexus.resource.api.IResourceManager;
028import org.onexus.resource.api.ORI;
029import org.onexus.resource.api.Project;
030import org.onexus.website.api.WebsiteApplication;
031
032import javax.inject.Inject;
033
034public class CustomCssBehavior extends Behavior {
035
036    private transient CssHeaderItem cssHeaderItem;
037
038    private ORI resourceUri;
039
040    @Inject
041    private IDataManager dataManager;
042
043    @Inject
044    private IResourceManager resourceManager;
045
046    public CustomCssBehavior(ORI parentOri, String cssTag) {
047        if (!Strings.isEmpty(cssTag)) {
048            resourceUri = new ORI(cssTag);
049            if (!resourceUri.isAbsolute()) {
050                resourceUri = resourceUri.toAbsolute(parentOri);
051            }
052        }
053    }
054
055    private CustomCssBehavior(ORI resourceUri) {
056        this.resourceUri = resourceUri;
057    }
058
059    @Override
060    public void renderHead(Component component, IHeaderResponse response) {
061
062        if (cssHeaderItem == null) {
063
064            if (resourceUri != null) {
065
066                String dataService = getDataManager().getMount();
067                Project project = getResourceManager().getProject(resourceUri.getProjectUrl());
068
069                String url;
070                if (project != null) {
071                    url = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName() + resourceUri.getPath());
072                } else {
073                    url = resourceUri.toString();
074                }
075                cssHeaderItem = CssHeaderItem.forUrl(url);
076
077            } else {
078                cssHeaderItem = CssHeaderItem.forReference(new CssResourceReference(component.getClass(), component.getClass().getSimpleName() + ".css"));
079            }
080        }
081
082        response.render(cssHeaderItem);
083    }
084
085    private IDataManager getDataManager() {
086
087        if (dataManager == null) {
088            WebsiteApplication.inject(this);
089        }
090
091        return dataManager;
092    }
093
094    private IResourceManager getResourceManager() {
095
096        if (resourceManager == null) {
097            WebsiteApplication.inject(this);
098        }
099
100        return resourceManager;
101    }
102
103}
