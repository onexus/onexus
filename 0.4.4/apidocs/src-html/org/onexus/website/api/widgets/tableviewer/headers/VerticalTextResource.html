018package org.onexus.website.api.widgets.tableviewer.headers;
019
020import org.apache.wicket.request.resource.DynamicImageResource;
021import org.apache.wicket.request.resource.caching.IResourceCachingStrategy;
022import org.apache.wicket.request.resource.caching.NoOpResourceCachingStrategy;
023
024import java.awt.*;
025import java.awt.image.BufferedImage;
026
027public class VerticalTextResource extends DynamicImageResource {
028
029    private static final Font DEFAULT_FONT = new Font("Arial", Font.BOLD, 12);
030    private static final IResourceCachingStrategy CACHING_STRATEGY = new NoOpResourceCachingStrategy();
031
032    private String text;
033    private int width;
034    private int height;
035    private Font font;
036
037    public VerticalTextResource(String text) {
038        this(text, DEFAULT_FONT);
039
040
041    }
042
043    public VerticalTextResource(String text, Font font) {
044        this(text, font.getSize() + 3, (int) (font.getSize() * 0.75 * text
045                .length()), font);
046    }
047
048    public VerticalTextResource(String text, int width, int height) {
049        this(text, width, height, DEFAULT_FONT);
050    }
051
052    public VerticalTextResource(String text, int width, int height, Font font) {
053        super();
054        this.text = text;
055        this.width = width;
056        this.height = height;
057        this.font = font;
058    }
059
060    @Override
061    protected byte[] getImageData(Attributes attributes) {
062        final BufferedImage image = new BufferedImage(width, height,
063                BufferedImage.TYPE_INT_RGB);
064        Graphics2D g2 = image.createGraphics();
065        g2.setBackground(Color.WHITE);
066        g2.clearRect(0, 0, 500, 500);
067        g2.setFont(font);
068        g2.setColor(Color.BLACK);
069        g2.translate((width + font.getSize()) / 2, height - 2);
070        g2.rotate(-Math.PI / 2.0);
071        g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
072                RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
073        g2.drawString(text, 0, 0);
074        g2.dispose();
075
076        return toImageData(image);
077    }
078
079    @Override
080    protected IResourceCachingStrategy getCachingStrategy() {
081        return CACHING_STRATEGY;
082    }
083
084
085}
