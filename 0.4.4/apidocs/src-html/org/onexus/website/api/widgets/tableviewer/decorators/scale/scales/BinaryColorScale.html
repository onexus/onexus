018package org.onexus.website.api.widgets.tableviewer.decorators.scale.scales;
019
020import java.awt.*;
021
022public class BinaryColorScale extends AbstractColorScale {
023
024    private double cutoff = 1.0;
025    private CutoffCmp cutoffCmp = CutoffCmp.EQ;
026
027    public BinaryColorScale(double minPoint, double maxPoint, double cutoff,
028                            Color minColor, Color maxColor) {
029        super(minPoint, maxPoint);
030        this.cutoff = cutoff;
031        this.minColor = minColor;
032        this.maxColor = maxColor;
033    }
034
035    public BinaryColorScale() {
036        super(0.0, 1.0);
037    }
038
039    @Override
040    public Color valueColor(Object obj) {
041
042        double value;
043        if (obj instanceof Double) {
044            value = (Double) obj;
045        } else {
046            value = Double.valueOf(String.valueOf(obj));
047        }
048
049        if (Double.isNaN(value)) {
050            return notANumberColor;
051        }
052        else if (value > maxPoint || value == Double.POSITIVE_INFINITY){
053            return posInfinityColor;
054        }
055        else if (value < minPoint || value == Double.NEGATIVE_INFINITY) {
056            return negInfinityColor;
057        }
058
059        boolean isSig = cutoffCmp.compare(value, cutoff);
060        return isSig ? maxColor : minColor;
061    }
062
063    public double getCutoff() {
064        return cutoff;
065    }
066
067    public void setCutoff(double cutoff) {
068        this.cutoff = cutoff;
069    }
070
071    public CutoffCmp getCutoffCmp() {
072        return cutoffCmp;
073    }
074
075    public void setCutoffCmp(CutoffCmp cutoffCmp) {
076        this.cutoffCmp = cutoffCmp;
077    }
078
079}
