018package org.onexus.website.api.widgets.views;
019
020import org.apache.commons.lang3.StringUtils;
021import org.apache.wicket.AttributeModifier;
022import org.apache.wicket.ajax.AjaxRequestTarget;
023import org.apache.wicket.ajax.markup.html.AjaxLink;
024import org.apache.wicket.event.Broadcast;
025import org.apache.wicket.markup.html.WebMarkupContainer;
026import org.apache.wicket.markup.html.basic.Label;
027import org.apache.wicket.markup.repeater.RepeatingView;
028import org.apache.wicket.model.IModel;
029import org.apache.wicket.model.Model;
030import org.onexus.website.api.events.EventViewChange;
031import org.onexus.website.api.pages.browser.BrowserPage;
032import org.onexus.website.api.pages.browser.BrowserPageConfig;
033import org.onexus.website.api.pages.browser.BrowserPageStatus;
034import org.onexus.website.api.pages.browser.ViewConfig;
035import org.onexus.website.api.utils.visible.VisiblePredicate;
036import org.onexus.website.api.widgets.Widget;
037
038import java.util.List;
039
040public class ViewsWidget extends Widget<ViewsWidgetConfig, ViewsWidgetStatus> {
041
042    public ViewsWidget(String componentId, IModel statusModel) {
043        super(componentId, statusModel);
044
045        onEventFireUpdate(EventViewChange.class);
046
047    }
048
049    @Override
050    protected void onBeforeRender() {
051
052        RepeatingView viewsContainer = new RepeatingView("views");
053        add(viewsContainer);
054
055        BrowserPageStatus browserStatus = getPageStatus();
056
057        setVisible(false);
058        if (browserStatus != null) {
059            BrowserPageConfig browserConfig = browserStatus.getConfig();
060
061            List<ViewConfig> views = browserConfig.getTab(browserStatus.getCurrentTabId()).getViews();
062
063            VisiblePredicate predicate = new VisiblePredicate(getPageBaseOri().getParent(), getPageStatus().getEntitySelections());
064
065            if (views != null && !views.isEmpty()) {
066                for (ViewConfig view : views) {
067
068                    WebMarkupContainer item = new WebMarkupContainer(viewsContainer.newChildId());
069                    viewsContainer.add(item);
070
071                    AjaxLink<String> link = new AjaxLink<String>("link", Model.of(view.getTitle())) {
072                        @Override
073                        public void onClick(AjaxRequestTarget target) {
074                            BrowserPageStatus browserStatus = getPageStatus();
075                            browserStatus.setCurrentView(getModelObject());
076
077                            send(getPage(), Broadcast.BREADTH, EventViewChange.EVENT);
078                        }
079                    };
080                    link.add(new Label("label", view.getTitle()));
081                    item.add(link);
082
083                    if (view.getTitle().equals(browserStatus.getCurrentView())) {
084                        item.add(new AttributeModifier("class", "active"));
085                    }
086
087                    // Disable non-visible views
088                    if (!predicate.evaluate(view)) {
089                        item.setEnabled(false);
090
091                        String msg = VisiblePredicate.getMessage(view);
092
093                        if (!StringUtils.isEmpty(msg)) {
094                            item.add(new AttributeModifier("rel", "tooltip"));
095                            item.add(new AttributeModifier("title", msg));
096                        }
097
098                    } else {
099                        item.setEnabled(true);
100                    }
101                }
102                setVisible(true);
103            }
104        }
105
106        super.onBeforeRender();
107    }
108
109    private BrowserPageStatus getPageStatus() {
110        return findParent(BrowserPage.class).getStatus();
111    }
112
113}
