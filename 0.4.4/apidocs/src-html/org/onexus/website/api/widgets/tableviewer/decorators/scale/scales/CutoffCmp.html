018package org.onexus.website.api.widgets.tableviewer.decorators.scale.scales;
019
020import java.io.Serializable;
021import java.util.HashMap;
022import java.util.Map;
023
024@SuppressWarnings("serial")
025public abstract class CutoffCmp implements Serializable {
026
027    private static final long serialVersionUID = 6256197968707025432L;
028
029    public static final CutoffCmp LT = new CutoffCmp("<", "less than") {
030        @Override
031        public boolean compare(double value, double cutoff) {
032            return value < cutoff;
033        }
034    };
035
036    public static final CutoffCmp LE = new CutoffCmp("<=", "less or equal") {
037        @Override
038        public boolean compare(double value, double cutoff) {
039            return value <= cutoff;
040        }
041    };
042
043    public static final CutoffCmp EQ = new CutoffCmp("=", "equal") {
044        @Override
045        public boolean compare(double value, double cutoff) {
046            return value == cutoff;
047        }
048    };
049
050    public static final CutoffCmp NE = new CutoffCmp("!=", "not equal") {
051        @Override
052        public boolean compare(double value, double cutoff) {
053            return value != cutoff;
054        }
055    };
056
057    public static final CutoffCmp GT = new CutoffCmp(">", "greater than") {
058        @Override
059        public boolean compare(double value, double cutoff) {
060            return value > cutoff;
061        }
062    };
063
064    public static final CutoffCmp GE = new CutoffCmp(">=", "greater or equal") {
065        @Override
066        public boolean compare(double value, double cutoff) {
067            return value >= cutoff;
068        }
069    };
070
071    public static final CutoffCmp ABS_LT = new CutoffCmp("abs <",
072            "absolute less than") {
073        @Override
074        public boolean compare(double value, double cutoff) {
075            return value < cutoff;
076        }
077    };
078
079    public static final CutoffCmp ABS_LE = new CutoffCmp("abs <=",
080            "absolute less or equal") {
081        @Override
082        public boolean compare(double value, double cutoff) {
083            return value <= cutoff;
084        }
085    };
086
087    public static final CutoffCmp ABS_EQ = new CutoffCmp("abs =",
088            "absolute equal") {
089        @Override
090        public boolean compare(double value, double cutoff) {
091            return Math.abs(value) == cutoff;
092        }
093    };
094
095    public static final CutoffCmp ABS_NE = new CutoffCmp("abs !=",
096            "absolute not equal") {
097        @Override
098        public boolean compare(double value, double cutoff) {
099            return Math.abs(value) != cutoff;
100        }
101    };
102
103    public static final CutoffCmp ABS_GT = new CutoffCmp("abs >",
104            "absolute greater than") {
105        @Override
106        public boolean compare(double value, double cutoff) {
107            return Math.abs(value) > cutoff;
108        }
109    };
110
111    public static final CutoffCmp ABS_GE = new CutoffCmp("abs >=",
112            "absolute greater or equal") {
113        @Override
114        public boolean compare(double value, double cutoff) {
115            return Math.abs(value) >= cutoff;
116        }
117    };
118
119    public static final CutoffCmp[] COMPARATORS = new CutoffCmp[]{LT, LE, GT,
120            GE, EQ, NE, ABS_LT, ABS_LE, ABS_GT, ABS_GE, ABS_EQ, ABS_NE};
121
122    public static final Map<String, CutoffCmp> ABBREVIATED_NAME_MAP = new HashMap<String, CutoffCmp>();
123    public static final Map<String, CutoffCmp> SHORT_NAME_MAP = new HashMap<String, CutoffCmp>();
124    public static final Map<String, CutoffCmp> LONG_NAME_MAP = new HashMap<String, CutoffCmp>();
125    public static final Map<String, CutoffCmp> NAME_MAP = new HashMap<String, CutoffCmp>();
126
127    static {
128        ABBREVIATED_NAME_MAP.put("lt", CutoffCmp.LT);
129        ABBREVIATED_NAME_MAP.put("le", CutoffCmp.LE);
130        ABBREVIATED_NAME_MAP.put("gt", CutoffCmp.GT);
131        ABBREVIATED_NAME_MAP.put("ge", CutoffCmp.GE);
132        ABBREVIATED_NAME_MAP.put("eq", CutoffCmp.EQ);
133        ABBREVIATED_NAME_MAP.put("ne", CutoffCmp.NE);
134        ABBREVIATED_NAME_MAP.put("alt", CutoffCmp.ABS_LT);
135        ABBREVIATED_NAME_MAP.put("ale", CutoffCmp.ABS_LE);
136        ABBREVIATED_NAME_MAP.put("agt", CutoffCmp.ABS_GT);
137        ABBREVIATED_NAME_MAP.put("age", CutoffCmp.ABS_GE);
138        ABBREVIATED_NAME_MAP.put("aeq", CutoffCmp.ABS_EQ);
139        ABBREVIATED_NAME_MAP.put("ane", CutoffCmp.ABS_NE);
140
141        for (CutoffCmp cmp : COMPARATORS) {
142            SHORT_NAME_MAP.put(cmp.getShortName(), cmp);
143            LONG_NAME_MAP.put(cmp.getLongName(), cmp);
144            NAME_MAP.put(cmp.getShortName(), cmp);
145            NAME_MAP.put(cmp.getLongName(), cmp);
146        }
147    }
148
149    private String shortName;
150    private String longName;
151
152    public CutoffCmp(String shortName, String longName) {
153        this.shortName = shortName;
154        this.longName = longName;
155    }
156
157    public String getShortName() {
158        return shortName;
159    }
160
161    public String getLongName() {
162        return longName;
163    }
164
165    public abstract boolean compare(double value, double cutoff);
166
167    @Override
168    public String toString() {
169        return shortName;
170    }
171}
