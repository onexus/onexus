018package org.onexus.website.api.widgets.tableviewer.decorators.box;
019
020import org.apache.wicket.markup.html.WebMarkupContainer;
021import org.apache.wicket.model.IModel;
022import org.apache.wicket.util.string.Strings;
023import org.onexus.collection.api.Collection;
024import org.onexus.collection.api.Field;
025import org.onexus.collection.api.IEntity;
026import org.onexus.resource.api.Parameters;
027import org.onexus.website.api.widgets.tableviewer.decorators.IDecorator;
028
029import java.util.ArrayList;
030import java.util.Collections;
031import java.util.HashMap;
032import java.util.List;
033import java.util.Map;
034import java.util.regex.Pattern;
035
036public class BoxDecorator implements IDecorator {
037
038    private static final Pattern COMMA_PATTERN = Pattern.compile(",");
039    private static final Pattern COLON_PATTERN = Pattern.compile(":");
040
041    private Field field;
042    private String fields;
043    private Map<String, String> decorators = new HashMap<String, String>();
044
045    public BoxDecorator(Field field, Parameters parameters) {
046        super();
047        this.field = field;
048        this.fields = parameters.get(BoxDecoratorParameters.FIELDS);
049
050        if (parameters.containsKey(BoxDecoratorParameters.DECORATORS)) {
051            String decoParameters[] = COMMA_PATTERN.split(parameters.get(BoxDecoratorParameters.DECORATORS));
052
053            for (String decoParam : decoParameters) {
054                String[] values = COLON_PATTERN.split(decoParam);
055                decorators.put(values[0], values[1]);
056            }
057        }
058    }
059
060    @Override
061    public void populateCell(WebMarkupContainer cellContainer, String componentId, IModel<IEntity> entity) {
062        cellContainer.add(new BoxEntityPanel(componentId, field, entity.getObject(), getFieldIds(fields), decorators));
063    }
064
065    private static List<String> getFieldIds(String fields) {
066
067        if (Strings.isEmpty(fields)) {
068            return new ArrayList<String>();
069        }
070
071        String[] fieldsSplit = fields.split(",");
072        List<String> fieldIds = new ArrayList<String>(fieldsSplit.length);
073
074        for (String id : fieldsSplit) {
075            fieldIds.add(id.trim());
076        }
077
078        return fieldIds;
079    }
080
081    @Override
082    public String getFormatValue(IEntity entity) {
083        if (entity == null) {
084            return null;
085        }
086
087        Object value = entity.get(field.getId());
088        if (value == null) {
089            return null;
090        }
091
092        return String.valueOf(value);
093    }
094
095    @Override
096    public String getColor(IEntity entity) {
097        return "#000000";
098    }
099
100    @Override
101    public String getTemplate() {
102        return null;
103    }
104
105    @Override
106    public void setTemplate(String template) {
107
108    }
109
110    @Override
111    public List<String> getExtraFields(Collection collection) {
112
113        List<String> fields = getFieldIds(this.fields);
114
115        if (fields.isEmpty()) {
116            for (Field field : collection.getFields()) {
117                fields.add(field.getId());
118            }
119        }
120
121        return fields;
122    }
123
124    @Override
125    public void detach() {
126    }
127}
