018package org.onexus.website.api.widgets.tableviewer.decorators.scale.scales;
019
020import java.awt.*;
021
022public class PValueColorScale extends CompositeColorScale {
023
024    private static final double EPSILON = 1e-16;
025
026    private double significanceLevel;
027
028    private UniformColorScale nonSigScale;
029
030    private ScaleRange nonSigScaleRange;
031
032    private LogColorScale scale;
033
034    private ScaleRange scaleRange;
035
036    public PValueColorScale(double significanceLevel, Color minColor,
037                            Color maxColor, Color nonSignificantColor) {
038
039        super(0.0, 1.0, minColor, maxColor);
040
041        this.significanceLevel = significanceLevel;
042
043        nonSigScale = new UniformColorScale(nonSignificantColor);
044
045        nonSigScaleRange = new ScaleRange(significanceLevel + EPSILON, 1.0,
046                nonSigScale);
047
048        scale = new LogColorScale(0.0, significanceLevel + EPSILON, minColor,
049                maxColor);
050
051        scaleRange = new ScaleRange(0.0, significanceLevel + EPSILON, scale);
052
053        ScaleRange[] scaleRanges = new ScaleRange[]{nonSigScaleRange,
054                scaleRange};
055
056        setScaleRanges(scaleRanges);
057    }
058
059    public PValueColorScale() {
060        this(0.05, ColorConstants.PVALUE_MIN_COLOR,
061                ColorConstants.PVALUE_MAX_COLOR,
062                ColorConstants.NON_SIGNIFICANT_COLOR);
063    }
064
065    public double getSignificanceLevel() {
066        return significanceLevel;
067    }
068
069    public void setSignificanceLevel(double significanceLevel) {
070        this.significanceLevel = significanceLevel;
071        nonSigScaleRange.setMinPoint(significanceLevel + EPSILON);
072        scaleRange.setMaxPoint(significanceLevel + EPSILON);
073        scale.setMaxPoint(significanceLevel + EPSILON);
074    }
075
076    @Override
077    public void setMinColor(Color color) {
078        super.setMinColor(color);
079        scale.setMinColor(color);
080    }
081
082    @Override
083    public void setMaxColor(Color color) {
084        super.setMaxColor(color);
085        scale.setMaxColor(color);
086    }
087
088    public Color getNonSignificantColor() {
089        return nonSigScale.getColor();
090    }
091
092    public void setNonSignificantColor(Color color) {
093        nonSigScale.setColor(color);
094    }
095}
