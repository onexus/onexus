018package org.onexus.website.api.widgets.jheatmap;
019
020import org.apache.wicket.extensions.markup.html.repeater.data.table.IColumn;
021import org.apache.wicket.markup.head.CssHeaderItem;
022import org.apache.wicket.markup.head.HeaderItem;
023import org.apache.wicket.markup.head.IHeaderResponse;
024import org.apache.wicket.markup.head.JavaScriptHeaderItem;
025import org.apache.wicket.markup.head.OnLoadHeaderItem;
026import org.apache.wicket.markup.html.image.Image;
027import org.apache.wicket.model.IModel;
028import org.apache.wicket.request.resource.CssResourceReference;
029import org.apache.wicket.request.resource.PackageResourceReference;
030import org.apache.wicket.request.resource.ResourceReference;
031import org.apache.wicket.resource.JQueryPluginResourceReference;
032import org.onexus.collection.api.ICollectionManager;
033import org.onexus.collection.api.IEntityTable;
034import org.onexus.collection.api.query.Query;
035import org.onexus.resource.api.ORI;
036import org.onexus.website.api.WebsiteApplication;
037import org.onexus.website.api.widgets.Widget;
038import org.onexus.website.api.widgets.tableviewer.columns.ColumnConfig;
039
040import javax.inject.Inject;
041import java.net.URLEncoder;
042import java.util.ArrayList;
043import java.util.List;
044
045public class HeatmapViewer extends Widget<HeatmapViewerConfig, HeatmapViewerStatus> {
046
047    private static final HeaderItem CSS = CssHeaderItem.forReference(new CssResourceReference(HeatmapViewer.class, "css/jheatmap-1.0.0-alpha.css"));
048    private static final HeaderItem JS_JHEATMAP = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(HeatmapViewer.class, "js/jheatmap-1.0.0-alpha.js"));
049    private static final ResourceReference LOADING_IMG = new PackageResourceReference(HeatmapViewer.class, "images/loading.gif");
050
051    @Inject
052    private ICollectionManager collectionManager;
053
054    public HeatmapViewer(String componentId, IModel<HeatmapViewerStatus> status) {
055        super(componentId, status);
056
057        add(new Image("loading", LOADING_IMG));
058    }
059
060    @Override
061    public void renderHead(IHeaderResponse response) {
062        response.render(CSS);
063        response.render(JS_JHEATMAP);
064        response.render(OnLoadHeaderItem.forScript(newJavaScriptHeatmap()));
065    }
066
067    private String newJavaScriptHeatmap() {
068        StringBuilder code = new StringBuilder();
069
070        HeatmapViewerConfig config = getConfig();
071
072        List<IColumn<IEntityTable, String>> columns = new ArrayList<IColumn<IEntityTable, String>>();
073
074        ORI parentUri = getPageBaseOri();
075        for (ColumnConfig column : getConfig().getColumns()) {
076            column.addColumns(columns, parentUri, false);
077        }
078        int lastColumnFields = columns.size();
079        columns.clear();
080
081        for (ColumnConfig row : getConfig().getRows()) {
082            row.addColumns(columns, parentUri, false);
083        }
084        int lastRowFields = lastColumnFields + columns.size();
085        columns.clear();
086
087        // Webservice URL
088        code.append(
089                "$('#heatmap').heatmap({\n" +
090                        "       data : {\n" +
091                        "                 values: new jheatmap.readers.TsvMatrixReader({ url: \"" + getQueryUrl() + "\"}), \n");
092
093        code.append("\t\t\t\tcols_annotations : [");
094        if (config.getColAnnotations() == null) {
095            for (int i = 0; i < lastColumnFields; i++) {
096                if (i != 0) {
097                    code.append(",");
098                }
099                code.append(i);
100            }
101        } else {
102            code.append(config.getColAnnotations());
103        }
104        code.append("], \n");
105        code.append("\t\t\t\trows_annotations : [");
106        if (config.getRowAnnotations() == null) {
107            for (int i = lastColumnFields; i < lastRowFields; i++) {
108                if (i != lastColumnFields) {
109                    code.append(",");
110                }
111                code.append(i);
112            }
113        } else {
114            code.append(config.getRowAnnotations());
115        }
116        code.append("]\n" +
117                "              },\n" +
118                "       init : function(heatmap) {\n");
119        code.append(config.getInit());
120        code.append("$('.footer').css( 'position', 'relative');");
121        code.append("}});");
122
123        return code.toString();
124    }
125
126    private String getQueryUrl() {
127        String serviceMount = collectionManager.getMount();
128        String webserviceUrl = WebsiteApplication.toAbsolutePath('/' + serviceMount);
129        Query query = getQuery();
130
131        String strQuery = query.toString();
132        String fileName = "file-" + Integer.toHexString(strQuery.hashCode()) + ".tsv";
133
134        return webserviceUrl + "?query=" + URLEncoder.encode(strQuery) + "&filename=" + fileName;
135    }
136}
