018package org.onexus.website.api.widgets.columnsets;
019
020import org.apache.wicket.ajax.AjaxRequestTarget;
021import org.apache.wicket.ajax.form.AjaxFormComponentUpdatingBehavior;
022import org.apache.wicket.markup.html.form.DropDownChoice;
023import org.apache.wicket.markup.html.form.Form;
024import org.apache.wicket.model.IModel;
025import org.apache.wicket.model.Model;
026import org.onexus.website.api.events.EventQueryUpdate;
027import org.onexus.website.api.events.EventViewChange;
028import org.onexus.website.api.pages.browser.BrowserPage;
029import org.onexus.website.api.pages.browser.BrowserPageConfig;
030import org.onexus.website.api.pages.browser.BrowserPageStatus;
031import org.onexus.website.api.widgets.Widget;
032import org.onexus.website.api.widgets.WidgetConfig;
033import org.onexus.website.api.widgets.tableviewer.ColumnSet;
034import org.onexus.website.api.widgets.tableviewer.TableViewerConfig;
035import org.onexus.website.api.widgets.tableviewer.TableViewerStatus;
036
037import java.util.List;
038
039public class ColumnsetsWidget extends Widget<ColumnsetsWidgetConfig, ColumnsetsWidgetStatus> {
040
041    public ColumnsetsWidget(String componentId, IModel statusModel) {
042        super(componentId, statusModel);
043
044        onEventFireUpdate(EventQueryUpdate.class);
045
046        Form<String> form = new Form<String>("form");
047        add(form);
048
049        final IModel<ColumnSet> csModel = new Model<ColumnSet>();
050        TableViewerStatus status = getTableViewerStatus();
051
052        int cs = status == null ? 0 : status.getCurrentColumnSet();
053        csModel.setObject(getTableViewerConfig().getColumnSets().get(cs));
054
055        form.add(new AjaxColumnSetSelector("columnsets", csModel, getTableViewerConfig().getColumnSets()));
056
057    }
058
059    private BrowserPageStatus getPageStatus() {
060        return findParent(BrowserPage.class).getStatus();
061    }
062
063    private BrowserPageConfig getPageConfig() {
064        return (BrowserPageConfig) getPageStatus().getConfig();
065    }
066
067    private TableViewerStatus getTableViewerStatus() {
068
069        String widgetId = getTableViewerConfig().getId();
070        return (TableViewerStatus) (widgetId == null ? null : getPageStatus().getWidgetStatus(widgetId));
071
072    }
073
074    private TableViewerConfig getTableViewerConfig() {
075
076        String currentTab = getPageStatus().getCurrentTabId();
077        String currentView = getPageStatus().getCurrentView();
078        String mainWidgetId = getPageConfig().getTab(currentTab).getView(currentView).getMain().trim();
079        WidgetConfig widgetConfig = getPageConfig().getWidget(mainWidgetId);
080
081        if (widgetConfig instanceof TableViewerConfig) {
082            return (TableViewerConfig) widgetConfig;
083        }
084
085        return null;
086    }
087
088    private final class AjaxColumnSetSelector extends DropDownChoice<ColumnSet> {
089
090        private AjaxColumnSetSelector(String id, final IModel<ColumnSet> selectItemModel, List<ColumnSet> listFilters) {
091            super(id, selectItemModel, listFilters);
092
093            setNullValid(false);
094            add(new AjaxFormComponentUpdatingBehavior("onchange") {
095                @Override
096                protected void onUpdate(AjaxRequestTarget target) {
097                    ColumnSet columnSet = (ColumnSet) getDefaultModelObject();
098                    int cs = getTableViewerConfig().getColumnSets().indexOf(columnSet);
099                    getTableViewerStatus().setCurrentColumnSet(cs);
100                    sendEvent(EventViewChange.EVENT);
101                }
102            });
103        }
104    }
105}
