018package org.onexus.website.api.widgets.tags.tagstore;
019
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023import javax.sql.DataSource;
024import java.io.Serializable;
025import java.sql.Connection;
026import java.sql.ResultSet;
027import java.sql.SQLException;
028import java.sql.Statement;
029import java.util.ArrayList;
030import java.util.Collection;
031import java.util.List;
032
033public class TagStore implements Serializable {
034
035    private static final Logger LOGGER = LoggerFactory.getLogger(TagStore.class);
036
037    private String namespace;
038    private String tableName;
039    private DataSource dataSource;
040
041    public TagStore(String namespace, DataSource dataSource) {
042        super();
043
044        this.namespace = namespace;
045        this.tableName = namespace.trim().replaceAll("[^a-zA-Z0-9]", "_");
046        this.dataSource = dataSource;
047
048        createNamespaceTable();
049
050    }
051
052    public String getNamespace() {
053        return namespace;
054    }
055
056    public Collection<String> getTagKeys() {
057
058        String sql = "SELECT DISTINCT `tagKey` AS `tagKey` FROM `" + tableName + "`";
059
060        List<String> tagKeys;
061
062        try {
063            tagKeys = executeCollection(sql, "tagKey");
064        } catch (SQLException e) {
065            String msg = "Error on getTagKeys() <" + sql + ">";
066            LOGGER.error(msg);
067            throw new RuntimeException(msg, e);
068        }
069
070        return tagKeys;
071    }
072
073    public Collection<String> getTagValues(String tagKey) {
074
075        String sql = "SELECT `tagValue` FROM `" + tableName + "` WHERE `tagKey` = '" + tagKey + "'";
076
077        List<String> tagValues;
078        try {
079            tagValues = executeCollection(sql, "tagValue");
080        } catch (SQLException e) {
081            String msg = "Error on getTagValues('" + tagKey + "') <" + sql + ">";
082            LOGGER.error(msg);
083            throw new RuntimeException(msg, e);
084        }
085
086        return tagValues;
087
088    }
089
090    public void putTagValue(String tagKey, String tagValue) {
091
092        Collection<String> values = getTagValues(tagKey);
093
094        // Skip already tagged values
095        if (values.contains(tagValue)) {
096            return;
097        }
098
099        String sql = "INSERT INTO `" + tableName + "` (`tagKey`, `tagValue`) VALUES ('" + tagKey + "', '" + tagValue
100                + "')";
101
102        try {
103            executeSQL(sql);
104        } catch (SQLException e) {
105            String msg = "Error on putTagValue('" + tagKey + "', '" + tagValue + "') <" + sql + ">";
106            LOGGER.error(msg);
107            throw new RuntimeException(msg, e);
108        }
109
110    }
111
112    public void removeTag(String tagKey) {
113
114        String sql = "DELETE FROM `" + tableName + "` WHERE `tagKey` = '" + tagKey + "'";
115
116        try {
117            executeSQL(sql);
118        } catch (SQLException e) {
119            String msg = "Error on removeTag('" + tagKey + "') <" + sql + ">";
120            LOGGER.error(msg);
121            throw new RuntimeException(msg, e);
122        }
123
124    }
125
126    public void putTagKey(String tagKey) {
127
128        String sql = "INSERT INTO `" + tableName + "` (`tagKey`, `tagValue`) VALUES ('" + tagKey + "', NULL)";
129
130        try {
131            executeSQL(sql);
132        } catch (SQLException e) {
133            String msg = "Error on putTagKey('" + tagKey + "') <" + sql + ">";
134            LOGGER.error(msg);
135            throw new RuntimeException(msg, e);
136        }
137
138    }
139
140    public List<String> getTagKeysByValue(String tagValue) {
141
142        String sql = "SELECT `tagKey` FROM `" + tableName + "` WHERE `tagValue` = '" + tagValue + "'";
143
144        List<String> tagKeys;
145        try {
146            tagKeys = executeCollection(sql, "tagKey");
147        } catch (SQLException e) {
148            String msg = "Error on getTagKeysByValues('" + tagValue + "') <" + sql + ">";
149            LOGGER.error(msg);
150            throw new RuntimeException(msg, e);
151        }
152
153        return tagKeys;
154    }
155
156    public void removeTagValue(String tagKey, String tagValue) {
157
158        String sql = "DELETE FROM `" + tableName + "` WHERE `tagKey` = '" + tagKey + "' AND `tagValue` = '"
159                + tagValue + "'";
160
161        try {
162            executeSQL(sql);
163        } catch (SQLException e) {
164            String msg = "Error on removeTagValue('" + tagKey + "', '" + tagValue + "') <" + sql + ">";
165            LOGGER.error(msg);
166            throw new RuntimeException(msg, e);
167        }
168
169    }
170
171    private void createNamespaceTable() {
172
173        StringBuilder sql = new StringBuilder();
174        sql.append("CREATE TABLE IF NOT EXISTS `")
175                .append(tableName)
176                .append("` ( `tagKey` varchar(128) NOT NULL, `tagValue` varchar(128) DEFAULT NULL, UNIQUE (`tagKey`, `tagValue`))");
177
178        try {
179            executeSQL(sql.toString());
180        } catch (SQLException e) {
181            String msg = "Error creating table for namespace '" + namespace + "' with SQL <" + sql.toString() + ">";
182            LOGGER.error(msg, e);
183            throw new RuntimeException(msg, e);
184        }
185    }
186
187    private void executeSQL(String sql) throws SQLException {
188
189        Connection conn = dataSource.getConnection();
190        Statement st = conn.createStatement();
191        st.execute(sql);
192        st.close();
193        conn.close();
194
195    }
196
197    private List<String> executeCollection(String sql, String fieldName) throws SQLException {
198
199        Connection conn = dataSource.getConnection();
200        Statement st = conn.createStatement();
201        ResultSet rs = st.executeQuery(sql);
202
203        List<String> result = new ArrayList<String>();
204        while (rs.next()) {
205            String value = rs.getString(fieldName);
206            if (value != null) {
207                result.add(value);
208            }
209        }
210
211        rs.close();
212        st.close();
213        conn.close();
214
215        return result;
216    }
217
218}
