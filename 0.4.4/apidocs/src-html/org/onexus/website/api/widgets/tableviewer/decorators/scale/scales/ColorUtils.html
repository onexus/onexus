018package org.onexus.website.api.widgets.tableviewer.decorators.scale.scales;
019
020import java.awt.*;
021
022public class ColorUtils {
023
024    public static Color mix(Color src, Color dst, double factor) {
025
026        double fs = factor / 255.0;
027        double fd = (1.0 - factor) / 255.0;
028
029        double r = src.getRed() * fs + dst.getRed() * fd;
030        double g = src.getGreen() * fs + dst.getGreen() * fd;
031        double b = src.getBlue() * fs + dst.getBlue() * fd;
032
033        int ir = Math.max(0, Math.min(255, (int) Math.round(r * 255)));
034        int ig = Math.max(0, Math.min(255, (int) Math.round(g * 255)));
035        int ib = Math.max(0, Math.min(255, (int) Math.round(b * 255)));
036
037        return new Color(ir, ig, ib);
038    }
039
040    public static String colorToRGBHtml(Color color) {
041        StringBuilder sb = new StringBuilder();
042
043        sb.append("rgb(");
044        sb.append(color.getRed()).append(',');
045        sb.append(color.getGreen()).append(',');
046        sb.append(color.getBlue()).append(')');
047
048        return sb.toString();
049    }
050
051    public static String colorToHexHtml(Color color) {
052        return "#"
053                + Integer.toHexString((color.getRGB() & 0xffffff) | 0x1000000)
054                .substring(1);
055    }
056
057    public static Color stringToColor(String color) {
058
059        if (color == null) {
060            return null;
061        }
062
063        int r = 0, g = 0, b = 0;
064        try {
065            String tmpcolor = color.trim();
066            tmpcolor = tmpcolor.substring(1, tmpcolor.length() - 1);
067            String rgb[] = tmpcolor.split(",");
068
069            r = Integer.valueOf(rgb[0]).intValue();
070            g = Integer.valueOf(rgb[1]).intValue();
071            b = Integer.valueOf(rgb[2]).intValue();
072
073            return new Color(r, g, b);
074        } catch (Exception e) {
075            throw new UnsupportedOperationException("The color '" + color + "' is malformed. Syntax: [r,g,b] Example: \"[128,12,34]\"", e);
076        }
077    }
078}
