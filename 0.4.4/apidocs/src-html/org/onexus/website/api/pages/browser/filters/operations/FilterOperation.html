001package org.onexus.website.api.pages.browser.filters.operations;
002
003import org.onexus.collection.api.query.*;
004
005import java.io.Serializable;
006
007public abstract class FilterOperation implements Serializable {
008
009    private String label;
010    private String symbol;
011    private boolean needsValue;
012
013    protected FilterOperation(String label, String symbol, boolean needsValue) {
014        this.label = label;
015        this.symbol = symbol;
016        this.needsValue = needsValue;
017    }
018
019    public String getLabel() {
020        return label;
021    }
022
023    public String getSymbol() {
024        return symbol;
025    }
026
027    public boolean isNeedsValue() {
028        return needsValue;
029    }
030
031    public String toString() {
032        return label;
033    }
034
035    public abstract Filter createFilter(String alias, String fieldId, Object value);
036
037    public String createTitle(String headerTitle, Object value) {
038        return headerTitle + " " + symbol + (needsValue ? " '" + value + "'" : "");
039    }
040}
