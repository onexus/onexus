055public class BrowserPage extends Page<BrowserPageConfig, BrowserPageStatus> {
056
057    public BrowserPage(String componentId, IModel<BrowserPageStatus> statusModel) {
058        super(componentId, statusModel);
059        onEventFireUpdate(EventTabSelected.class, EventAddFilter.class, EventRemoveFilter.class);
060
061        add(new SelectionPanel("selection", statusModel));
062
063        onEventFireUpdate(EventViewChange.class);
064    }

066    protected boolean isCurrentTab(String tabId) {
067        return tabId.equals(getStatus().getCurrentTabId());
068    }
069
070    protected TabConfig getCurrentTab() {
071        String currentTabId = getStatus().getCurrentTabId();
072        return getConfig().getTab(currentTabId);
073    }
074
075    @Override
076    protected void onBeforeRender() {
077
078        VisibleTabs visibleTabs = new VisibleTabs();
079
080        // Tabs can change when we fix/unfix entities
081        addOrReplace(new ListView<TabGroup>("tabs", visibleTabs) {
082
083            @Override
084            protected void populateItem(ListItem<TabGroup> item) {
085
086                TabGroup tabGroup = item.getModelObject();
087
088                if (tabGroup.hasSubMenu()) {
089
090                    WebMarkupContainer link = new WebMarkupContainer("link");
091                    link.add(new AttributeModifier("class", "dropdown-toggle"));
092                    link.add(new AttributeModifier("data-toggle", "dropdown"));
093                    link.add(new AttributeModifier("href", "#"));
094
095                    if (tabGroup.containsTab(getStatus().getCurrentTabId())) {
096                        item.add(new AttributeModifier("class", "dropdown active"));
097
098                        TabConfig currentTab = getConfig().getTab(getStatus().getCurrentTabId());
099                        String tabTitle = StringUtils.abbreviate(currentTab.getTitle(), 14);
100                        String currentLabel = "<div style=\"position:relative;\"><div style=\"white-space: nowrap; position:absolute; left:13px; top: 13px; font-size: 9px;\"><em>" + tabTitle + "</em></div></div>";
101                        link.add(new Label("label", currentLabel + "<b class='caret'></b>&nbsp;" + tabGroup.getGroupLabel()).setEscapeModelStrings(false));
102
103                    } else {
104                        item.add(new AttributeModifier("class", "dropdown"));
105                        link.add(new Label("label", "<b class='caret'></b>&nbsp;" + tabGroup.getGroupLabel()).setEscapeModelStrings(false));
106                    }
107                    item.add(link);
108
109
110                    WebMarkupContainer submenu = new WebMarkupContainer("submenu");
111                    submenu.add(new ListView<TabConfig>("item", tabGroup.getTabConfigs()) {
112
113                        @Override
114                        protected void populateItem(ListItem<TabConfig> item) {
115
116                            TabConfig tab = item.getModelObject();
117
118
119                            BrowserPageLink<String> link = new BrowserPageLink<String>("link", Model.of(tab.getId())) {
120
121                                @Override
122                                public void onClick(AjaxRequestTarget target) {
123                                    getStatus().setCurrentTabId(getModelObject());
124                                    sendEvent(EventTabSelected.EVENT);
125                                }
126
127                            };
128
129                            if (!Strings.isEmpty(tab.getHelp())) {
130                                item.add(new AttributeModifier("rel", "tooltip"));
131                                item.add(new AttributeModifier("data-placement", "right"));
132                                item.add(new AttributeModifier("title", tab.getHelp()));
133                            }
134
135                            link.add(new Label("label", tab.getTitle()));
136                            item.add(link);
137
138                            if (isCurrentTab(tab.getId())) {
139                                item.add(new AttributeModifier("class", new Model<String>("active")));
140                            }
141
142                        }
143                    });
144                    item.add(submenu);
145
146                } else {
147
148                    TabConfig tab = tabGroup.getTabConfigs().get(0);
149                    BrowserPageLink<String> link = new BrowserPageLink<String>("link", Model.of(tab.getId())) {
150
151                        @Override
152                        public void onClick(AjaxRequestTarget target) {
153                            getStatus().setCurrentTabId(getModelObject());
154                            sendEvent(EventTabSelected.EVENT);
155                        }
156
157                    };
158
159                    if (!Strings.isEmpty(tab.getHelp())) {
160                        link.add(new AttributeModifier("rel", "tooltip"));
161                        link.add(new AttributeModifier("title", tab.getHelp()));
162                    }
163
164                    link.add(new Label("label", tab.getTitle()));
165                    item.add(link);
166
167                    if (isCurrentTab(tab.getId())) {
168                        item.add(new AttributeModifier("class", new Model<String>("active")));
169                    }
170
171                    item.add(new WebMarkupContainer("submenu").setVisible(false));
172
173                }
174
175            }
176
177        });
178
179        // Check if the current selected tab is visible.
180        String currentTabId = getStatus().getCurrentTabId();
181
182        // Set a current tab if there is no one.
183        if (currentTabId == null) {
184
185            List<TabConfig> tabs = getConfig().getTabs();
186            if (tabs != null && !tabs.isEmpty()) {
187                currentTabId = tabs.get(0).getId();
188                getStatus().setCurrentTabId(currentTabId);
189
190                // Init currentPage
191                //TODO This is not a good solution
192                PageParameters pageParameters = getPage().getPageParameters();
193                pageParameters.set(Website.PARAMETER_CURRENT_PAGE, getConfig().getId());
194                pageParameters.set("ptab", currentTabId);
195                setResponsePage(Website.class, pageParameters);
196            }
197
198        }
199
200        List<TabGroup> tabs = visibleTabs.getObject();
201        boolean hiddenTab = true;
202        for (TabGroup tab : tabs) {
203            if (tab.containsTab(currentTabId)) {
204                hiddenTab = false;
205            }
206        }
207
208        if (hiddenTab && !tabs.isEmpty()) {
209            TabConfig firstTab = tabs.get(0).getTabConfigs().get(0);
210            getStatus().setCurrentTabId(firstTab.getId());
211        }
212
213
214        List<String> views = new ArrayList<String>();
215        if (getCurrentTab().getViews() != null) {
216            for (ViewConfig view : getCurrentTab().getViews()) {
217                views.add(view.getTitle());
218            }
219        }
220
221
222        if (getStatus().getCurrentView() == null && views.size() > 0) {
223            getStatus().setCurrentView(views.get(0));
224        }
225
226        // Check that the current view is visible
227        List<ViewConfig> filteredViews = new ArrayList<ViewConfig>();
228        VisiblePredicate predicate = new VisiblePredicate(getStatus().getORI().getParent(), getStatus().getEntitySelections());
229        CollectionUtils.select(getCurrentTab().getViews(), predicate, filteredViews);
230
231        boolean visibleView = false;
232        String currentView = getStatus().getCurrentView();
233        for (ViewConfig view : filteredViews) {
234            if (view.getTitle().equals(currentView)) {
235                visibleView = true;
236                break;
237            }
238        }
239
240        if (!visibleView) {
241            getStatus().setCurrentView(filteredViews.get(0).getTitle());
242        }
243
244        ViewConfig viewConfig = null;
245        if (getCurrentTab().getViews() != null) {
246            for (ViewConfig view : getCurrentTab().getViews()) {
247                if (view.getTitle().equals(getStatus().getCurrentView())) {
248                    viewConfig = view;
249                    break;
250                }
251            }
252        }
253
254        if (viewConfig == null) {
255            addOrReplace(new EmptyPanel("content"));
256        } else {
257            if (viewConfig.getLeft() != null && viewConfig.getTop() != null) {
258                addOrReplace(new TopleftLayout("content", viewConfig, getModel()));
259            } else if (viewConfig.getLeft() != null) {
260                addOrReplace(new LeftMainLayout("content", viewConfig, getModel()));
261            } else if (viewConfig.getTop() != null || viewConfig.getTopRight() != null) {
262                addOrReplace(new TopmainLayout("content", viewConfig, getModel()));
263            } else {
264                addOrReplace(new SingleLayout("content", viewConfig, getModel()));
265            }
266        }
267
268        boolean visible = !isEmbed();
269        get("selection").setVisible(visible);
270        get("tabs").setVisible(visible);
271        //TODO viewSelector.setVisible(visible && views.size() > 1);
272
273        super.onBeforeRender();
274    }
275

278        super.renderHead(response);
279
280        if (isEmbed()) {
281            response.render(CssHeaderItem.forCSS("div.tab-row { display: none; }", "embed-browser-page"));
282        }
283    }

285    private class VisibleTabs extends AbstractReadOnlyModel<List<TabGroup>> {
286
287        @Override
288        public List<TabGroup> getObject() {
289
290            List<TabConfig> allTabs = getConfig().getTabs();
291
292            // A predicate that filters the visible views
293            Predicate filter = new VisiblePredicate(getStatus().getORI().getParent(), getStatus().getEntitySelections());
294
295            // Return a new collection with only the visible tabs
296            List<TabConfig> tabs = new ArrayList<TabConfig>();
297            CollectionUtils.select(allTabs, filter, tabs);
298
299            // Tab groups
300            List<TabGroup> tabGroups = new ArrayList<TabGroup>();
301            Map<String, TabGroup> subMenus = new HashMap<String, TabGroup>();
302
303            for (TabConfig tab : tabs) {
304                if (tab.getGroup() == null) {
305                    tabGroups.add(new TabGroup(tab));
306                } else {
307                    TabGroup group;
308                    if (!subMenus.containsKey(tab.getGroup())) {
309                        group = new TabGroup(tab.getGroup());
310                        subMenus.put(tab.getGroup(), group);
311                        tabGroups.add(group);
312                    } else {
313                        group = subMenus.get(tab.getGroup());
314                    }
315                    group.add(tab);
316                }
317            }
318
319            return tabGroups;
320        }
321
322    }

324    private static class TabGroup implements Serializable {
325
326        private String groupLabel;
327        private List<TabConfig> tabConfigs = new ArrayList<TabConfig>();
328
329        public TabGroup(String groupLabel) {
330            super();
331            this.groupLabel = groupLabel;
332        }
333
334        public TabGroup(TabConfig tab) {
335            super();
336            this.groupLabel = null;
337            this.tabConfigs.add(tab);
338        }
339
340        public boolean hasSubMenu() {
341            return groupLabel != null;
342        }
343
344        public void add(TabConfig tabConfig) {
345            tabConfigs.add(tabConfig);
346        }
347
348        public boolean containsTab(String tabId) {
349            for (TabConfig tab : tabConfigs) {
350                if (tab.getId().equals(tabId)) {
351                    return true;
352                }
353            }
354            return false;
355        }
356
357        public String getGroupLabel() {
358            return groupLabel;
359        }
360
361        public List<TabConfig> getTabConfigs() {
362            return tabConfigs;
363        }
364    }
