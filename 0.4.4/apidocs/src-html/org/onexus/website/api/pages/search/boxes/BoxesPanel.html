054public class BoxesPanel extends Panel {
055
056
057
058    @Inject
059    private ICollectionManager collectionManager;
060
061    private boolean quickList;
062
063    public BoxesPanel(String id, SearchPageStatus status, ORI baseUri, FilterConfig filterConfig) {
064        super(id);
065        setMarkupId("boxes");
066        setOutputMarkupId(true);
067
068        add(new AttributeModifier("class", "accordion"));
069
070        RepeatingView boxes = new RepeatingView("boxes");
071
072        SearchType type = status.getType();
073
074        if (Strings.isEmpty(status.getSearch())) {
075
076            // Nothing selected
077            add(new EmptyPanel("disambiguation").setVisible(false));
078            List<SearchLink> links = type.getFixLinks();
079            if (links != null && !links.isEmpty()) {
080                boxes.add(new MainLinksBox(boxes.newChildId(), links));
081            }
082
083            if (type.getFigures() != null) {
084                for (FigureConfig figure : type.getFigures()) {
085                    if (!Strings.isEmpty(figure.getVisible()) && "NONE".equalsIgnoreCase(figure.getVisible())) {
086                        boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, null));
087                    }
088                }
089            }
090
091        } else {
092            ORI collectionUri = type.getCollection().toAbsolute(baseUri);
093            if (filterConfig == null && status.getSearch().indexOf(',') == -1) {
094
095
096                IEntity entity = null;
097                IEntityTable table = null;
098                if (status.getSearch().startsWith(DisambiguationPanel.PREFIX)) {
099
100                    String entityId = status.getSearch().replace(DisambiguationPanel.PREFIX, "");
101                    entity = new EntityModel(collectionUri, entityId).getObject();
102
103                } else {
104                    // Single entity selection
105                    table = getSingleEntityTable(collectionManager, type, collectionUri, baseUri, status.getSearch(), true);
106
107                    boolean found;
108                    if (table.next()) {
109                        found = true;
110                    } else {
111
112                        // If we don't found an exact match, look for a similar one
113                        table.close();
114                        table = getSingleEntityTable(collectionManager, type, collectionUri, baseUri, status.getSearch(), false);
115                        found = table.next();
116                    }
117
118                    if (found) {
119                        entity = table.getEntity(collectionUri);
120                    }
121                }
122
123                if (entity != null) {
124
125                    boxes.add(new LinksBox(boxes.newChildId(), status, entity));
126
127                    if (type.getFigures() != null) {
128                        for (FigureConfig figure : type.getFigures()) {
129                            if (Strings.isEmpty(figure.getVisible()) || "SINGLE".equalsIgnoreCase(figure.getVisible())) {
130                                boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, new SingleEntitySelection(entity)));
131                            }
132                        }
133                    }
134
135                    if (table != null && table.next()) {
136                        add(new DisambiguationPanel("disambiguation", table, collectionUri) {
137
138                            @Override
139                            protected void onSelection(AjaxRequestTarget target, String newSearch) {
140                                onDisambiguation(target, newSearch);
141                            }
142                        });
143                    } else {
144                        add(new EmptyPanel("disambiguation").setVisible(false));
145                    }
146
147                } else {
148                    add(new EmptyPanel("disambiguation").setVisible(false));
149                    boxes.add(new Label(boxes.newChildId(), "No results found").add(new AttributeModifier("class", "alert")));
150                }
151
152                if (table != null) {
153                    table.close();
154                }
155
156            } else {
157
158                // Multiple entities selection
159
160                add(new EmptyPanel("disambiguation").setVisible(false));
161
162                quickList = false;
163                if (filterConfig == null) {
164                    quickList = true;
165                    filterConfig = new FilterConfig(status.getSearch());
166
167                    filterConfig.setCollection(collectionUri);
168                    filterConfig.setDefine("fc='" + collectionUri + "'");
169                    String mainKey = type.getKeysList().get(0);
170                    In where = new In("fc", mainKey);
171                    String[] values = status.getSearch().split(",");
172                    for (String value : values) {
173                        where.addValue(value.trim());
174                    }
175                    filterConfig.setWhere(where.toString());
176                }
177
178                IEntityTable table = getMultipleEntityTable(collectionManager, type, collectionUri, baseUri, filterConfig);
179                boxes.add(new LinksBox(boxes.newChildId(), status, collectionUri, filterConfig, new EntityIterator(table, collectionUri)) {
180                    @Override
181                    protected void onNotFound(Set<String> valuesNotFound) {
182                        if (valuesNotFound.isEmpty() || !quickList) {
183                            BoxesPanel.this.addOrReplace(new EmptyPanel("disambiguation").setVisible(false));
184                        } else {
185                            BoxesPanel.this.addOrReplace(new DisambiguationPanel("disambiguation", valuesNotFound) {
186                                @Override
187                                protected void onSelection(AjaxRequestTarget target, String newSearch) {
188                                    onDisambiguation(target, newSearch);
189                                }
190                            });
191                        }
192                    }
193                });
194
195                if (type.getFigures() != null) {
196                    for (FigureConfig figure : type.getFigures()) {
197                        if (Strings.isEmpty(figure.getVisible()) || "LIST".equalsIgnoreCase(figure.getVisible())) {
198                            boxes.add(new FigureBox(boxes.newChildId(), figure, baseUri, new MultipleEntitySelection(filterConfig)));
199                        }
200                    }
201                }
202            }
203        }
204
205        add(boxes);
206
207    }
208
209    protected void onDisambiguation(AjaxRequestTarget target, String query) {
210    }
211
212    private static IEntityTable getMultipleEntityTable(ICollectionManager collectionManager, SearchType type, ORI collectionUri, ORI baseOri, FilterConfig filter) {
213
214        Query query = new Query();
215        query.setOn(baseOri);
216
217        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
218        query.setFrom(collectionAlias);
219
220
221        query.addSelect(collectionAlias, null);
222
223        IEntitySelection selection = new MultipleEntitySelection(filter);
224        query.setWhere(selection.buildFilter(query));
225
226        List<String> fieldList = type.getFieldsList();
227        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
228
229        return collectionManager.load(query);
230    }
231
232
233    public static IEntityTable getSingleEntityTable(ICollectionManager collectionManager, SearchType type, ORI collectionUri, ORI baseUri, String search, boolean equal) {
234
235        Query query = new Query();
236        query.setOn(baseUri);
237
238        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
239        query.setFrom(collectionAlias);
240
241        List<String> fieldList = type.getFieldsList();
242        query.addSelect(collectionAlias, null);
243
244        for (String field : fieldList) {
245            if (equal) {
246                QueryUtils.or(query, new Equal(collectionAlias, field, search));
247            } else {
248                QueryUtils.or(query, new Contains(collectionAlias, field, search));
249            }
250        }
251
252        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
253        query.setCount(20);
254
255        return collectionManager.load(query);
256    }
