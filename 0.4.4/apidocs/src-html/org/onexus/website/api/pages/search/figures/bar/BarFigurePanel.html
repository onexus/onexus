038import javax.inject.Inject;

045
046public class BarFigurePanel extends Panel {
047
048    @Inject
049    private ICollectionManager collectionManager;
050
051    private ORI parentOri;
052    private IEntitySelection selection;
053    private BarFigureConfig config;
054
055    private WebMarkupContainer chart;
056    private static final HeaderItem JS_HIGHCHARTS = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(BarFigurePanel.class, "highcharts.js"));
057
058    public BarFigurePanel(String id, ORI parentOri, IEntitySelection selection, BarFigureConfig config) {
059        super(id);
060
061        this.parentOri = parentOri;
062        this.selection = selection;
063        this.config = config;
064
065        chart = new WebMarkupContainer("chart");
066        chart.setOutputMarkupId(true);
067
068        add(chart);
069    }
070
071    @Override
072    public void renderHead(IHeaderResponse response) {
073        response.render(JS_HIGHCHARTS);
074        response.render(OnLoadHeaderItem.forScript(newJavaScript()));
075    }
076
077    private Query createQuery() {
078
079        Query query = new Query();
080        query.setOn(parentOri);
081
082        CollectionField value = config.getValue();
083        CollectionField xAxis = config.getxAxis();
084        CollectionField yAxis = config.getyAxis();
085
086        String valueAlias = QueryUtils.newCollectionAlias(query, value.getCollection());
087        String xAxisAlias = QueryUtils.newCollectionAlias(query, xAxis.getCollection());
088        String yAxisAlias = QueryUtils.newCollectionAlias(query, yAxis.getCollection());
089
090        query.setFrom(valueAlias);
091
092        query.addSelect(valueAlias, Arrays.asList(value.getField()));
093        query.addSelect(xAxisAlias, Arrays.asList(xAxis.getField()));
094        query.addSelect(yAxisAlias, Arrays.asList(yAxis.getField()));
095
096        if (selection != null) {
097            Filter filter = selection.buildFilter(query);
098            QueryUtils.and(query, filter);
099        }
100
101        return query;
102    }
103
104    private String newJavaScript() {
105        StringBuilder code = new StringBuilder();
106
107        Query query = createQuery();
108
109        CollectionField value = config.getValue();
110        CollectionField xAxis = config.getxAxis();
111        CollectionField yAxis = config.getyAxis();
112        IEntityTable table = collectionManager.load(query);
113
114
115        List<String> categories = new ArrayList<String>();
116        Map<String, Map<String, String>> values = new HashMap<String, Map<String, String>>();
117        while (table.next()) {
118
119            String xAxisData = String.valueOf(table.getEntity(xAxis.getCollection()).get(xAxis.getField()));
120            String yAxisData = String.valueOf(table.getEntity(yAxis.getCollection()).get(yAxis.getField()));
121            String valueData = String.valueOf(table.getEntity(value.getCollection()).get(value.getField()));
122
123            if (!categories.contains(xAxisData)) {
124                categories.add(xAxisData);
125            }
126
127            if (!values.containsKey(yAxisData)) {
128                values.put(yAxisData, new HashMap<String, String>());
129            }
130
131            values.get(yAxisData).put(xAxisData, valueData);
132        }
133        table.close();
134
135        code.append("$(function () { $('#");
136
137        code.append(chart.getMarkupId());
138
139        String initUrl = config.getInit();
140        String init;
141
142        if (Strings.isEmpty(initUrl)) {
143            init = "   chart: {\n" +
144                    "    type: 'bar'\n" +
145                    "    },\n" +
146                    "\n" +
147                    "    title: {\n" +
148                    "    text: null\n" +
149                    "    },\n" +
150                    "\n" +
151                    "    xAxis: {\n" +
152                    "    labels: {\n" +
153                    "                    enabled: " + (categories.size() > 30 ? "false" : "true") + "\n" +
154                    "                }," +
155                    "    categories: [ ${categories} ]\n" +
156                    "    },\n" +
157                    "    yAxis: {\n" +
158                    "    min: 0,\n" +
159                    "    title: {\n" +
160                    "    text: 'mutated samples'\n" +
161                    "    }\n" +
162                    "    },\n" +
163                    "    tooltip: {\n" +
164                    "    valueDecimals: 0,\n" +
165                    "    valueSuffix: ' samples'\n" +
166                    "    },\n" +
167                    "    plotOptions: {\n" +
168                    "    series: {\n" +
169                    "    stacking: 'normal'\n" +
170                    "    }\n" +
171                    "    },\n" +
172                    "    legend: {\n" +
173                    "    layout: 'vertical',\n" +
174                    "    align: 'right',\n" +
175                    "    verticalAlign: 'top',\n" +
176                    "    x: 0,\n" +
177                    "    y: 30,\n" +
178                    "    floating: false,\n" +
179                    "    borderWidth: 1,\n" +
180                    "    backgroundColor: '#FFFFFF',\n" +
181                    "    shadow: true,\n" +
182                    "    labelFormatter: function() {\n" +
183                    "                return this.name.length>20 ? this.name.substr(0,19)+'...' : this.name;\n" +
184                    "            }" +
185                    "    },\n" +
186                    "    credits: {\n" +
187                    "    enabled: false\n" +
188                    "    },\n" +
189                    "    series: [ ${series} ] ";
190        } else {
191            IModel<String> initModel = new HtmlDataResourceModel(parentOri, initUrl);
192            init = initModel.getObject();
193        }
194
195        // Build categories
196        StringBuilder categoriesText = new StringBuilder();
197        Iterator<String> categoriesIterator = categories.iterator();
198        while (categoriesIterator.hasNext()) {
199            categoriesText.append("'").append(categoriesIterator.next()).append("'");
200            if (categoriesIterator.hasNext()) {
201                categoriesText.append(", ");
202            }
203        }
204
205        init = init.replace("${categories}", categoriesText.toString());
206
207        // Build series
208        StringBuilder seriesText = new StringBuilder();
209        Iterator<String> seriesIterator = values.keySet().iterator();
210        while (seriesIterator.hasNext()) {
211            String serieName = seriesIterator.next();
212            seriesText.append("{ name: '").append(serieName).append("'\n, data: [");
213
214            categoriesIterator = categories.iterator();
215            while (categoriesIterator.hasNext()) {
216                seriesText.append(values.get(serieName).get(categoriesIterator.next()));
217                if (categoriesIterator.hasNext()) {
218                    seriesText.append(", ");
219                }
220            }
221
222            if (seriesIterator.hasNext()) {
223                seriesText.append("] }, \n");
224            } else {
225                seriesText.append("] }\n");
226            }
227        }
228        init = init.replace("${series}", seriesText.toString());
229
230        code.append("').highcharts({" + init + "});})");
231
232        return code.toString();
233    }
