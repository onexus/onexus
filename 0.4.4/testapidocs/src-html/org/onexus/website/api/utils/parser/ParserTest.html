018package org.onexus.website.api.utils.parser;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertFalse;
022import static org.junit.Assert.assertTrue;
023import org.junit.Test;
024
025public class ParserTest {
026
027    @Test
028    public void testStringTokenizer() {
029
030        StringTokenizer tokenizer = new StringTokenizer("  c1 (c2    AND c3),c4 OR (c5,c6)    ,     c7   ", "(),");
031
032        assertTrue(tokenizer.hasNext());
033        assertEquals(tokenizer.next(), "c1");
034        assertEquals(tokenizer.next(), "(");
035        assertEquals(tokenizer.next(), "c2");
036        assertEquals(tokenizer.next(), "AND");
037        assertEquals(tokenizer.next(), "c3");
038        assertEquals(tokenizer.next(), ")");
039        assertEquals(tokenizer.next(), ",");
040        assertEquals(tokenizer.next(), "c4");
041        assertEquals(tokenizer.next(), "OR");
042        assertEquals(tokenizer.next(), "(");
043        assertEquals(tokenizer.next(), "c5");
044        assertEquals(tokenizer.next(), ",");
045        assertEquals(tokenizer.next(), "c6");
046        assertEquals(tokenizer.next(), ")");
047        assertEquals(tokenizer.next(), ",");
048        assertEquals(tokenizer.next(), "c7");
049        assertFalse(tokenizer.hasNext());
050
051    }
052
053    @Test
054    public void testBooleanExpressionEvaluator() {
055
056        assertTrue(evaluate("c1 AND c2", "c1", "c2"));
057        assertTrue(evaluate("c1, c2", "c1", "c2"));
058        assertFalse(evaluate("c1 AND (NOT c2)", "c1", "c2"));
059        assertFalse(evaluate("c1, !c2", "c1", "c2"));
060        assertTrue(evaluate("(c1 AND c2)", "c1", "c2"));
061
062        assertTrue(evaluate("c1 OR c2", "c2"));
063        assertFalse(evaluate("c1 OR NOT c2", "c2"));
064        assertTrue(evaluate("(c1 OR c2)", "c2"));
065
066        assertTrue(evaluate("c1 AND c2 OR c3", "c3"));
067        assertTrue(evaluate("(c1 AND c2) OR c3", "c3"));
068        assertFalse(evaluate("c1 AND (c2 OR c3)", "c3"));
069        assertFalse(evaluate("c1 AND c2 OR NOT c3", "c3"));
070
071        assertTrue(evaluate("((c1 AND c2) OR (c3 AND (!c5 OR c6)))", "c2", "c5", "c6", "c3"));
072
073        assertTrue(evaluate("NOT !c1", "c1"));
074        assertTrue(evaluate("!!!c1"));
075        assertFalse(evaluate("!!!(!!!)!!!!"));
076        assertTrue(evaluate("", "c2"));
077
078        assertTrue(evaluate("NOT c1 OR c2"));
079        assertTrue(evaluate("NOT c1 OR c2", "c2"));
080
081        assertTrue(evaluate("(c2 OR !c1)"));
082        assertTrue(evaluate("(c2 OR !c1)", "c2"));
083
084    }
085
086    private boolean evaluate(String expression, String... trueValues) {
087        BooleanExpressionEvaluator evaluator = new BooleanExpressionEvaluator(expression, trueValues);
088        return evaluator.evaluate();
089    }
090}
