018package org.onexus.website.api;
019
020import org.apache.wicket.request.mapper.parameter.PageParameters;
021import org.apache.wicket.util.tester.WicketTester;
022import org.junit.Before;
023import org.junit.Test;
024import org.onexus.resource.api.DefaultResourceRegister;
025import org.onexus.resource.api.IResourceManager;
026import org.onexus.resource.api.IResourceRegister;
027import org.onexus.resource.api.IResourceSerializer;
028import org.onexus.resource.api.ORI;
029import org.onexus.resource.serializer.xstream.internal.ResourceSerializer;
030import org.onexus.ui.authentication.jaas.JaasSignInPage;
031import org.onexus.website.api.mocks.MockPaxWicketInjector;
032import org.onexus.website.api.mocks.MockResourceManager;
033import org.onexus.website.api.pages.DefaultPageManager;
034import org.onexus.website.api.pages.IPageManager;
035import org.onexus.website.api.pages.browser.BrowserPageCreator;
036import org.onexus.website.api.widgets.DefaultWidgetManager;
037import org.onexus.website.api.widgets.IWidgetManager;
038import org.onexus.website.api.widgets.text.TextWidgetCreator;
039import org.ops4j.pax.wicket.api.InjectorHolder;
040
041import java.util.Arrays;
042
043public class WebsiteTest {
044
045    private WicketTester tester;
046
047    @Before
048    public void setUp() {
049
050        // Create beans
051        IResourceSerializer resourceSerializer = new ResourceSerializer();
052        resourceSerializer.register(WebsiteConfig.class);
053
054        IResourceRegister resourceRegister = new DefaultResourceRegister(resourceSerializer);
055        IResourceManager resourceManager = new MockResourceManager();
056
057        IPageManager pageManager = new DefaultPageManager(
058                Arrays.asList(
059                    new BrowserPageCreator()
060                ),
061                resourceRegister
062        );
063
064        IWidgetManager widgetManager = new DefaultWidgetManager(
065                Arrays.asList(
066                        new TextWidgetCreator()
067                ),
068                resourceRegister
069        );
070
071        // Create a Website resource
072        WebsiteConfig website = resourceSerializer.unserialize(WebsiteConfig.class, getClass().getResourceAsStream("test1.onx"));
073        website.setORI(new ORI("http://localhost/test?website"));
074        resourceManager.save(website);
075
076        // Create and initialize a Wicket application
077        WebsiteApplication application = new WebsiteApplication();
078        application.setWebsiteOri(website.getORI());
079        application.setWebPath(website.getName());
080        application.setSignInPageClass(JaasSignInPage.class);
081
082        tester = new WicketTester(application);
083
084        // Create an injector
085        MockPaxWicketInjector injector = new MockPaxWicketInjector();
086        injector.addBean(IResourceManager.class, resourceManager);
087        injector.addBean(IPageManager.class, pageManager);
088        injector.addBean(IWidgetManager.class, widgetManager);
089
090        InjectorHolder.setInjector(application.getApplicationKey(), injector);
091        tester.getApplication().getComponentInstantiationListeners().add(injector);
092
093    }
094
095    @Test
096    public void renderPage(){
097        tester.startPage(new Website(new PageParameters()));
098        tester.assertRenderedPage(Website.class);
099        //tester.dumpPage();
100    }
101
102}
