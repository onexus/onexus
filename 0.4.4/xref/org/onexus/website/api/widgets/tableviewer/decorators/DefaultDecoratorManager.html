33  public class DefaultDecoratorManager implements IDecoratorManager {
34  
35      private static final Logger LOGGER = LoggerFactory.getLogger(DefaultDecoratorManager.class);
36      private static final String ANNOTATION_KEY = "BROWSER_DECORATOR";
37  
38      private List<IDecoratorCreator> creators;
39  
40  
41      @Override
42      public IDecorator getDecorator(String decorator, Collection collection, Field field) {
43  
44          if (decorator == null && field != null) {
45              decorator = field.getProperty(ANNOTATION_KEY);
46          }
47  
48          IDecoratorCreator creator = null;
49          Parameters parameters = new Parameters();
50  
51          // Look for a decorator creator
52          if (decorator != null) {
53  
54              decorator = decorator.trim();
55  
56              String decoratorId = decorator;
57  
58  
59              // Parse parameters
60              int parSep = decoratorId.indexOf('(');
61              if (parSep > 0) {
62  
63                  try {
64                      decoratorId = decorator.substring(0, parSep);
65                      int parEnd = decorator.indexOf(')');
66                      String allParameters = decorator.substring(parSep + 1, parEnd);
67                      String[] parametersNonTrim = splitParameters(allParameters);
68  
69                      // Get creator
70                      creator = getCreator(decoratorId);
71  
72                      if (creator == null) {
73                          LOGGER.warn("Unknown decorator '" + decorator + "'");
74                      } else {
75                          for (int i = 0; i < parametersNonTrim.length; i++) {
76                              String keyAndValue[] = parametersNonTrim[i].split(Pattern.quote("=\""));
77  
78                              ParameterKey key = null;
79                              String strKey = keyAndValue[0].trim();
80                              for (ParameterKey k : creator.getParameterKeys()) {
81                                  if (k.getKey().equalsIgnoreCase(strKey)) {
82                                      key = k;
83                                      break;
84                                  }
85                              }
86  
87                              if (key == null) {
88                                  throw new UnsupportedOperationException("Unknown parameter '" + keyAndValue[0] + "' on decorator '" + decorator + "'");
89                              }
90  
91                              String value = keyAndValue[1].trim();
92                              value = value.substring(0, value.length() - 1);
93                              parameters.put(key, value);
94                          }
95                      }
96                  } catch (UnsupportedOperationException e) {
97                      throw e;
98                  } catch (Exception e) {
99                      throw new UnsupportedOperationException("Malformed decorator parameters on '" + decorator + "'", e);
100                 }
101             } else {
102                 // Get creator
103                 creator = getCreator(decoratorId);
104             }
105         }
106 
107         if (creator == null) {
108             return getDefaultDecorator(field);
109         }
110 
111         return creator.createDecorator(collection, field, parameters);
112 
113     }
114 
115     private IDecorator getDefaultDecorator(Field field) {
116 
117         if (field.getType().equals(Double.class)) {
118             return new FieldDecorator(field, new DoubleFormater(3));
119         }
120 
121         return new FieldDecorator(field);
122     }
123 
124     private IDecoratorCreator getCreator(String decoratorId) {
125 
126         if (creators != null) {
127             for (IDecoratorCreator creator : creators) {
128                 if (creator.getDecoratorId().equalsIgnoreCase(decoratorId)) {
129                     return creator;
130                 }
131             }
132         }
133 
134         return null;
135     }
136 
137 
138     public List<IDecoratorCreator> getCreators() {
139         return creators;
140     }
141 
142     public void setCreators(List<IDecoratorCreator> creators) {
143         this.creators = creators;
144     }

146     private static String[] splitParameters(String allParameters) {
147 
148         List<String> parameters = new ArrayList<String>();
149 
150         boolean inQuote = false;
151         int last = 0;
152         for (int i = 0; i < allParameters.length(); i++) {
153             char c = allParameters.charAt(i);
154 
155             if (c == '"') {
156                 inQuote = !inQuote;
157             }
158 
159             if (c == ',' && !inQuote) {
160                 parameters.add(allParameters.substring(last, i).trim());
161                 last = i + 1;
162             }
163         }
164 
165         parameters.add(allParameters.substring(last, allParameters.length()).trim());
166 
167         return parameters.toArray(new String[parameters.size()]);
168     }
