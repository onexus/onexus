24  import org.onexus.website.api.widgets.tableviewer.decorators.IDecorator;

29  import java.awt.*;
30  import java.util.HashMap;
31  import java.util.Map;
32  
33  public class CategoricalDecoratorCreator implements IDecoratorCreator {
34  
35      @Override
36      public String getDecoratorId() {
37          return "CATEGORICAL";
38      }
39  
40      @Override
41      public ParameterKey[] getParameterKeys() {
42          return CategoricalDecoratorParameters.values();
43      }
44  
45      @Override
46      public IDecorator createDecorator(Collection collection, Field columnField, Parameters parameters) {
47  
48          Color defaultColor = ColorUtils.stringToColor(parameters.get(CategoricalDecoratorParameters.DEFAULT));
49  
50          if (defaultColor == null) {
51              defaultColor = new Color(255, 255, 255);
52          }
53  
54          Map<String, Color> colorsMap = new HashMap<String, Color>();
55          String mapValue = parameters.get(CategoricalDecoratorParameters.MAP);
56          String items[] = mapValue.split("\\|");
57  
58          for (String item : items) {
59              String pair[] = item.split("=");
60  
61              String key = pair[0].replace('[', ' ').replace(']', ' ').trim();
62              Color color = ColorUtils.stringToColor(pair[1]);
63  
64              colorsMap.put(key, color);
65          }
66  
67          return new ColorDecorator(columnField, new CategoricalColorScale(defaultColor, colorsMap), null, true);
68      }
69  }
