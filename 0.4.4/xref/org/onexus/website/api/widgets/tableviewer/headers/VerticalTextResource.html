18  package org.onexus.website.api.widgets.tableviewer.headers;
19  
20  import org.apache.wicket.request.resource.DynamicImageResource;
21  import org.apache.wicket.request.resource.caching.IResourceCachingStrategy;
22  import org.apache.wicket.request.resource.caching.NoOpResourceCachingStrategy;
23  
24  import java.awt.*;
25  import java.awt.image.BufferedImage;
26  
27  public class VerticalTextResource extends DynamicImageResource {
28  
29      private static final Font DEFAULT_FONT = new Font("Arial", Font.BOLD, 12);
30      private static final IResourceCachingStrategy CACHING_STRATEGY = new NoOpResourceCachingStrategy();
31  
32      private String text;
33      private int width;
34      private int height;
35      private Font font;
36  
37      public VerticalTextResource(String text) {
38          this(text, DEFAULT_FONT);
39  
40  
41      }
42  
43      public VerticalTextResource(String text, Font font) {
44          this(text, font.getSize() + 3, (int) (font.getSize() * 0.75 * text
45                  .length()), font);
46      }
47  
48      public VerticalTextResource(String text, int width, int height) {
49          this(text, width, height, DEFAULT_FONT);
50      }
51  
52      public VerticalTextResource(String text, int width, int height, Font font) {
53          super();
54          this.text = text;
55          this.width = width;
56          this.height = height;
57          this.font = font;
58      }
59  
60      @Override
61      protected byte[] getImageData(Attributes attributes) {
62          final BufferedImage image = new BufferedImage(width, height,
63                  BufferedImage.TYPE_INT_RGB);
64          Graphics2D g2 = image.createGraphics();
65          g2.setBackground(Color.WHITE);
66          g2.clearRect(0, 0, 500, 500);
67          g2.setFont(font);
68          g2.setColor(Color.BLACK);
69          g2.translate((width + font.getSize()) / 2, height - 2);
70          g2.rotate(-Math.PI / 2.0);
71          g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
72                  RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
73          g2.drawString(text, 0, 0);
74          g2.dispose();
75  
76          return toImageData(image);
77      }
78  
79      @Override
80      protected IResourceCachingStrategy getCachingStrategy() {
81          return CACHING_STRATEGY;
82      }
83  
84  
85  }
