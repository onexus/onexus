27  import org.onexus.website.api.events.EventViewChange;
28  import org.onexus.website.api.pages.browser.BrowserPage;
29  import org.onexus.website.api.pages.browser.BrowserPageConfig;
30  import org.onexus.website.api.pages.browser.BrowserPageStatus;
31  import org.onexus.website.api.widgets.Widget;
32  import org.onexus.website.api.widgets.WidgetConfig;
33  import org.onexus.website.api.widgets.tableviewer.ColumnSet;
34  import org.onexus.website.api.widgets.tableviewer.TableViewerConfig;
35  import org.onexus.website.api.widgets.tableviewer.TableViewerStatus;
36  
37  import java.util.List;
38  
39  public class ColumnsetsWidget extends Widget<ColumnsetsWidgetConfig, ColumnsetsWidgetStatus> {
40  
41      public ColumnsetsWidget(String componentId, IModel statusModel) {
42          super(componentId, statusModel);
43  
44          onEventFireUpdate(EventQueryUpdate.class);
45  
46          Form<String> form = new Form<String>("form");
47          add(form);
48  
49          final IModel<ColumnSet> csModel = new Model<ColumnSet>();
50          TableViewerStatus status = getTableViewerStatus();
51  
52          int cs = status == null ? 0 : status.getCurrentColumnSet();
53          csModel.setObject(getTableViewerConfig().getColumnSets().get(cs));
54  
55          form.add(new AjaxColumnSetSelector("columnsets", csModel, getTableViewerConfig().getColumnSets()));
56  
57      }
58  
59      private BrowserPageStatus getPageStatus() {
60          return findParent(BrowserPage.class).getStatus();
61      }
62  
63      private BrowserPageConfig getPageConfig() {
64          return (BrowserPageConfig) getPageStatus().getConfig();
65      }
66  
67      private TableViewerStatus getTableViewerStatus() {
68  
69          String widgetId = getTableViewerConfig().getId();
70          return (TableViewerStatus) (widgetId == null ? null : getPageStatus().getWidgetStatus(widgetId));
71  
72      }
73  
74      private TableViewerConfig getTableViewerConfig() {
75  
76          String currentTab = getPageStatus().getCurrentTabId();
77          String currentView = getPageStatus().getCurrentView();
78          String mainWidgetId = getPageConfig().getTab(currentTab).getView(currentView).getMain().trim();
79          WidgetConfig widgetConfig = getPageConfig().getWidget(mainWidgetId);
80  
81          if (widgetConfig instanceof TableViewerConfig) {
82              return (TableViewerConfig) widgetConfig;
83          }
84  
85          return null;
86      }
87  
88      private final class AjaxColumnSetSelector extends DropDownChoice<ColumnSet> {
89  
90          private AjaxColumnSetSelector(String id, final IModel<ColumnSet> selectItemModel, List<ColumnSet> listFilters) {
91              super(id, selectItemModel, listFilters);
92  
93              setNullValid(false);
94              add(new AjaxFormComponentUpdatingBehavior("onchange") {
95                  @Override
96                  protected void onUpdate(AjaxRequestTarget target) {
97                      ColumnSet columnSet = (ColumnSet) getDefaultModelObject();
98                      int cs = getTableViewerConfig().getColumnSets().indexOf(columnSet);
99                      getTableViewerStatus().setCurrentColumnSet(cs);
100                     sendEvent(EventViewChange.EVENT);
101                 }
102             });
103         }
104     }
