18  package org.onexus.website.api.utils;
19  
20  import org.apache.wicket.Component;
21  import org.apache.wicket.behavior.Behavior;
22  import org.apache.wicket.markup.head.CssHeaderItem;
23  import org.apache.wicket.markup.head.IHeaderResponse;
24  import org.apache.wicket.request.resource.CssResourceReference;
25  import org.apache.wicket.util.string.Strings;
26  import org.onexus.data.api.IDataManager;
27  import org.onexus.resource.api.IResourceManager;
28  import org.onexus.resource.api.ORI;
29  import org.onexus.resource.api.Project;
30  import org.onexus.website.api.WebsiteApplication;
31  
32  import javax.inject.Inject;
33  
34  public class CustomCssBehavior extends Behavior {
35  
36      private transient CssHeaderItem cssHeaderItem;
37  
38      private ORI resourceUri;
39  
40      @Inject
41      private IDataManager dataManager;
42  
43      @Inject
44      private IResourceManager resourceManager;
45  
46      public CustomCssBehavior(ORI parentOri, String cssTag) {
47          if (!Strings.isEmpty(cssTag)) {
48              resourceUri = new ORI(cssTag);
49              if (!resourceUri.isAbsolute()) {
50                  resourceUri = resourceUri.toAbsolute(parentOri);
51              }
52          }
53      }
54  
55      private CustomCssBehavior(ORI resourceUri) {
56          this.resourceUri = resourceUri;
57      }
58  
59      @Override
60      public void renderHead(Component component, IHeaderResponse response) {
61  
62          if (cssHeaderItem == null) {
63  
64              if (resourceUri != null) {
65  
66                  String dataService = getDataManager().getMount();
67                  Project project = getResourceManager().getProject(resourceUri.getProjectUrl());
68  
69                  String url;
70                  if (project != null) {
71                      url = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName() + resourceUri.getPath());
72                  } else {
73                      url = resourceUri.toString();
74                  }
75                  cssHeaderItem = CssHeaderItem.forUrl(url);
76  
77              } else {
78                  cssHeaderItem = CssHeaderItem.forReference(new CssResourceReference(component.getClass(), component.getClass().getSimpleName() + ".css"));
79              }
80          }
81  
82          response.render(cssHeaderItem);
83      }
84  
85      private IDataManager getDataManager() {
86  
87          if (dataManager == null) {
88              WebsiteApplication.inject(this);
89          }
90  
91          return dataManager;
92      }
93  
94      private IResourceManager getResourceManager() {
95  
96          if (resourceManager == null) {
97              WebsiteApplication.inject(this);
98          }
99  
100         return resourceManager;
101     }
102 
103 }
