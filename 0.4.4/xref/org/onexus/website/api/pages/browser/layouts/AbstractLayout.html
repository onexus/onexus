20  import org.apache.commons.collections.CollectionUtils;
21  import org.apache.wicket.markup.html.panel.Panel;
22  import org.apache.wicket.model.IModel;
23  import org.apache.wicket.util.string.StringValue;
24  import org.onexus.website.api.pages.PageConfig;
25  import org.onexus.website.api.pages.browser.BrowserPageStatus;
26  import org.onexus.website.api.pages.browser.ViewConfig;
27  import org.onexus.website.api.utils.visible.VisiblePredicate;
28  import org.onexus.website.api.widgets.WidgetConfig;
29  
30  import java.util.ArrayList;
31  import java.util.List;
32  
33  public class AbstractLayout extends Panel {
34  
35      public AbstractLayout(String panelId, IModel<BrowserPageStatus> statusModel) {
36          super(panelId, statusModel);
37      }
38  
39      public PageConfig getPageConfig() {
40          BrowserPageStatus status = getPageStatus();
41          return status == null ? null : status.getConfig();
42      }
43  
44      public BrowserPageStatus getPageStatus() {
45          return (BrowserPageStatus) getDefaultModelObject();
46      }
47  
48      public List<WidgetConfig> filterWidgets(String selectedWidgets) {
49          List<WidgetConfig> widgets = ViewConfig.getSelectedWidgetConfigs(getPageConfig(), selectedWidgets);
50          VisiblePredicate predicate = new VisiblePredicate(getPageStatus().getORI(), getPageStatus().getEntitySelections());
51          List<WidgetConfig> visibleWidgets = new ArrayList<WidgetConfig>();
52          CollectionUtils.select(widgets, predicate, visibleWidgets);
53          return visibleWidgets;
54      }
55  
56      protected boolean isEmbed() {
57          StringValue embed = getPage().getPageParameters().get("embed");
58          return embed.toBoolean(false);
59      }
