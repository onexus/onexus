48  public class FiltersToolbar extends Panel {
49  
50      private WebMarkupContainer widgetModal;
51  
52      private IModel<BrowserPageStatus> statusModel;
53  
54      public FiltersToolbar(String componentId, IModel<BrowserPageStatus> statusModel) {
55          super(componentId);
56          setOutputMarkupId(true);
57  
58          this.statusModel = statusModel;
59  
60          widgetModal = new WebMarkupContainer("widgetModal");
61          widgetModal.setOutputMarkupId(true);
62  
63          widgetModal.add(new EmptyPanel("widget"));
64  
65          widgetModal.add(new AjaxLink<String>("close") {
66  
67              @Override
68              public void onClick(AjaxRequestTarget target) {
69                  Component widget = widgetModal.get("widget");
70                  if (widget instanceof Widget) {
71                      ((Widget) widget).onClose(target);
72                  }
73                  target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
74              }
75          });
76  
77          add(widgetModal);
78  
79          widgetModal.add(new Label("modalHeader", "Add filter"));
80  
81      }
82  
83      @Override
84      public void onEvent(IEvent<?> event) {
85  
86          if (event.getPayload() instanceof EventFilterHeader) {
87              EventFilterHeader e = ((EventFilterHeader) event.getPayload());
88  
89              String filter = e.getHeader().getFilter();
90              String filterType;
91              String options;
92  
93              int separator = filter.indexOf('[');
94              if (separator != -1) {
95                  filterType = filter.substring(0, separator).trim();
96  
97                  int endSeparator = filter.indexOf(']', separator);
98                  if (endSeparator != -1) {
99                      options = filter.substring(separator+1, endSeparator).trim();
100                 } else {
101                     options = filter.substring(separator+1).trim();
102                 }
103             } else {
104                 filterType = filter.trim();
105                 options = "";
106             }
107 
108 
109             if ("STRING".equalsIgnoreCase(filterType)) {
110                 widgetModal.addOrReplace(new StringFilterPanel("widget", e.getHeader()) {
111                     @Override
112                     protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
113                         FiltersToolbar.this.addFilter(target, filterConfig);
114                     }
115                 });
116             } else if ("DOUBLE".equalsIgnoreCase(filterType)) {
117                 widgetModal.addOrReplace(new DoubleFilterPanel("widget", e.getHeader()) {
118 
119                     @Override
120                     protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
121                         FiltersToolbar.this.addFilter(target, filterConfig);
122                     }
123 
124                 });
125             } else if ("INTEGER".equalsIgnoreCase(filterType)) {
126                 widgetModal.addOrReplace(new IntegerFilterPanel("widget", e.getHeader()) {
127 
128                     @Override
129                     protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
130                         FiltersToolbar.this.addFilter(target, filterConfig);
131                     }
132 
133                 });
134             } else if ("CATEGORICAL".equalsIgnoreCase(filterType)) {
135 
136                 List<String> values = Arrays.asList(StringUtils.split(options, ','));
137 
138                 widgetModal.addOrReplace(new CategoricalFilterPanel("widget", e.getHeader(), values) {
139 
140                     @Override
141                     protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
142                         FiltersToolbar.this.addFilter(target, filterConfig);
143                     }
144 
145                 });
146             }
147 
148             widgetModal.addOrReplace(new Label("modalHeader", "Filter column '" + e.getHeader().getLabel() + "'"));
149 
150             e.getTarget().add(widgetModal);
151             e.getTarget().appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('show')");
152         }
153 
154     }
155 
156     protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
157 
158         statusModel.getObject().getCurrentFilters().add(filterConfig);
159 
160         target.add(this);
161         target.prependJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
162         send(getPage(), Broadcast.BREADTH, EventAddFilter.EVENT);
163     }
164 
165 
166     @Override
167     protected void onBeforeRender() {
168 
169         RepeatingView filtersView = new RepeatingView("filters");
170         Iterator<FilterConfig> itFilters = statusModel.getObject().getCurrentFilters().iterator();
171         while (itFilters.hasNext()) {
172             addFilter(filtersView, itFilters.next(), itFilters.hasNext());
173         }
174 
175         addOrReplace(filtersView);
176 
177         super.onBeforeRender();
178     }
179 
180     private void addFilter(RepeatingView filtersView, final FilterConfig filter, boolean hasNext) {
181 
182         WebMarkupContainer container = new WebMarkupContainer(filtersView.newChildId());
183         WebMarkupContainer labelContainer = new WebMarkupContainer("container");
184         labelContainer.add(new Label("title", filter.getName()).setEscapeModelStrings(false));
185         labelContainer.add(new AjaxLink<FilterConfig>("remove") {
186 
187             @Override
188             public void onClick(AjaxRequestTarget target) {
189                 statusModel.getObject().getCurrentFilters().remove(filter);
190                 target.add(FiltersToolbar.this);
191                 send(getPage(), Broadcast.BREADTH, EventRemoveFilter.EVENT);
192             }
193         });
194 
195         if (filter.isEnable()) {
196             labelContainer.add(new AttributeModifier("class", "label label-important"));
197         } else {
198             labelContainer.add(new AttributeModifier("class", "label"));
199         }
200 
201         container.add(labelContainer);
202         container.add(new WebMarkupContainer("operator").setVisible(hasNext));
203 
204         filtersView.add(container);
205     }
