18  package org.onexus.resource.serializer.xstream.internal;
19  
20  import com.thoughtworks.xstream.converters.SingleValueConverter;
21  import org.onexus.collection.api.types.Text;
22  
23  import java.util.HashMap;
24  import java.util.Map;
25  
26  public class ClassConverter implements SingleValueConverter {
27  
28      private static final Map<String, Class> STRING_TO_CLASS = new HashMap<String, Class>();
29      private static final Map<Class, String> CLASS_TO_STRING = new HashMap<Class, String>();
30  
31      static {
32          registerType("integer", Integer.class);
33          registerType("string", String.class);
34          registerType("double", Double.class);
35          registerType("boolean", Boolean.class);
36          registerType("text", Text.class);
37      }
38  
39      public static void registerType(String text, Class type) {
40          STRING_TO_CLASS.put(text, type);
41          CLASS_TO_STRING.put(type, text);
42      }
43  
44      @Override
45      public boolean canConvert(Class type) {
46          return Class.class.equals(type);
47      }
48  
49      @Override
50      public String toString(Object obj) {
51          if (obj == null) {
52              return null;
53          }
54  
55          if (CLASS_TO_STRING.containsKey(obj)) {
56              return CLASS_TO_STRING.get(obj);
57          }
58  
59          return ((Class) obj).getCanonicalName();
60      }
61  
62      @Override
63      public Object fromString(String str) {
64          if (str == null) {
65              return null;
66          }
67  
68          if (STRING_TO_CLASS.containsKey(str)) {
69              return STRING_TO_CLASS.get(str);
70          }
71  
72          try {
73              return getClass().getClassLoader().loadClass(str);
74          } catch (ClassNotFoundException e) {
75              throw new RuntimeException(e);
76          }
77      }
78  }
