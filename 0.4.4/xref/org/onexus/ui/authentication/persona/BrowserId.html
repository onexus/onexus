109     /**
110      * Create {@link BrowserId} object from success or failure JSON response
111      *
112      * @param json the response returned by https://browserid.org
113      * @return a {@link BrowserId} instance with status OK if the authentication passed, or with
114      *         status FAIL if the authentication failed, or {@code null} if the passed JSON response
115      *         is invalid
116      */
117     public static BrowserId of(String json) {
118         BrowserId browserId = null;
119 
120         if (json != null && json.startsWith("{") && json.endsWith("}")) {
121             json = json.substring(1);
122             json = json.substring(0, json.length() - 1);
123 
124             String[] settings = Strings.split(json, ',');
125             if (settings.length > 0) {
126                 browserId = new BrowserId();
127                 for (String setting : settings) {
128                     String[] pair = Strings.split(setting, ':');
129                     pair[0] = Strings.replaceAll(pair[0], "\"", "").toString();
130                     pair[1] = Strings.replaceAll(pair[1], "\"", "").toString();
131 
132                     if ("status".equals(pair[0])) {
133                         browserId.status = Status.parse(pair[1]);
134                     } else if ("email".equals(pair[0])) {
135                         browserId.email = pair[1];
136                     } else if ("audience".equals(pair[0])) {
137                         browserId.audience = pair[1];
138                     } else if ("issuer".equals(pair[0])) {
139                         browserId.issuer = pair[1];
140                     } else if ("expires".equals(pair[0])) {
141                         Long millis = Long.valueOf(pair[1]);
142                         browserId.validUntil = new Date(millis);
143                     } else if ("reason".equals(pair[0])) {
144                         browserId.reason = pair[1];
145                     } else {
146                         throw new IllegalArgumentException("Unknown setting: " + pair[0]);
147                     }
148                 }
149             }
150         }
151 
152         return browserId;
153     }
