18  package org.onexus.ui.api.progressbar.columns;
19  
20  
21  import org.apache.wicket.extensions.markup.html.repeater.data.grid.ICellPopulator;
22  import org.apache.wicket.extensions.markup.html.repeater.data.table.AbstractColumn;
23  import org.apache.wicket.markup.html.basic.Label;
24  import org.apache.wicket.markup.repeater.Item;
25  import org.apache.wicket.model.IModel;
26  import org.apache.wicket.model.Model;
27  import org.onexus.resource.api.Progress;
28  
29  public class StatusColumn extends AbstractColumn<Progress, String> {
30  
31  
32      public StatusColumn() {
33          super(Model.of("status"));
34      }
35  
36  
37      @Override
38      public void populateItem(Item<ICellPopulator<Progress>> cellItem, String componentId, IModel<Progress> rowModel) {
39  
40          String label = "<span class=\"label\">NA</span>";
41  
42          Progress progress = rowModel.getObject();
43  
44          if (progress != null && progress.getStatus() != null) {
45              switch (progress.getStatus()) {
46                  case DONE:
47                      label = "<span class=\"label label-success\">done</span>";
48                      break;
49                  case FAILED:
50                      label = "<span class=\"label label-important\">failed</span>";
51                      break;
52                  case RUNNING:
53                      label = "<span class=\"label label-warning\">running</span>";
54                      break;
55                  case CANCELED:
56                      label = "<span class=\"label label-inverse\">canceled</span>";
57                      break;
58                  case WAITING:
59                      label = "<span class=\"label label-info\">waiting</span>";
60                      break;
61                  default:
62                      label = "<span class=\"label\">Unknown</span>";
63              }
64          }
65  
66          cellItem.add(new Label(componentId, label).setEscapeModelStrings(false));
67      }
68  }
