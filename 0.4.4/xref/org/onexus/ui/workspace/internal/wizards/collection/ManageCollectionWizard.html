21  import org.apache.wicket.extensions.wizard.WizardStep;
22  import org.apache.wicket.markup.html.form.RadioChoice;
23  import org.apache.wicket.model.IModel;
24  import org.apache.wicket.model.PropertyModel;
25  import org.onexus.collection.api.ICollectionManager;
26  import org.onexus.collection.api.query.Query;
27  import org.onexus.resource.api.ORI;
28  import org.onexus.resource.api.Resource;
29  import org.onexus.ui.api.progressbar.ProgressBar;
30  import org.onexus.ui.api.wizards.AbstractWizard;
31  
32  import javax.inject.Inject;
33  import javax.inject.Named;
34  import java.util.Arrays;
35  import java.util.List;
36  
37  public class ManageCollectionWizard extends AbstractWizard {
38  
39      // Commands
40      private static final String LOAD = "Load collection";
41      private static final String UNLOAD = "Unload collection";
42      private static final List<String> COMMANDS = Arrays.asList(new String[]{LOAD, UNLOAD});
43  
44      private String selected;
45      private ORI resourceURI;
46  
47      @Inject
48      private ICollectionManager collectionManager;
49  
50      public ManageCollectionWizard(String id, IModel<? extends Resource> resourceModel) {
51          super(id);
52  
53          this.resourceURI = resourceModel.getObject().getORI();
54  
55          WizardModel model = new WizardModel();
56          model.add(new BasicOptions());
57  
58          init(model);
59      }
60  
61      @Override
62      public void onFinish() {
63  
64          super.onFinish();
65  
66          if (selected.equals(UNLOAD)) {
67              collectionManager.unload(resourceURI);
68              return;
69          }
70  
71          if (selected.equals(LOAD)) {
72              collectionManager.unload(resourceURI);
73              Query emptyQuery = new Query();
74              emptyQuery.addDefine("c", resourceURI);
75              emptyQuery.setFrom("c");
76              emptyQuery.setOffset(0);
77              emptyQuery.setCount(0);
78              ProgressBar.show(collectionManager.load(emptyQuery));
79              return;
80          }
81  
82      }
83  
84      public String getSelected() {
85          return selected;
86      }
87  
88      public void setSelected(String selected) {
89          this.selected = selected;
90      }
91  
92      private final class BasicOptions extends WizardStep {
93  
94  
95          public BasicOptions() {
96              super("Manage collection", "Basic collection tools");
97  
98              RadioChoice<String> commandOptions = new RadioChoice<String>("commands", new PropertyModel<String>(ManageCollectionWizard.this, "selected"), COMMANDS);
99              add(commandOptions);
100 
101         }
102 
103     }
