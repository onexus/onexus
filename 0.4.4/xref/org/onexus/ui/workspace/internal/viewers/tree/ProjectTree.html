18  package org.onexus.ui.workspace.internal.viewers.tree;
19  
20  import org.apache.wicket.Component;
21  import org.apache.wicket.ajax.AjaxRequestTarget;
22  import org.apache.wicket.event.Broadcast;
23  import org.apache.wicket.extensions.markup.html.repeater.tree.ITreeProvider;
24  import org.apache.wicket.extensions.markup.html.repeater.tree.NestedTree;
25  import org.apache.wicket.extensions.markup.html.repeater.tree.content.Folder;
26  import org.apache.wicket.extensions.markup.html.repeater.tree.theme.WindowsTheme;
27  import org.apache.wicket.extensions.markup.html.repeater.util.ProviderSubset;
28  import org.apache.wicket.markup.head.CssHeaderItem;
29  import org.apache.wicket.markup.head.IHeaderResponse;
30  import org.apache.wicket.model.AbstractReadOnlyModel;
31  import org.apache.wicket.model.IDetachable;
32  import org.apache.wicket.model.IModel;
33  import org.apache.wicket.model.PropertyModel;
34  import org.apache.wicket.request.resource.CssResourceReference;
35  import org.apache.wicket.request.resource.ResourceReference;
36  import org.onexus.resource.api.Project;
37  import org.onexus.resource.api.Resource;
38  import org.onexus.ui.api.events.EventResourceSelect;
39  
40  import java.util.Set;
41  
42  public class ProjectTree extends NestedTree<Resource> {
43  
44      private IModel<Resource> selected;
45  
46      private Set<Resource> state;
47  
48      private static final ResourceReference CSS = new CssResourceReference(ProjectTree.class, "css/tree.css");
49  
50      public ProjectTree(String id, ITreeProvider<Resource> treeProvider, IModel<? extends Resource> selectedResource) {
51          super(id, treeProvider);
52  
53          this.state = new InverseSet<Resource>(new ProviderSubset<Resource>(treeProvider));
54          this.selected = (IModel<Resource>) selectedResource;
55  
56          setOutputMarkupId(true);
57          setModel(newStateModel());
58          add(new WindowsTheme());
59  
60      }
61  
62      protected boolean isSelected(Resource resource) {
63  
64          if (resource != null && selected.getObject() != null) {
65              return resource.getORI().equals(selected.getObject().getORI());
66          }
67  
68          return false;
69      }
70  
71  
72      protected void select(Resource resource, final AjaxRequestTarget target) {
73  
74          // Update unselected node
75          updateNode(selected.getObject(), target);
76  
77          // Select new node
78          selected.setObject(resource);
79  
80          // Update new selection
81          updateNode(resource, target);
82  
83          send(getPage(), Broadcast.BREADTH, EventResourceSelect.EVENT);
84      }
85  
86      private IModel<Set<Resource>> newStateModel() {
87          return new AbstractReadOnlyModel<Set<Resource>>() {
88              @Override
89              public Set<Resource> getObject() {
90                  return state;
91              }
92  
93              /**
94               * Super class doesn't detach - would be nice though.
95               */
96              @Override
97              public void detach() {
98                  ((IDetachable) state).detach();
99              }
100         };
101     }
102 
103     @Override
104     protected void onDetach() {
105         if (selected != null) {
106             selected.detach();
107         }
108         super.onDetach();
109     }
110 
111     @Override
112     protected Component newContentComponent(String id, IModel<Resource> model) {
113         return new Folder<Resource>(id, this, model) {
114 
115             @Override
116             protected boolean isClickable() {
117                 return true;
118             }
119 
120 
121             @Override
122             protected void onClick(AjaxRequestTarget target) {
123                 ProjectTree.this.select(getModelObject(), target);
124             }
125 
126 
127             @Override
128             protected boolean isSelected() {
129                 return ProjectTree.this.isSelected(getModelObject());
130             }
131 
132             @Override
133             protected IModel<?> newLabelModel(IModel<Resource> resourceIModel) {
134                 return new PropertyModel<String>(resourceIModel, "name");
135             }
136 
137             @Override
138             protected String getStyleClass() {
139 
140                 Resource t = getModelObject();
141 
142                 String styleClass;
143                 if (getProvider().hasChildren(t)) {
144                     if (getState(t) == State.EXPANDED) {
145                         if (t instanceof Project) {
146                             styleClass = "tree-project-open";
147                         } else {
148                             styleClass = getOpenStyleClass();
149                         }
150                     } else {
151                         if (t instanceof Project) {
152                             styleClass = "tree-project-closed";
153                         } else {
154                             styleClass = getClosedStyleClass();
155                         }
156                     }
157                 } else {
158                     styleClass = getOtherStyleClass(t);
159                 }
160 
161                 if (isSelected()) {
162                     styleClass += " " + getSelectedStyleClass();
163                 }
164 
165                 return styleClass;
166             }
167 
168             @Override
169             protected String getOtherStyleClass(Resource resource) {
170                 return "resource-" + resource.getClass().getSimpleName().toLowerCase();
171             }
172         };
173     }
174 
175     @Override
176     public void renderHead(IHeaderResponse response) {
177         response.render(CssHeaderItem.forReference(CSS));
178     }
179 
180 
181 }
