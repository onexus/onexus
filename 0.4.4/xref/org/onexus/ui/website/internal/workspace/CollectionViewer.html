18  package org.onexus.ui.website.internal.workspace;
19  
20  import org.apache.wicket.markup.html.panel.EmptyPanel;
21  import org.apache.wicket.markup.html.panel.Panel;
22  import org.apache.wicket.model.IModel;
23  import org.apache.wicket.model.Model;
24  import org.onexus.collection.api.Collection;
25  import org.onexus.collection.api.query.Query;
26  import org.onexus.resource.api.Resource;
27  import org.onexus.website.api.widgets.tableviewer.ColumnSet;
28  import org.onexus.website.api.widgets.tableviewer.TableViewer;
29  import org.onexus.website.api.widgets.tableviewer.TableViewerConfig;
30  import org.onexus.website.api.widgets.tableviewer.TableViewerStatus;
31  import org.onexus.website.api.widgets.tableviewer.columns.ColumnConfig;
32  
33  public class CollectionViewer extends Panel {
34  
35      private static final String REGEXP_ALL_FIELDS = "*{(.*)}";
36  
37      public CollectionViewer(String id, IModel<? extends Resource> model) {
38          super(id, model);
39  
40          Resource resource = model.getObject();
41  
42          if (resource instanceof Collection) {
43  
44              Collection collection = (Collection) resource;
45  
46              TableViewerConfig viewerConfig = new TableViewerConfig("tableviewer", collection.getORI());
47  
48              ColumnSet columnSet = new ColumnSet();
49              columnSet.getColumns().add(new ColumnConfig(collection.getORI(), REGEXP_ALL_FIELDS));
50              viewerConfig.getColumnSets().add(columnSet);
51  
52              TableViewerStatus viewerStatus = new TableViewerStatus();
53              viewerStatus.setCurrentColumnSet(0);
54              viewerStatus.setConfig(viewerConfig);
55  
56              add(new TableViewer("table", Model.of(viewerStatus)) {
57                  @Override
58                  protected Query getQuery() {
59  
60                      Query query = new Query();
61  
62                      TableViewerStatus status = getStatus();
63  
64                      if (status != null) {
65                          status.beforeQueryBuild(query);
66                          status.onQueryBuild(query);
67                          status.afterQueryBuild(query);
68                      }
69  
70                      return query;
71  
72                  }
73              });
74  
75          } else {
76              add(new EmptyPanel("table"));
77          }
78  
79      }
80  
81  
82  }
