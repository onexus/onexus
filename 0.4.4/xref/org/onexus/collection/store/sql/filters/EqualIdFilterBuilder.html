18  package org.onexus.collection.store.sql.filters;
19  
20  import org.onexus.collection.api.Field;
21  import org.onexus.collection.api.query.EqualId;
22  import org.onexus.collection.api.query.Query;
23  import org.onexus.collection.api.utils.QueryUtils;
24  import org.onexus.collection.store.sql.SqlCollectionDDL;
25  import org.onexus.collection.store.sql.SqlCollectionStore;
26  import org.onexus.collection.store.sql.SqlDialect;
27  import org.onexus.resource.api.ORI;
28  
29  import java.util.ArrayList;
30  import java.util.List;
31  
32  public class EqualIdFilterBuilder extends AbstractFilterBuilder<EqualId> {
33  
34      public EqualIdFilterBuilder(SqlDialect dialect) {
35          super(dialect, EqualId.class);
36      }
37  
38  
39      @Override
40      protected void innerBuild(SqlCollectionStore store, Query query, StringBuilder where, EqualId filter) {
41  
42          String collectionAlias = filter.getCollectionAlias();
43          ORI collectionUri = QueryUtils.getCollectionOri(query, collectionAlias);
44          SqlCollectionDDL collection = store.getDDL(collectionUri);
45  
46          List<Field> primaryKey = new ArrayList<Field>();
47          for (Field field : collection.getCollection().getFields()) {
48              if (field.isPrimaryKey() != null && field.isPrimaryKey()) {
49                  primaryKey.add(field);
50              }
51          }
52  
53          Object id = filter.getId();
54          Object[] ids = String.valueOf(id).split("\t");
55  
56          if (primaryKey.size() != ids.length) {
57              throw new RuntimeException("Error on filter: " + filter);
58          }
59  
60          if (ids.length > 1) {
61              where.append("(");
62          }
63  
64          for (int i = 0; i < ids.length; i++) {
65  
66              Field keyField = primaryKey.get(i);
67  
68              String columnName = collection.getColumnInfoByFieldName(keyField.getId()).getColumnName();
69  
70              where.append("`").append(collectionAlias).append("`.`");
71              where.append(columnName).append("` = ");
72  
73              encodeValue(where, keyField.getType(), ids[i]);
74  
75              if (i + 1 < ids.length) {
76                  where.append(" AND ");
77              }
78          }
79  
80          if (ids.length > 1) {
81              where.append(")");
82          }
83  
84  
85      }
86  }
