18  package org.onexus.collection.store.sql.filters;
19  
20  import org.onexus.collection.api.query.IsNull;
21  import org.onexus.collection.api.query.Query;
22  import org.onexus.collection.api.utils.QueryUtils;
23  import org.onexus.collection.store.sql.SqlCollectionDDL;
24  import org.onexus.collection.store.sql.SqlCollectionStore;
25  import org.onexus.collection.store.sql.SqlDialect;
26  import org.onexus.resource.api.ORI;
27  
28  
29  public class IsNullFilterBuilder extends AbstractFilterBuilder<IsNull> {
30  
31      public IsNullFilterBuilder(SqlDialect dialect) {
32          super(dialect, IsNull.class);
33      }
34  
35      @Override
36      protected void innerBuild(SqlCollectionStore store, Query query, StringBuilder where, IsNull filter) {
37  
38          // Collection
39          String collectionAlias = filter.getCollectionAlias();
40          ORI collectionUri = QueryUtils.getCollectionOri(query, collectionAlias);
41          SqlCollectionDDL collection = store.getDDL(collectionUri);
42          SqlCollectionDDL.ColumnInfo column = collection.getColumnInfoByFieldName(filter.getFieldId());
43  
44          where.append("`").append(filter.getCollectionAlias()).append("`.`").append(column.getColumnName()).append("`");
45          where.append(" IS NULL");
46      }
47  }
