018package org.onexus.website.api.pages.browser.filters;
019
020import org.apache.commons.lang3.StringUtils;
021import org.apache.wicket.AttributeModifier;
022import org.apache.wicket.Component;
023import org.apache.wicket.ajax.AjaxRequestTarget;
024import org.apache.wicket.ajax.markup.html.AjaxLink;
025import org.apache.wicket.event.Broadcast;
026import org.apache.wicket.event.IEvent;
027import org.apache.wicket.markup.html.WebMarkupContainer;
028import org.apache.wicket.markup.html.basic.Label;
029import org.apache.wicket.markup.html.panel.EmptyPanel;
030import org.apache.wicket.markup.html.panel.Panel;
031import org.apache.wicket.markup.repeater.RepeatingView;
032import org.apache.wicket.model.IModel;
033import org.onexus.website.api.events.EventAddFilter;
034import org.onexus.website.api.events.EventFilterHeader;
035import org.onexus.website.api.events.EventRemoveFilter;
036import org.onexus.website.api.pages.browser.BrowserPageStatus;
037import org.onexus.website.api.pages.browser.filters.panels.CategoricalFilterPanel;
038import org.onexus.website.api.pages.browser.filters.panels.DoubleFilterPanel;
039import org.onexus.website.api.pages.browser.filters.panels.IntegerFilterPanel;
040import org.onexus.website.api.pages.browser.filters.panels.StringFilterPanel;
041import org.onexus.website.api.widgets.Widget;
042import org.onexus.website.api.widgets.selection.FilterConfig;
043
044import java.util.Arrays;
045import java.util.Iterator;
046import java.util.List;
047
048public class FiltersToolbar extends Panel {
049
050    private WebMarkupContainer widgetModal;
051
052    private IModel<BrowserPageStatus> statusModel;
053
054    public FiltersToolbar(String componentId, IModel<BrowserPageStatus> statusModel) {
055        super(componentId);
056        setOutputMarkupId(true);
057
058        this.statusModel = statusModel;
059
060        widgetModal = new WebMarkupContainer("widgetModal");
061        widgetModal.setOutputMarkupId(true);
062
063        widgetModal.add(new EmptyPanel("widget"));
064
065        widgetModal.add(new AjaxLink<String>("close") {
066
067            @Override
068            public void onClick(AjaxRequestTarget target) {
069                Component widget = widgetModal.get("widget");
070                if (widget instanceof Widget) {
071                    ((Widget) widget).onClose(target);
072                }
073                target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
074            }
075        });
076
077        add(widgetModal);
078
079        widgetModal.add(new Label("modalHeader", "Add filter"));
080
081    }
082
083    @Override
084    public void onEvent(IEvent<?> event) {
085
086        if (event.getPayload() instanceof EventFilterHeader) {
087            EventFilterHeader e = ((EventFilterHeader) event.getPayload());
088
089            String filter = e.getHeader().getFilter();
090            String filterType;
091            String options;
092
093            int separator = filter.indexOf('[');
094            if (separator != -1) {
095                filterType = filter.substring(0, separator).trim();
096
097                int endSeparator = filter.indexOf(']', separator);
098                if (endSeparator != -1) {
099                    options = filter.substring(separator+1, endSeparator).trim();
100                } else {
101                    options = filter.substring(separator+1).trim();
102                }
103            } else {
104                filterType = filter.trim();
105                options = "";
106            }
107
108
109            if ("STRING".equalsIgnoreCase(filterType)) {
110                widgetModal.addOrReplace(new StringFilterPanel("widget", e.getHeader()) {
111                    @Override
112                    protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
113                        FiltersToolbar.this.addFilter(target, filterConfig);
114                    }
115                });
116            } else if ("DOUBLE".equalsIgnoreCase(filterType)) {
117                widgetModal.addOrReplace(new DoubleFilterPanel("widget", e.getHeader()) {
118
119                    @Override
120                    protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
121                        FiltersToolbar.this.addFilter(target, filterConfig);
122                    }
123
124                });
125            } else if ("INTEGER".equalsIgnoreCase(filterType)) {
126                widgetModal.addOrReplace(new IntegerFilterPanel("widget", e.getHeader()) {
127
128                    @Override
129                    protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
130                        FiltersToolbar.this.addFilter(target, filterConfig);
131                    }
132
133                });
134            } else if ("CATEGORICAL".equalsIgnoreCase(filterType)) {
135
136                List<String> values = Arrays.asList(StringUtils.split(options, ','));
137
138                widgetModal.addOrReplace(new CategoricalFilterPanel("widget", e.getHeader(), values) {
139
140                    @Override
141                    protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
142                        FiltersToolbar.this.addFilter(target, filterConfig);
143                    }
144
145                });
146            }
147
148            widgetModal.addOrReplace(new Label("modalHeader", "Filter column '" + e.getHeader().getLabel() + "'"));
149
150            e.getTarget().add(widgetModal);
151            e.getTarget().appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('show')");
152        }
153
154    }
155
156    protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
157
158        statusModel.getObject().getCurrentFilters().add(filterConfig);
159
160        target.add(this);
161        target.prependJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
162        send(getPage(), Broadcast.BREADTH, EventAddFilter.EVENT);
163    }
164
165
166    @Override
167    protected void onBeforeRender() {
168
169        RepeatingView filtersView = new RepeatingView("filters");
170        Iterator<FilterConfig> itFilters = statusModel.getObject().getCurrentFilters().iterator();
171        while (itFilters.hasNext()) {
172            addFilter(filtersView, itFilters.next(), itFilters.hasNext());
173        }
174
175        addOrReplace(filtersView);
176
177        super.onBeforeRender();
178    }
179
180    private void addFilter(RepeatingView filtersView, final FilterConfig filter, boolean hasNext) {
181
182        WebMarkupContainer container = new WebMarkupContainer(filtersView.newChildId());
183        WebMarkupContainer labelContainer = new WebMarkupContainer("container");
184        labelContainer.add(new Label("title", filter.getName()).setEscapeModelStrings(false));
185        labelContainer.add(new AjaxLink<FilterConfig>("remove") {
186
187            @Override
188            public void onClick(AjaxRequestTarget target) {
189                statusModel.getObject().getCurrentFilters().remove(filter);
190                target.add(FiltersToolbar.this);
191                send(getPage(), Broadcast.BREADTH, EventRemoveFilter.EVENT);
192            }
193        });
194
195        if (filter.isEnable()) {
196            labelContainer.add(new AttributeModifier("class", "label label-important"));
197        } else {
198            labelContainer.add(new AttributeModifier("class", "label"));
199        }
200
201        container.add(labelContainer);
202        container.add(new WebMarkupContainer("operator").setVisible(hasNext));
203
204        filtersView.add(container);
205    }
206
207}
