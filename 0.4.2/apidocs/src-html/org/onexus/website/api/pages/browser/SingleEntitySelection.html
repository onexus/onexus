131    @Override
132    public boolean match(VisibleRule rule) {
133
134        //TODO Check projects
135
136        if (rule.getType() == VisibleRule.SelectionType.LIST) {
137            return false;
138        }
139
140        String filterPath = filteredCollection.getPath();
141        String rulePath = rule.getFilteredCollection().getPath();
142
143        boolean validCollection = (filterPath == null || rulePath == null) ? false : filterPath.endsWith(rulePath);
144
145
146        if (rule.getField() == null) {
147            return validCollection;
148        }
149
150
151        ORI collectionUri = filteredCollection.toAbsolute(rule.getParentURI());
152        IEntity entity = (new EntityModel(collectionUri, entityId)).getObject();
153
154        String fieldValue = String.valueOf(entity.get(rule.getField()));
155
156        return StringUtils.equals(fieldValue, rule.getValue());
157    }
158
159    private static final String SEPARATOR = "::";
160    private static final Pattern DOUBLE_COLON = Pattern.compile(SEPARATOR);
161
162    @Override
163    public String toUrlParameter(boolean global, ORI parentUri) {
164
165        StringBuilder str = new StringBuilder();
166
167        ORI uri;
168        if (global) {
169            uri = filteredCollection.toAbsolute(parentUri);
170        } else {
171            uri = filteredCollection;
172        }
173
174        str.append(uri).append(SEPARATOR);
175        str.append(entityId).append(SEPARATOR);
176        str.append(deletable ? "d" : "");
177        str.append(enable ? "e" : "");
178
179        return str.toString();
180    }
181
182    @Override
183    public void loadUrlPrameter(String parameter) {
184
185        String[] values = DOUBLE_COLON.split(parameter);
186
187        this.filteredCollection = new ORI(values[0]);
188        this.entityId = values[1];
189        String flags = values[2];
190        deletable = flags.contains("d");
191        enable = flags.contains("e");
192    }
193
194    @Override
195    public String getTitle(Query query) {
196
197        // Make collection URI absolute
198        ORI filteredCollection = getSelectionCollection().toAbsolute(query.getOn());
199        Collection collection = getResourceManager().load(Collection.class, filteredCollection);
200
201        String entityField = collection.getProperty("FIXED_ENTITY_FIELD");
202        String entityLabel = getEntityId();
203        if (entityField != null) {
204            IEntityTable entityTable = getCollectionManager().load(new SingleEntityQuery(filteredCollection, getEntityId()));
205
206            IEntity entity = new EntityIterator(entityTable, filteredCollection).next();
207
208            entityLabel = String.valueOf(entity.get(entityField));
209
210            if (entityLabel == null || entityLabel.isEmpty()) {
211                entityLabel = getEntityId();
212            }
213
214            entityTable.close();
215        }
216
217        String entityPattern = collection.getProperty("FIXED_ENTITY_PATTERN");
218        if (entityPattern != null) {
219            IEntityTable entityTable = getCollectionManager().load(new SingleEntityQuery(filteredCollection, getEntityId()));
220
221            IEntity entity = new EntityIterator(entityTable, filteredCollection).next();
222
223            if (entity != null) {
224                entityLabel = parseTemplate(entityPattern, entity);
225            }
226
227            entityTable.close();
228        }
229
230        return entityLabel;
231    }

248
249    private String parseTemplate(String entityPattern, IEntity entity) {
250
251
252        for (Field field : entity.getCollection().getFields()) {
253            String fieldName = field.getId();
254            entityPattern = entityPattern.replaceAll(
255                    Pattern.quote("${" + fieldName + "}"),
256                    String.valueOf(entity.get(fieldName))
257            );
258        }
259
260        return entityPattern;
261    }

275    @Override
276    public int hashCode() {
277        final int prime = 31;
278        int result = 1;
279        result = prime * result
280                + ((filteredCollection == null) ? 0 : filteredCollection.hashCode());
281        result = prime * result
282                + ((entityId == null) ? 0 : entityId.hashCode());
283        return result;
284    }

286    @Override
287    public boolean equals(Object obj) {
288        if (this == obj) {
289            return true;
290        }
291
292        if (obj == null) {
293            return false;
294        }
295
296        if (getClass() != obj.getClass()) {
297            return false;
298        }
299
300        SingleEntitySelection other = (SingleEntitySelection) obj;
301        if (filteredCollection == null) {
302            if (other.filteredCollection != null) {
303                return false;
304            }
305        } else if (!filteredCollection.equals(other.filteredCollection)) {
306            return false;
307        }
308
309        if (entityId == null) {
310            if (other.entityId != null) {
311                return false;
312            }
313        } else if (!entityId.equals(other.entityId)) {
314            return false;
315        }
316        return true;
317    }
