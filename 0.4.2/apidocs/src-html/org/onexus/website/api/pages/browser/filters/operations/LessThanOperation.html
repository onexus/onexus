001package org.onexus.website.api.pages.browser.filters.operations;
002
003import org.onexus.collection.api.query.Filter;
004import org.onexus.collection.api.query.LessThan;
005
006public class LessThanOperation extends FilterOperation {
007
008    public static LessThanOperation INSTANCE = new LessThanOperation();
009
010    private LessThanOperation() {
011        super("less than", "<", true);
012    }
013
014    @Override
015    public Filter createFilter(String alias, String fieldId, Object value) {
016        return new LessThan(alias, fieldId, value);
017    }
018}
