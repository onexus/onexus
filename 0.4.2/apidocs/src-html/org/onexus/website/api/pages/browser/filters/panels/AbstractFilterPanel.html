018package org.onexus.website.api.pages.browser.filters.panels;
019
020import org.apache.wicket.ajax.AjaxRequestTarget;
021import org.apache.wicket.ajax.form.OnChangeAjaxBehavior;
022import org.apache.wicket.ajax.markup.html.form.AjaxSubmitLink;
023import org.apache.wicket.markup.html.form.DropDownChoice;
024import org.apache.wicket.markup.html.form.Form;
025import org.apache.wicket.markup.html.form.FormComponent;
026import org.apache.wicket.markup.html.panel.FeedbackPanel;
027import org.apache.wicket.markup.html.panel.Panel;
028import org.apache.wicket.model.CompoundPropertyModel;
029import org.apache.wicket.model.IModel;
030import org.apache.wicket.model.Model;
031import org.onexus.collection.api.query.Filter;
032import org.onexus.website.api.pages.browser.filters.operations.FilterOperation;
033import org.onexus.website.api.widgets.selection.FilterConfig;
034import org.onexus.website.api.widgets.tableviewer.headers.FieldHeader;
035
036import java.util.List;
037
038public abstract class AbstractFilterPanel<T> extends Panel {
039
040    private IModel<FilterOption> option;
041    private FieldHeader header;
042    private FilterOption<T> defaultOption;
043    private List<FilterOperation> operations;
044
045    public AbstractFilterPanel(String id, FieldHeader fieldHeader, FilterOption<T> defaultOption, List<FilterOperation> operations) {
046        super(id);
047
048        this.header = fieldHeader;
049        this.option = new CompoundPropertyModel<FilterOption>(new Model<FilterOption>(defaultOption));
050        this.defaultOption = defaultOption;
051        this.operations = operations;
052    }
053
054    @Override
055    protected void onInitialize() {
056        super.onInitialize();
057
058        Form<FilterOption> form = new Form("form", option);
059
060        // Feedback panel
061        final FeedbackPanel feedbackPanel = new FeedbackPanel("feedback");
062        feedbackPanel.setOutputMarkupId(true);
063        form.add(feedbackPanel);
064
065        // Value component
066        final FormComponent<T> valueComponent = createValueFormComponent("value");
067        valueComponent.setOutputMarkupPlaceholderTag(true);
068        valueComponent.setVisible(defaultOption.getOperation().isNeedsValue());
069        form.add(valueComponent);
070
071        // Operations drop down
072        final DropDownChoice<FilterOperation> operationsDropDown = new DropDownChoice<FilterOperation>("operation", operations);
073        operationsDropDown.setNullValid(false);
074        operationsDropDown.add(new OnChangeAjaxBehavior() {
075            @Override
076            protected void onUpdate(AjaxRequestTarget target) {
077                valueComponent.setVisible(option.getObject().getOperation().isNeedsValue());
078                target.add(valueComponent);
079            }
080        });
081        form.add(operationsDropDown);
082
083        // Create filter link
084        form.add(new AjaxSubmitLink("submit") {
085            @Override
086            protected void onSubmit(AjaxRequestTarget target, Form<?> form) {
087
088                FilterOption<T> fo = option.getObject();
089                T value = fo.getValue();
090
091                FilterOperation operation = fo.getOperation();
092                String title = operation.createTitle(header.getLabel(), value);
093                FilterConfig fc = new FilterConfig(title);
094
095                fc.setCollection(header.getCollection().getORI());
096                fc.setDefine("fc='" + fc.getCollection() + "'");
097
098
099                Filter where = operation.createFilter("fc", header.getField().getId(), value);
100                fc.setWhere(where.toString());
101
102                addFilter(target, fc);
103            }
104        });
105
106        add(form);
107
108    }
109
110    protected abstract FormComponent<T> createValueFormComponent(String componentId);
111
112    protected abstract void addFilter(AjaxRequestTarget target, FilterConfig filterConfig);
113
114}
