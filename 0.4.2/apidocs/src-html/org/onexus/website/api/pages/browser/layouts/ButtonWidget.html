018package org.onexus.website.api.pages.browser.layouts;
019
020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.Component;
022import org.apache.wicket.ajax.AjaxEventBehavior;
023import org.apache.wicket.ajax.AjaxRequestTarget;
024import org.apache.wicket.ajax.markup.html.AjaxLink;
025import org.apache.wicket.event.IEvent;
026import org.apache.wicket.markup.html.WebMarkupContainer;
027import org.apache.wicket.markup.html.basic.Label;
028import org.apache.wicket.markup.html.panel.EmptyPanel;
029import org.apache.wicket.model.IModel;
030import org.apache.wicket.model.PropertyModel;
031import org.apache.wicket.request.cycle.RequestCycle;
032import org.apache.wicket.util.string.Strings;
033import org.onexus.website.api.WebsiteApplication;
034import org.onexus.website.api.events.AbstractEvent;
035import org.onexus.website.api.events.EventCloseModal;
036import org.onexus.website.api.events.EventPanel;
037import org.onexus.website.api.events.EventQueryUpdate;
038import org.onexus.website.api.pages.browser.BrowserPageStatus;
039import org.onexus.website.api.widgets.IWidgetManager;
040import org.onexus.website.api.widgets.Widget;
041import org.onexus.website.api.widgets.WidgetConfig;
042import org.onexus.website.api.widgets.WidgetModel;
043import org.onexus.website.api.widgets.WidgetStatus;
044
045import javax.inject.Inject;
046
047public class ButtonWidget extends EventPanel {
048
049    @Inject
050    private IWidgetManager widgetManager;
051
052    private WidgetConfig widgetConfig;
053    private IModel<BrowserPageStatus> pageModel;
054    private WebMarkupContainer widgetModal;
055
056    public ButtonWidget(String id, final WidgetConfig widgetConfig, final IModel<BrowserPageStatus> pageModel) {
057        super(id);
058        onEventFireUpdate(EventQueryUpdate.class);
059
060
061        this.widgetConfig = widgetConfig;
062        this.pageModel = pageModel;
063
064        this.widgetModal = new WebMarkupContainer("widgetModal");
065        widgetModal.setOutputMarkupId(true);
066        widgetModal.add(new EmptyPanel("widget"));
067
068        widgetModal.add(new AjaxLink<String>("close") {
069
070            @Override
071            public void onClick(AjaxRequestTarget target) {
072                Component widget = widgetModal.get("widget");
073                if (widget instanceof Widget) {
074                    ((Widget) widget).onClose(target);
075                }
076                target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
077            }
078        });
079
080        if (!Strings.isEmpty(widgetConfig.getWidth())) {
081            int width = Integer.valueOf(widgetConfig.getWidth());
082            int marginLeft = width / 2;
083            widgetModal.add(new AttributeModifier("style", "width: " + width + "px; margin-left: -" + marginLeft + "px;"));
084        }
085
086        add(widgetModal);
087
088
089        Label button = new Label("button", new PropertyModel<String>(this, "buttonText"));
090        button.setEscapeModelStrings(false);
091        button.setOutputMarkupId(true);
092
093        if (widgetConfig.getTitle() != null) {
094            button.add(new AttributeModifier("title", widgetConfig.getTitle()));
095            button.add(new AttributeModifier("rel", "tooltip"));
096            widgetModal.add(new Label("modalHeader", widgetConfig.getTitle()));
097        } else {
098            widgetModal.add(new Label("modalHeader", ""));
099        }
100
101        button.add(new AjaxEventBehavior("onclick") {
102            @Override
103            protected void onEvent(AjaxRequestTarget target) {
104                Widget<?, ?> widgetPanel = getWidgetManager().create("widget", new WidgetModel(widgetConfig.getId(), pageModel));
105                widgetModal.addOrReplace(widgetPanel);
106                target.add(widgetModal);
107                target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('show')");
108            }
109        });
110
111        add(button);
112
113    }
114
115    private IWidgetManager getWidgetManager() {
116
117        if (widgetManager == null) {
118            WebsiteApplication.inject(this);
119        }
120
121        return widgetManager;
122    }
123
124    public String getButtonText() {
125        WidgetStatus status = pageModel.getObject().getWidgetStatus(widgetConfig.getId());
126        String buttonText = widgetConfig.getButton();
127        if (status != null) {
128            buttonText = status.getButton();
129        }
130        return buttonText;
131    }
132
133    @Override
134    protected void onRegisteredEvent(AjaxRequestTarget target, AbstractEvent event) {
135        target.add(get("button"));
136    }
137
138    @Override
139    public void onEvent(IEvent<?> event) {
140
141        if (event.getPayload() instanceof EventCloseModal) {
142            AjaxRequestTarget target = RequestCycle.get().find(AjaxRequestTarget.class);
143
144            if (target != null) {
145                target.prependJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
146            }
147        }
148
149        super.onEvent(event);
150    }
151}
