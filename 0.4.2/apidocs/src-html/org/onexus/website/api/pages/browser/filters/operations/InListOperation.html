001package org.onexus.website.api.pages.browser.filters.operations;
002
003import org.onexus.collection.api.query.Filter;
004import org.onexus.collection.api.query.In;
005
006import java.util.*;
007
008public class InListOperation extends FilterOperation {
009
010    public static InListOperation INSTANCE = new InListOperation();
011
012    private InListOperation() {
013        super("in list", "in", true);
014    }
015
016    @Override
017    public Filter createFilter(String alias, String fieldId, Object value) {
018
019        In filter = new In(alias, fieldId);
020        for (String val : parseValues(value)) {
021            filter.addValue(val.trim());
022        }
023
024        return filter;
025    }
026
027    private List<String> parseValues(Object value) {
028
029        if (value == null) {
030            return Collections.EMPTY_LIST;
031        }
032
033        return Arrays.asList(String.valueOf(value).split(","));
034    }
035
036    @Override
037    public String createTitle(String headerTitle, Object value) {
038
039        StringBuilder title = new StringBuilder();
040
041        title.append(headerTitle);
042        title.append(" ");
043        title.append(getSymbol());
044        title.append(" (");
045
046        Iterator<String> values = parseValues(value).iterator();
047
048        while (values.hasNext()) {
049            title.append("'").append(values.next().trim()).append("'");
050            if (values.hasNext()) {
051                title.append(",");
052            }
053        }
054
055        title.append(")");
056        return title.toString();
057    }
058}
