018package org.onexus.website.api.widgets.tableviewer.decorators;
019
020import org.onexus.collection.api.Collection;
021import org.onexus.collection.api.Field;
022import org.onexus.resource.api.ParameterKey;
023import org.onexus.resource.api.Parameters;
024import org.onexus.website.api.widgets.tableviewer.decorators.utils.FieldDecorator;
025import org.onexus.website.api.widgets.tableviewer.formaters.DoubleFormater;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;
028
029import java.util.ArrayList;
030import java.util.List;
031import java.util.regex.Pattern;
032
033public class DefaultDecoratorManager implements IDecoratorManager {
034
035    private static final Logger LOGGER = LoggerFactory.getLogger(DefaultDecoratorManager.class);
036    private static final String ANNOTATION_KEY = "BROWSER_DECORATOR";
037
038    private List<IDecoratorCreator> creators;
039
040
041    @Override
042    public IDecorator getDecorator(String decorator, Collection collection, Field field) {
043
044        if (decorator == null && field != null) {
045            decorator = field.getProperty(ANNOTATION_KEY);
046        }
047
048        IDecoratorCreator creator = null;
049        Parameters parameters = new Parameters();
050
051        // Look for a decorator creator
052        if (decorator != null) {
053
054            decorator = decorator.trim();
055
056            String decoratorId = decorator;
057
058
059            // Parse parameters
060            int parSep = decoratorId.indexOf('(');
061            if (parSep > 0) {
062
063                try {
064                    decoratorId = decorator.substring(0, parSep);
065                    int parEnd = decorator.indexOf(')');
066                    String allParameters = decorator.substring(parSep + 1, parEnd);
067                    String[] parametersNonTrim = splitParameters(allParameters);
068
069                    // Get creator
070                    creator = getCreator(decoratorId);
071
072                    if (creator == null) {
073                        LOGGER.warn("Unknown decorator '" + decorator + "'");
074                    } else {
075                        for (int i = 0; i < parametersNonTrim.length; i++) {
076                            String keyAndValue[] = parametersNonTrim[i].split(Pattern.quote("=\""));
077
078                            ParameterKey key = null;
079                            String strKey = keyAndValue[0].trim();
080                            for (ParameterKey k : creator.getParameterKeys()) {
081                                if (k.getKey().equalsIgnoreCase(strKey)) {
082                                    key = k;
083                                    break;
084                                }
085                            }
086
087                            if (key == null) {
088                                throw new UnsupportedOperationException("Unknown parameter '" + keyAndValue[0] + "' on decorator '" + decorator + "'");
089                            }
090
091                            String value = keyAndValue[1].trim();
092                            value = value.substring(0, value.length() - 1);
093                            parameters.put(key, value);
094                        }
095                    }
096                } catch (UnsupportedOperationException e) {
097                    throw e;
098                } catch (Exception e) {
099                    throw new UnsupportedOperationException("Malformed decorator parameters on '" + decorator + "'", e);
100                }
101            } else {
102                // Get creator
103                creator = getCreator(decoratorId);
104            }
105        }
106
107        if (creator == null) {
108            return getDefaultDecorator(field);
109        }
110
111        return creator.createDecorator(collection, field, parameters);
112
113    }
114
115    private IDecorator getDefaultDecorator(Field field) {
116
117        if (field.getType().equals(Double.class)) {
118            return new FieldDecorator(field, new DoubleFormater(3));
119        }
120
121        return new FieldDecorator(field);
122    }
123
124    private IDecoratorCreator getCreator(String decoratorId) {
125
126        if (creators != null) {
127            for (IDecoratorCreator creator : creators) {
128                if (creator.getDecoratorId().equalsIgnoreCase(decoratorId)) {
129                    return creator;
130                }
131            }
132        }
133
134        return null;
135    }
136
137
138    public List<IDecoratorCreator> getCreators() {
139        return creators;
140    }
141
142    public void setCreators(List<IDecoratorCreator> creators) {
143        this.creators = creators;
144    }
145
146    private static String[] splitParameters(String allParameters) {
147
148        List<String> parameters = new ArrayList<String>();
149
150        boolean inQuote = false;
151        int last = 0;
152        for (int i = 0; i < allParameters.length(); i++) {
153            char c = allParameters.charAt(i);
154
155            if (c == '"') {
156                inQuote = !inQuote;
157            }
158
159            if (c == ',' && !inQuote) {
160                parameters.add(allParameters.substring(last, i).trim());
161                last = i + 1;
162            }
163        }
164
165        parameters.add(allParameters.substring(last, allParameters.length()).trim());
166
167        return parameters.toArray(new String[parameters.size()]);
168    }
169}
