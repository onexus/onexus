020import org.apache.commons.collections.CollectionUtils;
021import org.apache.commons.collections.Predicate;

025import org.apache.wicket.markup.head.CssHeaderItem;
026import org.apache.wicket.markup.head.IHeaderResponse;
027import org.apache.wicket.markup.html.panel.Panel;
028import org.apache.wicket.model.IModel;
029import org.apache.wicket.request.resource.CssResourceReference;
030import org.apache.wicket.util.string.Strings;
031import org.onexus.collection.api.IEntityTable;
032import org.onexus.collection.api.query.Query;
033import org.onexus.resource.api.ORI;
034import org.onexus.resource.api.Progress;
035import org.onexus.website.api.events.EventAddFilter;
036import org.onexus.website.api.events.EventQueryUpdate;
037import org.onexus.website.api.events.EventRemoveFilter;
038import org.onexus.website.api.pages.browser.BrowserPageStatus;
039import org.onexus.website.api.utils.panels.ondomready.OnDomReadyPanel;
040import org.onexus.website.api.utils.visible.VisiblePredicate;
041import org.onexus.website.api.widgets.Widget;
042import org.onexus.website.api.widgets.tableviewer.columns.IColumnConfig;
043
044import java.util.ArrayList;
045import java.util.Collections;
046import java.util.List;
047
048public class TableViewer extends Widget<TableViewerConfig, TableViewerStatus> {
049
050    public static final CssResourceReference TABLE_VIEWER_CSS = new CssResourceReference(TableViewer.class,
051            "TableViewer.css");
052
053    public static final MetaDataKey<Integer> DEFAULT_ROWS_PER_PAGE = new MetaDataKey<Integer>() {
054    };
055
056    // Model objects
057    private EntitiesRowProvider dataProvider;
058
059    public TableViewer(String componentId, IModel<TableViewerStatus> status) {
060        super(componentId, status);
061
062        onEventFireUpdate(EventQueryUpdate.class, EventAddFilter.class, EventRemoveFilter.class);
063
064        if (status.getObject() == null) {
065            status.setObject(getStatus());
066        }
067
068        Integer sessionRowsPerPage = getSession().getMetaData(DEFAULT_ROWS_PER_PAGE);
069        final Integer rowsPerPage = sessionRowsPerPage == null ? 20 : sessionRowsPerPage;
070
071        this.dataProvider = new EntitiesRowProvider(status, rowsPerPage) {
072
073            @Override
074            protected Query getQuery() {
075                return TableViewer.this.getQuery();
076            }
077
078            @Override
079            protected void addTaskStatus(Progress progressStatus) {
080                //TODO
081            }
082
083        };
084
085        // Create the columns from the config
086        int ccs = getStatus().getCurrentColumnSet();
087        List<IColumnConfig> columnsConfig = getConfig().getColumnSets().get(ccs).getColumns();
088        final List<IColumn<IEntityTable, String>> columns = new ArrayList<IColumn<IEntityTable, String>>();
089
090        ORI parentURI = getQuery().getOn();
091
092        List<IColumnConfig> visibleColumnsConfig = new ArrayList<IColumnConfig>(columnsConfig.size());
093        BrowserPageStatus pageStatus = findParentStatus(BrowserPageStatus.class);
094
095        Predicate sortablePredicate;
096        if (pageStatus != null) {
097            sortablePredicate = new VisiblePredicate(getPageBaseOri(), pageStatus.getEntitySelections());
098            Predicate filter = new VisiblePredicate(getPageBaseOri(), pageStatus.getEntitySelections());
099            CollectionUtils.select(columnsConfig, filter, visibleColumnsConfig);
100        } else {
101            sortablePredicate = new VisiblePredicate(getPageBaseOri(), Collections.EMPTY_LIST);
102            visibleColumnsConfig = columnsConfig;
103        }
104
105        boolean tableSort = sortablePredicate.evaluate(getConfig().getSortable());
106        for (IColumnConfig columnConfig : visibleColumnsConfig) {
107            String columnSortStr = columnConfig.getSortable();
108            boolean columnSort = sortablePredicate.evaluate(columnSortStr);
109            columnConfig.addColumns(columns, parentURI, Strings.isEmpty(columnSortStr) ? tableSort : columnSort);
110        }
111
112        // Disable default status order if the table is not sortable.
113        if (!tableSort) {
114            getStatus().setOrder(null);
115        }
116
117        add(new OnDomReadyPanel("datatable") {
118            @Override
119            protected Panel onDomReadyPanel(String componentId) {
120
121                Boolean forceCount = getConfig().getForceCount();
122                if (forceCount != null && forceCount.booleanValue()) {
123                    dataProvider.forceCount();
124                }
125
126                return new DataTablePanel("datatable", columns, dataProvider, rowsPerPage);
127            }
128        });
129
130    }
131
132    @Override
133    public void onEvent(IEvent<?> event) {
134        super.onEvent(event);
135
136        if (event.getPayload() instanceof EventQueryUpdate) {
137            this.dataProvider.clearCount();
138        }
139    }
140
141    @Override
142    public void renderHead(IHeaderResponse response) {
143        super.renderHead(response);
144        response.render(CssHeaderItem.forReference(TABLE_VIEWER_CSS));
145    }
146
147}
