030import java.util.Map;
031
032public class CtDecorator extends FieldDecorator {
033
034    private static final String EMPTY_COLOR = "#FFFFFF";
035

037

048        // SSNP - #F5B800
049        COLORS.put("SYNONYMOUS_CODING", "#F5B800");

071
072    public static final Map<String, String> SHORT_NAMES = new HashMap<String, String>();
073
074    static {
075        SHORT_NAMES.put("3PRIME_UTR", "3-UTR");
076        SHORT_NAMES.put("NON_SYNONYMOUS_CODING", "NS-SNP");
077        SHORT_NAMES.put("SYNONYMOUS_CODING", "S-SNP");
078        SHORT_NAMES.put("WITHIN_NON_CODING_GENE", "non-Cod");
079        SHORT_NAMES.put("STOP_GAINED", "stp-gain");
080        SHORT_NAMES.put("STOP_LOST", "stp-lost");
081        SHORT_NAMES.put("NMD_TRANSCRIPT", "NMD");
082        SHORT_NAMES.put("SPLICE_SITE", "splice");
083        SHORT_NAMES.put("5PRIME_UTR", "5-UTR");
084        SHORT_NAMES.put("DOWNSTREAM", "downst");
085        SHORT_NAMES.put("REGULATORY_REGION", "Reg-SNP");
086        SHORT_NAMES.put("ESSENTIAL_SPLICE_SITE", "e-splice");
087        SHORT_NAMES.put("UPSTREAM", "upst");
088        SHORT_NAMES.put("INTRONIC", "intro");
089        SHORT_NAMES.put("INTERGENIC", "intergen");
090        SHORT_NAMES.put("FRAMESHIFT_CODING", "frameshift");
091        SHORT_NAMES.put("WITHIN_MATURE_MIRNA", "mature miRNA");
092        SHORT_NAMES.put("NONCODING_RNA", "non-rna");
093
094    }
095
096    private String cssClass;
097
098    public CtDecorator(Field valueProperty, String cssClass) {
099        super(valueProperty);
100        this.cssClass = cssClass;
101    }
102
103    protected String getTooltip(IEntity data) {
104        Object value = data.get(getField().getId());
105        return value == null ? "" : value.toString();
106    }
107
108    @Override
109    public String getColor(IEntity data) {
110        Object value = getValue(data);
111
112        return value == null ? EMPTY_COLOR : COLORS.get(value.toString().toUpperCase());
113    }
114
115    @Override
116    public String getTemplate() {
117        return null;
118    }
119
120    @Override
121    public void setTemplate(String template) {
122
123    }
124
125    @Override
126    public void populateCell(WebMarkupContainer cellContainer,
127                             String componentId, IModel<IEntity> data) {
128        IEntity entity = data.getObject();
129
130        Object value = entity != null ? entity.get(getField().getId()) : null;
131        String shortName = value != null ? SHORT_NAMES.get(value.toString().toUpperCase()) : "";
132        shortName = shortName == null ? value.toString().toLowerCase() : shortName;
133        Label empty = new Label(componentId, shortName);
134        // empty.setVisible(false);
135
136        cellContainer.add(empty);
137        cellContainer.add(new AttributeModifier("style", new Model<String>(
138                "background-color: " + getColor(data.getObject()) + ";")));
139        cellContainer.add(new AttributeModifier("title", new Model<String>(
140                getTooltip(data.getObject()))));
141        if (cssClass != null) {
142            cellContainer.add(new AttributeModifier("class", new Model<String>(
143                    cssClass)));
144        }
145
146    }
147
148}
