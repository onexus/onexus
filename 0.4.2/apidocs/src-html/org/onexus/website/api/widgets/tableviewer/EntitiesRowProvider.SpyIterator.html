035
036public class EntitiesRowProvider implements
037        ISortableDataProvider<IEntityTable, String> {
038
039    @Inject
040    private ICollectionManager collectionManager;
041
042    private IModel<TableViewerStatus> statusModel;
043    private transient EntitiesRow rows;
044    private SortState sortState = new SortState();
045    private int rowsPerPage;
046    private long knownSize;
047    private long realSize;
048    private Query query;
049
050    public EntitiesRowProvider(IModel<TableViewerStatus> status, int rowsPerPage) {
051        WebsiteApplication.inject(this);
052        this.statusModel = status;
053        this.rowsPerPage = rowsPerPage;
054        this.knownSize = rowsPerPage + 2;
055        this.realSize = -1;
056    }
057
058    public EntitiesRowProvider(Query query, IModel<TableViewerStatus> status, int rowsPerPage) {
059        this(status, rowsPerPage);
060        this.query = query;
061    }
062
063
064    protected TableViewerStatus getTableViewerStatus() {
065        return statusModel.getObject();
066    }
067
068    public void close() {
069        if (rows != null) {
070            rows.close();
071        }
072    }
073
074    @Override
075    public Iterator<IEntityTable> iterator(long first, long total) {
076
077        Query query = getQuery();
078        query.setOffset(first);
079        query.setCount(total);
080        return new SpyIterator<IEntityTable>(first, total, loadIterator(query));
081    }
082
083    @Override
084    public long size() {
085        if (realSize != -1) {
086            return realSize;
087        }
088
089        return knownSize;
090    }
091
092    public void clearCount() {
093        this.realSize = -1;
094        this.knownSize = rowsPerPage + 2;
095    }
096
097    public void forceCount() {
098
099        if (realSize == -1) {
100            Query query = getQuery();
101            IEntityTable entityTable = collectionManager.load(query);
102
103            Progress progress = entityTable.getProgress();
104            if (progress != null && !progress.isDone()) {
105                addTaskStatus(entityTable.getProgress());
106            }
107
108            this.realSize = entityTable.size();
109            this.knownSize = this.realSize;
110        }
111    }
112
113    public long getRealSize() {
114        return realSize;
115    }
116
117    public void setRealSize(long realSize) {
118        this.realSize = realSize;
119    }
120
121    public long getKnownSize() {
122        return knownSize;
123    }
124
125    public void setKnownSize(long knownSize) {
126        this.knownSize = knownSize;
127    }
128
129    protected Query getQuery() {
130        return this.query;
131    }
132
133    protected void addTaskStatus(Progress progressStatus) {
134
135    }
136
137    private Iterator<IEntityTable> loadIterator(Query query) {
138        if (rows == null) {
139            //TODO IEntityTable entityTable = ProgressBar.show(collectionManager.load(query));
140            IEntityTable entityTable = collectionManager.load(query);
141
142            Progress progress = entityTable.getProgress();
143            if (progress != null && !progress.isDone()) {
144                addTaskStatus(entityTable.getProgress());
145            }
146
147            rows = new EntitiesRow(entityTable);
148        }
149        return rows;
150    }
151
152    @Override
153    public IModel<IEntityTable> model(IEntityTable object) {
154        return new EntityMatrixModel(object);
155    }
156
157    @Override
158    public void detach() {
159        if (rows != null) {
160            statusModel.detach();
161            rows = null;
162        }
163    }
164
165    @Override
166    public ISortState getSortState() {
167        return sortState;
168    }

186    }

189    public class SortState implements ISortState<String> {
190
191        @Override
192        public void setPropertySortOrder(String property, SortOrder order) {
193            String[] values = property
194                    .split(FieldHeader.SORT_PROPERTY_SEPARATOR);
195            String collectionId = values[0];
196            String fieldName = values[1];
197
198            EntitiesRowProvider.this.realSize = -1;
199            EntitiesRowProvider.this.knownSize = EntitiesRowProvider.this.rowsPerPage + 2;
200
201            getTableViewerStatus().setOrder(new OrderBy(collectionId, fieldName, order == SortOrder.ASCENDING));
202
203        }
204
205        @Override
206        public SortOrder getPropertySortOrder(String property) {
207            String[] values = property.split(FieldHeader.SORT_PROPERTY_SEPARATOR);
208            String collectionId = values[0];
209            String fieldId = values[1];
210
211            OrderBy order = getTableViewerStatus().getOrder();
212            if (order != null && collectionId.endsWith(order.getCollection()) && order.getField().equals(fieldId)) {
213                return order.isAscendent() ? SortOrder.ASCENDING : SortOrder.DESCENDING;
214            }
215
216            return SortOrder.NONE;
217        }
218
219    }
220
221    public class SpyIterator<T> implements Iterator<T> {
222
223        private long first, total;
224        private Iterator<T> it;
225
226        private long currentPos;
227
228        public SpyIterator(long first, long total, Iterator<T> it) {
229            this.first = first;
230            this.it = it;
231            this.total = total;
232            this.currentPos = first - 1;
233        }
234
235        @Override
236        public boolean hasNext() {
237            boolean hasNext = it.hasNext();
238            if (!hasNext && (currentPos - first < total)) {
239                EntitiesRowProvider.this.knownSize = currentPos + 1;
240                EntitiesRowProvider.this.realSize = currentPos + 1;
241            }
242            return hasNext;
243        }
244
245        @Override
246        public T next() {
247            currentPos++;
248            return it.next();
249        }
250
251        @Override
252        public void remove() {
253            it.remove();
254        }
255    }
