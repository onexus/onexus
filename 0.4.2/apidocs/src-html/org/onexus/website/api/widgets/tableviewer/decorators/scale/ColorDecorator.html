018package org.onexus.website.api.widgets.tableviewer.decorators.scale;
019
020import org.apache.commons.lang3.StringUtils;
021import org.apache.wicket.AttributeModifier;
022import org.apache.wicket.markup.html.WebMarkupContainer;
023import org.apache.wicket.markup.html.basic.Label;
024import org.apache.wicket.model.IModel;
025import org.apache.wicket.model.Model;
026import org.onexus.collection.api.*;
027import org.onexus.collection.api.Collection;
028import org.onexus.website.api.widgets.tableviewer.decorators.scale.scales.ColorUtils;
029import org.onexus.website.api.widgets.tableviewer.decorators.scale.scales.IColorScaleHtml;
030import org.onexus.website.api.widgets.tableviewer.decorators.utils.FieldDecorator;
031import org.onexus.website.api.widgets.tableviewer.formaters.ITextFormater;
032
033import java.awt.*;
034import java.util.*;
035import java.util.List;
036
037public class ColorDecorator extends FieldDecorator {
038
039    private static final Color EMPTY_COLOR = new Color(255, 255, 255);
040    private static final String DEFAULT_URL_TITLE = "Click to see more details about this value";
041
042    private boolean showValue = false;
043    private Field tooltipField;
044    private IColorScaleHtml colorScale;
045    private String cssClass;
046    private String urlLink;
047    private String urlTitle;
048
049
050    public ColorDecorator(Field valueField, IColorScaleHtml colorScale) {
051        this(valueField, colorScale, null);
052    }
053
054    public ColorDecorator(Field valueField, IColorScaleHtml colorScale,
055                          String cssClass) {
056        this(valueField, valueField, colorScale, cssClass, false, null, null, DEFAULT_URL_TITLE);
057    }
058
059    public ColorDecorator(Field valueField, IColorScaleHtml colorScale,
060                          String cssClass, ITextFormater textFormater) {
061        this(valueField, valueField, colorScale, cssClass, false, textFormater, null, DEFAULT_URL_TITLE);
062    }
063
064    public ColorDecorator(Field valueField, IColorScaleHtml colorScale,
065                          String cssClass, boolean showValue) {
066        this(valueField, valueField, colorScale, cssClass, showValue, null, null, DEFAULT_URL_TITLE);
067    }
068
069    public ColorDecorator(Field valueField, Field tooltipField,
070                          IColorScaleHtml colorScale, String cssClass, boolean showValue,
071                          ITextFormater textFormater, String urlLink, String urlTitle) {
072        super(valueField, textFormater);
073        this.tooltipField = tooltipField;
074        this.colorScale = colorScale;
075        this.cssClass = cssClass;
076        this.showValue = showValue;
077        this.urlLink = urlLink;
078        this.urlTitle = urlTitle;
079    }
080
081    protected String getTooltip(IEntity entity) {
082
083        if (tooltipField == null || entity == null) {
084            return "";
085        }
086
087        Object value = entity.get(tooltipField.getId());
088
089        if (value == null) {
090            return "";
091        }
092
093        return value.toString();
094
095    }
096
097    public boolean isShowValue() {
098        return showValue;
099    }
100
101    public void setShowValue(boolean showValue) {
102        this.showValue = showValue;
103    }
104
105    public String getColor(IEntity entity) {
106        return ColorUtils.colorToHexHtml(getRealColor(entity));
107    }
108
109    public Color getRealColor(IEntity entity) {
110
111        Object value = getValue(entity);
112
113        if (value == null) {
114            return EMPTY_COLOR;
115        }
116
117        return colorScale.valueColor(value);
118    }
119
120    @Override
121    public List<String> getExtraFields(Collection collection) {
122
123        if (this.urlLink != null) {
124            return extractFields(this.urlLink);
125        }
126
127        return Collections.EMPTY_LIST;
128    }
129
130    @Override
131    public void populateCell(WebMarkupContainer cellContainer,
132                             String componentId, IModel<IEntity> entityModel) {
133
134        IEntity rowEntity = entityModel.getObject();
135        Color bkgColor = getRealColor(rowEntity);
136
137        if (!showValue) {
138            WebMarkupContainer empty = new WebMarkupContainer(componentId);
139            empty.setVisible(false);
140            cellContainer.add(empty);
141        } else {
142            String value = getFormatValue(rowEntity);
143            boolean black = lum(bkgColor) >= 128;
144            String styleCss = black ? "color: #000;" : "color: #FFF;";
145            if (urlLink != null && !StringUtils.isEmpty(value)) {
146                String url = fixLinkUrl(urlLink);
147                url = replaceParameters(getField(), value, rowEntity, url, false);
148                String link = "<a href=\"" + url + "\" rel=\"tooltip\" title=\"" + urlTitle + "\"><i class=\"icon-zoom-in" + (black ? "" : " icon-white") + "\"></i></a>&nbsp;";
149                value = link + value;
150            }
151            Label textValue = new Label(componentId, value);
152            textValue.setEscapeModelStrings(false);
153            textValue.add(new AttributeModifier("style", new Model<String>(styleCss)));
154            cellContainer.add(textValue);
155        }
156
157        cellContainer
158                .add(new AttributeModifier("style", new Model<String>(
159                        "background-color: " + ColorUtils.colorToHexHtml(bkgColor) + ";")));
160        cellContainer.add(new AttributeModifier("title", new Model<String>(
161                getTooltip(rowEntity))));
162        if (cssClass != null) {
163            cellContainer.add(new AttributeModifier("class", new Model<String>(
164                    cssClass)));
165        }
166
167    }
168
169    public static double lum(Color color) {
170        int r = color.getRed();
171        int g = color.getGreen();
172        int b = color.getBlue();
173        return .299 * r + .587 * g + .114 * b;
174    }
