018package org.onexus.website.api.widgets.tableviewer;
019
020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.ajax.form.OnChangeAjaxBehavior;
023import org.apache.wicket.ajax.markup.html.AjaxLink;
024import org.apache.wicket.event.Broadcast;
025import org.apache.wicket.extensions.ajax.markup.html.IndicatingAjaxLink;
026import org.apache.wicket.extensions.markup.html.repeater.data.table.AbstractToolbar;
027import org.apache.wicket.extensions.markup.html.repeater.data.table.DataTable;
028import org.apache.wicket.markup.html.WebMarkupContainer;
029import org.apache.wicket.markup.html.basic.Label;
030import org.apache.wicket.markup.html.form.ChoiceRenderer;
031import org.apache.wicket.markup.html.form.DropDownChoice;
032import org.apache.wicket.markup.html.form.IChoiceRenderer;
033import org.apache.wicket.model.PropertyModel;
034import org.onexus.website.api.events.EventFiltersUpdate;
035import org.onexus.website.api.events.EventQueryUpdate;
036import org.onexus.website.api.events.EventViewChange;
037
038import java.util.Arrays;
039import java.util.List;
040
041public class NavigationToolbar extends AbstractToolbar {
042
043    private Integer rowsPerPage;
044
045    private static List<Integer> PAGE_OPTIONS = Arrays.asList(20, 40, 60, 100);
046
047    public NavigationToolbar(final DataTable<?, ?> table) {
048        super(table);
049
050
051    }
052
053    @Override
054    protected void onBeforeRender() {
055
056        WebMarkupContainer span = new WebMarkupContainer("span");
057        addOrReplace(span);
058        span.add(AttributeModifier.replace("colspan", String.valueOf(getTable().getColumns().size())));
059
060        DropDownChoice<Integer> dropDown = new DropDownChoice<Integer>("rowsxpage",
061                new PropertyModel<Integer>(this, "rowsPerPage"),
062                PAGE_OPTIONS,
063                new RowsPerPageChoiceRenderer()
064        );
065
066        dropDown.add(new OnChangeAjaxBehavior() {
067            @Override
068            protected void onUpdate(AjaxRequestTarget target) {
069                send(getPage(), Broadcast.BREADTH, EventViewChange.EVENT);
070            }
071        });
072
073        span.add(dropDown);
074
075        span.add(new PrevLink("prev"));
076
077        span.add(new NextLink("next"));
078
079        AjaxLink<String> countLink = new IndicatingAjaxLink<String>("count") {
080            @Override
081            public void onClick(AjaxRequestTarget target) {
082                getDataProvider().forceCount();
083                target.add(getTable());
084            }
085        };
086        span.add(countLink);
087        countLink.add(new Label("from", new PropertyModel<Long>(this, "from")));
088        countLink.add(new Label("to", new PropertyModel<Long>(this, "to")));
089        countLink.add(new Label("of", new PropertyModel<String>(this, "of")));
090
091        super.onBeforeRender();
092    }
093
094    public long getFrom() {
095
096        if (getDataProvider().getRealSize() == 0) {
097            return 0;
098        }
099
100        long itemsPerPage = getTable().getItemsPerPage();
101        long currentPage = getTable().getCurrentPage();
102        return (currentPage * itemsPerPage) + 1;
103    }
104
105    public long getTo() {
106        long itemsPerPage = getTable().getItemsPerPage();
107        long from = getFrom();
108        return from + Math.min(itemsPerPage, getDataProvider().size() - from + 1) - 1;
109    }
110
111    public String getOf() {
112
113        long realSize = getDataProvider().getRealSize();
114
115        if (realSize == -1) {
116            return "(click to count)";
117        }
118
119        return Long.toString(realSize);
120    }
121
122    private EntitiesRowProvider getDataProvider() {
123        return (EntitiesRowProvider) getTable().getDataProvider();
124    }
125
126    public Integer getRowsPerPage() {
127
128        if (rowsPerPage == null) {
129            Integer sessionRowsPerPage = getSession().getMetaData(TableViewer.DEFAULT_ROWS_PER_PAGE);
130            rowsPerPage = sessionRowsPerPage == null ? 20 : sessionRowsPerPage;
131        }
132
133        return rowsPerPage;
134    }
135
136    public void setRowsPerPage(Integer rowsPerPage) {
137        getSession().setMetaData(TableViewer.DEFAULT_ROWS_PER_PAGE, rowsPerPage);
138        this.rowsPerPage = rowsPerPage;
139    }
140
141    private class PrevLink extends AjaxLink<String> {
142
143        public PrevLink(String id) {
144            super(id);
145        }
146
147        @Override
148        public void onClick(AjaxRequestTarget target) {
149            getTable().setCurrentPage(getTable().getCurrentPage() - 1);
150            target.add(getTable());
151        }
152
153        @Override
154        protected void onBeforeRender() {
155            setVisible(getFrom() > 1);
156            super.onBeforeRender();
157        }
158
159    }
160
161    private class NextLink extends AjaxLink<String> {
162
163        public NextLink(String id) {
164            super(id);
165        }
166
167        @Override
168        public void onClick(AjaxRequestTarget target) {
169
170            long page = getTable().getCurrentPage() + 1;
171            getTable().setCurrentPage(page);
172
173            getDataProvider().setKnownSize(((page + 1) * getTable().getItemsPerPage()) + 2);
174
175            target.add(getTable());
176
177        }
178
179        @Override
180        protected void onBeforeRender() {
181            long to = getTo();
182            long size = getDataProvider().size();
183
184            setVisible(size > to);
185            super.onBeforeRender();
186        }
187    }
188
189    private class RowsPerPageChoiceRenderer implements IChoiceRenderer<Integer> {
190
191        @Override
192        public Object getDisplayValue(Integer object) {
193            return String.valueOf(object) + " rows";
194        }
195
196        @Override
197        public String getIdValue(Integer object, int index) {
198            return String.valueOf(object);
199        }
200    }
201
202}
