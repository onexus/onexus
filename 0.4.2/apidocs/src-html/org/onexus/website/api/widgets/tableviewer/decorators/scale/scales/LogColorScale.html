018package org.onexus.website.api.widgets.tableviewer.decorators.scale.scales;
019
020import java.awt.*;
021
022public class LogColorScale extends AbstractColorScale {
023
024    public static final double DEFAULT_LOG_FACTOR = 0.25;
025
026    protected Color nonSignificantColor;
027
028    private double logFactor;
029
030    public LogColorScale(double minPoint, double maxPoint, double logFactor) {
031
032        super(minPoint, maxPoint);
033        this.logFactor = logFactor;
034    }
035
036    public LogColorScale(double minPoint, double maxPoint) {
037
038        this(minPoint, maxPoint, DEFAULT_LOG_FACTOR);
039    }
040
041    public LogColorScale() {
042        this(0.0, 1.0, DEFAULT_LOG_FACTOR);
043    }
044
045    public LogColorScale(double minPoint, double maxPoint, Color minColor,
046                         Color maxColor) {
047
048        this(minPoint, maxPoint);
049        setMinColor(minColor);
050        setMaxColor(maxColor);
051    }
052
053    public double getLogFactor() {
054        return logFactor;
055    }
056
057    public void setLogFactor(double logFactor) {
058        this.logFactor = logFactor;
059    }
060
061    public Color valueColor(Object obj) {
062
063        double value;
064        if (obj instanceof Double) {
065            value = (Double) obj;
066        } else {
067            value = Double.valueOf(String.valueOf(obj));
068        }
069
070
071        if (Double.isNaN(value)) {
072            return notANumberColor;
073        }
074        else if (value > maxPoint || value == Double.POSITIVE_INFINITY) {
075            return posInfinityColor;
076        }
077        else if (value < minPoint || value == Double.NEGATIVE_INFINITY) {
078            return negInfinityColor;
079        }
080
081        double range = maxPoint - minPoint;
082
083        double f = value / range;
084
085        f = f > 0.0 ? 1.0 + logFactor * Math.log10(f) : f < 0.0 ? 1.0
086                + logFactor * Math.log10(-f) : 0.0;
087
088        return ColorUtils.mix(maxColor, minColor, f);
089    }
090}
