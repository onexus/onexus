018package org.onexus.website.api.widgets.selection;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.ajax.form.AjaxFormComponentUpdatingBehavior;
023import org.apache.wicket.ajax.markup.html.AjaxLink;
024import org.apache.wicket.event.Broadcast;
025import org.apache.wicket.markup.html.basic.Label;
026import org.apache.wicket.markup.html.form.ChoiceRenderer;
027import org.apache.wicket.markup.html.form.DropDownChoice;
028import org.apache.wicket.markup.html.form.Form;
029import org.apache.wicket.markup.html.list.ListItem;
030import org.apache.wicket.markup.html.list.ListView;
031import org.apache.wicket.markup.html.panel.EmptyPanel;
032import org.apache.wicket.model.IModel;
033import org.apache.wicket.model.PropertyModel;
034import org.onexus.website.api.events.EventAddFilter;
035import org.onexus.website.api.events.EventCloseModal;
036import org.onexus.website.api.pages.browser.BrowserPage;
037import org.onexus.website.api.pages.browser.BrowserPageStatus;
038import org.onexus.website.api.widgets.Widget;
039import org.onexus.website.api.widgets.selection.custom.CustomFilter;
040import org.onexus.website.api.widgets.selection.custom.ListCustomFilterPanel;
041import org.onexus.website.api.widgets.selection.custom.NumericCustomFilterPanel;
042
043public class FiltersWidget extends Widget<FiltersWidgetConfig, FiltersWidgetStatus> {
044
045    private CustomFilter currentFilter;
046
047    private static final Component EMPTY_CUSTOM_PANEL = new EmptyPanel("customPanel").setOutputMarkupId(true);
048
049    public FiltersWidget(String componentId, IModel<FiltersWidgetStatus> statusModel) {
050        super(componentId, statusModel);
051
052        // Filters list
053        final Form<String> filtersForm = new Form<String>("filtersForm");
054        filtersForm.setOutputMarkupId(true);
055        filtersForm.add(new ListView<FilterConfig>("filters", getConfig().getFilters()) {
056
057            @Override
058            protected void populateItem(final ListItem<FilterConfig> item) {
059
060                final FilterConfig filterConfig = item.getModelObject();
061
062                item.setOutputMarkupId(true);
063
064                // Add button
065                item.add(new AjaxLink<String>("apply") {
066
067                    @Override
068                    public void onClick(AjaxRequestTarget target) {
069                        applyFilter(filterConfig, target);
070                    }
071
072                });
073
074                // Title
075                item.add(new Label("name", new TextFormaterPropertyModel(item.getModel(), "name", 65, true)));
076
077
078            }
079        });
080        add(filtersForm);
081
082        // Add custom filter select
083        Form<String> selectForm = new Form<String>("selectForm");
084        DropDownChoice<CustomFilter> selector = new DropDownChoice<CustomFilter>(
085                "select",
086                new PropertyModel<CustomFilter>(this, "currentFilter"),
087                getConfig().getCustomFilters(),
088                new ChoiceRenderer<CustomFilter>("title"));
089
090        selector.add(new AjaxFormComponentUpdatingBehavior("onchange") {
091            @Override
092            protected void onUpdate(AjaxRequestTarget target) {
093
094                CustomFilter customFilter = getCurrentFilter();
095
096                if (customFilter == null || customFilter.getType() == null) {
097                    FiltersWidget.this.addOrReplace(EMPTY_CUSTOM_PANEL);
098                } else if (customFilter.getType().equalsIgnoreCase("list")) {
099                    FiltersWidget.this.addOrReplace(new ListCustomFilterPanel("customPanel", customFilter) {
100                        @Override
101                        protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
102                            addCustomFilter(target, filterConfig);
103                        }
104                    });
105                } else if (customFilter.getType().equalsIgnoreCase("numeric")) {
106                    FiltersWidget.this.addOrReplace(new NumericCustomFilterPanel("customPanel", customFilter, FiltersWidget.this.getPageBaseOri()) {
107                        @Override
108                        protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
109                            addCustomFilter(target, filterConfig);
110                        }
111                    });
112                }
113
114                target.add(FiltersWidget.this.get("customPanel"));
115
116            }
117
118            private void addCustomFilter(AjaxRequestTarget target, FilterConfig filter) {
119                applyFilter(filter, target);
120            }
121
122        });
123        selectForm.add(selector);
124        add(selectForm);
125
126        // Add custom filter panel
127        add(EMPTY_CUSTOM_PANEL);
128
129
130    }
131
132    protected void applyFilter(FilterConfig filterConfig, AjaxRequestTarget target) {
133        getPageStatus().addEntitySelection(new MultipleEntitySelection(filterConfig));
134        send(getPage(), Broadcast.BREADTH, EventAddFilter.EVENT);
135        send(getPage(), Broadcast.BREADTH, EventCloseModal.EVENT);
136    }
137
138    public CustomFilter getCurrentFilter() {
139        return currentFilter;
140    }
141
142    public void setCurrentFilter(CustomFilter currentFilter) {
143        this.currentFilter = currentFilter;
144    }
145
146    protected BrowserPageStatus getPageStatus() {
147        return findParent(BrowserPage.class).getStatus();
148    }
149}
