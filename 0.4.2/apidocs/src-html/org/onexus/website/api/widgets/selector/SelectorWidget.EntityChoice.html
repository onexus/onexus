018package org.onexus.website.api.widgets.selector;
019
020import org.apache.wicket.ajax.AjaxRequestTarget;
021import org.apache.wicket.ajax.form.OnChangeAjaxBehavior;
022import org.apache.wicket.extensions.ajax.markup.html.AjaxIndicatorAppender;
023import org.apache.wicket.markup.html.form.DropDownChoice;
024import org.apache.wicket.markup.html.form.Form;
025import org.apache.wicket.model.IModel;
026import org.apache.wicket.model.PropertyModel;

028import org.apache.wicket.util.string.Strings;

030import org.onexus.collection.api.IEntity;
031import org.onexus.collection.api.query.Filter;

037import org.onexus.resource.api.ORI;
038import org.onexus.website.api.events.EventAddFilter;
039import org.onexus.website.api.events.EventFiltersUpdate;
040import org.onexus.website.api.pages.browser.BrowserPageStatus;
041import org.onexus.website.api.pages.browser.SingleEntitySelection;

043
044import javax.inject.Inject;
045import java.io.Serializable;
046import java.util.ArrayList;
047import java.util.Iterator;
048import java.util.List;
049
050public class SelectorWidget extends Widget<SelectorWidgetConfig, SelectorWidgetStatus> {
051
052    @Inject
053    private ICollectionManager collectionManager;
054
055    @Inject
056    private IQueryParser queryParser;
057
058    private transient EntityChoice selection;
059    private transient List<EntityChoice> choices;
060
061    public SelectorWidget(String componentId, IModel<SelectorWidgetStatus> statusModel) {
062        super(componentId, statusModel);
063
064        Form form = new Form("form");
065        DropDownChoice<EntityChoice> dropDown = new DropDownChoice<EntityChoice>(
066                "select",
067                new PropertyModel<EntityChoice>(this, "selection"),
068                getChoices()
069        ) {
070            @Override
071            protected CharSequence getDefaultChoice(String selectedValue) {
072
073                String option = SelectorWidget.this.getConfig().getTitle();
074
075                if (Strings.isEmpty(option)) {
076                    option = "Choose one...";
077                }
078
079                // The <option> tag buffer
080                final AppendingStringBuffer buffer = new AppendingStringBuffer(64 + option.length());
081
082                // Add option tag
083                buffer.append("\n<option");
084
085                // If null is selected, indicate that
086                if ("".equals(selectedValue)) {
087                    buffer.append(" selected=\"selected\"");
088                }
089
090                // Add body of option tag
091                buffer.append(" value=\"\">").append(option).append("</option>");
092
093                return buffer;
094            }
095        };
096
097        dropDown.add(new OnChangeAjaxBehavior() {
098            @Override
099            protected void onUpdate(AjaxRequestTarget target) {
100
101                Boolean selection = getConfig().getSelection();
102
103                if (selection == null || !selection) {
104                    sendEvent(EventFiltersUpdate.EVENT);
105                } else {
106                    EntityChoice choice = getSelection();
107
108                    if (choice != null) {
109                        BrowserPageStatus pageStatus = findParentStatus(BrowserPageStatus.class);
110                        pageStatus.addEntitySelection(new SingleEntitySelection(getConfig().getCollection(), getSelection().getId()));
111                        setSelection(null);
112                        sendEvent(EventAddFilter.EVENT);
113                    }
114                }
115            }
116        });
117
118        dropDown.setNullValid(true);
119
120        add(new AjaxIndicatorAppender());
121
122        form.add(dropDown);
123        add(form);
124
125    }
126
127    public EntityChoice getSelection() {
128
129        if (selection == null) {
130            SelectorWidgetStatus status = getStatus();
131            if (status != null && status.getSelection() != null) {
132                for (EntityChoice choice : getChoices()) {
133                    if (choice.getId().equals(status.getSelection())) {
134                        this.selection = choice;
135                        break;
136                    }
137                }
138            }
139        }
140
141        return selection;
142    }
143
144    public void setSelection(EntityChoice entityChoice) {
145        selection = entityChoice;
146        String id = entityChoice == null ? null : entityChoice.getId();
147        getStatus().setSelection(id);
148    }
149
150    private List<EntityChoice> getChoices() {
151
152        if (choices == null) {
153
154            String field = getConfig().getField();
155            ORI collection = getConfig().getCollection();
156
157            Query query = new Query();
158            query.setOn(getWebsiteOri());
159
160            String collectionAlias = QueryUtils.newCollectionAlias(query, collection);
161            query.setFrom(collectionAlias);
162            query.addSelect(collectionAlias, null);
163            query.addOrderBy(new OrderBy(collectionAlias, field));
164
165            String oqlWhere = getConfig().getWhere();
166
167            if (oqlWhere != null && !oqlWhere.isEmpty()) {
168                Filter where = queryParser.parseWhere(collectionAlias + "." + getConfig().getWhere().trim());
169                query.setWhere(where);
170            }
171
172            Iterator<IEntity> choicesIt = new EntityIterator(collectionManager.load(query), collection);
173
174            choices = new ArrayList<EntityChoice>();
175
176            while (choicesIt.hasNext()) {
177                IEntity entity = choicesIt.next();
178                choices.add(new EntityChoice(entity.getId(), String.valueOf(entity.get(field))));
179            }
180
181        }
182
183        return choices;
184    }
185
186    public static class EntityChoice implements Serializable {
187
188        private String id;
189        private String field;
190
191        public EntityChoice(String id, String field) {
192            this.id = id;
193            this.field = field;
194        }
195
196        public String getId() {
197            return id;
198        }
199
200        public void setId(String id) {
201            this.id = id;
202        }
203
204        public String getField() {
205            return field;
206        }
207
208        public void setField(String field) {
209            this.field = field;
210        }
211
212        public String toString() {
213            return getField();
214        }
215    }
