018package org.onexus.website.api.widgets.tableviewer.decorators.scale;
019
020import org.onexus.resource.api.ParameterKey;
021
022public enum CategoricalDecoratorParameters implements ParameterKey {
023
024    DEFAULT("default", "Set the default color. Syntax: [r,g,b]", null, false),
025    MAP("map", "Set the color map. Example: [blue value]=[0,0,255] | [red value]=[255,0,0]", null, false);
026
027    private final String key;
028    private final String description;
029    private final String defaultValue;
030    private final boolean optional;
031
032    private CategoricalDecoratorParameters(String key, String description, String defaultValue, boolean optional) {
033        this.key = key;
034        this.description = description;
035        this.defaultValue = defaultValue;
036        this.optional = optional;
037    }
038
039    @Override
040    public String getKey() {
041        return key;
042    }
043
044    @Override
045    public String getDescription() {
046        return description;
047    }
048
049    @Override
050    public String getDefault() {
051        return defaultValue;
052    }
053
054    @Override
055    public boolean isOptional() {
056        return optional;
057    }
058}
