018package org.onexus.website.api.widgets.tableviewer;
019
020import org.apache.wicket.ajax.IAjaxIndicatorAware;
021import org.apache.wicket.extensions.markup.html.repeater.data.table.DataTable;
022import org.apache.wicket.extensions.markup.html.repeater.data.table.IColumn;
023import org.apache.wicket.extensions.markup.html.repeater.data.table.NoRecordsToolbar;
024import org.apache.wicket.markup.html.WebMarkupContainer;
025import org.apache.wicket.markup.html.image.Image;
026import org.apache.wicket.markup.html.panel.Panel;
027import org.onexus.collection.api.IEntityTable;
028import org.onexus.website.api.utils.panels.ondomready.OnDomReadyPanel;
029
030import java.util.List;
031
032public class DataTablePanel extends Panel implements IAjaxIndicatorAware {
033
034    private WebMarkupContainer indicator;
035
036    private transient EntitiesRowProvider dataProvider;
037
038    public DataTablePanel(String id, final List<? extends IColumn<IEntityTable, String>> columns,
039                          final EntitiesRowProvider dataProvider, final long defaultRowsPerPage) {
040        super(id);
041
042        this.dataProvider = dataProvider;
043
044        indicator = new WebMarkupContainer("indicator");
045        indicator.setOutputMarkupId(true);
046        indicator.add(new Image("loading", OnDomReadyPanel.LOADING_IMAGE));
047        add(indicator);
048
049        DataTable<IEntityTable, String> resultTable = new DataTable<IEntityTable, String>("datatable", columns, dataProvider, defaultRowsPerPage);
050        resultTable.setOutputMarkupId(true);
051        resultTable.setVersioned(false);
052        resultTable.addTopToolbar(new HeadersToolbar(resultTable, dataProvider));
053        resultTable.addBottomToolbar(new NoRecordsToolbar(resultTable));
054        resultTable.addBottomToolbar(new NavigationToolbar(resultTable));
055        add(resultTable);
056    }
057
058    @Override
059    public String getAjaxIndicatorMarkupId() {
060        return indicator.getMarkupId();
061    }
062
063    @Override
064    protected void onAfterRender() {
065        super.onAfterRender();
066        if (dataProvider != null) {
067            dataProvider.close();
068        }
069    }
070}
