018package org.onexus.website.api;
019
020import org.apache.commons.lang3.SerializationUtils;
021import org.onexus.resource.api.Resource;
022import org.onexus.resource.api.annotations.ResourceAlias;
023import org.onexus.resource.api.annotations.ResourceImplicitList;
024import org.onexus.website.api.pages.PageConfig;
025import org.onexus.website.api.pages.PageStatus;
026import org.onexus.website.api.utils.authorization.IAuthorization;
027
028import java.util.ArrayList;
029import java.util.List;
030
031@ResourceAlias("website")
032public class WebsiteConfig extends Resource implements IAuthorization {
033
034    private WebsiteStatus defaultStatus;
035
036    private String authorization;
037
038    private Boolean login;
039
040    private String signInPage;
041
042    private List<PageConfig> pages;
043
044    private String header;
045
046    private String css;
047
048    private String bottom;
049
050    @ResourceImplicitList("connection")
051    private List<Connection> connections;
052
053    public WebsiteConfig() {
054        super();
055    }
056
057    public List<Connection> getConnections() {
058        return connections;
059    }
060
061    public void setConnections(List<Connection> connections) {
062        this.connections = connections;
063    }
064
065    @Override
066    public String getAuthorization() {
067        return authorization;
068    }
069
070    public void setAuthorization(String authorization) {
071        this.authorization = authorization;
072    }
073
074    public Boolean getLogin() {
075        return login;
076    }
077
078    public void setLogin(Boolean login) {
079        this.login = login;
080    }
081
082    public String getHeader() {
083        return header;
084    }
085
086    public void setHeader(String header) {
087        this.header = header;
088    }
089
090    public String getCss() {
091        return css;
092    }
093
094    public void setCss(String css) {
095        this.css = css;
096    }
097
098    public String getBottom() {
099        return bottom;
100    }
101
102    public void setBottom(String bottom) {
103        this.bottom = bottom;
104    }
105
106    public String getSignInPage() {
107        return signInPage;
108    }
109
110    public void setSignInPage(String signInPage) {
111        this.signInPage = signInPage;
112    }
113
114    public List<PageConfig> getPages() {
115        return pages;
116    }
117
118    public PageConfig getPage(String pageId) {
119        if (pages != null) {
120            for (PageConfig page : pages) {
121                if (pageId.equals(page.getId())) {
122                    return page;
123                }
124            }
125        }
126
127        return null;
128    }
129
130    public void setPages(List<PageConfig> pages) {
131        this.pages = pages;
132    }
133
134    public WebsiteStatus getDefaultStatus() {
135        return defaultStatus;
136    }
137
138    public WebsiteStatus createEmptyStatus() {
139        return new WebsiteStatus();
140    }
141
142    public void setDefault(WebsiteStatus defaultStatus) {
143        this.defaultStatus = defaultStatus;
144    }
145
146    public WebsiteStatus newStatus() {
147
148        WebsiteStatus status = getDefaultStatus();
149
150        if (status != null) {
151            status = SerializationUtils.clone(status);
152        } else {
153            status = createEmptyStatus();
154        }
155
156        status.setConfig(this);
157
158        // Add page status
159        if (getPages() != null) {
160            List<PageStatus> pageStatuses = new ArrayList<PageStatus>();
161            for (PageConfig page : getPages()) {
162                page.setWebsiteConfig(this);
163                pageStatuses.add(page.newStatus());
164            }
165            status.setPageStatuses(pageStatuses);
166        }
167
168        return status;
169    }
170
171}
