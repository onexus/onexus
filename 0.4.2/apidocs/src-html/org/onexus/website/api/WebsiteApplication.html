018package org.onexus.website.api;
019
020import org.apache.wicket.Application;
021import org.apache.wicket.Page;
022import org.apache.wicket.RestartResponseAtInterceptPageException;
023import org.apache.wicket.RuntimeConfigurationType;
024import org.apache.wicket.authroles.authentication.AbstractAuthenticatedWebSession;
025import org.apache.wicket.authroles.authentication.AuthenticatedWebApplication;
026import org.apache.wicket.core.request.handler.PageProvider;
027import org.apache.wicket.core.request.handler.RenderPageRequestHandler;
028import org.apache.wicket.markup.html.WebPage;
029import org.apache.wicket.protocol.http.RequestUtils;
030import org.apache.wicket.request.IRequestHandler;
031import org.apache.wicket.request.cycle.AbstractRequestCycleListener;
032import org.apache.wicket.request.cycle.RequestCycle;
033import org.onexus.resource.api.ORI;
034import org.onexus.resource.api.session.LoginContext;
035import org.onexus.website.api.utils.error.ExceptionErrorPage;
036import org.onexus.website.api.utils.panels.NotAuthorizedPage;
037import org.onexus.website.api.utils.panels.SignOutPage;
038import org.ops4j.pax.wicket.api.InjectorHolder;
039
040import javax.servlet.http.HttpServletRequest;
041import javax.servlet.http.HttpSession;
042
043public class WebsiteApplication extends AuthenticatedWebApplication {
044
045    private static final String SERVER_URL = System.getProperty("onexus.server.url");
046
047    private String webPath;
048    private ORI websiteOri;
049    private Class<? extends WebPage> signInPageClass;
050
051    public WebsiteApplication() {
052        super();
053    }
054
055    @Override
056    public RuntimeConfigurationType getConfigurationType() {
057        return RuntimeConfigurationType.DEPLOYMENT;
058    }
059
060    public Class<? extends Page> getHomePage() {
061        return Website.class;
062    }
063
064    public boolean usePersonSignIn() {
065        return Boolean.parseBoolean(System.getProperty("org.onexus.ui.authentication.persona", "false"));
066    }
067
068    @Override
069    protected void init() {
070        super.init();
071
072        // Authentication
073        getApplicationSettings().setAccessDeniedPage(getSignInPageClass());
074
075                /* In case of unhandled exception redirect it to a custom page */
076        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
077            @Override
078            public IRequestHandler onException(RequestCycle cycle, Exception e) {
079                return new RenderPageRequestHandler(new PageProvider(new ExceptionErrorPage(e)));
080            }
081        });
082
083        // Set the login context on each thread
084        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
085            @Override
086            public void onBeginRequest(RequestCycle cycle) {
087                HttpServletRequest webRequest = (HttpServletRequest) cycle.getRequest().getContainerRequest();
088                HttpSession session = webRequest.getSession();
089
090                if (session != null && LoginContext.get(session.getId()) != null) {
091                    LoginContext.set(LoginContext.get(session.getId()), null);
092                } else {
093                    LoginContext.set(LoginContext.ANONYMOUS_CONTEXT, null);
094                }
095            }
096        });
097
098        mountPage(webPath + "/#{" + Website.PARAMETER_CURRENT_PAGE + "}/#{ptab}", Website.class);
099        mountPage("/login", getSignInPageClass());
100        mountPage("/logout", SignOutPage.class);
101        mountPage("/forbbiden", NotAuthorizedPage.class);
102
103    }
104
105
106    @Override
107    protected Class<? extends AbstractAuthenticatedWebSession> getWebSessionClass() {
108        return WebsiteSession.class;
109    }
110
111    @Override
112    public void restartResponseAtSignInPage() {
113        throw new RestartResponseAtInterceptPageException(getSignInPageClass());
114    }
115
116    @Override
117    protected Class<? extends WebPage> getSignInPageClass() {
118        return signInPageClass;
119    }
120
121    public void setSignInPageClass(Class<? extends WebPage> signInPageClass) {
122        this.signInPageClass = signInPageClass;
123    }
124
125    public static WebsiteApplication get() {
126        return (WebsiteApplication) Application.get();
127    }
128
129    public ORI getWebsiteOri() {
130        return websiteOri;
131    }
132
133    public void setWebsiteOri(ORI websiteOri) {
134        this.websiteOri = websiteOri;
135    }
136
137    public String getWebPath() {
138        return webPath;
139    }
140
141    public void setWebPath(String webPath) {
142        this.webPath = webPath;
143    }
144
145
146    public static String toAbsolutePath(String relativePagePath) {
147
148        HttpServletRequest request = (HttpServletRequest) RequestCycle.get().getRequest().getContainerRequest();
149        String serverUrl = SERVER_URL;
150        if (serverUrl == null) {
151            serverUrl = request.getScheme() + "://" + request.getServerName();
152            if (request.getServerPort() != 80) {
153                serverUrl = serverUrl + ":" + request.getServerPort();
154            }
155        }
156        return toAbsolutePath(relativePagePath, serverUrl);
157    }
158
159    public static String toAbsolutePath(String relativePagePath, String serverUrl) {
160
161        HttpServletRequest request = (HttpServletRequest) RequestCycle.get().getRequest().getContainerRequest();
162
163        if (relativePagePath.charAt(0) == '/') {
164            return serverUrl + relativePagePath;
165        }
166
167        String fullPath = RequestUtils.toAbsolutePath(request.getRequestURI(), relativePagePath);
168        return serverUrl + fullPath;
169    }
170
171    public static void inject(Object obj) {
172        if (obj != null) {
173            InjectorHolder.getInjector().inject(obj, obj.getClass());
174        }
175    }
176
177}
