043
044public class DataManager implements IDataManager {
045
046    private static final Logger LOGGER = LoggerFactory.getLogger(DataManager.class);
047
048    private IResourceManager resourceManager;
049
050
051    public DataManager() {
052        super();
053    }
054
055    @Override
056    public IDataStreams load(ORI dataURI) {
057
058        Project project = resourceManager.getProject(dataURI.getProjectUrl());
059        Data data = null;
060        boolean uncompress = false;
061
062        try {
063            data = resourceManager.load(Data.class, dataURI);
064        } catch (ResourceNotFoundException e) {
065        }
066
067        // Look for the gzip version
068        if (data == null) {
069
070            try {
071                data = resourceManager.load(Data.class, new ORI(dataURI.getProjectUrl(), dataURI.getPath() + ".gz"));
072            } catch (ResourceNotFoundException e) {
073                throw new ResourceNotFoundException(dataURI);
074            }
075
076            uncompress = true;
077        }
078
079        Loader loader = data.getLoader();
080        Progress progress = new Progress(Integer.toHexString(dataURI.hashCode()), "Retrive '" + dataURI.getPath() + "'");
081        progress.run();
082
083        Plugin plugin = project.getPlugin(loader.getPlugin());
084
085        // If there is no plugin to load the data
086        // check for the 'data-url' parameters directly
087        if (plugin == null) {
088            List<String> dataUrls = loader.getParameterList("data-url");
089            List<URL> urls = new ArrayList<URL>(dataUrls.size());
090            for (String dataUrl : dataUrls) {
091                try {
092                    URL url = new URL(dataUrl);
093                    urls.add(url);
094                } catch (MalformedURLException e) {
095                    progress.error("Malformed URL " + dataUrl);
096                    progress.fail();
097                    return new EmptyDataStreams(progress);
098                }
099            }
100
101            progress.done();
102            return new UrlDataStreams(progress, urls, uncompress);
103        }
104
105
106        IDataLoader dataLoader = resourceManager.getLoader(IDataLoader.class, plugin, data.getLoader());
107
108        //TODO use a IDataStore as a cache
109        //TODO run loaders asynchronously
110
111        Callable<IDataStreams> callable = dataLoader.newCallable(progress, plugin, data);
112
113        IDataStreams dataStreams = null;
114        try {
115            dataStreams = callable.call();
116        } catch (Exception e) {
117            LOGGER.error("Error loading '" + dataURI.toString() + "'", e);
118            progress.error(e.getMessage());
119            progress.fail();
120            return new EmptyDataStreams(progress);
121        }
122
123        return dataStreams;
124    }
125
126
127    public IResourceManager getResourceManager() {
128        return resourceManager;
129    }
130
131    public void setResourceManager(IResourceManager resourceManager) {
132        this.resourceManager = resourceManager;
133    }
134
135    @Override
136    public long size(ORI dataURI) {
137
138        Project project = resourceManager.getProject(dataURI.getProjectUrl());
139        Data data = resourceManager.load(Data.class, dataURI);
140
141        Loader loader = data.getLoader();
142
143        Plugin plugin = project.getPlugin(loader.getPlugin());
144
145        // If there is no plugin to load the data
146        // check for the 'data-url' parameters directly
147        if (plugin == null) {
148            List<String> dataUrls = loader.getParameterList("data-url");
149            List<URL> urls = new ArrayList<URL>(dataUrls.size());
150            for (String dataUrl : dataUrls) {
151                try {
152                    URL url = new URL(dataUrl);
153                    urls.add(url);
154                } catch (MalformedURLException e) {
155                    LOGGER.error("Malformed URL '" + dataUrl + "'", e);
156                }
157            }
158
159            long size = 0;
160
161            for (URL url : urls) {
162
163                long oneSize = getContentLength(url);
164                if (oneSize == -1) {
165                    return -1;
166                }
167
168                size += oneSize;
169            }
170
171            return size;
172        }
173
174        IDataLoader dataLoader = resourceManager.getLoader(IDataLoader.class, plugin, data.getLoader());
175
176        return dataLoader.size();
177    }
178
179    private long getContentLength(URL url) {
180        File file = null;
181        try {
182            file = new File(url.toURI());
183        } catch (URISyntaxException e) {
184            return -1;
185        }
186        return file.length();
187    }
188
189    @Override
190    public String getMount() {
191        return "ds";
192    }
