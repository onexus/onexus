018package org.onexus.resource.manager.internal;
019
020import org.apache.commons.io.filefilter.FileFilterUtils;
021import org.apache.commons.io.monitor.FileAlterationListenerAdaptor;
022import org.apache.commons.io.monitor.FileAlterationMonitor;
023import org.apache.commons.io.monitor.FileAlterationObserver;
024import org.onexus.resource.api.IResourceListener;
025import org.onexus.resource.api.IResourceSerializer;
026import org.onexus.resource.api.Project;
027import org.onexus.resource.manager.internal.providers.AbstractProjectProvider;
028import org.onexus.resource.manager.internal.providers.ProjectProviderFactory;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import java.io.File;
033import java.io.FileInputStream;
034import java.io.FileNotFoundException;
035import java.io.FileOutputStream;
036import java.io.IOException;
037import java.util.ArrayList;
038import java.util.Collection;
039import java.util.List;
040import java.util.Map;
041import java.util.Properties;
042import java.util.concurrent.ConcurrentHashMap;
043
044public class ProjectsContainer {
045
046    private static final Logger LOGGER = LoggerFactory.getLogger(ProjectsContainer.class);
047
048    public static final String ONEXUS_FOLDER = System.getProperty("user.home") + File.separator + ".onexus";
049    public static final String ONEXUS_PROJECTS_SETTINGS = "projects.ini";
050    public static final String ONEXUS_PROJECTS_FOLDER = ONEXUS_FOLDER + File.separator + "projects";
051
052    private File propertiesFile;
053    private FileAlterationMonitor monitor;
054
055    private Properties properties;
056    private ProjectProviderFactory providerFactory;
057    private Map<String, AbstractProjectProvider> providers;
058
059    // Listeners
060    private List<IResourceListener> listeners = new ArrayList<IResourceListener>();
061
062    public ProjectsContainer(IResourceSerializer serializer, PluginLoader pluginLoader) {
063        super();
064
065        File onexusFolder = new File(ONEXUS_FOLDER);
066        File projectsFolder = new File(ONEXUS_PROJECTS_FOLDER);
067        propertiesFile = new File(onexusFolder, ONEXUS_PROJECTS_SETTINGS);
068        monitor = new FileAlterationMonitor(2000);
069
070        if (!onexusFolder.exists()) {
071            onexusFolder.mkdir();
072        }
073
074        if (!projectsFolder.exists()) {
075            projectsFolder.mkdir();
076        }
077
078        try {
079            if (!propertiesFile.exists()) {
080                propertiesFile.createNewFile();
081            }
082
083            this.providerFactory = new ProjectProviderFactory(serializer, pluginLoader, monitor, listeners);
084            this.providers = new ConcurrentHashMap<String, AbstractProjectProvider>();
085
086            init();
087        } catch (IOException e) {
088            throw new RuntimeException("Loading projects file '" + ONEXUS_PROJECTS_SETTINGS + "'", e);
089        }
090
091
092        FileAlterationObserver observer = new FileAlterationObserver(onexusFolder, FileFilterUtils.nameFileFilter(ONEXUS_PROJECTS_SETTINGS));
093        observer.addListener(new FileAlterationListenerAdaptor() {
094            @Override
095            public void onFileChange(File file) {
096                try {
097                    ProjectsContainer.this.init();
098                } catch (IOException e) {
099                    LOGGER.error("Loading projects file '" + ONEXUS_PROJECTS_SETTINGS + "'", e);
100                }
101            }
102        });
103
104        monitor.addObserver(observer);
105        try {
106            monitor.start();
107        } catch (Exception e) {
108            LOGGER.error("On start projects file monitor", e);
109        }
110
111    }
112
113    public void destroy() {
114        try {
115            monitor.stop();
116
117
118        } catch (Exception e) {
119            LOGGER.error("On stop projects file monitor", e);
120        }
121    }
122
123    private void init() throws IOException {
124        this.properties = new Properties();
125
126        properties.load(new FileInputStream(propertiesFile));
127
128        for (String projectUrl : getProjectUrls()) {
129
130            String projectPath = null;
131            String projectName = null;
132            try {
133
134                String projectProperty[] = properties.getProperty(projectUrl).split(",");
135                projectPath = projectProperty[0];
136                projectName = projectProperty.length == 2 ? projectProperty[1] : Integer.toHexString(projectUrl.hashCode());
137
138                File projectFolder = new File(projectPath);
139
140                AbstractProjectProvider previousProvider = providers.get(projectUrl);
141
142                if (previousProvider != null
143                        && previousProvider.getProjectFolder().equals(projectFolder)
144                        && previousProvider.getProject().getName().equals(projectName)
145                        ) {
146
147                    // This project is already registered, skip it
148                    continue;
149                }
150
151                AbstractProjectProvider provider = providerFactory.newProjectProvider(projectName, projectUrl, projectFolder);
152                providers.put(projectUrl, provider);
153
154                if (previousProvider == null) {
155                    onProjectCreate(provider.getProject());
156                } else {
157                    onProjectChange(provider.getProject());
158                }
159            } catch (Exception e) {
160                LOGGER.error("Loading project '" + projectUrl + "' named '" + projectName + "' at " + projectPath, e);
161            }
162        }
163
164        // Remove deleted projects
165        List<String> deletedProjects = new ArrayList<String>(providers.keySet());
166        deletedProjects.removeAll(getProjectUrls());
167        for (String deletedProject : deletedProjects) {
168            onProjectDelete(providers.get(deletedProject).getProject());
169            providers.remove(deletedProject);
170        }
171    }
172
173    public Collection<String> getProjectUrls() {
174        return properties.stringPropertyNames();
175    }
176
177    public AbstractProjectProvider getProjectProvider(String projectUri) {
178        return providers.get(projectUri);
179    }
180
181    public AbstractProjectProvider importProject(String projectName, String projectUri) {
182
183        File defaultProjectFolder = newProjectFolder(projectUri);
184
185        AbstractProjectProvider provider = providerFactory.newProjectProvider(projectName, projectUri, defaultProjectFolder);
186
187        if (provider != null) {
188            providers.put(projectUri, provider);
189            properties.setProperty(projectUri, provider.getProjectFolder().getAbsolutePath() + "," + projectName);
190
191            try {
192                properties.store(new FileOutputStream(new File(new File(ONEXUS_FOLDER), ONEXUS_PROJECTS_SETTINGS)), null);
193            } catch (FileNotFoundException e) {
194                LOGGER.error(e.getMessage());
195            } catch (IOException e) {
196                LOGGER.error(e.getMessage());
197            }
198        }
199
200        onProjectCreate(provider.getProject());
201        return provider;
202    }
203
204    private File newProjectFolder(String projectUri) {
205        return new File(new File(ONEXUS_PROJECTS_FOLDER), Integer.toHexString(projectUri.hashCode()));
206    }
207
208    void addResourceListener(IResourceListener resourceListener) {
209        listeners.add(resourceListener);
210    }
211
212    private void onProjectCreate(final Project project) {
213        LOGGER.info("Project '" + project.getName() + "' created.");
214
215        new Thread(new Runnable() {
216            @Override
217            public void run() {
218                for (IResourceListener listener : listeners) {
219                    listener.onProjectCreate(project);
220                }
221            }
222        }).start();
223
224    }
225
226    private void onProjectChange(final Project project) {
227        LOGGER.info("Project '" + project.getName() + "' changed.");
228
229        new Thread(new Runnable() {
230            @Override
231            public void run() {
232                for (IResourceListener listener : listeners) {
233                    listener.onProjectChange(project);
234                }
235            }
236        }).start();
237    }
238
239    private void onProjectDelete(final Project project) {
240        LOGGER.info("Project '" + project.getName() + "' deleted.");
241
242        for (IResourceListener listener : listeners) {
243            listener.onProjectDelete(project);
244        }
245    }
246
247    public void bundleCreated(long bundleId) {
248
249        for (AbstractProjectProvider provider : providers.values()) {
250
251            if (provider.dependsOnBundle(bundleId)) {
252                provider.loadProject();
253                onProjectChange(provider.getProject());
254            }
255
256        }
257
258    }
259
260    public void bundleUninstalled(long bundleId) {
261
262    }
