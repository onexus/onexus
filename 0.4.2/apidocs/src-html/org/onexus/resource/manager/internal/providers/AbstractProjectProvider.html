071    public AbstractProjectProvider(String projectName, String projectUrl, File projectFolder, FileAlterationMonitor monitor, List<IResourceListener> listeners) {
072        super();
073        this.projectName = projectName;
074        this.projectUrl = projectUrl;
075        this.projectFolder = projectFolder;
076        this.listeners = listeners;
077
078
079        // Don't watch hidden folders and files
080        IOFileFilter notHiddenDirectoryFilter = FileFilterUtils.notFileFilter(
081                FileFilterUtils.or(
082                        FileFilterUtils.and(
083                                FileFilterUtils.directoryFileFilter(),
084                                HiddenFileFilter.HIDDEN
085                        ),
086                        HiddenFileFilter.HIDDEN
087                )
088        );
089
090        String MONITOR_PROJECTS = System.getProperty("onexus.monitor.projects");
091        if (MONITOR_PROJECTS == null || Boolean.valueOf(MONITOR_PROJECTS)) {
092            observer = new FileAlterationObserver(projectFolder, notHiddenDirectoryFilter);
093            observer.addListener(new FileAlterationListenerAdaptor() {
094
095                @Override
096                public void onDirectoryCreate(File directory) {
097                    LOGGER.info("Creating folder '" + directory.getName() + "'");
098                    onResourceCreate(loadFile(directory));
099                }
100
101                @Override
102                public void onDirectoryDelete(File directory) {
103                    LOGGER.info("Deleting folder '" + directory.getName() + "'");
104                    ORI resourceOri = convertFileToORI(directory);
105                    onResourceDelete(resources.remove(resourceOri));
106                }
107
108                @Override
109                public void onFileChange(File file) {
110                    LOGGER.info("Reloading file '" + file.getName() + "'");
111                    onResourceChange(loadFile(file));
112                }
113
114                @Override
115                public void onFileCreate(File file) {
116                    LOGGER.info("Creating file '" + file.getName() + "'");
117                    onResourceCreate(loadFile(file));
118                }
119
120                @Override
121                public void onFileDelete(File file) {
122                    LOGGER.info("Deleting file '" + file.getName() + "'");
123                    ORI resourceOri = convertFileToORI(file);
124                    onResourceDelete(resources.remove(resourceOri));
125                }
126
127            });
128
129            monitor.addObserver(observer);
130        }
131
132        // Initialize template engine
133        try {
134            freemarkerConfig.setDirectoryForTemplateLoading(projectFolder);
135        } catch (IOException e) {
136            LOGGER.error("At template engine configuration. Project folder: '" + projectFolder + "'", e);
137            throw new RuntimeException(e);
138        }
139
140        freemarkerConfig.setObjectWrapper(new DefaultObjectWrapper());
141        freemarkerConfig.setDefaultEncoding("UTF-8");
142        freemarkerConfig.setTemplateExceptionHandler(TemplateExceptionHandler.HTML_DEBUG_HANDLER);
143        freemarkerConfig.setIncompatibleImprovements(new Version(2, 3, 20));  // FreeMarker 2.3.20
144
145    }

163    public synchronized void loadProject() {
164        File projectOnx = new File(projectFolder, ONEXUS_PROJECT_FILE);
165
166        if (!projectOnx.exists()) {
167            throw new InvalidParameterException("No Onexus project in " + projectFolder.getAbsolutePath());
168        }
169
170        this.project = (Project) loadResource(projectOnx);
171        this.project.setName(projectName);
172
173        this.bundleDependencies.clear();
174
175        if (project.getPlugins() != null) {
176            for (Plugin plugin : project.getPlugins()) {
177
178                //TODO Remove this property
179                if (plugin.getParameter("location") == null && plugin.getParameters() != null) {
180                    plugin.getParameters().add(new Parameter("location", projectFolder.getAbsolutePath()));
181                }
182
183                try {
184                    long bundleId = pluginLoader.load(plugin);
185                    bundleDependencies.add(bundleId);
186                } catch (InvalidParameterException e) {
187                    LOGGER.error(e.getMessage());
188                }
189            }
190        }
191
192        if (project.getAlias() != null && !project.getAlias().isEmpty()) {
193
194            this.projectAlias = new Properties();
195            try {
196                this.projectAlias.load(new StringReader(project.getAlias()));
197            } catch (IOException e) {
198                LOGGER.error("Malformed project alias", e);
199                this.projectAlias = null;
200            }
201        }
202
203        this.resources = null;
204    }
205
206    private synchronized void loadResources() {
207
208        this.resources = new ConcurrentHashMap<ORI, Resource>();
209
210        Collection<File> files = addFilesRecursive(new ArrayList<File>(), projectFolder);
211
212        for (File file : files) {
213
214            // Skip project file
215            if (ONEXUS_PROJECT_FILE.equals(file.getName())) {
216                continue;
217            }
218
219            Resource resource = loadResource(file);
220
221            if (resource != null) {
222
223                if (resources == null) {
224                    return;
225                }
226
227                resources.put(resource.getORI(), resource);
228            }
229        }
230    }
231
232    public Resource getResource(ORI resourceUri) {
233
234        if (resourceUri.getPath() == null && projectUrl.equals(resourceUri.getProjectUrl())) {
235            return getProject();
236        }
237
238        if (resources == null) {
239            loadResources();
240        }
241
242        if (!resources.containsKey(resourceUri)) {
243            throw new ResourceNotFoundException(resourceUri);
244        }
245
246        return resources.get(resourceUri);
247
248    }

250    public <T extends Resource> List<T> getResourceChildren(IAuthorizationManager authorizationManager, Class<T> resourceType, ORI parentURI) {
251
252        if (resources == null) {
253            loadResources();
254        }
255
256        List<T> children = new ArrayList<T>();
257        if (resources != null) {
258            for (Resource resource : resources.values()) {
259                if (parentURI.isChild(resource.getORI())
260                        && resourceType.isAssignableFrom(resource.getClass())
261                        && authorizationManager.check(IAuthorizationManager.READ, resource.getORI())) {
262                    children.add((T) resource);
263                }
264            }
265        }
266
267        return children;
268    }

282    private Resource loadResource(File resourceFile) {
283
284        Resource resource;
285
286        if (ONEXUS_EXTENSION.equalsIgnoreCase(FilenameUtils.getExtension(resourceFile.getName()))) {
287
288            StringWriter out = null;
289
290            try {
291
292                String relativePath = projectFolder.toURI().relativize(resourceFile.toURI()).getPath();
293                Template resourceTemplate = freemarkerConfig.getTemplate(relativePath);
294
295                out = new StringWriter((int) resourceFile.length() / 4);
296                resourceTemplate.process(projectAlias, out);
297
298                InputStream input = new ByteArrayInputStream(out.toString().getBytes("UTF-8"));
299                resource = serializer.unserialize(Resource.class, input);
300
301                for (String include : getIncludes(resourceTemplate)) {
302                    addIncludeDependency(include, resourceFile);
303                }
304
305            } catch (FileNotFoundException e) {
306                resource = createErrorResource(resourceFile, "File '" + resourceFile.getPath() + "' not found.");
307            } catch (UnserializeException e) {
308
309                String msg = "Parsing file " + resourceFile.getPath() + " at line " + e.getLine() + " on " + e.getPath();
310
311                if (out != null) {
312                    String[] lines = out.toString().split(System.getProperty("line.separator"));
313
314                    int error = Integer.valueOf(e.getLine());
315                    int from = (error-1 > 0 ? error-1 : 0);
316                    int to = (error+1 < lines.length ? error+1 : lines.length);
317
318                    for (int i=from ; i <= to; i++) {
319                        msg = msg + "\n " + i + " >> " + lines[i];
320                    }
321                }
322
323                resource = createErrorResource(resourceFile, msg);
324            } catch (Exception e) {
325                resource = createErrorResource(resourceFile, e.getMessage());
326            }
327
328        } else {
329
330            if (resourceFile.isDirectory()) {
331                resource = new Folder();
332            } else {
333                resource = createDataResource(resourceFile);
334            }
335
336        }
337
338        if (resource == null) {
339            return null;
340        }
341
342        resource.setORI(convertFileToORI(resourceFile));
343        return resource;
344
345    }
346
347    private ORI convertFileToORI(File file) {
348
349        String projectPath = projectFolder.getAbsolutePath() + File.separator;
350        String filePath = file.getAbsolutePath();
351        String relativePath = filePath.replace(projectPath, "");
352
353        if (relativePath.equals(ONEXUS_PROJECT_FILE)) {
354            relativePath = null;
355        } else {
356            relativePath = relativePath.replace("." + ONEXUS_EXTENSION, "");
357        }
358
359        return new ORI(projectUrl, relativePath);
360
361    }

363    private Resource createErrorResource(File resourceFile, String msg) {
364        LOGGER.error(msg);
365        Resource errorResource = createDataResource(resourceFile);
366        errorResource.setDescription("ERROR: " + msg);
367        return errorResource;
368    }

370    private Resource createDataResource(File resourceFile) {
371        Data data = new Data();
372        Loader loader = new Loader();
373        loader.setParameters(new ArrayList<Parameter>());
374        loader.getParameters().add(new Parameter("data-url", resourceFile.toURI().toString()));
375        data.setLoader(loader);
376        return data;
377    }

379    private static Collection<File> addFilesRecursive(Collection<File> files, File parentFolder) {
380
381        if (parentFolder.isDirectory()) {
382            File[] inFiles = parentFolder.listFiles();
383            if (inFiles != null) {
384                for (File file : inFiles) {
385                    if (!file.isHidden()) {
386                        files.add(file);
387                        if (file.isDirectory()) {
388                            addFilesRecursive(files, file);
389                        }
390                    }
391                }
392            }
393        }
394
395        return files;
396    }

398    public void save(Resource resource) {
399
400        if (resource == null) {
401            return;
402        }
403
404        if (resource instanceof Project) {
405            throw new IllegalArgumentException("Cannot create a project '" + resource.getORI() + "' inside project '" + projectUrl + "'");
406        }
407
408        String resourcePath = resource.getORI().getPath();
409
410        File file;
411        if (resource instanceof Folder) {
412            file = new File(projectFolder, resourcePath);
413            file.mkdirs();
414
415            return;
416        }
417
418
419        file = new File(projectFolder, resourcePath + "." + ONEXUS_EXTENSION);
420
421        try {
422            if (!file.exists()) {
423                file.createNewFile();
424            }
425
426            FileOutputStream os = new FileOutputStream(file);
427            serializer.serialize(resource, os);
428            os.close();
429
430        } catch (IOException e) {
431            LOGGER.error("Saving resource '" + resource.getORI() + "' in file '" + file.getAbsolutePath() + "'", e);
432        }
433
434        observer.checkAndNotify();
435
436    }

458    protected void onResourceCreate(Resource resource) {
459
460        if (resource == null) {
461            return;
462        }
463
464        LOGGER.info("Resource '" + resource.getName() + "' created.");
465
466        for (IResourceListener listener : listeners) {
467            listener.onResourceCreate(resource);
468        }
469    }
470
471    protected void onResourceChange(Resource resource) {
472
473        if (resource == null) {
474            return;
475        }
476
477        LOGGER.info("Resource '" + resource.getName() + "' changed.");
478
479        for (IResourceListener listener : listeners) {
480            listener.onResourceChange(resource);
481        }
482
483        // If it's an include dependency reload parent resources
484        String resourcePath = resource.getORI().getPath().substring(1);
485        if (includeDependencies.containsKey(resourcePath)) {
486            freemarkerConfig.clearTemplateCache();
487            for (File parentResourceFile : includeDependencies.get(resourcePath)) {
488                onResourceChange(loadFile(parentResourceFile));
489            }
490        }
491
492    }
493
494    protected void onResourceDelete(Resource resource) {
495
496        if (resource == null) {
497            return;
498        }
499
500        LOGGER.info("Resource '" + resource.getName() + "' deleted.");
501
502        for (IResourceListener listener : listeners) {
503            listener.onResourceDelete(resource);
504        }
505    }

507    public boolean dependsOnBundle(long bundleId) {
508        return bundleDependencies.contains(Long.valueOf(bundleId));
509    }

511    private List<String> getIncludes(Template resourceTemplate) {
512
513        List<String> includes = new ArrayList<String>();
514        TemplateElement root = resourceTemplate.getRootTreeNode();
515        for (int i=0; i < root.getChildCount(); i++) {
516            TreeNode node = root.getChildAt(i);
517            if (node instanceof TemplateElement) {
518                String tag = ((TemplateElement) node).getCanonicalForm();
519                if (tag != null && tag.startsWith("<#include")) {
520                    String include = tag.replace("<#include \"", "").replace("\"/>", "");
521                    String templatePath = FilenameUtils.getFullPath(resourceTemplate.getName());
522                    includes.add(templatePath + include);
523                    try {
524                        Template includeTemplate = freemarkerConfig.getTemplate(templatePath + include);
525                        includes.addAll( getIncludes(includeTemplate));
526                    } catch (IOException e) {
527                        LOGGER.error(e.getMessage(), e);
528                    }
529                }
530            }
531        }
532
533        return includes;
534    }
535
536    private void addIncludeDependency(String templateName, File parentResourceFile) {
537        if (!includeDependencies.containsKey(templateName)) {
538            includeDependencies.put(templateName, new HashSet<File>());
539        }
540        includeDependencies.get(templateName).add(parentResourceFile);
541    }
542
543    private Resource loadFile(File file) {
544
545        // Skip project file
546        if (ONEXUS_PROJECT_FILE.equals(file.getName())) {
547            return null;
548        }
549
550        Resource resource = loadResource(file);
551
552        if (resource != null) {
553            resources.put(resource.getORI(), resource);
554        }
555
556        return resource;
557    }
