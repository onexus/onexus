018package org.onexus.resource.profile.internal;
019
020import org.apache.commons.lang3.StringUtils;
021import org.onexus.resource.api.IProfileManager;
022import org.onexus.resource.api.session.LoginContext;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026import java.io.File;
027import java.io.FileReader;
028import java.io.FileWriter;
029import java.io.IOException;
030import java.util.Collection;
031import java.util.Collections;
032import java.util.Properties;
033
034public class ProfileManager implements IProfileManager {
035
036    private static final Logger LOGGER = LoggerFactory.getLogger(ProfileManager.class);
037
038    public static final String ONEXUS_FOLDER = System.getProperty("user.home") + File.separator + ".onexus";
039    public static final String ONEXUS_PROFILES_FOLDER = ONEXUS_FOLDER + File.separator + "profiles";
040
041
042    @Override
043    public Collection<String> getKeys() {
044        Properties properties = null;
045        try {
046            properties = loadProperties(getUserName());
047        } catch (IOException e) {
048            LOGGER.error("Loading '" + getUserName() + "' profile properties file.", e);
049            return Collections.EMPTY_LIST;
050        }
051        return properties.stringPropertyNames();
052    }
053
054    @Override
055    public String getValue(String key) {
056        Properties properties = null;
057        try {
058            properties = loadProperties(getUserName());
059        } catch (IOException e) {
060            LOGGER.error("Loading '" + getUserName() + "' profile properties file.", e);
061            return null;
062        }
063        return properties.getProperty(key);
064    }
065
066    @Override
067    public String[] getValueArray(String key) {
068        return splitValue(getValue(key));
069    }
070
071    @Override
072    public void putValue(String key, String value) {
073        String userName = getUserName();
074        Properties properties = null;
075
076        try {
077            properties = loadProperties(userName);
078        } catch (IOException e) {
079            LOGGER.error("Loading '" + getUserName() + "' profile properties file.", e);
080            return;
081        }
082
083        properties.put(key, value);
084        try {
085            storeProperties(userName, properties);
086        } catch (IOException e) {
087            LOGGER.error("Storing '" + getUserName() + "' profile properties file.", e);
088        }
089    }
090
091    @Override
092    public void putValueArray(String key, String[] values) {
093        putValue(key, joinValues(values));
094    }
095
096    public void load() {
097
098    }
099
100    private String getUserName() {
101        LoginContext ctx = LoginContext.get();
102        return ctx.getUserName();
103    }
104
105    private Properties loadProperties(String userName) throws IOException {
106
107        Properties properties = new Properties();
108
109        File file = new File(ONEXUS_PROFILES_FOLDER + File.separator + userName);
110
111        //TODO
112        //if (!file.exists()) {
113        //}
114
115        properties.load(new FileReader(file));
116
117        return properties;
118    }
119
120    private void storeProperties(String userName, Properties properties) throws IOException {
121        properties.store(new FileWriter(ONEXUS_PROFILES_FOLDER + File.separator + userName), "");
122    }
123
124    private String[] splitValue(String value) {
125        return StringUtils.split(value, '\t');
126    }
127
128    private String joinValues(String[] values) {
129        return StringUtils.join(values, '\t');
130    }
131
132}
