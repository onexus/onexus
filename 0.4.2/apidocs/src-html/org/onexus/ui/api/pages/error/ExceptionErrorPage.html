018package org.onexus.ui.api.pages.error;
019
020import org.apache.wicket.markup.html.basic.MultiLineLabel;
021import org.apache.wicket.markup.html.pages.AbstractErrorPage;
022import org.apache.wicket.request.http.WebResponse;
023import org.apache.wicket.util.lang.Generics;
024import org.onexus.resource.api.exceptions.OnexusException;
025import org.onexus.ui.api.pages.theme.DefaultTheme;
026
027import javax.servlet.http.HttpServletResponse;
028import java.util.List;
029
030
031public class ExceptionErrorPage extends AbstractErrorPage {
032    private static final long serialVersionUID = 1L;
033
034    /**
035     * Keep a reference to the root cause. WicketTester will use it
036     */
037    private final transient Throwable throwable;
038
039    /**
040     * Constructor.
041     *
042     * @param throwable The exception to show
043     */
044    public ExceptionErrorPage(final Throwable throwable) {
045        this.throwable = throwable;
046
047        add(new DefaultTheme());
048
049        // Add exception label
050        add(new MultiLineLabel("exception", getErrorMessage(throwable)));
051
052        add(new MultiLineLabel("stacktrace", getStackTrace(throwable)));
053
054    }
055
056    /**
057     * Converts a Throwable to a string.
058     *
059     * @param throwable The throwable
060     * @return The string
061     */
062    public String getErrorMessage(final Throwable throwable) {
063        if (throwable != null) {
064            StringBuilder sb = new StringBuilder(256);
065
066            // first print the last cause or the first onexus exception
067            Throwable cause = null;
068            List<Throwable> al = convertToList(throwable);
069            for (Throwable e : al) {
070                if (OnexusException.class.isAssignableFrom(e.getClass())) {
071                    cause = e;
072                    break;
073                }
074            }
075
076            if (cause == null) {
077                int length = al.size() - 1;
078                cause = al.get(length);
079            }
080
081            sb.append(cause.getMessage());
082            return sb.toString();
083        } else {
084            return "[Unknown]";
085        }
086    }
087
088    /**
089     * Converts a Throwable to a string.
090     *
091     * @param throwable The throwable
092     * @return The string
093     */
094    public String getStackTrace(final Throwable throwable) {
095        if (throwable != null) {
096            List<Throwable> al = convertToList(throwable);
097
098            StringBuilder sb = new StringBuilder(256);
099
100            // first print the last cause
101            int length = al.size() - 1;
102            Throwable cause = al.get(length);
103
104            outputThrowable(cause, sb, false);
105
106            if (length > 0) {
107                sb.append("\n\nComplete stack:\n\n");
108                for (int i = 0; i < length; i++) {
109                    outputThrowable(al.get(i), sb, true);
110                    sb.append("\n");
111                }
112            }
113            return sb.toString();
114        } else {
115            return "<Null Throwable>";
116        }
117    }
118
119    /**
120     * @param throwable
121     * @return xxx
122     */
123    private List<Throwable> convertToList(final Throwable throwable) {
124        List<Throwable> al = Generics.newArrayList();
125        Throwable cause = throwable;
126        al.add(cause);
127        while (cause.getCause()!=null && cause!=cause.getCause()) {
128            cause = cause.getCause();
129            al.add(cause);
130        }
131        return al;
132    }
133
134    /**
135     * Outputs the throwable and its stacktrace to the stringbuffer. If stopAtWicketSerlvet is true
136     * then the output will stop when the org.apache.wicket servlet is reached. sun.reflect.
137     * packages are filtered out.
138     *
139     * @param cause
140     * @param sb
141     * @param stopAtWicketServlet
142     */
143    private void outputThrowable(Throwable cause, StringBuilder sb, boolean stopAtWicketServlet) {
144        sb.append(cause);
145        sb.append("\n");
146        StackTraceElement[] trace = cause.getStackTrace();
147        for (int i = 0; i < trace.length; i++) {
148            String traceString = trace[i].toString();
149            if (!(traceString.startsWith("sun.reflect.") && i > 1)) {
150                sb.append("     at ");
151                sb.append(traceString);
152                sb.append("\n");
153                if (stopAtWicketServlet &&
154                        (traceString.startsWith("org.apache.wicket.protocol.http.WicketServlet") || traceString.startsWith("org.apache.wicket.protocol.http.WicketFilter"))) {
155                    return;
156                }
157            }
158        }
159    }
160
161    @Override
162    protected void setHeaders(final WebResponse response) {
163        response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
164    }
165
166    /**
167     * Get access to the exception
168     *
169     * @return The exception
170     */
171    public Throwable getThrowable() {
172        return throwable;
173    }
174}
