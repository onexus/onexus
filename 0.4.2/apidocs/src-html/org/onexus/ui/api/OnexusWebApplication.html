018package org.onexus.ui.api;
019
020import org.apache.wicket.Application;
021import org.apache.wicket.Page;
022import org.apache.wicket.RuntimeConfigurationType;
023import org.apache.wicket.authroles.authentication.AbstractAuthenticatedWebSession;
024import org.apache.wicket.authroles.authentication.AuthenticatedWebApplication;
025import org.apache.wicket.core.request.handler.PageProvider;
026import org.apache.wicket.core.request.handler.RenderPageRequestHandler;
027import org.apache.wicket.markup.html.WebPage;
028import org.apache.wicket.request.IRequestHandler;
029import org.apache.wicket.request.cycle.AbstractRequestCycleListener;
030import org.apache.wicket.request.cycle.RequestCycle;
031import org.onexus.resource.api.session.LoginContext;
032import org.onexus.ui.api.pages.SignOutPage;
033import org.onexus.ui.api.pages.error.ExceptionErrorPage;
034import org.onexus.ui.api.pages.resource.ResourcesPage;
035import org.onexus.ui.authentication.jaas.JaasSignInPage;
036import org.onexus.ui.authentication.persona.PersonaSignInPage;
037import org.ops4j.pax.wicket.api.InjectorHolder;
038
039import javax.servlet.http.HttpServletRequest;
040import javax.servlet.http.HttpSession;
041
042public class OnexusWebApplication extends AuthenticatedWebApplication {
043
044    private static final String SERVER_URL = System.getenv("ONEXUS_SERVER_URL");
045
046    public Class<? extends Page> getHomePage() {
047        return ResourcesPage.class;
048    }
049
050    @Override
051    public RuntimeConfigurationType getConfigurationType() {
052        return RuntimeConfigurationType.DEPLOYMENT;
053    }
054
055    @Override
056    protected void init() {
057        super.init();
058
059        // Authentication
060        getApplicationSettings().setAccessDeniedPage(getSignInPageClass());
061
062        /* In case of unhandled exception redirect it to a custom page */
063        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
064            @Override
065            public IRequestHandler onException(RequestCycle cycle, Exception e) {
066                return new RenderPageRequestHandler(new PageProvider(new ExceptionErrorPage(e)));
067            }
068        });
069
070        // Set the login context on each thread
071        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
072            @Override
073            public void onBeginRequest(RequestCycle cycle) {
074                HttpServletRequest webRequest = (HttpServletRequest) cycle.getRequest().getContainerRequest();
075                HttpSession session = webRequest.getSession();
076
077                if (session != null && LoginContext.get(session.getId()) != null) {
078                    LoginContext.set(LoginContext.get(session.getId()), null);
079                } else {
080                    LoginContext.set(LoginContext.ANONYMOUS_CONTEXT, null);
081                }
082            }
083        });
084
085        // Mount pages
086        mountPage("/login", getSignInPageClass());
087        mountPage("/logout", SignOutPage.class);
088        mountPage("/admin", ResourcesPage.class);
089
090    }
091
092    public boolean usePersonSignIn() {
093        return Boolean.parseBoolean(System.getProperty("org.onexus.ui.authentication.persona", "false"));
094    }
095
096    @Override
097    protected Class<? extends WebPage> getSignInPageClass() {
098        if (usePersonSignIn()) {
099            return PersonaSignInPage.class;
100        }
101
102        return JaasSignInPage.class;
103    }
104
105    @Override
106    protected Class<? extends AbstractAuthenticatedWebSession> getWebSessionClass() {
107        return OnexusWebSession.class;
108    }
109
110    public static OnexusWebApplication get() {
111        return (OnexusWebApplication) Application.get();
112    }
113
114    public static String getRequestUrl() {
115
116        HttpServletRequest request = (HttpServletRequest) RequestCycle.get().getRequest().getContainerRequest();
117        String requestUrl = request.getRequestURL().toString();
118
119        if (SERVER_URL != null) {
120            String baseUrl = RequestCycle.get().getUrlRenderer().getBaseUrl().toString();
121            return SERVER_URL + '/' + baseUrl;
122        }
123
124        return requestUrl;
125    }
126
127    public static void inject(Object obj) {
128        if (obj != null) {
129            InjectorHolder.getInjector().inject(obj, obj.getClass());
130        }
131    }
132
133
134}
