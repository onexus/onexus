018package org.onexus.ui.workspace.internal.wizards.folder;
019
020import org.apache.wicket.extensions.wizard.WizardModel;
021import org.apache.wicket.extensions.wizard.WizardStep;
022import org.apache.wicket.markup.html.form.RadioChoice;
023import org.apache.wicket.model.IModel;
024import org.apache.wicket.model.PropertyModel;
025import org.onexus.collection.api.Collection;
026import org.onexus.collection.api.ICollectionManager;
027import org.onexus.collection.api.query.Query;
028import org.onexus.resource.api.IResourceManager;
029import org.onexus.resource.api.ORI;
030import org.onexus.resource.api.Resource;
031import org.onexus.ui.api.progressbar.ProgressBar;
032import org.onexus.ui.api.wizards.AbstractWizard;
033
034import javax.inject.Inject;
035import javax.inject.Named;
036import java.util.Arrays;
037import java.util.List;
038
039public class ManageCollectionsWizard extends AbstractWizard {
040
041    // Commands
042    private static final String LOAD = "Load all the collections";
043    private static final String UNLOAD = "Unload all the collections";
044    private static final List<String> COMMANDS = Arrays.asList(new String[]{LOAD, UNLOAD});
045
046    private String selected;
047    private ORI resourceURI;
048
049    @Inject
050    private IResourceManager resourceManager;
051
052    @Inject
053    private ICollectionManager collectionManager;
054
055    public ManageCollectionsWizard(String id, IModel<? extends Resource> resourceModel) {
056        super(id);
057
058        this.resourceURI = resourceModel.getObject().getORI();
059
060        WizardModel model = new WizardModel();
061        model.add(new BasicOptions());
062
063        init(model);
064    }
065
066    @Override
067    public void onFinish() {
068
069        super.onFinish();
070
071        if (selected.equals(UNLOAD)) {
072            List<Collection> collections = resourceManager.loadChildren(Collection.class, resourceURI);
073            for (Collection collection : collections) {
074                collectionManager.unload(collection.getORI());
075            }
076            return;
077        }
078
079        if (selected.equals(LOAD)) {
080            List<Collection> collections = resourceManager.loadChildren(Collection.class, resourceURI);
081            for (Collection collection : collections) {
082                collectionManager.unload(collection.getORI());
083                Query emptyQuery = new Query();
084                emptyQuery.addDefine("c", collection.getORI());
085                emptyQuery.setFrom("c");
086                emptyQuery.setOffset(0);
087                emptyQuery.setCount(0);
088                ProgressBar.show(collectionManager.load(emptyQuery));
089            }
090            return;
091        }
092
093    }
094
095    public String getSelected() {
096        return selected;
097    }
098
099    public void setSelected(String selected) {
100        this.selected = selected;
101    }
102
103    private final class BasicOptions extends WizardStep {
104
105
106        public BasicOptions() {
107            super("Manage collections", "Basic collection tools");
108
109            RadioChoice<String> commandOptions = new RadioChoice<String>("commands", new PropertyModel<String>(ManageCollectionsWizard.this, "selected"), COMMANDS);
110            add(commandOptions);
111
112        }
113
114    }
115
116
117}
