018package org.onexus.ui.workspace.internal.viewers.tree;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.event.Broadcast;
023import org.apache.wicket.extensions.markup.html.repeater.tree.ITreeProvider;
024import org.apache.wicket.extensions.markup.html.repeater.tree.NestedTree;
025import org.apache.wicket.extensions.markup.html.repeater.tree.content.Folder;
026import org.apache.wicket.extensions.markup.html.repeater.tree.theme.WindowsTheme;
027import org.apache.wicket.extensions.markup.html.repeater.util.ProviderSubset;
028import org.apache.wicket.markup.head.CssHeaderItem;
029import org.apache.wicket.markup.head.IHeaderResponse;
030import org.apache.wicket.model.AbstractReadOnlyModel;
031import org.apache.wicket.model.IDetachable;
032import org.apache.wicket.model.IModel;
033import org.apache.wicket.model.PropertyModel;
034import org.apache.wicket.request.resource.CssResourceReference;
035import org.apache.wicket.request.resource.ResourceReference;
036import org.onexus.resource.api.Project;
037import org.onexus.resource.api.Resource;
038import org.onexus.ui.api.events.EventResourceSelect;
039
040import java.util.Set;
041
042public class ProjectTree extends NestedTree<Resource> {
043
044    private IModel<Resource> selected;
045
046    private Set<Resource> state;
047
048    private static final ResourceReference CSS = new CssResourceReference(ProjectTree.class, "css/tree.css");
049
050    public ProjectTree(String id, ITreeProvider<Resource> treeProvider, IModel<? extends Resource> selectedResource) {
051        super(id, treeProvider);
052
053        this.state = new InverseSet<Resource>(new ProviderSubset<Resource>(treeProvider));
054        this.selected = (IModel<Resource>) selectedResource;
055
056        setOutputMarkupId(true);
057        setModel(newStateModel());
058        add(new WindowsTheme());
059
060    }
061
062    protected boolean isSelected(Resource resource) {
063
064        if (resource != null && selected.getObject() != null) {
065            return resource.getORI().equals(selected.getObject().getORI());
066        }
067
068        return false;
069    }
070
071
072    protected void select(Resource resource, final AjaxRequestTarget target) {
073
074        // Update unselected node
075        updateNode(selected.getObject(), target);
076
077        // Select new node
078        selected.setObject(resource);
079
080        // Update new selection
081        updateNode(resource, target);
082
083        send(getPage(), Broadcast.BREADTH, EventResourceSelect.EVENT);
084    }
085
086    private IModel<Set<Resource>> newStateModel() {
087        return new AbstractReadOnlyModel<Set<Resource>>() {
088            @Override
089            public Set<Resource> getObject() {
090                return state;
091            }
092
093            /**
094             * Super class doesn't detach - would be nice though.
095             */
096            @Override
097            public void detach() {
098                ((IDetachable) state).detach();
099            }
100        };
101    }
102
103    @Override
104    protected void onDetach() {
105        if (selected != null) {
106            selected.detach();
107        }
108        super.onDetach();
109    }
110
111    @Override
112    protected Component newContentComponent(String id, IModel<Resource> model) {
113        return new Folder<Resource>(id, this, model) {
114
115            @Override
116            protected boolean isClickable() {
117                return true;
118            }
119
120
121            @Override
122            protected void onClick(AjaxRequestTarget target) {
123                ProjectTree.this.select(getModelObject(), target);
124            }
125
126
127            @Override
128            protected boolean isSelected() {
129                return ProjectTree.this.isSelected(getModelObject());
130            }
131
132            @Override
133            protected IModel<?> newLabelModel(IModel<Resource> resourceIModel) {
134                return new PropertyModel<String>(resourceIModel, "name");
135            }
136
137            @Override
138            protected String getStyleClass() {
139
140                Resource t = getModelObject();
141
142                String styleClass;
143                if (getProvider().hasChildren(t)) {
144                    if (getState(t) == State.EXPANDED) {
145                        if (t instanceof Project) {
146                            styleClass = "tree-project-open";
147                        } else {
148                            styleClass = getOpenStyleClass();
149                        }
150                    } else {
151                        if (t instanceof Project) {
152                            styleClass = "tree-project-closed";
153                        } else {
154                            styleClass = getClosedStyleClass();
155                        }
156                    }
157                } else {
158                    styleClass = getOtherStyleClass(t);
159                }
160
161                if (isSelected()) {
162                    styleClass += " " + getSelectedStyleClass();
163                }
164
165                return styleClass;
166            }
167
168            @Override
169            protected String getOtherStyleClass(Resource resource) {
170                return "resource-" + resource.getClass().getSimpleName().toLowerCase();
171            }
172        };
173    }
174
175    @Override
176    public void renderHead(IHeaderResponse response) {
177        response.render(CssHeaderItem.forReference(CSS));
178    }
179
180
181}
