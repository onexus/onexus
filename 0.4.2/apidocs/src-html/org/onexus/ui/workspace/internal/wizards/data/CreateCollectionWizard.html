074    private static final int MAXIMUM_UNIQUE_VALUES = 1500;

086    private int nullNA = 0;

099    private void readData() throws IOException {
100
101        String separator = " ";
102        if (selected.equals(CSV)) {
103            separator = ",";
104        }
105
106        if (selected.equals(TSV)) {
107            separator = "\t";
108        }
109
110        Iterator<InputStream> streams = dataManager.load(sourceURI).iterator();
111
112        if (!streams.hasNext()) {
113            return;
114        }
115
116        BufferedReader fr = new BufferedReader(new InputStreamReader(streams.next()));
117
118        // Get headers
119        headers = fr.readLine().split(separator);
120
121        // Build values map
122        sampleData = new HashMap<String, Set<String>>();
123        for (String header : headers) {
124            sampleData.put(header, new HashSet<String>());
125        }
126        String line = fr.readLine();
127
128        for (int i = 0; i < MAXIMUM_LINES && line != null; i++) {
129            String values[] = line.split(separator);
130            for (int h = 0; h < headers.length && h < values.length; h++) {
131
132                if (values[h] == null || values[h].isEmpty()) {
133                    nullEmpty++;
134                } else if (values[h].equalsIgnoreCase("-")) {
135                    nullDash++;
136                } else if (values[h].equalsIgnoreCase("NULL")) {
137                    nullString++;
138                } else if (values[h].equalsIgnoreCase("NA")) {
139                    nullNA++;
140                }
141
142                Set<String> columnValues = sampleData.get(headers[h]);
143                if (columnValues.size() < MAXIMUM_UNIQUE_VALUES) {
144                    sampleData.get(headers[h]).add(values[h]);
145                }
146            }
147            line = fr.readLine();
148        }
149        fr.close();
150
151    }

153    @Override
154    public void onFinish() {
155        super.onFinish();
156
157        // Create collection
158        Collection collection = newCollection();
159
160        // Collect fields from other collections in the same folder
161        Map<String, Field> otherFields = collectFields();
162
163        List<Field> fields = new ArrayList<Field>();
164        for (String header : headers) {
165            String shortName, title;
166            if (otherFields.containsKey(header)) {
167                Field field = otherFields.get(header);
168                shortName = field.getLabel();
169                title = field.getTitle();
170            } else {
171                String lower = StringUtils.lowerCase(header);
172                shortName = StringUtils.abbreviate(lower, 20);
173                title = StringUtils.capitalize(lower);
174            }
175
176            Field field = new Field(header, shortName, title, deduceClass(sampleData.get(header)));
177
178            if (header.toLowerCase().contains("pvalue") || header.toLowerCase().contains("qvalue")) {
179                field.setProperties(Arrays.asList(new Property[]{
180                        new Property("BROWSER_DECORATOR", "PVALUE2")
181                }));
182            }
183
184            if (primaryKeys.contains(header)) {
185                field.setPrimaryKey(Boolean.TRUE);
186            }
187
188            fields.add(field);
189
190        }
191        collection.setFields(fields);
192
193        // Deduce links from other collections in the same folder
194        Map<String, Link> otherLinks = collectLinks();
195        List<Link> links = new ArrayList<Link>();
196
197        List<Collection> allProjectCollections = new ArrayList<Collection>();
198        addAllCollections(allProjectCollections, resourceManager.getProject(sourceURI.getProjectUrl()).getORI());
199
200        for (String header : headers) {
201            if (otherLinks.containsKey(header)) {
202                Link otherLink = otherLinks.get(header);
203                Link link = new Link();
204                link.setCollection(otherLink.getCollection());
205                link.getFields().add(otherLink.getFields().get(0));
206                links.add(link);
207            } else {
208
209                for (Collection col : allProjectCollections) {
210                    Field field = col.getField(header);
211
212                    if (field != null && (header.toLowerCase().endsWith("id") || header.toLowerCase().endsWith("key"))) {
213
214                        // Only link to collections without any link
215                        if (col.getLinks() == null || col.getLinks().isEmpty()) {
216                            Link link = new Link();
217                            link.setCollection(new ORI((String) null, col.getORI().getPath()));
218                            link.getFields().add(header);
219                            links.add(link);
220                        }
221                    }
222                }
223
224
225            }
226        }
227        collection.setLinks(links);
228
229
230        Loader loader = new Loader();
231        loader.setPlugin("tsv-loader");
232        List<Parameter> parameters = new ArrayList<Parameter>();
233        parameters.add(new Parameter("data", sourceURI.getPath()));
234
235        if (nullEmpty > nullDash && nullEmpty > nullString && nullEmpty > nullNA) {
236            parameters.add(new Parameter("NULL_VALUE", ""));
237        }
238
239        if (nullString > nullDash && nullString > nullEmpty && nullString > nullNA) {
240            parameters.add(new Parameter("NULL_VALUE", "NULL"));
241        }
242
243        if (nullNA > nullDash && nullNA > nullString && nullNA > nullEmpty) {
244            parameters.add(new Parameter("NULL_VALUE", "NA"));
245        }
246
247        loader.setParameters(parameters);
248        collection.setLoader(loader);
249
250        resourceManager.save(collection);
251
252        PageParameters params = new PageParameters().add(ResourcesPage.PARAMETER_RESOURCE, collection.getORI());
253        setResponsePage(ResourcesPage.class, params);
254
255
256    }

259    private Map<String, Link> collectLinks() {
260        Map<String, Link> links = new HashMap<String, Link>();
261
262        List<Collection> collections = new ArrayList<Collection>();
263        addAllCollections(collections, sourceURI.getParent());
264
265        for (Collection collection : collections) {
266            if (collection.getLinks() != null) {
267                for (Link link : collection.getLinks()) {
268
269                    // Only simple links (not composed)
270                    if (link.getFields().size() == 1) {
271                        String field = LinkUtils.getFromFieldName(link.getFields().get(0));
272                        links.put(field, link);
273                    }
274                }
275            }
276        }
277
278        return links;
279    }
280
281    private Map<String, Field> collectFields() {
282        Map<String, Field> fields = new HashMap<String, Field>();
283
284        List<Collection> collections = new ArrayList<Collection>();
285        addAllCollections(collections, sourceURI.getParent());
286
287        for (Collection collection : collections) {
288            for (Field field : collection.getFields()) {
289                fields.put(field.getId(), field);
290            }
291        }
292
293        return fields;
294    }

296    private Collection newCollection() {
297
298        String sourceName = sourceURI.getPath();
299        String collectionName;
300
301        int punt = sourceName.lastIndexOf('.');
302        if (punt != -1) {
303            collectionName = sourceName.substring(0, punt);
304        } else {
305            collectionName = sourceName + ".col";
306        }
307
308        ORI collectionURI = new ORI(sourceURI.getProjectUrl(), collectionName);
309
310        Collection collection = new Collection();
311        collection.setORI(collectionURI);
312        collection.setTitle(collectionName);
313
314        return collection;
315    }

317    private void addAllCollections(List<Collection> collections, ORI parentUri) {
318        collections.addAll(resourceManager.loadChildren(Collection.class, parentUri));
319        List<Folder> folders = resourceManager.loadChildren(Folder.class, parentUri);
320        for (Folder folder : folders) {
321            addAllCollections(collections, folder.getORI());
322        }
323    }

333    private static Class<?> deduceClass(Set<String> values) {
334
335        boolean longString = false;
336        boolean integerType = true;
337        boolean doubleType = false;
338
339        for (String value : values) {
340
341            // Skip null and empty values
342            if (value == null || value.isEmpty() || value.equals("-")) {
343                continue;
344            }
345
346            if (integerType) {
347                try {
348                    Integer.valueOf(value);
349                } catch (NumberFormatException e) {
350                    integerType = false;
351                    doubleType = true;
352                }
353            }
354
355            if (doubleType) {
356                try {
357                    Double.valueOf(value);
358                } catch (NumberFormatException e) {
359                    doubleType = false;
360                }
361            }
362
363            if (value.length() > 127) {
364                longString = true;
365            }
366        }
367
368        if (integerType) {
369            return Integer.class;
370        }
371
372        if (doubleType) {
373            return Double.class;
374        }
375
376        if (longString) {
377            return Text.class;
378        } else {
379            return String.class;
380        }
381    }

391    private final class ChooseFormat extends WizardStep {
392
393        public ChooseFormat() {
394            super("Create collection", "Choose one file format");
395
396            RadioChoice<String> commandOptions = new RadioChoice<String>("formats", new PropertyModel<String>(CreateCollectionWizard.this, "selected"), FORMATS);
397            add(commandOptions);
398
399        }
400
401        @Override
402        public void applyState() {
403            try {
404                readData();
405            } catch (IOException e) {
406                error(e.getMessage());
407            }
408        }
