018package org.onexus.ui.authentication.persona;
019
020
021import org.apache.wicket.AttributeModifier;
022import org.apache.wicket.Component;
023import org.apache.wicket.ajax.AjaxRequestTarget;
024import org.apache.wicket.markup.head.IHeaderResponse;
025import org.apache.wicket.markup.head.JavaScriptHeaderItem;
026import org.apache.wicket.markup.html.image.Image;
027import org.apache.wicket.markup.html.panel.Panel;
028import org.apache.wicket.request.resource.PackageResourceReference;
029import org.apache.wicket.request.resource.ResourceReference;
030
031/**
032 * A panel that is used for non-authenticated users
033 */
034public class GuestPanel extends Panel {
035
036    private static final long serialVersionUID = 1L;
037
038    public static final JavaScriptHeaderItem INCLUDE = JavaScriptHeaderItem.forUrl("https://login.persona.org/include.js");
039
040    private static final ResourceReference RED_ICON = new PackageResourceReference(
041            GuestPanel.class, "sign_in_red.png");
042    private static final ResourceReference BLUE_ICON = new PackageResourceReference(
043            GuestPanel.class, "sign_in_blue.png");
044    private static final ResourceReference GREEN_ICON = new PackageResourceReference(
045            GuestPanel.class, "sign_in_green.png");
046    private static final ResourceReference ORANGE_ICON = new PackageResourceReference(
047            GuestPanel.class, "sign_in_orange.png");
048    private static final ResourceReference GREY_ICON = new PackageResourceReference(
049            GuestPanel.class, "sign_in_grey.png");
050
051    /**
052     * The possible styles of the default "Sign In" image button provided by https://browserid.org
053     */
054    public enum Style {
055        RED, BLUE, GREEN, ORANGE, GREY
056    }
057
058    private final Style style;
059
060    public GuestPanel(String id, Style style) {
061        super(id);
062
063        this.style = style;
064
065        Component signInImage = createSignInButton("signInImage");
066        signInImage.add(new VerifyBehavior() {
067
068            private static final long serialVersionUID = 1L;
069
070            @Override
071            protected void onSuccess(AjaxRequestTarget target) {
072                GuestPanel.this.onSuccess(target);
073            }
074
075            @Override
076            protected void onFailure(AjaxRequestTarget target, String failureReason) {
077                GuestPanel.this.onFailure(target, failureReason);
078            }
079
080        });
081        add(signInImage);
082    }
083
084    protected Component createSignInButton(String componentId) {
085        Image image = new Image(componentId, getImage(style));
086        image.add(AttributeModifier.replace("alt", "Sign In"));
087        return image;
088    }
089
090    protected ResourceReference getImage(final Style style) {
091        final ResourceReference imageIcon;
092        switch (style) {
093            case RED:
094                imageIcon = RED_ICON;
095                break;
096            case GREEN:
097                imageIcon = GREEN_ICON;
098                break;
099            case ORANGE:
100                imageIcon = ORANGE_ICON;
101                break;
102            case GREY:
103                imageIcon = GREY_ICON;
104                break;
105            case BLUE:
106            default:
107                imageIcon = BLUE_ICON;
108                break;
109        }
110
111        return imageIcon;
112    }
113
114    @Override
115    public void renderHead(IHeaderResponse response) {
116        super.renderHead(response);
117
118        renderBrowserIdJavaScript(response);
119    }
120
121    /**
122     * Renders a reference for external browserid.js (loaded from browserid.org). <br/>
123     * Can be overridden with local reference to browserid.js if needed.
124     *
125     * @param response the current header response
126     */
127    protected void renderBrowserIdJavaScript(final IHeaderResponse response) {
128        response.render(INCLUDE);
129    }
130
131    /**
132     * A callback called when the authentication is successful
133     *
134     * @param target the current request target
135     */
136    protected void onSuccess(final AjaxRequestTarget target) {
137
138    }
139
140    /**
141     * A callback called when the authentication has failed
142     *
143     * @param target        the current request target
144     * @param target
145     * @param failureReason
146     */
147    protected void onFailure(final AjaxRequestTarget target, final String failureReason) {
148
149    }
150}
