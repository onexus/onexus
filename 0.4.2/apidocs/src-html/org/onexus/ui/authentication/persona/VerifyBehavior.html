020import org.apache.wicket.Component;
021import org.apache.wicket.Session;
022import org.apache.wicket.ajax.AbstractDefaultAjaxBehavior;
023import org.apache.wicket.ajax.AjaxRequestTarget;
024import org.apache.wicket.markup.head.IHeaderResponse;
025import org.apache.wicket.markup.head.OnDomReadyHeaderItem;
026import org.apache.wicket.request.Request;
027import org.apache.wicket.request.cycle.RequestCycle;
028import org.apache.wicket.util.io.IOUtils;
029import org.apache.wicket.util.string.StringValue;
030import org.apache.wicket.util.template.PackageTextTemplate;
031import org.apache.wicket.util.template.TextTemplate;
032
033import java.io.IOException;
034import java.io.OutputStream;
035import java.net.URL;

040/**
041 * The behavior that should be attached to the "Sign In" button. It cares about loading the
042 * authentication window and notifying the caller via {@link #onSuccess(org.apache.wicket.ajax.AjaxRequestTarget)} or
043 * {@link #onFailure(org.apache.wicket.ajax.AjaxRequestTarget, String)}
044 */
045public abstract class VerifyBehavior extends AbstractDefaultAjaxBehavior {
046
047    private static final long serialVersionUID = 1L;
048
049    @Override
050    public void renderHead(final Component component, final IHeaderResponse response) {
051        component.setOutputMarkupId(true);
052        super.renderHead(component, response);
053
054        final Map<String, Object> variables = new HashMap<String, Object>();
055        variables.put("componentId", component.getMarkupId());
056        variables.put("callbackUrl", getCallbackUrl());
057
058        final TextTemplate verifyTemplate = new PackageTextTemplate(VerifyBehavior.class,
059                "verify.js.tmpl");
060        String asString = verifyTemplate.asString(variables);
061        response.render(OnDomReadyHeaderItem.forScript(asString));
062    }
063
064    @Override
065    protected void respond(AjaxRequestTarget target) {
066        RequestCycle cycle = RequestCycle.get();
067        Request request = cycle.getRequest();
068        StringValue assertionParam = request.getQueryParameters().getParameterValue("assertion");
069        StringValue audienceParam = request.getQueryParameters().getParameterValue("audience");
070
071        if (!assertionParam.isEmpty() && !audienceParam.isEmpty()) {
072            String failureReason = verify(assertionParam.toString(), audienceParam.toString());
073            if (failureReason == null) {
074                onSuccess(target);
075            } else {
076                onFailure(target, failureReason);
077            }
078        }
079    }
080
081    private String verify(final String assertion, final String audience) {
082        String failureReason = null;
083        try {
084            URL verifyUrl = new URL("https://verifier.login.persona.org/verify");
085            URLConnection urlConnection = verifyUrl.openConnection();
086            urlConnection.setDoOutput(true);
087            OutputStream outputStream = urlConnection.getOutputStream();
088            String postParams = "assertion=" + assertion + "&audience=" + audience;
089            outputStream.write(postParams.getBytes());
090            outputStream.close();
091
092            String response = IOUtils.toString(urlConnection.getInputStream(), "UTF-8");
093
094            BrowserId browserId = BrowserId.of(response);
095            if (browserId != null) {
096                if (BrowserId.Status.OK.equals(browserId.getStatus())) {
097                    SessionHelper.logIn(Session.get(), browserId);
098                } else {
099                    failureReason = browserId.getReason();
100                }
101            }
102        } catch (IOException e) {
103            failureReason = e.getMessage();
104        }
105
106        return failureReason;
107    }
108
109    protected abstract void onSuccess(AjaxRequestTarget target);
110
111    protected abstract void onFailure(AjaxRequestTarget target, String failureReason);
112
113}
