064    static final String CREATE_TABLE_SQL = "CREATE TABLE IF NOT EXISTS system_properties (name varchar(128), version VARCHAR(128), primary key(name))";
065    static final String DELETE_OBJECT_SQL = "DELETE FROM system_properties WHERE name = ?";
066    static final String INSERT_OBJECT_SQL = "INSERT INTO system_properties (name, version) VALUES (?, ?)";
067    static final String SELECT_OBJECT_SQL = "SELECT version FROM system_properties WHERE name = ?";
068    static final String SELECT_PROPERTIES_KEYS = "SELECT name FROM system_properties";
069    static final String SELECT_PROPERTIES_MAP = "SELECT name, version FROM system_properties";

076    public SqlDialect() {
077        super();
078        this.sqlAdapters = registerAdapters();
079        this.builders = registerFilterBuilders();
080        this.columnTypes = registerColumnTypes();
081    }

083    protected Map<Class<?>, String> registerColumnTypes() {
084        Map<Class<?>, String> columnTypes = new HashMap<Class<?>, String>();
085        columnTypes.put(String.class, "VARCHAR(128)");
086        columnTypes.put(Text.class, "TEXT");
087        columnTypes.put(Boolean.class, "TINYINT(1)");
088        columnTypes.put(Date.class, "TIMESTAMP");
089        columnTypes.put(Integer.class, "INT(11)");
090        columnTypes.put(Long.class, "BIGINT");
091        columnTypes.put(Double.class, "DOUBLE");
092        return columnTypes;
093    }
094
095    protected Map<Class<?>, SqlAdapter> registerAdapters() {
096        Map<Class<?>, SqlAdapter> sqlAdapters = new HashMap<Class<?>, SqlAdapter>();
097        sqlAdapters.put(Double.class, new DoubleAdapter());
098        sqlAdapters.put(Integer.class, new IntegerAdapter());
099        sqlAdapters.put(Long.class, new LongAdapter());
100        sqlAdapters.put(String.class, new StringAdapter(this));
101        sqlAdapters.put(Boolean.class, new BooleanAdapter());
102        sqlAdapters.put(Text.class, new TextAdapter(this));
103        return sqlAdapters;
104    }
105
106    protected List<FilterBuilder> registerFilterBuilders() {
107        List<FilterBuilder> builders = new ArrayList<FilterBuilder>();
108        builders.add(new AtomicFilterBuilder(this));
109        builders.add(new EqualIdFilterBuilder(this));
110        builders.add(new BinaryFilterBuilder(this));
111        builders.add(new NotFilterBuilder(this));
112        builders.add(new InFilterBuilder(this));
113        builders.add(new IsNullFilterBuilder(this));
114        return builders;
115    }

122    public void createSystemPropertiesTable(Connection conn,
123                                            boolean closeConnection) throws SQLException {
124        Statement st = null;
125        try {
126            st = conn.createStatement();
127            st.execute(CREATE_TABLE_SQL);
128        } finally {
129
130            if (st != null) {
131                st.close();
132            }
133
134            if (conn != null && closeConnection) {
135                conn.close();
136            }
137        }
138    }

140    public List<String> loadPropertyKeys(Connection conn) throws Exception {
141        List<String> keys = new ArrayList<String>();
142        PreparedStatement pstmt = null;
143        ResultSet rs = null;
144        try {
145            pstmt = conn.prepareStatement(SELECT_PROPERTIES_KEYS);
146            rs = pstmt.executeQuery();
147
148            while (rs.next()) {
149                keys.add(rs.getString("name"));
150            }
151
152        } finally {
153            if (rs != null){
154                rs.close();
155            }
156
157            if (pstmt != null) {
158                pstmt.close();
159            }
160
161        }
162        return keys;
163    }

190    public void saveProperty(Connection conn, String name, String version)
191            throws Exception {
192        PreparedStatement pstmt = null;
193        try {
194            pstmt = conn.prepareStatement(DELETE_OBJECT_SQL);
195            pstmt.setString(1, name);
196            pstmt.execute();
197            pstmt.close();
198
199            pstmt = conn.prepareStatement(INSERT_OBJECT_SQL);
200            pstmt.setString(1, name);
201            pstmt.setString(2, version);
202            pstmt.executeUpdate();
203        } finally {
204            if (pstmt != null) {
205                pstmt.close();
206            }
207        }
208    }

225    public String loadProperty(Connection conn, String name)
226            throws Exception {
227        PreparedStatement pstmt = null;
228        ResultSet rs = null;
229        try {
230            pstmt = conn.prepareStatement(SELECT_OBJECT_SQL);
231
232            pstmt.setString(1, name);
233            rs = pstmt.executeQuery();
234
235            if (!rs.next()) {
236                return null;
237            }
238
239            return rs.getString(1);
240
241        } finally {
242            if (rs != null) {
243                rs.close();
244            }
245
246            if (pstmt != null) {
247                pstmt.close();
248            }
249        }
250    }

253            throws SQLException {
254        Statement st = null;
255        try {
256            LOGGER.trace("Execute: " + query);
257            st = conn.createStatement();
258            st.execute(query);
259        } finally {
260            if (st != null) {
261                st.close();
262            }
263        }

266    public void openInsert(StringBuilder sql, SqlCollectionDDL ddl) {
267
268        // Open insert
269        sql.append("INSERT INTO `").append(ddl.getTableName())
270                .append("` (`");
271
272        // Column names
273        Iterator<ColumnInfo> it = ddl.getColumnInfos().iterator();
274        while (it.hasNext()) {
275            sql.append(it.next().getColumnName());
276            if (it.hasNext()) {
277                sql.append("`, `");
278            }
279        }
280
281        sql.append("`) VALUES ");
282    }

284    public void addValues(StringBuilder sql, SqlCollectionDDL ddl,
285                          IEntity entity) {
286
287        sql.append("(");
288        Iterator<ColumnInfo> itColumns = ddl.getColumnInfos().iterator();
289        while (itColumns.hasNext()) {
290            ColumnInfo column = itColumns.next();
291            Object value = entity.get(column.getField().getId());
292            if (value == null) {
293
294                if (column.getField().isPrimaryKey() != null && column.getField().isPrimaryKey()) {
295                    if (column.getField().getType().equals(String.class)) {
296                        sql.append("\"NONE\"");
297                    } else {
298                        sql.append("0");
299                    }
300                } else {
301                    sql.append("NULL");
302                }
303            } else {
304                SqlAdapter adapter = column.getAdapter();
305
306                try {
307                    adapter.append(sql, value);
308                } catch (Exception e) {
309                    throw new RuntimeException(e);
310                }
311
312            }
313
314            if (itColumns.hasNext()) {
315                sql.append(",");
316            }
317        }
318        sql.append(")");
319    }

321    public SqlEntity loadEntity(Query query, SqlCollectionDDL ddl, ResultSet rs,
322                                SqlEntity dstObj, String columnPrefix) {
323
324        Set<String> columnWithError = new HashSet<String>();
325
326        String collectionAlias = QueryUtils.newCollectionAlias(query, ddl.getCollection().getORI());
327        List<String> queryFields = query.getSelect().get(collectionAlias);
328
329        for (ColumnInfo column : ddl.getColumnInfos()) {
330
331            if (queryFields != null && !queryFields.contains(column.getField().getId())) {
332                continue;
333            }
334
335            String columnName = column.getColumnName();
336            SqlAdapter adapter = column.getAdapter();
337            try {
338                String sqlColumnName = (columnPrefix == null ? columnName
339                        : columnPrefix + "_" + columnName);
340                dstObj.put(column.getField().getId(),
341                        adapter.extract(rs, sqlColumnName));
342            } catch (Exception e) {
343
344                if (!columnWithError.contains(columnName)) {
345                    LOGGER.error("Error loading column '" + columnName + "'");
346                    try {
347                        ResultSetMetaData m = rs.getMetaData();
348
349                        List<String> columnNames = new ArrayList<String>();
350                        for (int i = 1; i <= m.getColumnCount(); i++) {
351                            columnNames.add(m.getColumnName(i));
352                        }
353                        LOGGER.error("Valid column names: " + columnNames + "'");
354
355                    } catch (SQLException e1) {
356                        LOGGER.error("Error getting table metadata.", e1);
357                    }
358                    columnWithError.add(columnName);
359                }
360
361            }
362        }
363
364        return dstObj;
365    }

371    public String quoteString(Object value) {
372
373        if (value == null) {
374            return nullValue();
375        }
376
377        String s;
378
379        if (value instanceof String) {
380            s = (String) value;
381        } else {
382            s = String.valueOf(value);
383        }
384
385        int length = s.length();
386        StringBuilder buff = new StringBuilder(length + 2);
387        buff.append('\'');
388        for (int i = 0; i < length; i++) {
389            char c = s.charAt(i);
390            if (c == '\'' || c == '\\') {
391                buff.append(c);
392            }
393            buff.append(c);
394        }
395        buff.append('\'');
396        return buff.toString();
397    }

400    public FilterBuilder getFilterBuilder(Filter filter) {
401
402        for (FilterBuilder builder : builders) {
403            if (builder.canBuild(filter)) {
404                return builder;
405            }
406        }
407
408        return new UnknownFilterBuilder();
409
410    }
