046    private static final Logger log = LoggerFactory.getLogger(SqlQuery.class);
047    private List<String> select = new ArrayList<String>();
048    private String from;
049    private String where;
050    protected List<String> orderBy = new ArrayList<String>();
051    private String limit;
052
053    private List<String> leftJoins = new ArrayList<String>();
054
055    private StringBuilder sqlCount;
056    private StringBuilder sqlSelect;

058    protected Query query;
059    private SqlCollectionStore manager;

061    public SqlQuery(SqlCollectionStore manager, Query query) {
062        super();
063
064        this.manager = manager;
065        this.query = query;
066
067        try {
068            addSelect();
069            addFrom();
070            addJoins();
071            addWhere();
072            addOrderBy();
073            addLimit();
074        } catch (Exception e) {
075            StringBuilder msg = new StringBuilder();
076            msg.append(e.getMessage()).append(". Rendering query:\n");
077            query.toString(msg, true);
078            throw new OnexusException(msg.toString(), e);
079        }
080
081    }
082
083    protected void addSelect() {
084
085        for (Map.Entry<String, List<String>> selectCollection : query.getSelect().entrySet()) {
086
087            String collectionAlias = selectCollection.getKey();
088            ORI collectionUri = QueryUtils.getCollectionOri(query, collectionAlias);
089            SqlCollectionDDL collectionDDL = manager.getDDL(collectionUri);
090
091            List<String> fields = selectCollection.getValue();
092            if (fields == null) {
093                fields = new ArrayList<String>();
094                for (SqlCollectionDDL.ColumnInfo column : collectionDDL.getColumnInfos()) {
095                    fields.add(column.getField().getId());
096                }
097            }
098            for (String fieldId : fields) {
099
100                SqlCollectionDDL.ColumnInfo columnInfo = collectionDDL.getColumnInfoByFieldName(fieldId);
101                if (columnInfo == null) {
102                    throw new UnsupportedOperationException("Unknown field '" + fieldId + "'");
103                }
104                String fieldName = columnInfo.getColumnName();
105                select.add("`" + collectionAlias + "`.`" + fieldName + "` " + collectionAlias + "_" + fieldName);
106            }
107
108
109        }
110
111    }
112
113    protected void addFrom() {
114
115        String collectionAlias = query.getFrom();
116        ORI collectionUri = QueryUtils.getCollectionOri(query, collectionAlias);
117        String collectionTable = manager.getDDL(collectionUri).getTableName();
118
119        this.from = "`" + collectionTable + "` AS " + collectionAlias;
120    }
121
122    protected void addJoins() {
123
124        Collection fromCollection = manager.getCollection(QueryUtils.getCollectionOri(query, query.getFrom()));
125
126        // Fixed entities
127        List<EqualId> equalIds = new ArrayList<EqualId>();
128        addEqualIdFilters(equalIds, query.getWhere());
129
130        List<ORI> equalIdsCollections = new ArrayList<ORI>();
131        for (EqualId se : equalIds) {
132            equalIdsCollections.add(QueryUtils.getCollectionOri(query, se.getCollectionAlias()));
133        }
134
135        Map<ORI, List<ORI>> networkFixedCollection = new HashMap<ORI, List<ORI>>();
136        Map<ORI, StringBuilder> networkFixedJoins = new HashMap<ORI, StringBuilder>();
137        Map<ORI, List<FieldLink>> networkLinks = new HashMap<ORI, List<FieldLink>>();
138
139        for (Map.Entry<String, ORI> define : query.getDefine().entrySet()) {
140
141            String collectionAlias = define.getKey();
142            ORI collectionUri = define.getValue().toAbsolute(query.getOn());
143
144            // Skip from collection
145            if (collectionUri.equals(fromCollection.getORI())) {
146                continue;
147            }
148
149            String collectionTable = manager.getDDL(collectionUri).getTableName();
150
151            Collection joinCollection = manager.getCollection(collectionUri);
152
153            StringBuilder leftJoin = new StringBuilder(50);
154            leftJoin.append(" LEFT JOIN `");
155            leftJoin.append(collectionTable).append("` AS `").append(collectionAlias).append("` ON ");
156
157            // Fixed entities
158            List<ORI> collectionsFixed = new ArrayList<ORI>();
159            for (EqualId se : equalIds) {
160                if (joinCollection.getLinks() != null) {
161                    for (Link colLink : joinCollection.getLinks()) {
162
163                        ORI fixedCollection = QueryUtils.getCollectionOri(query, se.getCollectionAlias());
164                        ORI linkCollection = colLink.getCollection().toAbsolute(query.getOn());
165
166                        if (fixedCollection.equals(linkCollection)) {
167                            collectionsFixed.add(fixedCollection);
168
169                            List<String> fields = colLink.getFields();
170
171                            Object id = se.getId();
172                            //TODO manage composed keys
173                            Object[] ids = new Object[]{id};
174
175                            if (fields.size() != ids.length) {
176                                throw new RuntimeException("Bad entity id: " + se);
177                            }
178                            for (int i = 0; i < ids.length; i++) {
179
180                                String fromFieldName = LinkUtils.getFromFieldName(fields.get(i));
181
182                                Field field = joinCollection.getField(fromFieldName);
183                                SqlCollectionDDL.ColumnInfo fieldColumn = manager.getDDL(joinCollection.getORI()).getColumnInfoByFieldName(fromFieldName);
184
185                                leftJoin.append("`").append(collectionAlias).append("`.`")
186                                        .append(fieldColumn.getColumnName())
187                                        .append("` = ");
188
189                                SqlAdapter adapter = manager.getSqlDialect().getAdapter(field.getType());
190                                try {
191                                    adapter.append(leftJoin, ids[i]);
192                                } catch (Exception e) {
193                                    //TODO
194                                    log.error("Appending join value " + ids[i], e);
195                                }
196
197                                leftJoin.append(
198                                        " AND ");
199                            }
200                        }
201                    }
202                }
203            }
204
205            // Add links
206            ORI parentURI = query.getOn();
207            List<FieldLink> linkFields = LinkUtils.getLinkFields(parentURI, joinCollection, fromCollection, equalIdsCollections);
208
209            // Add thirdParty joins
210            if (linkFields.isEmpty()) {
211
212                // Check if it's possible to link with another joined collection
213
214                for (Map.Entry<String, ORI> tpDefine : query.getDefine().entrySet()) {
215
216                    // Skip current joined collection
217                    if (tpDefine.getKey().equals(define.getKey())) {
218                        continue;
219                    }
220
221                    ORI tpCollectionUri = tpDefine.getValue().toAbsolute(query.getOn());
222
223                    // Skip from collection
224                    if (tpCollectionUri.equals(fromCollection.getORI())) {
225                        continue;
226                    }
227
228                    Collection tpJoinCollection = manager.getCollection(tpCollectionUri);
229
230                    linkFields.addAll(LinkUtils.getLinkFields(parentURI, joinCollection, tpJoinCollection, equalIdsCollections));
231                }
232            }
233
234            // Check if it's possible to join this collection
235            if (linkFields.isEmpty()) {
236                throw new UnsupportedOperationException("Impossible to link collection '" + collectionUri + "' to '" + fromCollection.getORI() + "'.");
237            }
238
239            networkFixedJoins.put(collectionUri, leftJoin);
240            networkFixedCollection.put(collectionUri, collectionsFixed);
241            networkLinks.put(collectionUri, linkFields);
242
243
244        }
245
246        // Sort the network to include JOINS in the correct order
247
248        List<ORI> keys = new ArrayList<ORI>(networkLinks.keySet());
249        keys = sort(keys, new LinksNetworkComparator(networkLinks, fromCollection.getORI()));
250
251        for (ORI collectionUri : keys) {
252
253            StringBuilder leftJoin = networkFixedJoins.get(collectionUri);
254
255            for (FieldLink fieldLink : networkLinks.get(collectionUri)) {
256
257                ORI aCollection = fieldLink.getFromCollection();
258                SqlCollectionDDL aDDL = manager.getDDL(aCollection);
259                String aAlias = QueryUtils.newCollectionAlias(query, aCollection);
260                String aField = aDDL.getColumnInfoByFieldName(
261                        fieldLink.getFromFieldName()).getColumnName();
262                leftJoin.append("`").append(aAlias).append("`.`")
263                        .append(aField).append("` = `");
264
265                ORI bCollection = fieldLink.getToCollection();
266                SqlCollectionDDL bDDL = manager.getDDL(bCollection);
267                String bAlias = QueryUtils.newCollectionAlias(query, bCollection);
268
269                SqlCollectionDDL.ColumnInfo columnInfo = bDDL.getColumnInfoByFieldName(fieldLink.getToFieldName());
270
271                if (columnInfo == null) {
272                    throw new UnsupportedOperationException("Malformed link '" + fieldLink.toString() + "'");
273                }
274
275
276                String bField = columnInfo.getColumnName();
277                leftJoin.append(bAlias).append("`.`").append(bField)
278                        .append("` AND ");
279
280            }
281
282            // Remove last ' AND '
283            leftJoin.delete(leftJoin.length() - 5, leftJoin.length() - 1);
284            this.leftJoins.add(leftJoin.toString());
285
286        }
287
288    }

290    private static List<ORI> sort(List<ORI> keys, LinksNetworkComparator comparator) {
291
292        List<ORI> sortedKeys = new ArrayList<ORI>(keys);
293
294        int l = keys.size();
295        boolean swaped = true;
296        while (swaped) {
297            swaped = false;
298            for (int i = 0; i < l - 1; i++) {
299                for (int f = i + 1; f < l; f++) {
300                    int value = comparator.compare(sortedKeys.get(i), sortedKeys.get(f));
301                    if (value > 0) {
302                        ORI vI = sortedKeys.get(f);
303                        ORI vF = sortedKeys.get(i);
304                        sortedKeys.set(i, vI);
305                        sortedKeys.set(f, vF);
306                        swaped = true;
307                    }
308                }
309            }
310        }
311
312        return sortedKeys;
313    }

315    protected void addEqualIdFilters(List<EqualId> equalIds, Filter filter) {
316
317        if (filter instanceof EqualId) {
318            equalIds.add((EqualId) filter);
319            return;
320        }
321
322        if (filter instanceof And) {
323            addEqualIdFilters(equalIds, ((And) filter).getLeft());
324            addEqualIdFilters(equalIds, ((And) filter).getRight());
325        }
326    }

329    protected void addWhere() {
330
331        Filter filter = query.getWhere();
332
333        if (filter != null) {
334            FilterBuilder builder = manager.getSqlDialect().getFilterBuilder(filter);
335
336            StringBuilder where = new StringBuilder();
337            builder.build(manager, query, where, filter);
338            this.where = where.toString();
339        }
340
341    }
342
343
344    protected void addOrderBy() {
345
346        List<OrderBy> ordersOql = query.getOrders();
347
348        if (ordersOql == null || ordersOql.isEmpty()) {
349            return;
350        }
351
352        Iterator<OrderBy> orderIt = ordersOql.iterator();
353
354        while (orderIt.hasNext()) {
355            OrderBy order = orderIt.next();
356
357            String collectionAlias = order.getCollection();
358            ORI collectionOri = QueryUtils.getCollectionOri(query, collectionAlias);
359            SqlCollectionDDL collectionDDL = manager.getDDL(collectionOri);
360            SqlCollectionDDL.ColumnInfo columnInfo = collectionDDL.getColumnInfoByFieldName(order.getField());
361
362            String field = "`" + collectionAlias + "`.`" + columnInfo.getColumnName() + "`";
363            if (order.isAscendent()) {
364                this.orderBy.add("ISNULL(" + field + ") ASC");
365            }
366            this.orderBy.add(field + (order.isAscendent() ? " ASC" : "DESC"));
367        }
368
369    }
370
371    protected void addLimit() {
372
373        if (query.getCount() != null) {
374            Long offset = query.getOffset();
375            this.limit = (offset == null ? "0" : offset) + ", " + query.getCount();
376        }
377    }
378
379    protected String toSelectSQL() {
380        if (sqlSelect == null) {
381            sqlSelect = new StringBuilder();
382
383            sqlSelect.append("SELECT ");
384
385            if (select != null && !select.isEmpty()) {
386                flatCollection(sqlSelect, select, ",");
387            } else {
388                sqlSelect.append("*");
389            }
390
391            sqlSelect.append(" FROM ").append(from);
392
393            if (!leftJoins.isEmpty()) {
394                flatCollection(sqlSelect, leftJoins, " ");
395            }
396
397            if (where != null) {
398                sqlSelect.append(" WHERE ").append(where);
399            }
400
401            if (orderBy != null && !orderBy.isEmpty()) {
402                sqlSelect.append(" ORDER BY ");
403                flatCollection(sqlSelect, this.orderBy, ",");
404            }
405
406            if (limit != null) {
407                sqlSelect.append(" LIMIT ").append(limit);
408            }
409        }
410
411        return sqlSelect.toString();
412    }
413
414    protected String toCountSQL() {
415        if (sqlCount == null) {
416            sqlCount = new StringBuilder();
417            sqlCount.append("SELECT COUNT(*) AS `size` FROM ").append(from);
418
419            if (!leftJoins.isEmpty()) {
420                flatCollection(sqlCount, leftJoins, " ");
421            }
422
423            if (where != null) {
424                sqlCount.append(" WHERE ").append(where);
425            }
426
427        }
428
429        return sqlCount.toString();
430    }
431
432    private static void flatCollection(StringBuilder builder, java.util.Collection<?> collection, String separator) {
433        Iterator<?> it = collection.iterator();
434        while (it.hasNext()) {
435            builder.append(it.next());
436            if (it.hasNext()) {
437                builder.append(String.valueOf(separator));
438            }
439        }
440    }
