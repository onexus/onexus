018package org.onexus.collection.store.sql;
019
020import org.onexus.collection.api.Collection;
021import org.onexus.collection.api.IEntity;
022import org.onexus.collection.api.IEntityTable;
023import org.onexus.collection.api.query.Query;
024import org.onexus.collection.api.utils.QueryUtils;
025import org.onexus.resource.api.ORI;
026import org.onexus.resource.api.Progress;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030import java.sql.Connection;
031import java.sql.ResultSet;
032import java.sql.SQLException;
033import java.sql.Statement;
034
035public class SqlEntityTable implements IEntityTable {
036
037    private final static Logger LOGGER = LoggerFactory
038            .getLogger(SqlEntityTable.class);
039
040    private Query query;
041    private Progress progress;
042
043    private SqlQuery mysqlQuery;
044
045    private SqlCollectionStore store;
046
047    private Connection dataConn = null;
048    private Statement dataSt = null;
049    private ResultSet dataRS = null;
050    private Long size = null;
051
052    public SqlEntityTable(SqlCollectionStore store, Query query,
053                          Connection conn) {
054
055        this.store = store;
056        this.query = query;
057        this.mysqlQuery = store.newSqlQuery(query);
058        this.dataConn = conn;
059
060    }
061
062    private void initDataRS() {
063        String sql = null;
064        try {
065            dataSt = store.createReadStatement(dataConn);
066
067            sql = mysqlQuery.toSelectSQL();
068            LOGGER.debug(sql);
069            dataRS = dataSt.executeQuery(sql);
070        } catch (Exception e) {
071
072            try {
073                // Give a second change (Broken pipe exception)
074                dataSt = store.createReadStatement(dataConn);
075
076                sql = mysqlQuery.toSelectSQL();
077                dataRS = dataSt.executeQuery(sql);
078            } catch (Exception e2) {
079                throw new RuntimeException(e2);
080            }
081        }
082    }
083
084    @Override
085    public boolean next() {
086
087        if (dataRS == null) {
088            initDataRS();
089        }
090
091        try {
092            boolean hasNext = dataRS == null ? false : dataRS.next();
093
094            if (!hasNext) {
095                close();
096                return hasNext;
097            }
098
099            return hasNext;
100        } catch (SQLException e) {
101            return false;
102        }
103    }
104
105    @Override
106    public long size() {
107
108        if (size == null) {
109
110            String sql = null;
111            try {
112                // Count all
113                sql = mysqlQuery.toCountSQL();
114                LOGGER.debug(sql);
115                Statement dataSt = dataConn.createStatement();
116                ResultSet dataRS = dataSt.executeQuery(sql);
117                if (dataRS.next()) {
118                    size = dataRS.getLong("size");
119                } else {
120                    size = Long.valueOf(0);
121                }
122                dataRS.close();
123                dataSt.close();
124            } catch (SQLException e) {
125                throw new RuntimeException(e);
126            }
127
128        }
129
130        return size;
131    }
132
133    @Override
134    public void close() {
135        try {
136            if (dataRS != null && !dataRS.isClosed()) {
137                dataRS.close();
138            }
139            if (dataSt != null && !dataSt.isClosed()) {
140                dataSt.close();
141            }
142            if (dataConn != null && !dataConn.isClosed()) {
143                dataConn.close();
144            }
145        } catch (SQLException e) {
146            throw new RuntimeException(e);
147        }
148    }
149
150    @Override
151    public Query getQuery() {
152        return query;
153    }
154
155    @Override
156    public IEntity getEntity(ORI collectionURI) {
157
158        collectionURI = collectionURI.toAbsolute(query.getOn());
159
160        Collection collection = store.getCollection(collectionURI);
161        SqlEntity entity = new SqlEntity(collection);
162        store.getSqlDialect().loadEntity(query, store.getDDL(collectionURI), dataRS, entity,
163                QueryUtils.newCollectionAlias(query, collectionURI));
164
165        return entity;
166    }
167
168    @Override
169    public Progress getProgress() {
170        return progress;
171    }
172
173}
