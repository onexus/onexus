046    private List<String> createIndex;

049    public SqlCollectionDDL(SqlDialect sqlDialect, Collection collection, String tableName) {
050
051        this.sqlDialect = sqlDialect;
052        this.collection = collection;
053        this.columnInfos = new HashMap<String, ColumnInfo>();
054        this.tableName = tableName == null ? convertURItoTableName(collection) : tableName;
055
056        prepareFieldInfoMap(collection);
057        prepareCreateTable(collection);
058        prepareCreateIndex(collection);
059
060    }
061
062    private void prepareCreateIndex(Collection collection) {
063
064        this.createIndex = new ArrayList<String>();
065        this.dropIndex = new ArrayList<String>();
066        if (collection.getLinks() != null) {
067            StringBuilder linkSQL, dropIndexSQL;
068            int i = 0;
069            for (Link link : collection.getLinks()) {
070                linkSQL = new StringBuilder();
071                dropIndexSQL = new StringBuilder();
072
073                String indexName = getTableName() + "-" + Integer.toString(i);
074                linkSQL.append("CREATE INDEX `").append(indexName);
075                linkSQL.append("` ON `").append(getTableName()).append("` (");
076
077                dropIndexSQL.append("DROP INDEX `").append(indexName).append("`");
078
079                Iterator<String> fieldIt = link.getFields().iterator();
080                while (fieldIt.hasNext()) {
081                    String fromFieldId = LinkUtils.getFromFieldName(fieldIt.next());
082                    ColumnInfo indexField = getColumnInfoByFieldName(fromFieldId);
083
084                    if (indexField == null) {
085                        throw new UnsupportedOperationException("Impossible to link field '" + fromFieldId + "' at '" + collection.getORI() + "'.");
086                    }
087
088                    linkSQL.append("`").append(indexField.getColumnName()).append("`");
089                    if (fieldIt.hasNext()) {
090                        linkSQL.append(", ");
091                    }
092                }
093                linkSQL.append(")");
094                this.createIndex.add(linkSQL.toString());
095                this.dropIndex.add(dropIndexSQL.toString());
096                i++;
097            }
098        }
099
100    }
101
102    private static String convertURItoTableName(Collection collection) {
103        String hashCode = Integer.toHexString(collection.getORI().getProjectUrl().hashCode());
104        String tableName = removeNonValidChars(collection.getORI().getPath());
105
106        // Check that the table name is no longer than 64 characters (the
107        // maximum allowed)
108        int totalLength = tableName.length() + hashCode.length() + 1;
109        if (totalLength > 64) {
110            return tableName.substring(0, tableName.length()
111                    - (totalLength - 64))
112                    + "_" + hashCode;
113        } else {
114            return hashCode + "_" + tableName;
115        }
116    }

142    public ColumnInfo getColumnInfoByFieldName(String fieldName) {
143        for (ColumnInfo ci : columnInfos.values()) {
144            if (ci.getField().getId().equals(fieldName)) {
145                return ci;
146            }
147        }
148        return null;
149    }

151    private void prepareFieldInfoMap(Collection collection) {
152        if (collection == null) {
153            return;
154        }
155
156        for (Field field : collection.getFields()) {
157            ColumnInfo fi = new ColumnInfo(field);
158            columnInfos.put(fi.getColumnName(), fi);
159        }
160    }
161
162    private void prepareCreateTable(Collection collection) {
163        createTable = new StringBuilder();
164        createTable.append("CREATE TABLE IF NOT EXISTS `").append(tableName)
165                .append("` (");
166
167        Iterator<ColumnInfo> ifi = columnInfos.values().iterator();
168        while (ifi.hasNext()) {
169            ColumnInfo fi = ifi.next();
170            createTable.append("`").append(fi.getColumnName()).append("` ")
171                    .append(fi.getColumnType());
172
173            if (ifi.hasNext()) {
174                createTable.append(", ");
175            }
176        }
177
178        // Add primary key
179
180        // TODO Use common collections filters
181        List<String> keys = new ArrayList<String>();
182        for (Field field : collection.getFields()) {
183            if (field.isPrimaryKey() != null && field.isPrimaryKey()) {
184                keys.add(field.getId());
185            }
186        }
187
188        if (!keys.isEmpty()) {
189            createTable.append(", PRIMARY KEY (`");
190            Iterator<String> keyFields = keys.iterator();
191            while (keyFields.hasNext()) {
192                String fieldURI = keyFields.next();
193                String columnName = null;
194                for (ColumnInfo column : columnInfos.values()) {
195                    if (column.getField().getId().equals(fieldURI)) {
196                        columnName = column.getColumnName();
197                        break;
198                    }
199                }
200                if (columnName == null) {
201                    String msg = String.format(
202                            "Key field '%s' not fount in collection '%s'.",
203                            fieldURI, collection);
204                    LOGGER.error(msg);
205                    throw new RuntimeException(msg);
206                }
207
208                createTable.append(columnName);
209                if (keyFields.hasNext()) {
210                    createTable.append("`, `");
211                }
212            }
213            createTable.append("`)");
214        }
215
216        createTable.append(")");
217    }

219    private static String removeNonValidChars(String id) {
220        return id.toLowerCase().trim().replaceAll("[^a-z0-9]", "_");
221    }

231    public class ColumnInfo {
232
233        private Field field;
234        private String columnName;
235        private String columnType = null;
236
237        private ColumnInfo(Field field) {
238            this.columnName = removeNonValidChars(field.getId());
239            this.field = field;
240
241            this.columnType = field.getProperty("SQL_COLUMN_TYPE");
242            if (this.columnType == null) {
243                this.columnType = sqlDialect.getColumnType(field.getType());
244            }
245            if (this.columnType == null) {
246                String msg = "Column type not found for the field: '" + field
247                        + "'";
248                LOGGER.error(msg);
249                throw new RuntimeException(msg);
250            }
251        }
252
253        public Field getField() {
254            return field;
255        }
256
257        public String getColumnName() {
258            return columnName;
259        }
260
261        public String getColumnType() {
262            return columnType;
263        }
264
265        @Override
266        public String toString() {
267            StringBuilder builder = new StringBuilder();
268            builder.append("ColumnInfo [field=");
269            builder.append(field);
270            builder.append(", columnName=");
271            builder.append(columnName);
272            builder.append(", columnType=");
273            builder.append(columnType);
274            builder.append("]");
275            return builder.toString();
276        }
277
278        public SqlAdapter getAdapter() {
279            return sqlDialect.getAdapter(field.getType());
280        }
281
282    }
