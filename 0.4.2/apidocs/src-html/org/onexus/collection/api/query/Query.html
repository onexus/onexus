029public class Query implements Serializable {
030
031    public static final String END_LINE_AND_TAB = "\n\t";
032
033    private Map<String, ORI> define = new LinkedHashMap<String, ORI>();
034
035    private ORI on;
036
037    private Map<String, List<String>> select = new LinkedHashMap<String, List<String>>();
038
039    private String from;
040
041    private Filter where;
042
043    private List<OrderBy> orders = new ArrayList<OrderBy>();
044
045    private Long offset;
046
047    private Long count;
048

051    }

054        return define;
055    }

058        define.put(key, resourceURI.toRelative(on));
059    }
060
061    public Map<String, List<String>> getSelect() {
062        return select;

066        return from;
067    }
068

071    }

074        return offset;

077    public Long getCount() {

081    public void addSelect(String key, List<String> fieldNames) {
082        if (select.containsKey(key) && select.get(key) != null) {
083            select.get(key).addAll(fieldNames);
084        } else {
085            select.put(key, fieldNames == null ? null : new ArrayList<String>(fieldNames));
086        }
087    }

131    @Override
132    public String toString() {
133        return toString(new StringBuilder(), true).toString();
134    }
135
136    public StringBuilder toString(StringBuilder oql, boolean prettyPrint) {
137
138
139        // Define section
140        if (define != null && !define.isEmpty()) {
141            oql.append("DEFINE");
142
143            Iterator<Map.Entry<String, ORI>> itDefine = define.entrySet().iterator();
144            while (itDefine.hasNext()) {
145                oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
146
147                Map.Entry<String, ORI> entry = itDefine.next();
148                oql.append(entry.getKey()).append("=").append(escapeString(entry.getValue().toString()));
149
150                if (itDefine.hasNext()) {
151                    oql.append(',');
152                }
153            }
154        }
155
156        // On section
157        if (on != null) {
158            oql.append(prettyPrint ? "\n" : " ");
159            oql.append("ON");
160            oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
161            oql.append(escapeString(on.toString()));
162        }
163
164
165        // Select section
166        if (select != null && !select.isEmpty()) {
167            oql.append(prettyPrint ? "\n" : " ");
168            oql.append("SELECT");
169            Iterator<Map.Entry<String, List<String>>> itSelect = select.entrySet().iterator();
170            while (itSelect.hasNext()) {
171                oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
172                Map.Entry<String, List<String>> entry = itSelect.next();
173                oql.append(entry.getKey());
174                if (entry.getValue() != null) {
175                    Iterator<String> fields = entry.getValue().iterator();
176                    if (fields.hasNext()) {
177                        oql.append(" (");
178                        while (fields.hasNext()) {
179                            oql.append(fields.next());
180                            if (fields.hasNext()) {
181                                oql.append(", ");
182                            }
183                        }
184                        oql.append(")");
185                    }
186                }
187
188                if (itSelect.hasNext()) {
189                    oql.append(",");
190                }
191            }
192        }
193
194        // From section
195        if (from != null && !from.isEmpty()) {
196            oql.append(prettyPrint ? "\n" : " ");
197            oql.append("FROM");
198            oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
199            oql.append(from);
200        }
201
202        // Where section
203        if (where != null) {
204            oql.append(prettyPrint ? "\n" : " ");
205            oql.append("WHERE");
206            oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
207            where.toString(oql, prettyPrint);
208        }
209
210        // OrderBy section
211        if (orders != null && !orders.isEmpty()) {
212            oql.append(prettyPrint ? "\n" : " ");
213            oql.append("ORDER BY");
214
215            Iterator<OrderBy> itOrders = orders.iterator();
216            while (itOrders.hasNext()) {
217                oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
218                OrderBy orderBy = itOrders.next();
219                orderBy.toString(oql, prettyPrint);
220                if (itOrders.hasNext()) {
221                    oql.append(',');
222                }
223            }
224        }
225
226        // Limit section
227        if (offset != null && count != null) {
228            oql.append(prettyPrint ? "\n" : " ");
229            oql.append("LIMIT");
230            oql.append(prettyPrint ? END_LINE_AND_TAB : " ");
231            oql.append("'").append(offset).append("', '").append(count).append("'");
232        }
233
234        return oql;
235    }
236
237    public static String unescapeString(String value) {
238
239        if (value == null) {
240            return null;
241        }
242
243        if (value.length() < 3) {
244            return "";
245        }
246
247        char quote = value.charAt(0);
248
249        value = value.substring(1, value.length() - 1);
250        if (quote == '"') {
251            value = value.replace("\\\"", "\"");
252        } else {
253            value = value.replace("\\'", "'");
254        }
255
256        return value;
257
258    }
259
260
261    public static String escapeString(String value) {
262        if (value == null) {
263            return null;
264        }
265        value = "'" + value.replace("'", "\\'") + "'";
266        return value;
267    }
