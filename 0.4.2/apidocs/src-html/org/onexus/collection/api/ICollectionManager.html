018package org.onexus.collection.api;
019
020import org.onexus.collection.api.query.Query;
021import org.onexus.resource.api.IResourceService;
022import org.onexus.resource.api.ORI;
023
024/**
025 * <p>A ICollectionManager manage the life-cycle of a collection. We can have a non-blocking
026 * ICollectionManager, so it will return partial {@link IEntityTable} result with a {@link org.onexus.resource.api.Progress}
027 * in process. It is the user that have to keep calling getTaskStatus() method until the task
028 * is done and then call again the load() method with the same query to get all the results.</p>
029 * <p/>
030 * <p>This is the typical event sequence the first time that one collection is required:</p>
031 * <ul>
032 * <li>CM = ICollectionManager</li>
033 * <li>CS = ICollectionStore</li>
034 * <li>TM = ITaskManager</li>
035 * <li>TE = ITaskExecutor</li>
036 * <li>TC = ITaskCallable</li>
037 * </ul>
038 * <img src="doc-files/basic-collection-loading.svg" width="100%" />
039 *
040 * @author Jordi Deu-Pons
041 */
042public interface ICollectionManager extends IResourceService {
043
044    /**
045     * @param query The query
046     * @return The result of the query. If it's not a blocking ICollectionManager
047     *         it can return a partial result but with a {@link org.onexus.resource.api.Progress} under process.
048     */
049    IEntityTable load(Query query);
050
051    /**
052     * Unload the collection from origin store.
053     *
054     * @param collectionOri
055     */
056    void unload(ORI collectionOri);
057
058    boolean isLinkable(Query query, ORI collectionOri);
059}
