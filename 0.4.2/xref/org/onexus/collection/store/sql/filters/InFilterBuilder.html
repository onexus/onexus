18  package org.onexus.collection.store.sql.filters;
19  
20  import org.onexus.collection.api.query.In;
21  import org.onexus.collection.api.query.Query;
22  import org.onexus.collection.api.utils.QueryUtils;
23  import org.onexus.collection.store.sql.SqlCollectionDDL;
24  import org.onexus.collection.store.sql.SqlCollectionStore;
25  import org.onexus.collection.store.sql.SqlDialect;
26  import org.onexus.resource.api.ORI;
27  
28  import java.util.Iterator;
29  
30  
31  public class InFilterBuilder extends AbstractFilterBuilder<In> {
32  
33      public InFilterBuilder(SqlDialect dialect) {
34          super(dialect, In.class);
35      }
36  
37      @Override
38      protected void innerBuild(SqlCollectionStore store, Query query, StringBuilder where, In filter) {
39  
40          // Collection
41          String collectionAlias = filter.getCollectionAlias();
42          ORI collectionUri = QueryUtils.getCollectionOri(query, collectionAlias);
43          SqlCollectionDDL collection = store.getDDL(collectionUri);
44  
45          // Field
46          String fieldId = filter.getFieldId();
47          SqlCollectionDDL.ColumnInfo column = collection.getColumnInfoByFieldName(fieldId);
48  
49          // Values
50          Iterator<Object> values = filter.getValues().iterator();
51  
52          where.append('`').append(collectionAlias).append("`.`").append(column.getColumnName()).append('`');
53          where.append(" IN (");
54  
55          while (values.hasNext()) {
56              encodeValue(where, column.getField().getType(), values.next());
57  
58              if (values.hasNext()) {
59                  where.append(',');
60              }
61          }
62          where.append(")");
63  
64      }
65  }
