79      @Override
80      protected void onInitialize() {
81          super.onInitialize();
82  
83          // Prepare accordion containers
84          RepeatingView links = new RepeatingView("links");
85          WebMarkupContainer accordionBody = new WebMarkupContainer("accordion-body");
86          String bodyId = getMarkupId() + "-body";
87          accordionBody.setMarkupId(bodyId);
88  
89          add(links);
90          add(accordionBody);
91  
92          if (entityIterator != null) {
93              entity = entityIterator.next();
94          }
95  
96          // Label
97          Set<String> labels = new HashSet<String>();
98          String labelField = entity.getCollection().getProperty("FIXED_ENTITY_FIELD");
99  
100         WebMarkupContainer item = new WebMarkupContainer(links.newChildId());
101         item.add(new AttributeModifier("class", "active"));
102         links.add(item);
103 
104         WebMarkupContainer activeLink = new WebMarkupContainer("link");
105         activeLink.add(new AttributeModifier("href", "#" + entity.getId()));
106         activeLink.add(new AttributeModifier("data-toggle", "pill"));
107         item.add(activeLink);
108 
109         String label = getLabel(entity, labelField);
110         labels.add(label.toUpperCase());
111         activeLink.add(new Label("label", label).setEscapeModelStrings(false));
112 
113         // Fields values
114         RepeatingView fields = new RepeatingView("fields");
115         accordionBody.add(fields);
116 
117         Component first = newFieldsBox(fields.newChildId(), labelField, entity, true);
118         fields.add(first);
119 
120         // Complete the label
121         if (entityIterator != null) {
122             int count = 1;
123             int remaining = 0;
124             while (entityIterator.hasNext()) {
125 
126                 entity = entityIterator.next();
127 
128                 count++;
129 
130                 label = getLabel(entity, labelField);
131                 labels.add(label.toUpperCase());
132 
133                 if (count > MAX_LABELS) {
134                     remaining++;
135                     continue;
136                 }
137 
138                 item = new WebMarkupContainer(links.newChildId());
139                 links.add(item);
140 
141                 activeLink = new WebMarkupContainer("link");
142                 activeLink.add(new AttributeModifier("href", "#" + entity.getId()));
143                 activeLink.add(new AttributeModifier("data-toggle", "pill"));
144                 item.add(activeLink);
145                 activeLink.add(new Label("label", label));
146 
147                 fields.add(newFieldsBox(fields.newChildId(), labelField, entity, false));
148 
149             }
150 
151             if (remaining > 0) {
152                 item = new WebMarkupContainer(links.newChildId());
153                 links.add(item);
154                 activeLink = new ExternalLink("link", "");
155                 activeLink.add(new Label("label", "and " + remaining + " more"));
156                 activeLink.setEnabled(false);
157                 item.add(activeLink);
158             }
159         }
160 
161         // Discover not matched labels
162         Set<String> notFound = new HashSet<String>();
163         String[] values = status.getSearch().split(",");
164         for (String value : values) {
165             if (!labels.contains(value.trim().toUpperCase())) {
166                 notFound.add(value);
167             }
168         }
169         onNotFound(notFound);
170 
171         // Prepare links variables
172         SearchType searchType = status.getType();
173         String varEntity = filterConfig == null ? entity.getId() : "";
174 
175         String varFilter = createVarFilter(entity, filterConfig);
176 
177         // Links
178         accordionBody.add(createLinks(collection, searchType, varEntity, varFilter));
179     }

181     private Component newFieldsBox(String componentId, String labelField, IEntity entity, boolean active) {
182 
183         Component box;
184         if (status.getType().getTemplate() == null || status.getType().getTemplate().isEmpty()) {
185             box = new FieldsPanel(componentId, labelField, entity);
186         } else {
187             box = new Label(componentId, replaceEntityValues(status.getType().getTemplate(), entity)).setEscapeModelStrings(false);
188         }
189 
190         if (active) {
191             box.add(new AttributeAppender("class", " active"));
192         }
193 
194         box.setMarkupId(entity.getId());
195         return box;

199     private static String createVarFilter(IEntity entity, FilterConfig filterConfig) {
200 
201         if (filterConfig != null) {
202             MultipleEntitySelection browserSelection = new MultipleEntitySelection(filterConfig);
203             return "pfc=" + UrlEncoder.QUERY_INSTANCE.encode(browserSelection.toUrlParameter(false, null), "UTF-8");
204         }
205 
206         if (entity != null) {
207             SingleEntitySelection singleEntitySelection = new SingleEntitySelection(entity);
208             return "pf=" + UrlEncoder.QUERY_INSTANCE.encode(singleEntitySelection.toUrlParameter(false, null), "UTF-8");
209         }
210 
211         return "";
212     }

225     private WebMarkupContainer createLinks(ORI collectionORI, SearchType searchType, String varEntity, String varFilter) {
226         WebMarkupContainer linksContainer = new WebMarkupContainer("linksContainer");
227         RepeatingView links = new RepeatingView("links");
228 
229         if (searchType.getLinks() != null) {
230 
231             String prefix = getPage().getPageParameters().get(Website.PARAMETER_CURRENT_PAGE).isEmpty() ? WebsiteApplication.get().getWebPath() + "/" : "";
232 
233             linksContainer.setVisible(false);
234             boolean isList = entityIterator != null || filterConfig != null;
235 
236             for (SearchLink searchLink : searchType.getLinks()) {
237 
238                 String visible = searchLink.getVisible();
239                 if (!Strings.isEmpty(visible)) {
240 
241                     if (isList && !"LIST".equalsIgnoreCase(visible)) {
242                         continue;
243                     }
244 
245                     if (!isList && !"SINGLE".equalsIgnoreCase(visible)) {
246                         continue;
247                     }
248                 }
249 
250                 linksContainer.setVisible(true);
251                 WebMarkupContainer item = new WebMarkupContainer(links.newChildId());
252                 WebMarkupContainer link = new WebMarkupContainer("link");
253                 link.add(new AttributeModifier("href", prefix + createLink(searchLink.getUrl(), searchType.getCollection(), varEntity, varFilter)));
254                 link.add(new Label("label", searchLink.getTitle()).setEscapeModelStrings(false));
255                 item.add(link);
256                 links.add(item);
257             }
258         } else {
259             linksContainer.setVisible(false);
260         }
261 
262         linksContainer.add(links);
263         return linksContainer;
264     }

266     public static String createLink(String url, ORI collection, String varEntity, String varFilter) {
267 
268         String link = url;
269 
270         //TODO Remove deprecated variables
271         link = link.replace("$collection", collection.toString());
272         link = link.replace("$entity", varEntity);
273         link = link.replace("$filter", varFilter);
274 
275         link = link.replace("$[collection]", collection.toString());
276         link = link.replace("$[entity]", varEntity);
277         link = link.replace("$[filter]", varFilter);
278 
279         return link;
280     }

282     private static String replaceEntityValues(String template, IEntity entity) {
283 
284         Collection collection = entity.getCollection();
285 
286         for (Field field : collection.getFields()) {
287             template = template.replaceAll("\\$\\[" + field.getId() + "\\]", String.valueOf(entity.get(field.getId())));
288         }
289 
290         return template;
