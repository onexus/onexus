314     private Iterator<IEntity> getAutocompleteChoices(String in) {
315 
316         int lastComma = in.lastIndexOf(',');
317         String input = lastComma > -1 ? in.substring(lastComma + 1).trim() : in.trim();
318 
319         Query query = new Query();
320         SearchType type = getStatus().getType();
321 
322         ORI collectionUri = getAbsoluteUri(type.getCollection());
323         String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
324         query.setFrom(collectionAlias);
325 
326         List<String> fieldList = type.getFieldsList();
327         query.addSelect(collectionAlias, fieldList);
328 
329         for (String field : fieldList) {
330             QueryUtils.or(query, new Contains(collectionAlias, field, input));
331         }
332         query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
333 
334         query.setCount(10);
335 
336         return new EntityIterator(collectionManager.load(query), collectionUri);
337     }

376     /**
377      * Generic IEntity renderer that show all the fields.
378      */
379     private class EntityRenderer implements IAutoCompleteRenderer<IEntity> {
380 
381         public final void render(final IEntity object, final Response response, final String criteria) {
382             String textValue = getTextValue(object, criteria);
383             if (textValue == null) {
384                 throw new IllegalStateException(
385                         "A call to textValue(Object) returned an illegal value: null for object: " +
386                                 object.toString());
387             }
388             textValue = textValue.replaceAll("\\\"", "&quot;");
389 
390             response.write("<li textvalue=\"" + textValue + "\"");
391             response.write(">");
392             renderChoice(object, response, criteria);
393             response.write("</li>");
394         }
395 
396         private String getTextValue(IEntity object, String in) {
397 
398             int lastComma = in.lastIndexOf(',');
399             String criteria = lastComma > -1 ? in.substring(lastComma + 1).trim() : in.trim();
400             String previous = lastComma > -1 ? in.substring(0, lastComma) + ", " : "";
401 
402 
403             SearchType type = getStatus().getType();
404             List<String> fields = type.getKeysList();
405 
406             for (String field : fields) {
407                 String value = String.valueOf(object.get(field));
408 
409                 if (StringUtils.containsIgnoreCase(value, criteria)) {
410                     return previous + value;
411                 }
412             }
413 
414             return previous + object.get(fields.get(0));
415         }
416 
417         public final void renderHeader(final Response response) {
418             response.write("<ul>");
419         }
420 
421         public final void renderFooter(final Response response, int count) {
422             response.write("</ul>");
423         }
424 
425         protected void renderChoice(IEntity object, Response response, String criteria) {
426 
427             SearchType type = getStatus().getType();
428             List<String> keys = type.getKeysList();
429             List<String> fields = type.getFieldsList();
430 
431 
432             boolean keyFieldAdded = false;
433             for (String field : fields) {
434                 String value = String.valueOf(object.get(field));
435 
436                 if (StringUtils.containsIgnoreCase(value, criteria)) {
437                     if (keys.contains(field)) {
438                         keyFieldAdded = true;
439                     }
440 
441                     renderValue(response, value, criteria, field);
442 
443                 }
444             }
445 
446             if (!keyFieldAdded) {
447                 String field = keys.get(0);
448                 String value = String.valueOf(object.get(field));
449                 renderValue(response, value, criteria, field);
450             }
451         }
452 
453         private void renderValue(Response response, String value, String criteria, String field) {
454 
455             String hlValue = value.replaceAll("(?i)(" + criteria + ")", "<strong>$1</strong>");
456             response.write("<span class='f'>" + field.toLowerCase() + ":</span>" + hlValue);
457             response.write("<br />");
458         }

464     private final class ExamplesModel extends AbstractReadOnlyModel<List<SearchExample>> {
465 
466         private IModel<SearchType> model;
467 
468         private ExamplesModel(IModel<SearchType> model) {
469             this.model = model;
470         }
471 
472         @Override
473         public List<SearchExample> getObject() {
474 
475             SearchType searchType = model.getObject();
476 
477             if (searchType == null || searchType.getExamples() == null) {
478                 return Collections.EMPTY_LIST;
479             }
480 
481             List<SearchExample> values = new ArrayList<SearchExample>();
482             for (String value : searchType.getExamples().split(",")) {
483 
484                 String label = value.trim();
485                 String query = label;
486 
487                 if (label.contains("|")) {
488                     String[] split = label.split("\\|");
489                     label = split[0].trim();
490                     query = split[1].replace(';', ',');
491                 }
492 
493                 values.add(new SearchExample(label, query));
494             }
495             return values;
496         }
497     }
