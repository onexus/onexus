25  import org.apache.wicket.markup.html.form.FormComponent;
26  import org.apache.wicket.markup.html.panel.FeedbackPanel;
27  import org.apache.wicket.markup.html.panel.Panel;
28  import org.apache.wicket.model.CompoundPropertyModel;

30  import org.apache.wicket.model.Model;
31  import org.onexus.collection.api.query.Filter;
32  import org.onexus.website.api.pages.browser.filters.operations.FilterOperation;
33  import org.onexus.website.api.widgets.selection.FilterConfig;
34  import org.onexus.website.api.widgets.tableviewer.headers.FieldHeader;

36  import java.util.List;

38  public abstract class AbstractFilterPanel<T> extends Panel {
39  
40      private IModel<FilterOption> option;
41      private FieldHeader header;
42      private FilterOption<T> defaultOption;
43      private List<FilterOperation> operations;
44  
45      public AbstractFilterPanel(String id, FieldHeader fieldHeader, FilterOption<T> defaultOption, List<FilterOperation> operations) {
46          super(id);
47  
48          this.header = fieldHeader;
49          this.option = new CompoundPropertyModel<FilterOption>(new Model<FilterOption>(defaultOption));
50          this.defaultOption = defaultOption;
51          this.operations = operations;
52      }
53  
54      @Override
55      protected void onInitialize() {
56          super.onInitialize();
57  
58          Form<FilterOption> form = new Form("form", option);
59  
60          // Feedback panel
61          final FeedbackPanel feedbackPanel = new FeedbackPanel("feedback");
62          feedbackPanel.setOutputMarkupId(true);
63          form.add(feedbackPanel);
64  
65          // Value component
66          final FormComponent<T> valueComponent = createValueFormComponent("value");
67          valueComponent.setOutputMarkupPlaceholderTag(true);
68          valueComponent.setVisible(defaultOption.getOperation().isNeedsValue());
69          form.add(valueComponent);
70  
71          // Operations drop down
72          final DropDownChoice<FilterOperation> operationsDropDown = new DropDownChoice<FilterOperation>("operation", operations);
73          operationsDropDown.setNullValid(false);
74          operationsDropDown.add(new OnChangeAjaxBehavior() {
75              @Override
76              protected void onUpdate(AjaxRequestTarget target) {
77                  valueComponent.setVisible(option.getObject().getOperation().isNeedsValue());
78                  target.add(valueComponent);
79              }
80          });
81          form.add(operationsDropDown);
82  
83          // Create filter link
84          form.add(new AjaxSubmitLink("submit") {
85              @Override
86              protected void onSubmit(AjaxRequestTarget target, Form<?> form) {
87  
88                  FilterOption<T> fo = option.getObject();
89                  T value = fo.getValue();
90  
91                  FilterOperation operation = fo.getOperation();
92                  String title = operation.createTitle(header.getLabel(), value);
93                  FilterConfig fc = new FilterConfig(title);
94  
95                  fc.setCollection(header.getCollection().getORI());
96                  fc.setDefine("fc='" + fc.getCollection() + "'");
97  
98  
99                  Filter where = operation.createFilter("fc", header.getField().getId(), value);
100                 fc.setWhere(where.toString());
101 
102                 addFilter(target, fc);
103             }
104         });
105 
106         add(form);
107 
108     }
109 
110     protected abstract FormComponent<T> createValueFormComponent(String componentId);
111 
112     protected abstract void addFilter(AjaxRequestTarget target, FilterConfig filterConfig);
113 
114 }
