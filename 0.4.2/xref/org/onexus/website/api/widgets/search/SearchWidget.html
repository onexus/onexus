18  package org.onexus.website.api.widgets.search;
19  
20  import org.apache.commons.lang3.StringUtils;
21  import org.apache.wicket.ajax.AjaxRequestTarget;
22  import org.apache.wicket.ajax.form.OnChangeAjaxBehavior;
23  import org.apache.wicket.ajax.markup.html.form.AjaxButton;
24  import org.apache.wicket.markup.html.form.Form;
25  import org.apache.wicket.markup.html.form.TextField;
26  import org.apache.wicket.model.CompoundPropertyModel;
27  import org.apache.wicket.model.IModel;
28  import org.onexus.website.api.events.EventFiltersUpdate;
29  import org.onexus.website.api.widgets.Widget;
30  
31  public class SearchWidget extends Widget<SearchWidgetConfig, SearchWidgetStatus> {
32  
33      public SearchWidget(String componentId, IModel<SearchWidgetStatus> statusModel) {
34          super(componentId, statusModel);
35  
36          Form<SearchWidgetStatus> form = new Form<SearchWidgetStatus>("toolsForms", new CompoundPropertyModel<SearchWidgetStatus>((IModel<SearchWidgetStatus>) statusModel));
37  
38          // Search field & button
39          TextField<String> searchField = new TextField<String>("search");
40          searchField.add(new OnChangeAjaxBehavior() {
41              @Override
42              protected void onUpdate(AjaxRequestTarget target) {
43  
44                  String search = SearchWidget.this.getStatus().getSearch();
45  
46                  if (StringUtils.isEmpty(search)) {
47                      sendEvent(EventFiltersUpdate.EVENT);
48                  }
49  
50              }
51          });
52  
53          form.add(searchField);
54  
55          form.add(new AjaxButton("searchButton") {
56  
57              @Override
58              protected void onSubmit(AjaxRequestTarget target, Form<?> form) {
59                  sendEvent(EventFiltersUpdate.EVENT);
60              }
61  
62              @Override
63              protected void onError(AjaxRequestTarget target, Form<?> form) {
64  
65              }
66          });
67  
68          add(form);
69  
70      }
71  
72  }
