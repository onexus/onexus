42  public class LinkDecorator extends FieldDecorator {
43  
44      private Map<ParameterKey, String> parameters;
45  
46      public LinkDecorator(Field field, Map<ParameterKey, String> parameters) {
47          super(field);
48          this.parameters = parameters;
49      }
50  
51      @Override
52      public void populateCell(WebMarkupContainer cellContainer, String componentId, IModel<IEntity> entityModel) {
53  
54          IEntity entity = entityModel.getObject();
55          Object description = getValue(entity);
56  
57          if (description != null) {
58  
59              String currentColumnValue = String.valueOf(entity.get(getField().getId()));
60  
61              List<String> columnValues = new ArrayList<String>();
62              if (parameters.containsKey(LinkDecoratorParameters.SEPARATOR)) {
63                  for (String value : currentColumnValue.split(parameters.get(LinkDecoratorParameters.SEPARATOR).trim())) {
64                      columnValues.add(value.trim());
65                  }
66              } else {
67                  columnValues.add(getFormatValue(entity));
68              }
69  
70              StringBuilder content = new StringBuilder();
71              Iterator<String> columnValueIt = columnValues.iterator();
72  
73              while (columnValueIt.hasNext()) {
74                  String columnValue = columnValueIt.next();
75  
76                  String href = parameters.get(LinkDecoratorParameters.URL);
77  
78                  href = replaceParameters(getField(), columnValue, entity, href, false);
79                  href = fixLinkUrl(href);
80  
81                  content.append("<a href=\"").append(href).append("\"");
82                  if (parameters.containsKey(LinkDecoratorParameters.TARGET)) {
83                      content.append(" target=\"").append(parameters.get(LinkDecoratorParameters.TARGET)).append("\"");
84                  }
85  
86                  if (parameters.containsKey(LinkDecoratorParameters.LENGTH)) {
87                      columnValue = StringUtils.abbreviate(columnValue, Integer.valueOf(parameters.get(LinkDecoratorParameters.LENGTH)));
88                  }
89  
90                  if (parameters.containsKey(LinkDecoratorParameters.ICON)) {
91  
92                      String iconTitle = parameters.get(LinkDecoratorParameters.ICON_TITLE);
93                      if (iconTitle == null) {
94                          iconTitle = "Click to see more details about this value";
95                      }
96                      content.append(" rel=\"tooltip\" title=\"").append(iconTitle).append("\"><i class=\"");
97                      content.append(parameters.get(LinkDecoratorParameters.ICON)).append("\"></i></a>&nbsp;");
98                      content.append(columnValue);
99                  } else {
100                     content.append(">").append(columnValue).append("</a>");
101                 }
102 
103                 if (columnValueIt.hasNext()) {
104                     content.append(", ");
105                 }
106             }
107 
108             cellContainer.add(new Label(componentId, content.toString()).setEscapeModelStrings(false));
109 
110         } else {
111             cellContainer.add(new EmptyPanel(componentId));
112         }
113 
114         cellContainer.add(new AttributeModifier("title", new Model<String>(description == null ? "No data" : description.toString())));
115     }
116 
117     @Override
118     public List<String> getExtraFields(Collection collection) {
119 
120         String href = parameters.get(LinkDecoratorParameters.URL);
121 
122         if (href != null) {
123             return extractFields(href);
124         }
125 
126         return Collections.EMPTY_LIST;
127     }
