146     public WebsiteStatus newStatus() {
147

149 
150         if (status != null) {
151             status = SerializationUtils.clone(status);
152         } else {
153             status = createEmptyStatus();
154         }
155 
156         status.setConfig(this);
157 
158         // Add page status
159         if (getPages() != null) {
160             List<PageStatus> pageStatuses = new ArrayList<PageStatus>();
161             for (PageConfig page : getPages()) {
162                 page.setWebsiteConfig(this);
163                 pageStatuses.add(page.newStatus());

166         }
