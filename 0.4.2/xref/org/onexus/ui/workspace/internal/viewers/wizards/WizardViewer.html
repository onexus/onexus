18  package org.onexus.ui.workspace.internal.viewers.wizards;

20  import org.apache.wicket.Component;
21  import org.apache.wicket.ajax.AjaxRequestTarget;
22  import org.apache.wicket.ajax.markup.html.AjaxLink;
23  import org.apache.wicket.markup.html.WebMarkupContainer;
24  import org.apache.wicket.markup.html.basic.Label;
25  import org.apache.wicket.markup.html.list.ListItem;
26  import org.apache.wicket.markup.html.list.ListView;
27  import org.apache.wicket.markup.html.panel.EmptyPanel;
28  import org.apache.wicket.markup.html.panel.Panel;
29  import org.apache.wicket.model.AbstractReadOnlyModel;
30  import org.apache.wicket.model.IModel;
31  import org.onexus.resource.api.Resource;
32  import org.onexus.ui.api.utils.panels.HelpMark;
33  import org.onexus.ui.api.wizards.IWizardCreator;
34  import org.onexus.ui.api.wizards.IWizardsManager;
35  
36  import javax.inject.Inject;
37  import javax.inject.Named;
38  import java.util.List;
39  
40  public class WizardViewer extends Panel {
41  
42      @Inject
43      private IWizardsManager wizardsManager;
44  
45      private IModel<? extends Resource> resourceModel;
46      private WebMarkupContainer wizardContainer;
47  
48      public WizardViewer(String id, IModel<? extends Resource> model) {
49          super(id);
50          this.resourceModel = model;
51  
52          add(new ListView<IWizardCreator>("wizardList", new WizardCreatorsModel()) {
53  
54              @Override
55              protected void populateItem(final ListItem<IWizardCreator> item) {
56  
57                  AjaxLink<String> link = new AjaxLink<String>("link") {
58  
59                      @Override
60                      public void onClick(AjaxRequestTarget target) {
61                          IWizardCreator wizard = item.getModelObject();
62                          wizardContainer.addOrReplace(wizard.getPanel("wizard", resourceModel));
63                          target.add(wizardContainer);
64                      }
65                  };
66  
67                  IWizardCreator wizard = item.getModelObject();
68                  String label = wizard.getLabel();
69                  String title = wizard.getTitle();
70                  String description = wizard.getDescription();
71  
72                  link.add(new Label("label", label));
73                  item.add(link);
74  
75                  item.add(new Label("title", title));
76  
77                  if (description != null) {
78                      item.add(new HelpMark("description", title, description));
79                  } else {
80                      item.add(new EmptyPanel("description"));
81                  }
82  
83              }
84          });
85  
86          wizardContainer = new WebMarkupContainer("wizardContainer") {
87  
88              @Override
89              public boolean isVisible() {
90                  Component wizard = get("wizard");
91  
92                  if (wizard == null || !wizard.isVisible()) {
93                      return false;
94                  }
95  
96                  return true;
97              }
98  
99          };
100         wizardContainer.setOutputMarkupPlaceholderTag(true);
101         wizardContainer.add(new EmptyPanel("wizard").setVisible(false));
102         add(wizardContainer);
103     }
104 
105     private class WizardCreatorsModel extends AbstractReadOnlyModel<List<? extends IWizardCreator>> {
106 
107         @Override
108         public List<? extends IWizardCreator> getObject() {
109             return wizardsManager.getWizardCreators(resourceModel.getObject());
110         }
111 
112     }
113 
114 }
