18  package org.onexus.ui.authentication.jaas;
19  
20  import org.apache.wicket.Session;
21  import org.apache.wicket.markup.head.CssHeaderItem;
22  import org.apache.wicket.markup.head.IHeaderResponse;
23  import org.apache.wicket.markup.html.WebPage;
24  import org.apache.wicket.request.flow.RedirectToUrlException;
25  import org.apache.wicket.request.mapper.parameter.PageParameters;
26  import org.apache.wicket.request.resource.CssResourceReference;
27  import org.apache.wicket.util.string.StringValue;
28  import org.onexus.resource.api.session.IAuthenticatedSession;
29  
30  public class JaasSignInPage extends WebPage {
31      public static final CssResourceReference CSS = new CssResourceReference(JaasSignInPage.class, "JaasSignInPage.css");
32      private static final long serialVersionUID = 1L;
33  
34      /**
35       * Construct
36       */
37      public JaasSignInPage() {
38          this(null);
39      }
40  
41      /**
42       * Constructor
43       *
44       * @param parameters The page parameters
45       */
46      public JaasSignInPage(final PageParameters parameters) {
47          super(parameters);
48  
49          if (parameters != null) {
50              StringValue userName = parameters.get("username");
51              StringValue password = parameters.get("password");
52              StringValue redirect = parameters.get("redirect");
53  
54              if (!userName.isEmpty()) {
55                  boolean valid = getAuthenticatedSession().authenticate(userName.toString(), password.toString());
56  
57                  if (valid) {
58                      throw new RedirectToUrlException(redirect.toString());
59                  }
60              }
61          }
62  
63          SignInPanel signPanel = new SignInPanel("signInPanel");
64          signPanel.setRememberMe(false);
65          add(signPanel);
66  
67      }
68  
69      @Override
70      public void renderHead(IHeaderResponse response) {
71          response.render(CssHeaderItem.forReference(CSS));
72      }
73  
74      public static IAuthenticatedSession getAuthenticatedSession() {
75          return (IAuthenticatedSession) Session.get();
76      }
77  }
