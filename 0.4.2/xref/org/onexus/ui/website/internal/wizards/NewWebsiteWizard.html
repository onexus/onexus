167     private final class MainTabCollection extends WizardStep {
168 
169         public MainTabCollection() {
170             super("New website", "Main tab collection");
171 
172             List<ORI> projectCollections = new ArrayList<ORI>();
173             addAllCollections(projectCollections, getParentUri());
174             add(new DropDownChoice<ORI>("mainTabCollection", projectCollections));
175 
176         }
177 
178         @Override
179         public void applyState() {
180 
181             BrowserPageConfig browserPage = getBrowserPage();
182             ViewConfig view = new ViewConfig();
183             view.setTitle("default");
184 
185             String tableWidgetId = currentTab.getId() + "-table";
186             String searchWidgetId = currentTab.getId() + "-search";
187             view.setMain(tableWidgetId);
188             view.setTop(searchWidgetId);
189             view.setTopRight("download, share");
190 
191             // Create search widget
192             SearchWidgetConfig searchWidget = new SearchWidgetConfig(searchWidgetId);
193             List<SearchField> searchFields = new ArrayList<SearchField>();
194             searchWidget.setFields(searchFields);
195             browserPage.getWidgets().add(searchWidget);
196 
197             // Create table widget
198             TableViewerConfig tableWidget = new TableViewerConfig(tableWidgetId, mainTabCollection);
199             tableWidget.setColumnSets(new ArrayList<ColumnSet>());
200 
201             // Main collection columns
202             Collection collection = resourceManager.load(Collection.class, mainTabCollection.toAbsolute(getParentUri()));
203 
204             List<String> keyFields = new ArrayList<String>();
205             List<String> fields = new ArrayList<String>();
206             List<String> search = new ArrayList<String>();
207 
208             for (Field field : collection.getFields()) {
209 
210                 if (field.isPrimaryKey() != null && field.isPrimaryKey()) {
211                     keyFields.add(field.getId());
212                 } else {
213                     fields.add(field.getId());
214                 }
215 
216                 if (String.class.equals(field.getType())) {
217                     search.add(field.getId());
218                 }
219             }
220 
221             searchFields.add(new SearchField(mainTabCollection, StringUtils.join(search, ',')));
222 
223             ColumnSet columnSet = new ColumnSet();
224             columnSet.setTitle("default");
225             columnSet.setColumns(new ArrayList<IColumnConfig>());
226 
227             if (!keyFields.isEmpty()) {
228                 columnSet.getColumns().add(new ColumnConfig(mainTabCollection, StringUtils.join(keyFields, ','), "LINK"));
229             }
230 
231             if (!fields.isEmpty()) {
232                 columnSet.getColumns().add(new ColumnConfig(mainTabCollection, StringUtils.join(fields, ',')));
233             }
234 
235             tableWidget.getColumnSets().add(columnSet);
236             browserPage.getWidgets().add(tableWidget);
237 
238             // Add view
239             currentTab.setViews(new ArrayList<ViewConfig>());
240             currentTab.getViews().add(view);
241 
242             // Add current tab
243             browserPage.getTabs().add(currentTab);
244 
245         }

248     private final class LinkedCollections extends WizardStep {
249 
250         private List<ORI> linkedCollections;
251 
252         public LinkedCollections() {
253             super("New website", "Linked collections");
254 
255         }
256 
257         @Override
258         protected void onBeforeRender() {
259 
260             List<ORI> projectCollections = new ArrayList<ORI>();
261             addAllCollections(projectCollections, getParentUri());
262 
263             linkedCollections = new ArrayList<ORI>();
264 
265 
266             Collection colA = resourceManager.load(Collection.class, getMainTabCollection().toAbsolute(getParentUri()));
267 
268             for (ORI collection : projectCollections) {
269 
270 
271                 Collection colB = resourceManager.load(Collection.class, collection.toAbsolute(getParentUri()));
272 
273                 if (colA.getORI().equals(colB.getORI())) {
274                     continue;
275                 }
276 
277                 List<FieldLink> links = LinkUtils.getLinkFields(getParentUri(), colA, colB);
278 
279                 if (!links.isEmpty()) {
280                     linkedCollections.add(collection);
281                 }
282             }
283 
284             addOrReplace(new ListMultipleChoice<ORI>("linkCollections", linkedCollections));
285 
286             super.onBeforeRender();
287         }
288 
289         @Override
290         public void applyState() {
291 
292             BrowserPageConfig browserPage = getBrowserPage();
293 
294             if (!linkCollections.isEmpty()) {
295 
296                 String tableWidgetId = currentTab.getId() + "-table";
297                 TableViewerConfig tableWidget = (TableViewerConfig) browserPage.getWidget(tableWidgetId);
298                 ColumnSet columnSet = tableWidget.getColumnSets().get(0);
299 
300                 for (ORI linkCollection : linkCollections) {
301 
302                     ORI resourceUri = linkCollection.toAbsolute(getParentUri());
303                     Collection collection = resourceManager.load(Collection.class, resourceUri);
304 
305                     List<String> otherFields = new ArrayList<String>();
306 
307                     for (Field field : collection.getFields()) {
308                         otherFields.add(field.getId());
309                     }
310 
311                     if (!otherFields.isEmpty()) {
312                         columnSet.getColumns().add(new ColumnConfig(linkCollection, StringUtils.join(otherFields, ',')));
313                     }
314                 }
315             }
316 
317             // Prepare next tab iteration
318             tabName = "Tab-" + browserPage.getTabs().size();
319             mainTabCollection = null;
320             linkCollections.clear();
321             currentTab = null;
322             WizardModel model = (WizardModel) getWizardModel();
323             IWizardStep firstStep = model.stepIterator().next();
324             model.setActiveStep(firstStep);
325         }

353     private void addAllCollections(List<ORI> collectionUris, ORI parentUri) {
354 
355         List<Collection> collections = resourceManager.loadChildren(Collection.class, parentUri);
356         for (Collection collection : collections) {
357             collectionUris.add(collection.getORI());
358         }
359 
360         List<Folder> folders = resourceManager.loadChildren(Folder.class, parentUri);
361         for (Folder folder : folders) {
362             addAllCollections(collectionUris, folder.getORI());
363         }
