18  package org.onexus.resource.api.utils.string;
19  
20  
21  import java.io.Serializable;
22  
23  /**
24   * Original code extract from Apache Wicket core
25   */
26  public abstract class VariableInterpolator implements Serializable
27  {
28      /** The <code>String</code> to interpolate into */
29      protected final String string;
30  
31      private final boolean exceptionOnNullVarValue;
32  
33      /**
34       * Constructor.
35       *
36       * @param string
37       *            a <code>String</code> to interpolate with variable values
38       */
39      public VariableInterpolator(final String string)
40      {
41          this(string, false);
42      }
43  
44      /**
45       * Constructor.
46       *
47       * @param string
48       *            a <code>String</code> to interpolate with variable values
49       * @param exceptionOnNullVarValue
50       *            if <code>true</code> an {@link IllegalStateException} will be thrown if
51       *            {@link #getValue(String)} returns <code>null</code>, otherwise the
52       *            <code>${varname}</code> string will be left in the <code>String</code> so that
53       *            multiple interpolators can be chained
54       */
55      public VariableInterpolator(final String string, final boolean exceptionOnNullVarValue)
56      {
57          this.string = string;
58          this.exceptionOnNullVarValue = exceptionOnNullVarValue;
59      }
60  
61      /**
62       * Retrieves a value for a variable name during interpolation.
63       *
64       * @param variableName
65       *            a variable name
66       * @return the value
67       */
68      protected abstract String getValue(String variableName);
69  
70      private int lowerPositive(final int i1, final int i2)
71      {
72          if (i2 < 0)
73          {
74              return i1;
75          }
76          else if (i1 < 0)
77          {
78              return i2;
79          }
80          else
81          {
82              return i1 < i2 ? i1 : i2;
83          }
84      }
85  
86      /**
87       * Interpolates using variables.
88       *
89       * @return the interpolated <code>String</code>
90       */
91      @Override
92      public String toString()
93      {
94  
95          if (string == null)
96          {
97              return null;
98          }
99  
100         // If there's any reason to go to the expense of property expressions
101         if (!string.contains("${"))
102         {
103             return string;
104         }
105 
106         // Result buffer
107         final StringBuilder buffer = new StringBuilder();
108 
109         // For each occurrences of "${"or "$$"
110         int start;
111         int pos = 0;
112 
113         while ((start = lowerPositive(string.indexOf("$$", pos), string.indexOf("${", pos))) != -1)
114         {
115             // Append text before possible variable
116             buffer.append(string.substring(pos, start));
117 
118             if (string.charAt(start + 1) == '$')
119             {
120                 buffer.append("$");
121                 pos = start + 2;
122                 continue;
123             }
124 
125 
126             // Position is now where we found the "${"
127             pos = start;
128 
129             // Get start and end of variable name
130             final int startVariableName = start + 2;
131             final int endVariableName = string.indexOf('}', startVariableName);
132 
133             // Found a close brace?
134             if (endVariableName != -1)
135             {
136                 // Get variable name inside brackets
137                 final String variableName = string.substring(startVariableName, endVariableName);
138 
139                 // Get value of variable
140                 final String value = getValue(variableName);
141 
142                 // If there's no value
143                 if (value == null)
144                 {
145                     if (exceptionOnNullVarValue)
146                     {
147                         throw new IllegalArgumentException("Value of variable [[" + variableName +
148                                 "]] could not be resolved while interpolating [[" + string + "]]");
149                     }
150                     else
151                     {
152                         // Leave variable uninterpolated, allowing multiple
153                         // interpolators to
154                         // do their work on the same string
155                         buffer.append("${").append(variableName).append("}");
156                     }
157                 }
158                 else
159                 {
160                     // Append variable value
161                     buffer.append(value);
162                 }
163 
164                 // Move past variable
165                 pos = endVariableName + 1;
166             }
167             else
168             {
169                 break;
170             }
171         }
172 
173         // Append anything that might be left
174         if (pos < string.length())
175         {
176             buffer.append(string.substring(pos));
177         }
178 
179         // Convert result to String
180         return buffer.toString();
181     }
