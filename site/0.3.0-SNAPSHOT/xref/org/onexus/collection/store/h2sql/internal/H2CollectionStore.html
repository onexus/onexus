41  public class H2CollectionStore extends SqlCollectionStore {
42  
43      private static final Logger log = LoggerFactory.getLogger(H2CollectionStore.class);
44      private String database = "onexus-h2-database;LOG=0;CACHE_SIZE=65536;LOCK_MODE=0"; // ;TRACE_LEVEL_FILE=3;TRACE_LEVEL_SYSTEM_OUT=3";
45  
46      private String username = "sa";

52      public H2CollectionStore() {
53          super(new H2Dialect());
54      }
55  
56      @Override
57      public SqlQuery newSqlQuery(Query query) {
58          return new H2Query(this, query);
59      }
60  
61      protected DataSource newDataSource() {
62  
63          Driver.load();
64  
65          // Initialize the DataSource with a connection pool
66          ConnectionFactory connectionFactory = new H2ConnectionFactory();
67          ObjectPool connectionPool = new GenericObjectPool(null, GenericObjectPool.DEFAULT_MAX_ACTIVE,
68                  GenericObjectPool.WHEN_EXHAUSTED_GROW, GenericObjectPool.DEFAULT_MAX_WAIT);
69          @SuppressWarnings("unused")
70          PoolableConnectionFactory poolableConnectionFactory = new PoolableConnectionFactory(connectionFactory,
71                  connectionPool, null, null, false, true);
72          return new PoolingDataSource(connectionPool);
73      }
74  
75      public void stop(ICollectionStore store, @SuppressWarnings("rawtypes") Map properties) {
76          try {
77              if (getDataSource() != null) {
78                  Connection conn = getDataSource().getConnection();
79                  Statement stat = conn.createStatement();
80  
81                  stat.execute("SHUTDOWN");
82                  stat.close();
83              }
84  
85          } catch (SQLException e) {
86              log.error("At H2CollectionStore close", e);
87          }
88  
89      }
90  
91      public class H2ConnectionFactory implements ConnectionFactory {
92  
93          @Override
94          public Connection createConnection() throws SQLException {
95              return DriverManager.getConnection(
96                      "jdbc:h2:" + database,
97                      username,
98                      password);
99          }
100     }
