18  package org.onexus.ui.website.internal.workspace;

20  import org.apache.wicket.AttributeModifier;
21  import org.apache.wicket.markup.html.WebMarkupContainer;
22  import org.apache.wicket.markup.html.panel.Panel;
23  import org.apache.wicket.model.IModel;
24  import org.onexus.resource.api.IResourceManager;
25  import org.onexus.resource.api.Project;
26  import org.onexus.resource.api.Resource;
27  import org.onexus.website.api.IWebsiteService;
28  import org.onexus.website.api.WebsiteApplication;
29  import org.onexus.website.api.WebsiteConfig;
30  import org.ops4j.pax.wicket.api.PaxWicketBean;
31  
32  public class WebsiteViewer extends Panel {
33  
34      @PaxWicketBean(name = "websiteService")
35      private IWebsiteService websiteService;
36  
37      @PaxWicketBean(name = "resourceManager")
38      private IResourceManager resourceManager;
39  
40      public WebsiteViewer(String id, final IModel<? extends Resource> model) {
41          super(id);
42  
43          WebsiteConfig website = (WebsiteConfig) model.getObject();
44          Project project = resourceManager.getProject(website.getORI().getProjectUrl());
45  
46          String path = '/' + websiteService.getMount() + '/' + project.getName() + '/' + website.getName() + '/';
47          String src = WebsiteApplication.toAbsolutePath(path);
48  
49          WebMarkupContainer iframe = new WebMarkupContainer("browser");
50          iframe.add(new AttributeModifier("src", src));
51  
52          add(iframe);
53      }
54  
55  }
