89                  ORI resourceUri = resource.getORI();
90                  getResourceManager().syncProject(resourceUri.getProjectUrl());
91  
92                  Resource newVersion = getResourceManager().load(Resource.class, resourceUri);

172         public ToolOriBreadCrumb(String id, IModel<Resource> resource) {
173             super(id, resource);
174 
175             add(new AjaxUpdateOnEvent(EventResourceSelect.EVENT));
176 
177             // Add list items
178             add(new ListView<String>("ori", new BreadCrumbItemsModel(resource)) {
179 
180                 private transient List<String> links;
181 
182                 @Override
183                 protected void onBeforeRender() {
184 
185                     BreadCrumbItemsModel model = (BreadCrumbItemsModel) getModel();
186 
187                     List<String> oriItems = model.getObject();
188 
189                     if (oriItems == null) {
190                         links = Collections.EMPTY_LIST;
191                     }
192 
193                     links = new ArrayList<String>(oriItems.size());
194                     Resource resource = model.getWrappedModel().getObject();
195                     String projectUri = (resource == null ? null : resource.getORI().getProjectUrl().toString());
196 
197                     links.add(projectUri);
198                     for (int i = 1; i < oriItems.size(); i++) {
199                         StringBuilder link = new StringBuilder();
200                         link.append(projectUri).append("?");
201                         link.append(oriItems.get(1));
202                         for (int t = 2; t <= i; t++) {
203                             link.append(ORI.SEPARATOR);
204                             link.append(oriItems.get(t));
205                         }
206                         links.add(link.toString());
207 
208                     }
209 
210                     super.onBeforeRender();
211                 }
212 
213                 @Override
214                 protected void populateItem(ListItem<String> item) {
215 
216                     // Link
217                     PageParameters parameters = new PageParameters();
218                     String resourceUri = links.get(item.getIndex());
219                     parameters.set(ResourcesPage.PARAMETER_RESOURCE, resourceUri);
220                     Link<String> link = new BookmarkablePageLink<String>("link", ResourcesPage.class, parameters);
221                     link.setEnabled(resourceUri != null);
222 
223                     // Label
224                     link.add(new Label("name", item.getModelObject()));
225 
226                     // Divider
227                     boolean showDivider = !(getModelObject().size() - 1 == item.getIndex());
228                     Label divider = new Label("divider", (item.getIndex() == 0 ? "?" : "/"));
229                     item.add(divider.setVisible(showDivider));
230                     item.add(link);
231 
232                 }
233             });
234 
235         }

296     private class ViewerCreatorsModel extends AbstractWrapModel<List<IViewerCreator>> {
297 
298         private IModel<Resource> resourceModel;
299 
300         private ViewerCreatorsModel(IModel<Resource> resourceModel) {
301             this.resourceModel = resourceModel;
302         }
303 
304         @Override
305         public List<IViewerCreator> getObject() {
306 
307             Resource resource = resourceModel.getObject();
308 
309             if (resource == null) {
310                 return Collections.EMPTY_LIST;
311             }
312 
313             return viewersManager.getViewerCreators(resource);
314         }
315 
316         @Override
317         public IModel<?> getWrappedModel() {
318             return resourceModel;
319         }
320 
321         public IViewerCreator getViewerCreatorAt(int pos) {
322             List<IViewerCreator> viewerCreators = getObject();
323 
324             if (viewerCreators == null || viewerCreators.isEmpty()) {
325                 return new EmptyPanelViewerCreator();
326             }
327 
328             if (viewerCreators.size() <= pos) {
329                 return viewerCreators.get(0);
330             }
331 
332             return viewerCreators.get(pos);
333         }

336     private static class BreadCrumbItemsModel extends AbstractWrapModel<List<String>> {
337 
338         private IModel<Resource> resource;
339         private ORI lastResourceUri;
340         private transient List<String> object;
341 
342         private BreadCrumbItemsModel(IModel<Resource> resource) {
343             super();
344             this.resource = resource;
345         }
346 
347         @Override
348         public List<String> getObject() {
349 
350 
351             if (resource.getObject() == null) {
352                 return Collections.EMPTY_LIST;
353             }
354 
355             if (!resource.getObject().getORI().equals(lastResourceUri)) {
356                 this.object = null;
357             }
358 
359             if (object == null) {
360 
361                 Resource resource = this.resource.getObject();
362                 lastResourceUri = resource.getORI();
363 
364                 if (resource == null || resource.getORI() == null) {
365                     return Collections.EMPTY_LIST;
366                 }
367 
368                 ORI ori = resource.getORI();
369 
370                 object = new ArrayList<String>();
371 
372                 ORI resourceOri = resource.getORI();
373                 object.add(resourceOri.getProjectUrl());
374 
375                 if (!Strings.isEmpty(ori.getPath())) {
376                     // Remove the first separator
377                     String path = resourceOri.getPath().substring(1);
378                     object.addAll(Arrays.asList(path.split(Pattern.quote(String.valueOf(ORI.SEPARATOR)))));
379                 }
380 
381             }
382 
383             return object;
384         }
385 
386         @Override
387         public void detach() {
388             this.object = null;
389         }
390 
391         @Override
392         public IModel<Resource> getWrappedModel() {
393             return resource;
394         }
