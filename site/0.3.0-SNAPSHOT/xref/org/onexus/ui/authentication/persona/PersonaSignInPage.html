33              @Override

35                  BrowserId browserId = SessionHelper.getBrowserId(Session.get());
36  
37                  if (browserId != null) {
38                      if (AuthenticatedWebSession.get().signIn(browserId.getEmail(), null)) {
39  
40                          // logon successful. Continue to the original destination
41                          continueToOriginalDestination();
42                          // Ups, no original destination. Go to the home page
43                          throw new RestartResponseException(getSession().getPageFactory().newPage(
44                                  getApplication().getHomePage()));
45  
46                      } else {
47                          onFailure(target, "You are not authorized");
48                      }
49                  }
50              }
51  
52              @Override
53              protected void onFailure(AjaxRequestTarget target, final String failureReason) {
54                  error("The authentication failed: " + failureReason);
55                  target.addChildren(getPage(), IFeedback.class);
56              }
57  
58          });
59
