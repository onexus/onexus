137     @Override
138     public void addColumns(List<IColumn<IEntityTable, String>> columns, ORI parentURI) {
139 
140         ORI collectionURI = collection.toAbsolute(parentURI);
141         Collection collection = getResourceManager().load(Collection.class, collectionURI);
142 
143         if (collection != null) {
144             List<Field> fields = getFields(collection, parentURI);
145 
146             if (Strings.isEmpty(template)) {
147                 for (Field field : fields) {
148                     IDecorator decoratorImpl = getDecoratorManager().getDecorator(decorator, collection, field);
149                     List<IDecorator> actionsImpl = createActions(collection, field);
150 
151                     columns.add(new CollectionColumn(collectionURI, new FieldHeader(label, title, collection, field, new CollectionHeader(collection)), decoratorImpl, actionsImpl));
152                 }
153             } else {
154                 Field field = fields.get(0);
155 
156                 IDecorator decoratorImpl = getDecoratorManager().getDecorator(decorator, collection, field);
157                 List<IDecorator> actionsImpl = createActions(collection, field);
158                 decoratorImpl.setTemplate(template);
159 
160                 columns.add(new CollectionColumn(collectionURI, new FieldHeader(label, title, collection, field, new CollectionHeader(collection)), decoratorImpl, actionsImpl));
161             }
162         }
163     }

165     private List<IDecorator> createActions(Collection collection, Field field) {
166 
167         if (actions == null) {
168             return Collections.emptyList();
169         }
170 
171         List<IDecorator> actionsImpl = new ArrayList<IDecorator>();
172         for (String action : actions.split(",")) {
173             IDecorator actionImpl = getDecoratorManager().getDecorator(action.trim(), collection, field);
174             if (actionImpl != null) {
175                 actionsImpl.add(actionImpl);
176             }
177         }
178 
179         return actionsImpl;
180 
181     }

183     protected List<Field> getFields(Collection collection, ORI parentOri) {
184         List<Field> fields;
185 
186         if (this.fields == null) {
187             fields = new ArrayList<Field>();
188             for (Field field : collection.getFields()) {
189                 fields.add(field);
190             }
191         } else {
192             fields = new ArrayList<Field>();
193 
194             for (String fieldName : this.fields.split(",")) {
195                 if (fieldName.trim().startsWith("*{")) {
196                     String regExp = fieldName.trim().substring(2);
197                     regExp = regExp.substring(0, regExp.lastIndexOf("}"));
198                     Pattern pattern = Pattern.compile(regExp);
199                     for (Field field : collection.getFields()) {
200                         if (pattern.matcher(field.getId()).matches()) {
201                             fields.add(field);
202                         }
203                     }
204                 } else {
205 
206                     Field field = null;
207 
208                     if (collection != null && fieldName != null) {
209                         field = collection.getField(fieldName.trim());
210                     }
211 
212                     if (field == null) {
213                         throw new RuntimeException("Field '" + fieldName + "' not found on collection '" + collection.getORI() + "'.");
214                     } else {
215                         fields.add(field);
216                     }
217                 }
218 
219             }
220         }
221         return fields;
222     }
