45  public class HeatmapViewer extends Widget<HeatmapViewerConfig, HeatmapViewerStatus> {
46  
47      private final static HeaderItem CSS = CssHeaderItem.forReference(new CssResourceReference(HeatmapViewer.class, "css/jheatmap-1.0.0-alpha.css"));
48      private final static HeaderItem JS_JHEATMAP = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(HeatmapViewer.class, "js/jheatmap-1.0.0-alpha.js"));

50      private final static ResourceReference LOADING_IMG = new PackageResourceReference(HeatmapViewer.class, "images/loading.gif");

53      @PaxWicketBean(name = "collectionManager")
54      private ICollectionManager collectionManager;

56      public HeatmapViewer(String componentId, IModel<HeatmapViewerStatus> status) {
57          super(componentId, status);
58  
59          add(new Image("loading", LOADING_IMG));
60      }

62      @Override
63      public void renderHead(IHeaderResponse response) {
64          response.render(CSS);
65          response.render(JS_JHEATMAP);
66          response.render(OnLoadHeaderItem.forScript(newJavaScriptHeatmap()));
67      }

69      private String newJavaScriptHeatmap() {
70          StringBuilder code = new StringBuilder();
71  
72          HeatmapViewerConfig config = getConfig();
73  
74          List<IColumn<IEntityTable, String>> columns = new ArrayList<IColumn<IEntityTable, String>>();
75  
76          ORI parentUri = getPageBaseOri();
77          for (ColumnConfig column : getConfig().getColumns()) {
78              column.addColumns(columns, parentUri);
79          }
80          int lastColumnFields = columns.size();
81          columns.clear();
82  
83          for (ColumnConfig row : getConfig().getRows()) {
84              row.addColumns(columns, parentUri);
85          }
86          int lastRowFields = lastColumnFields + columns.size();
87          columns.clear();
88  
89          // Webservice URL
90          code.append(
91                  "$('#heatmap').heatmap({\n" +
92                          "       data : {\n" +
93                          "                 type : \"tdm\",\n" +
94                          "                 values : '" + getQueryUrl() + "', \n");
95  
96          code.append("\t\t\t\tcols_annotations : [");
97          if (config.getColAnnotations() == null) {
98              for (int i = 0; i < lastColumnFields; i++) {
99                  if (i != 0) {
100                     code.append(",");
101                 }
102                 code.append(i);
103             }
104         } else {
105             code.append(config.getColAnnotations());
106         }
107         code.append("], \n");
108         code.append("\t\t\t\trows_annotations : [");
109         if (config.getRowAnnotations() == null) {
110             for (int i = lastColumnFields; i < lastRowFields; i++) {
111                 if (i != lastColumnFields) {
112                     code.append(",");
113                 }
114                 code.append(i);
115             }
116         } else {
117             code.append(config.getRowAnnotations());
118         }
119         code.append("]\n" +
120                 "              },\n" +
121                 "       init : function(heatmap) {\n");
122         code.append(config.getInit());
123         code.append("}});");
124 
125         return code.toString();
126     }
127 
128     private String getQueryUrl() {
129         String serviceMount = collectionManager.getMount();
130         String webserviceUrl = WebsiteApplication.toAbsolutePath('/' + serviceMount);
131         Query query = getQuery();
132 
133         String strQuery = query.toString();
134         String fileName = "file-" + Integer.toHexString(strQuery.hashCode()) + ".tsv";
135 
136         return webserviceUrl + "?query=" + URLEncoder.encode(strQuery) + "&filename=" + fileName;
137     }
