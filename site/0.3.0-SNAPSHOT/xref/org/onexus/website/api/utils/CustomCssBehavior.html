18  package org.onexus.website.api.utils;
19  
20  import org.apache.wicket.Component;
21  import org.apache.wicket.behavior.Behavior;
22  import org.apache.wicket.markup.head.CssHeaderItem;
23  import org.apache.wicket.markup.head.IHeaderResponse;
24  import org.apache.wicket.request.resource.CssResourceReference;
25  import org.apache.wicket.util.string.Strings;
26  import org.onexus.data.api.IDataManager;
27  import org.onexus.resource.api.IResourceManager;
28  import org.onexus.resource.api.ORI;
29  import org.onexus.resource.api.Project;
30  import org.onexus.website.api.WebsiteApplication;
31  import org.ops4j.pax.wicket.api.PaxWicketBean;
32  
33  public class CustomCssBehavior extends Behavior {
34  
35      private transient CssHeaderItem CSS;
36  
37      private ORI resourceUri;
38  
39      @PaxWicketBean(name = "dataManager")
40      private IDataManager dataManager;
41  
42      @PaxWicketBean(name = "resourceManager")
43      private IResourceManager resourceManager;
44  
45      public CustomCssBehavior(ORI parentOri, String cssTag) {
46          if (!Strings.isEmpty(cssTag)) {
47              resourceUri = new ORI(cssTag);
48              if (!resourceUri.isAbsolute()) {
49                  resourceUri = resourceUri.toAbsolute(parentOri);
50              }
51          }
52      }
53  
54      private CustomCssBehavior(ORI resourceUri) {
55          this.resourceUri = resourceUri;
56      }
57  
58      @Override
59      public void renderHead(Component component, IHeaderResponse response) {
60  
61          if (CSS == null) {
62  
63              if (resourceUri != null) {
64  
65                  String dataService = getDataManager().getMount();
66                  Project project = getResourceManager().getProject(resourceUri.getProjectUrl());
67  
68                  String url;
69                  if (project!=null) {
70                      url = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName() + resourceUri.getPath());
71                  } else {
72                      url = resourceUri.toString();
73                  }
74                  CSS = CssHeaderItem.forUrl(url);
75  
76              } else {
77                  CSS = CssHeaderItem.forReference(new CssResourceReference(component.getClass(), component.getClass().getSimpleName() + ".css"));
78              }
79          }
80  
81          response.render(CSS);
82      }
83  
84      private IDataManager getDataManager() {
85  
86          if (dataManager == null) {
87              WebsiteApplication.inject(this);
88          }
89  
90          return dataManager;
91      }
92  
93      private IResourceManager getResourceManager() {
94  
95          if (resourceManager == null) {
96              WebsiteApplication.inject(this);
97          }
98  
99          return resourceManager;
100     }
