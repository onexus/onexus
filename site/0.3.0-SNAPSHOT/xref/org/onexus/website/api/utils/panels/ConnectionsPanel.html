18  package org.onexus.website.api.utils.panels;
19  
20  import org.apache.wicket.AttributeModifier;
21  import org.apache.wicket.markup.html.WebMarkupContainer;
22  import org.apache.wicket.markup.html.basic.Label;
23  import org.apache.wicket.markup.html.link.ExternalLink;
24  import org.apache.wicket.markup.repeater.RepeatingView;
25  import org.apache.wicket.request.Url;
26  import org.apache.wicket.request.mapper.parameter.PageParameters;
27  import org.onexus.website.api.Connection;
28  import org.onexus.website.api.Website;
29  import org.onexus.website.api.WebsiteSession;
30  import org.onexus.website.api.WebsiteStatus;
31  import org.onexus.website.api.events.EventPanel;
32  import org.onexus.website.api.events.EventQueryUpdate;
33  
34  import java.util.Collections;
35  import java.util.List;
36  
37  public class ConnectionsPanel extends EventPanel {
38  
39      private List<Connection> connections;
40  
41      public ConnectionsPanel(String id, List<Connection> connections) {
42          super(id);
43  
44          onEventFireUpdate(EventQueryUpdate.class);
45  
46          if (connections == null) {
47              this.connections = Collections.emptyList();
48          } else {
49              this.connections = connections;
50          }
51  
52          setVisible(!this.connections.isEmpty() && Boolean.parseBoolean(System.getProperty("org.onexus.website.connections", "false")));
53  
54      }
55  
56      @Override
57      protected void onBeforeRender() {
58          super.onBeforeRender();
59  
60          String urlPath = "";
61          Website website = findParent(Website.class);
62          if (!connections.isEmpty()) {
63              PageParameters params = new PageParameters();
64  
65              if (website != null) {
66                  WebsiteStatus status = website.getStatus();
67                  status.encodeParameters(params, true);
68              }
69  
70              Url url = getRequestCycle().mapUrlFor(getPage().getClass(), params);
71              urlPath = url.getPath() + url.getQueryString();
72  
73              int firstSlash = urlPath.indexOf('/');
74              urlPath = urlPath.substring(firstSlash);
75          }
76  
77          RepeatingView connectionsView = new RepeatingView("projects");
78          for (Connection connection : connections) {
79              WebMarkupContainer connectionItem = new WebMarkupContainer(connectionsView.newChildId());
80              if (connection.getActive() != null && connection.getActive()) {
81                  connectionItem.add(new AttributeModifier("class", "active"));
82              }
83              ExternalLink link = new ExternalLink("url", connection.getUrl() + urlPath);
84              link.add(new Label("title", connection.getTitle()));
85              connectionItem.add(link);
86              connectionsView.add(connectionItem);
87          }
88          addOrReplace(connectionsView);
89  
90          WebMarkupContainer divider = new WebMarkupContainer("divider");
91          addOrReplace(divider);
92          RepeatingView userProjects = new RepeatingView("user");
93          addOrReplace(userProjects);
94  
95          //TODO Use an user projects manager
96          if (WebsiteSession.get().getRoles().hasRole("tester")) {
97              WebMarkupContainer connectionItem = new WebMarkupContainer(userProjects.newChildId());
98              if (website.getConfig().getORI().getProjectUrl().endsWith("snp_hopkins_breast")) {
99                  connectionItem.add(new AttributeModifier("class", "active"));
100             }
101             ExternalLink link = new ExternalLink("url", "/web/snp_hopkins_breast/website" + urlPath);
102             link.add(new Label("title", "Breast and colorectal cancer"));
103             connectionItem.add(link);
104             userProjects.add(connectionItem);
105         } else {
106             divider.setVisible(false);
107         }
108 
109     }
110 
111 }
