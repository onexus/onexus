24  import org.onexus.data.api.IDataStreams;

31  import org.slf4j.LoggerFactory;

33  import java.util.regex.Matcher;
34  import java.util.regex.Pattern;

36  public class HtmlDataResourceModel extends LoadableDetachableModel<String> {
37  
38      private static final Logger log = LoggerFactory.getLogger(HtmlDataResourceModel.class);
39  
40      @PaxWicketBean(name = "dataManager")
41      private IDataManager dataManager;
42  
43      @PaxWicketBean(name = "resourceManager")
44      private IResourceManager resourceManager;
45  
46      private ORI contentUri;
47      private String dataResourceUrl;
48  
49      public HtmlDataResourceModel(ORI parentUri, String htmlTag) {
50          if (!Strings.isEmpty(htmlTag)) {
51              contentUri = new ORI(htmlTag);
52              if (!contentUri.isAbsolute()) {
53                  contentUri = contentUri.toAbsolute(parentUri);
54              }
55  
56              String dataService = getDataManager().getMount();
57              Project project = getResourceManager().getProject(contentUri.getProjectUrl());
58              this.dataResourceUrl = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName());
59          }
60      }
61  
62      @Override
63      protected String load() {
64  
65          if (contentUri == null || Strings.isEmpty(contentUri.getPath())) {
66              return "";
67          }
68  
69          try {
70              IDataStreams stream = getDataManager().load(contentUri);
71              return filterRelativeUrls(IOUtils.toString(stream.iterator().next()));
72          } catch (Exception e) {
73              log.error("Error loading HTML source '" + contentUri + "'");
74              return "";
75          }
76      }
77  
78      private IDataManager getDataManager() {
79  
80          if (dataManager == null) {
81              WebsiteApplication.inject(this);
82          }
83  
84          return dataManager;
85      }
86  
87      private IResourceManager getResourceManager() {
88  
89          if (resourceManager == null) {
90              WebsiteApplication.inject(this);
91          }
92  
93          return resourceManager;
94      }
95  
96      private final static Pattern PATTERN_HREF = Pattern.compile("(<a\\s+href\\s*=\\s*[\"|']?)(.*?)([\"|'])", Pattern.CASE_INSENSITIVE);
97      private final static Pattern PATTERN_SRC = Pattern.compile("(src\\s*=\\s*\"?)(.*?)(\")", Pattern.CASE_INSENSITIVE);
98      private final static Pattern PATTERN_ANTICACHE = Pattern.compile("\\$\\{anticache\\}", Pattern.CASE_INSENSITIVE);
99  
100 
101     private String filterRelativeUrls(String html) {
102 
103         // Replace anticache token
104         html = PATTERN_ANTICACHE.matcher(html).replaceAll(Long.toHexString(System.currentTimeMillis()));
105 
106         String resourcePath = contentUri.getPath();
107         resourcePath = resourcePath.substring(1, resourcePath.lastIndexOf('/'));
108 
109         Matcher mHref = PATTERN_HREF.matcher(html);
110         StringBuffer sbHref = new StringBuffer();
111         while (mHref.find()) {
112             String link = mHref.group(2).trim();
113 
114             if (!link.contains("://") && !link.startsWith("#")) {
115                 if (!link.startsWith("/")) {
116                     link = dataResourceUrl + '/' + resourcePath + '/' + link;
117                 } else {
118                     link = dataResourceUrl + '/' + link;
119                 }
120                 mHref.appendReplacement(sbHref, mHref.group(1) + link + mHref.group(3));
121             }
122         }
123         mHref.appendTail(sbHref);
124 
125         Matcher mSrc = PATTERN_SRC.matcher(sbHref);
126         StringBuffer sbSrc = new StringBuffer();
127         while (mSrc.find()) {
128             String link = mSrc.group(2).trim();
129 
130             if (!link.contains("://")) {
131                 if (!link.startsWith("/")) {
132                     link = dataResourceUrl + '/' + resourcePath + '/' + link;
133                 } else {
134                     link = dataResourceUrl + '/' + link;
135                 }
136 
137                 mSrc.appendReplacement(sbSrc, mSrc.group(1) + link + mSrc.group(3));
138             }
139         }
140         mSrc.appendTail(sbSrc);
141 
142         return sbSrc.toString();
143     }
