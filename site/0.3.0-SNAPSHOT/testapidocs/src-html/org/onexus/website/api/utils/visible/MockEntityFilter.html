018package org.onexus.website.api.utils.visible;
019
020
021import org.h2.util.StringUtils;
022import org.onexus.collection.api.query.Filter;
023import org.onexus.collection.api.query.Query;
024import org.onexus.resource.api.ORI;
025import org.onexus.website.api.pages.browser.IFilter;
026import org.onexus.website.api.widgets.filters.FilterConfig;
027
028public class MockEntityFilter implements IFilter {
029
030    private ORI filteredCollection;
031    private String fieldValue;
032
033    public MockEntityFilter(ORI filteredCollection, String fieldValue) {
034        this.fieldValue = fieldValue;
035        this.filteredCollection = filteredCollection;
036    }
037
038    @Override
039    public boolean match(VisibleRule rule) {
040
041        String filterPath = filteredCollection.getPath();
042        String rulePath = rule.getFilteredCollection().getPath();
043
044        boolean validCollection = (filterPath == null || rulePath == null) ? false : filterPath.endsWith(rulePath);
045
046        if (rule.getField() == null) {
047            return validCollection;
048        }
049
050        return StringUtils.equals(fieldValue, rule.getValue());
051
052    }
053
054
055    @Override
056    public ORI getFilteredCollection() {
057        return filteredCollection;
058    }
059
060    @Override
061    public FilterConfig getFilterConfig() {
062        throw new UnsupportedOperationException();
063    }
064
065    @Override
066    public boolean isEnable() {
067        throw new UnsupportedOperationException();
068    }
069
070    @Override
071    public void setEnable(boolean enabled) {
072        throw new UnsupportedOperationException();
073    }
074
075    @Override
076    public boolean isDeletable() {
077        throw new UnsupportedOperationException();
078    }
079
080    @Override
081    public void setDeletable(boolean deletable) {
082        throw new UnsupportedOperationException();
083    }
084
085    @Override
086    public Filter buildFilter(Query query) {
087        throw new UnsupportedOperationException();
088    }
089
090    @Override
091    public String getTitle(Query query) {
092        throw new UnsupportedOperationException();
093    }
094
095    @Override
096    public String toUrlParameter(boolean global, ORI parent) {
097        throw new UnsupportedOperationException();
098    }
099
100    @Override
101    public void loadUrlPrameter(String parameter) {
102        throw new UnsupportedOperationException();
103    }
104
105}
