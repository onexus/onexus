001package org.onexus.collection.api;
002
003import static org.junit.Assert.assertArrayEquals;
004import static org.junit.Assert.assertEquals;
005import org.junit.Test;
006import org.onexus.collection.api.utils.FieldLink;
007import org.onexus.collection.api.utils.LinkUtils;
008import org.onexus.resource.api.ORI;
009
010import java.util.Arrays;
011import java.util.Collections;
012import java.util.List;
013
014public class LinkUtilsTest {
015
016    @Test
017    public void testLinkFields() {
018
019        ORI parentORI = new ORI("http://test.onexus.org");
020
021        ORI aOri = new ORI("http://test.onexus.org?collections/a");
022        ORI bOri = new ORI("http://test.onexus.org?collections/b");
023
024        String aFieldId = "aTestField";
025        String bFieldId = "bTestField";
026
027        Collection a = new Collection();
028        a.setORI(aOri);
029        a.setFields(Arrays.asList(new Field[]{ new Field(aFieldId, aFieldId, aFieldId, String.class, true)}));
030        a.setLinks(Arrays.asList(new Link[]{
031                new Link(bOri, aFieldId + " == " + bFieldId)
032        }));
033        Collection b = new Collection();
034        b.setORI(bOri);
035        b.setFields(Arrays.asList(new Field[]{ new Field(bFieldId, bFieldId, bFieldId, String.class, false)}));
036        b.setLinks(Collections.EMPTY_LIST);
037
038        List<FieldLink> links = LinkUtils.getLinkFields(parentORI, a, b);
039
040        assertEquals(links.size(), 1);
041        assertArrayEquals(links.toArray(), new FieldLink[] { new FieldLink(aOri, aFieldId, bOri, bFieldId)});
042
043        links = LinkUtils.getLinkFields(parentORI, b, a);
044        assertEquals(links.size(), 0);
045
046    }
047}
