001package org.onexus.resource.api.utils.string;
002
003import java.util.Map;
004
005/**
006 * Original code extract from Apache Wicket core *
007 */
008public class MapVariableInterpolator extends VariableInterpolator
009{
010    /** Map of variables */
011    private Map<?, ?> variables;
012
013    /**
014     * Constructor.
015     *
016     * @param string
017     *            a <code>String</code> to interpolate into
018     * @param variables
019     *            the variables to substitute
020     */
021    public MapVariableInterpolator(final String string, final Map<?, ?> variables)
022    {
023        super(string);
024        this.variables = variables;
025    }
026
027    /**
028     * Constructor.
029     *
030     * @param string
031     *            a <code>String</code> to interpolate into
032     * @param variables
033     *            the variables to substitute
034     * @param exceptionOnNullVarValue
035     *            if <code>true</code> an {@link IllegalStateException} will be thrown if
036     *            {@link #getValue(String)} returns <code>null</code>, otherwise the
037     *            <code>${varname}</code> string will be left in the <code>String</code> so that
038     *            multiple interpolators can be chained
039     */
040    public MapVariableInterpolator(final String string, final Map<?, ?> variables,
041                                   final boolean exceptionOnNullVarValue)
042    {
043        super(string, exceptionOnNullVarValue);
044        this.variables = variables;
045    }
046
047    /**
048     * Sets the <code>Map</code> of variables.
049     *
050     * @param variables
051     *            the <code>Map</code> of variables
052     */
053    public final void setVariables(final Map<?, ?> variables)
054    {
055        this.variables = variables;
056    }
057
058    /**
059     * Retrieves a value for a variable name during interpolation.
060     *
061     * @param variableName
062     *            the variable name
063     * @return the value
064     */
065    @Override
066    protected String getValue(final String variableName)
067    {
068        Object value = variables.get(variableName);
069
070        if (value == null) {
071            return null;
072        }
073
074        return value.toString();
075    }
076
077    /**
078     * Interpolates a <code>String</code> with the arguments defined in the given <code>Map</code>.
079     *
080     * @param string
081     *            a <code>String</code> to interpolate into
082     * @param variables
083     *            the variables to substitute
084     * @return the interpolated <code>String</code>
085     */
086    public static String interpolate(final String string, final Map<?, ?> variables)
087    {
088        return new MapVariableInterpolator(string, variables).toString();
089    }
090
091}
