020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.markup.html.WebMarkupContainer;
022import org.apache.wicket.markup.html.basic.Label;
023import org.apache.wicket.model.IModel;
024import org.apache.wicket.model.Model;
025import org.onexus.collection.api.Field;
026import org.onexus.collection.api.IEntity;
027import org.onexus.website.api.widgets.tableviewer.decorators.utils.FieldDecorator;
028
029import java.util.HashMap;
030import java.util.Map;
031
032public class CtDecorator extends FieldDecorator {
033
034    private static final String emptyColor = "#FFFFFF";
035
036    public static final Map<String, String> colors = new HashMap<String, String>();
037
038    static {
039
040        // NSSNP - #FFCCE6
041        colors.put("NON_SYNONYMOUS_CODING", "#FFCCE6");
042
043        // RSNP - #002EB8 (#cdddff)
044        colors.put("REGULATORY_REGION", "#cdddff");
045        colors.put("DOWNSTREAM", "#cdddff");
046        colors.put("UPSTREAM", "#cdddff");
047
048        // SSNP - #F5B800
049        colors.put("SYNONYMOUS_CODING", "#F5B800");
050
051        // STOP_SNP - #CCFF33
052        colors.put("STOP_GAINED", "#CCFF33");
053        colors.put("STOP_LOST", "#CCFF33");
054
055        // SPLICE_SNP - #FF6633
056        colors.put("SPLICE_SITE", "#FF6633");
057        colors.put("ESSENTIAL_SPLICE_SITE", "#FF6633");
058
059        // OTRSNP - #FFFFCC
060        colors.put("3PRIME_UTR", "#FFFFCC");
061        colors.put("WITHIN_NON_CODING_GENE", "#FFFFCC");
062        colors.put("NMD_TRANSCRIPT", "#FFFFCC");
063        colors.put("5PRIME_UTR", "#FFFFCC");
064
065        colors.put("INTRONIC", "#FFFFCC");
066        colors.put("INTERGENIC", "#FFFFCC");
067        colors.put("FRAMESHIFT_CODING", "#FFFFCC");
068        colors.put("WITHIN_MATURE_MIRNA", "#FFFFCC");
069
070    }
071
072    public static final Map<String, String> shortNames = new HashMap<String, String>();
073
074    static {
075        shortNames.put("3PRIME_UTR", "3-UTR");
076        shortNames.put("NON_SYNONYMOUS_CODING", "NS-SNP");
077        shortNames.put("SYNONYMOUS_CODING", "S-SNP");
078        shortNames.put("WITHIN_NON_CODING_GENE", "non-Cod");
079        shortNames.put("STOP_GAINED", "stp-gain");
080        shortNames.put("STOP_LOST", "stp-lost");
081        shortNames.put("NMD_TRANSCRIPT", "NMD");
082        shortNames.put("SPLICE_SITE", "splice");
083        shortNames.put("5PRIME_UTR", "5-UTR");
084        shortNames.put("DOWNSTREAM", "downst");
085        shortNames.put("REGULATORY_REGION", "Reg-SNP");
086        shortNames.put("ESSENTIAL_SPLICE_SITE", "e-splice");
087        shortNames.put("UPSTREAM", "upst");
088        shortNames.put("INTRONIC", "intro");
089        shortNames.put("INTERGENIC", "intergen");
090        shortNames.put("FRAMESHIFT_CODING", "frameshift");
091        shortNames.put("WITHIN_MATURE_MIRNA", "mature miRNA");
092        shortNames.put("NONCODING_RNA", "non-rna");
093
094    }
095
096    private String cssClass;
097
098    public CtDecorator(Field valueProperty, String cssClass) {
099        super(valueProperty);
100        this.cssClass = cssClass;
101    }
102
103    protected String getTooltip(IEntity data) {
104        Object value = data.get(getField().getId());
105        return (value == null ? "" : value.toString());
106    }
107
108    @Override
109    public String getColor(IEntity data) {
110        Object value = getValue(data);
111
112        return (value == null ? emptyColor : colors.get(value.toString()
113                .toUpperCase()));
114    }
115
116    @Override
117    public String getTemplate() {
118        return null;
119    }
120
121    @Override
122    public void setTemplate(String template) {
123
124    }
125
126    @Override
127    public void populateCell(WebMarkupContainer cellContainer,
128                             String componentId, IModel<IEntity> data) {
129        IEntity entity = data.getObject();
130
131        Object value = (entity != null ? entity.get(getField()
132                .getId()) : null);
133        String shortName = (value != null ? shortNames.get(value.toString()
134                .toUpperCase()) : "");
135        shortName = (shortName == null ? value.toString().toLowerCase()
136                : shortName);
137        Label empty = new Label(componentId, shortName);
138        // empty.setVisible(false);
139
140        cellContainer.add(empty);
141        cellContainer.add(new AttributeModifier("style", new Model<String>(
142                "background-color: " + getColor(data.getObject()) + ";")));
143        cellContainer.add(new AttributeModifier("title", new Model<String>(
144                getTooltip(data.getObject()))));
145        if (cssClass != null) {
146            cellContainer.add(new AttributeModifier("class", new Model<String>(
147                    cssClass)));
148        }
149
150    }
151
152}
