018package org.onexus.website.api.utils;
019
020import org.apache.wicket.model.LoadableDetachableModel;
021import org.apache.wicket.util.io.IOUtils;
022import org.apache.wicket.util.string.Strings;
023import org.onexus.data.api.IDataManager;
024import org.onexus.data.api.IDataStreams;
025import org.onexus.resource.api.IResourceManager;
026import org.onexus.resource.api.ORI;
027import org.onexus.resource.api.Project;
028import org.onexus.website.api.WebsiteApplication;
029import org.ops4j.pax.wicket.api.PaxWicketBean;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033import java.util.regex.Matcher;
034import java.util.regex.Pattern;
035
036public class HtmlDataResourceModel extends LoadableDetachableModel<String> {
037
038    private static final Logger log = LoggerFactory.getLogger(HtmlDataResourceModel.class);
039
040    @PaxWicketBean(name = "dataManager")
041    private IDataManager dataManager;
042
043    @PaxWicketBean(name = "resourceManager")
044    private IResourceManager resourceManager;
045
046    private ORI contentUri;
047    private String dataResourceUrl;
048
049    public HtmlDataResourceModel(ORI parentUri, String htmlTag) {
050        if (!Strings.isEmpty(htmlTag)) {
051            contentUri = new ORI(htmlTag);
052            if (!contentUri.isAbsolute()) {
053                contentUri = contentUri.toAbsolute(parentUri);
054            }
055
056            String dataService = getDataManager().getMount();
057            Project project = getResourceManager().getProject(contentUri.getProjectUrl());
058            this.dataResourceUrl = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName());
059        }
060    }
061
062    @Override
063    protected String load() {
064
065        if (contentUri == null || Strings.isEmpty(contentUri.getPath())) {
066            return "";
067        }
068
069        try {
070            IDataStreams stream = getDataManager().load(contentUri);
071            return filterRelativeUrls(IOUtils.toString(stream.iterator().next()));
072        } catch (Exception e) {
073            log.error("Error loading HTML source '" + contentUri + "'");
074            return "";
075        }
076    }
077
078    private IDataManager getDataManager() {
079
080        if (dataManager == null) {
081            WebsiteApplication.inject(this);
082        }
083
084        return dataManager;
085    }
086
087    private IResourceManager getResourceManager() {
088
089        if (resourceManager == null) {
090            WebsiteApplication.inject(this);
091        }
092
093        return resourceManager;
094    }
095
096    private final static Pattern PATTERN_HREF = Pattern.compile("(<a\\s+href\\s*=\\s*[\"|']?)(.*?)([\"|'])", Pattern.CASE_INSENSITIVE);
097    private final static Pattern PATTERN_SRC = Pattern.compile("(src\\s*=\\s*\"?)(.*?)(\")", Pattern.CASE_INSENSITIVE);
098    private final static Pattern PATTERN_ANTICACHE = Pattern.compile("\\$\\{anticache\\}", Pattern.CASE_INSENSITIVE);
099
100
101    private String filterRelativeUrls(String html) {
102
103        // Replace anticache token
104        html = PATTERN_ANTICACHE.matcher(html).replaceAll(Long.toHexString(System.currentTimeMillis()));
105
106        String resourcePath = contentUri.getPath();
107        resourcePath = resourcePath.substring(1, resourcePath.lastIndexOf('/'));
108
109        Matcher mHref = PATTERN_HREF.matcher(html);
110        StringBuffer sbHref = new StringBuffer();
111        while (mHref.find()) {
112            String link = mHref.group(2).trim();
113
114            if (!link.contains("://") && !link.startsWith("#")) {
115                if (!link.startsWith("/")) {
116                    link = dataResourceUrl + '/' + resourcePath + '/' + link;
117                } else {
118                    link = dataResourceUrl + '/' + link;
119                }
120                mHref.appendReplacement(sbHref, mHref.group(1) + link + mHref.group(3));
121            }
122        }
123        mHref.appendTail(sbHref);
124
125        Matcher mSrc = PATTERN_SRC.matcher(sbHref);
126        StringBuffer sbSrc = new StringBuffer();
127        while (mSrc.find()) {
128            String link = mSrc.group(2).trim();
129
130            if (!link.contains("://")) {
131                if (!link.startsWith("/")) {
132                    link = dataResourceUrl + '/' + resourcePath + '/' + link;
133                } else {
134                    link = dataResourceUrl + '/' + link;
135                }
136
137                mSrc.appendReplacement(sbSrc, mSrc.group(1) + link + mSrc.group(3));
138            }
139        }
140        mSrc.appendTail(sbSrc);
141
142        return sbSrc.toString();
143    }
144}
