018package org.onexus.website.api.theme;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.behavior.Behavior;
023import org.apache.wicket.event.IEvent;
024import org.apache.wicket.markup.head.CssHeaderItem;
025import org.apache.wicket.markup.head.HeaderItem;
026import org.apache.wicket.markup.head.IHeaderResponse;
027import org.apache.wicket.markup.head.JavaScriptHeaderItem;
028import org.apache.wicket.markup.head.OnLoadHeaderItem;
029import org.apache.wicket.request.resource.CssResourceReference;
030import org.apache.wicket.resource.JQueryPluginResourceReference;
031
032public class DefaultTheme extends Behavior {
033
034    private final static HeaderItem BOOTSTRAP_CSS = CssHeaderItem.forReference(new CssResourceReference(DefaultTheme.class, "css/bootstrap.min.css"));
035    private final static HeaderItem BOOTSTRAP_JS = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(DefaultTheme.class, "js/bootstrap.min.js"));
036    private final static HeaderItem STYLE_CSS = CssHeaderItem.forReference(new CssResourceReference(DefaultTheme.class, "css/style.css"));
037
038    private final static HeaderItem COLORBOX_JS = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(DefaultTheme.class, "colorbox/jquery.colorbox-min.js"));
039    private final static HeaderItem COLORBOX_CSS = CssHeaderItem.forReference(new CssResourceReference(DefaultTheme.class, "colorbox/colorbox.css"));
040
041    private final static HeaderItem THEME_JS = JavaScriptHeaderItem.forReference(new JQueryPluginResourceReference(DefaultTheme.class, "defaulttheme.js"));
042
043    @Override
044    public void renderHead(Component component, IHeaderResponse response) {
045
046        response.render(BOOTSTRAP_CSS);
047        response.render(STYLE_CSS);
048        response.render(COLORBOX_CSS);
049
050        response.render(BOOTSTRAP_JS);
051        response.render(COLORBOX_JS);
052
053        response.render(THEME_JS);
054
055        response.render(OnLoadHeaderItem.forScript(
056                getTooltipJavascript() +
057                        getModalJavascript() +
058                        getPopoverJavascript() +
059                        getMoveFooter() +
060                        getColorBoxJavascript()));
061    }
062
063    private static String getTooltipJavascript() {
064        return "$(\"[rel=tooltip]\").tooltip({ placement: 'bottom'});";
065    }
066
067    private static String getTooltipHideJavascript() {
068        return "$(\"[rel=tooltip]\").tooltip('hide');";
069    }
070
071    private static String getModalJavascript() {
072        return "$(\"div.modal\").modal({ show: false });";
073    }
074
075    private static String getPopoverJavascript() {
076        return "$(\"[rel=popover]\").popover({ placement: 'bottom'});";
077    }
078
079    private static String getMoveFooter() {
080        return "moveFooter();";
081    }
082
083    private static String getColorBoxJavascript() {
084        return "$(\".iframe\").colorbox({iframe:true, width:\"80%\", height:\"80%\"});";
085    }
086
087    @Override
088    public void onEvent(Component component, IEvent<?> event) {
089
090        if (event.getPayload() instanceof AjaxRequestTarget) {
091            AjaxRequestTarget target = (AjaxRequestTarget) event.getPayload();
092            target.prependJavaScript(getTooltipHideJavascript());
093            target.appendJavaScript(getTooltipJavascript());
094            target.appendJavaScript(getMoveFooter());
095            target.appendJavaScript(getColorBoxJavascript());
096        }
097
098    }
099}
