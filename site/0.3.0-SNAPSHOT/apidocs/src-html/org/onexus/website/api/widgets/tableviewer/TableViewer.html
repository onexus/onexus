020import org.apache.commons.collections.CollectionUtils;

023import org.apache.wicket.event.IEvent;
024import org.apache.wicket.extensions.markup.html.repeater.data.table.IColumn;
025import org.apache.wicket.markup.head.CssHeaderItem;
026import org.apache.wicket.markup.head.IHeaderResponse;
027import org.apache.wicket.markup.html.panel.Panel;
028import org.apache.wicket.model.IModel;
029import org.apache.wicket.request.resource.CssResourceReference;
030import org.onexus.collection.api.IEntityTable;
031import org.onexus.collection.api.query.Query;
032import org.onexus.resource.api.ORI;
033import org.onexus.resource.api.Progress;
034import org.onexus.website.api.events.EventAddFilter;
035import org.onexus.website.api.events.EventQueryUpdate;
036import org.onexus.website.api.events.EventRemoveFilter;
037import org.onexus.website.api.pages.browser.BrowserPageStatus;
038import org.onexus.website.api.utils.panels.ondomready.OnDomReadyPanel;
039import org.onexus.website.api.utils.visible.VisiblePredicate;
040import org.onexus.website.api.widgets.Widget;
041import org.onexus.website.api.widgets.tableviewer.columns.IColumnConfig;
042
043import java.util.ArrayList;
044import java.util.List;
045
046public class TableViewer extends Widget<TableViewerConfig, TableViewerStatus> {
047
048    public static final CssResourceReference TABLE_VIEWER_CSS = new CssResourceReference(TableViewer.class,
049            "TableViewer.css");
050
051    private final static MetaDataKey<Integer> DEFAULT_ROWS_PER_PAGE = new MetaDataKey<Integer>() {
052    };
053
054    // Model objects
055    private EntitiesRowProvider dataProvider;
056
057    public TableViewer(String componentId, IModel<TableViewerStatus> status) {
058        super(componentId, status);
059
060        onEventFireUpdate(EventQueryUpdate.class, EventAddFilter.class, EventRemoveFilter.class);
061
062        if (status.getObject() == null) {
063            status.setObject(getStatus());
064        }
065
066        this.getVariation();
067
068        Integer sessionRowsPerPage = getSession().getMetaData(DEFAULT_ROWS_PER_PAGE);
069        final Integer rowsPerPage = (sessionRowsPerPage == null ? 30 : sessionRowsPerPage);
070
071        this.dataProvider = new EntitiesRowProvider(status, rowsPerPage) {
072
073            @Override
074            protected Query getQuery() {
075                return TableViewer.this.getQuery();
076            }
077
078            @Override
079            protected void addTaskStatus(Progress progressStatus) {
080
081                /*TODO
082                ActiveProgress activeProgress = Session.get().getMetaData(ProgressBar.TASKS);
083                if (activeProgress == null) {
084                    activeProgress = new ActiveProgress();
085                    Session.get().setMetaData(ProgressBar.TASKS, activeProgress);
086                }
087                for (Progress progress : progressStatus.getSubProgresses()) {
088                    activeProgress.addTask(progress);
089                } */
090            }
091
092        };
093
094        // Create the columns from the config
095        int ccs = getStatus().getCurrentColumnSet();
096        List<IColumnConfig> columnsConfig = getConfig().getColumnSets().get(ccs).getColumns();
097        final List<IColumn<IEntityTable, String>> columns = new ArrayList<IColumn<IEntityTable, String>>();
098
099        ORI parentURI = getQuery().getOn();
100
101        List<IColumnConfig> visibleColumnsConfig = new ArrayList<IColumnConfig>(columnsConfig.size());
102        BrowserPageStatus pageStatus = findParentStatus(BrowserPageStatus.class);
103
104        if (pageStatus != null) {
105            Predicate filter = new VisiblePredicate(getPageBaseOri(), pageStatus.getFilters());
106            CollectionUtils.select(columnsConfig, filter, visibleColumnsConfig);
107        } else {
108            visibleColumnsConfig = columnsConfig;
109        }
110
111        for (IColumnConfig columnConfig : visibleColumnsConfig) {
112            columnConfig.addColumns(columns, parentURI);
113        }
114
115        add(new OnDomReadyPanel("datatable") {
116            @Override
117            protected Panel onDomReadyPanel(String componentId) {
118                return new DataTablePanel("datatable", columns, dataProvider, rowsPerPage);
119            }
120        });
121
122    }
123
124    @Override
125    public void onEvent(IEvent<?> event) {
126        super.onEvent(event);
127
128        if (event.getPayload() instanceof EventQueryUpdate) {
129            this.dataProvider.clearCount();
130        }
131    }
132
133    @Override
134    public void renderHead(IHeaderResponse response) {
135        super.renderHead(response);
136        response.render(CssHeaderItem.forReference(TABLE_VIEWER_CSS));
137    }
138
139}
