070        WebsiteApplication.inject(this);

073        this.entityId = entityId;
074        this.deletable = deletable;
075        this.enable = true;

121    @Override
122    public Filter buildFilter(Query query) {
123        String collectionAlias = QueryUtils.newCollectionAlias(query, filteredCollection);
124        return new EqualId(collectionAlias, entityId);
125    }
126
127    @Override
128    public boolean match(VisibleRule rule) {
129
130        //TODO Check projects
131
132        String filterPath = filteredCollection.getPath();
133        String rulePath = rule.getFilteredCollection().getPath();
134
135        boolean validCollection = (filterPath == null || rulePath == null) ? false : filterPath.endsWith(rulePath);
136
137        if (rule.getField() == null) {
138            return validCollection;
139        }
140
141        ORI collectionUri = filteredCollection.toAbsolute(rule.getParentURI());
142        IEntity entity = (new EntityModel(collectionUri, entityId)).getObject();
143
144        String fieldValue = String.valueOf(entity.get(rule.getField()));
145
146        return StringUtils.equals(fieldValue, rule.getValue());
147    }
148
149    private static String SEPARATOR = "::";
150    private static Pattern DOUBLE_COLON = Pattern.compile(SEPARATOR);
151
152    @Override
153    public String toUrlParameter(boolean global, ORI parentUri) {
154
155        StringBuilder str = new StringBuilder();
156
157        ORI uri;
158        if (global) {
159            uri = filteredCollection.toAbsolute(parentUri);
160        } else {
161            uri = filteredCollection;
162        }
163
164        str.append(uri).append(SEPARATOR);
165        str.append(entityId).append(SEPARATOR);
166        str.append((deletable ? "d" : ""));
167        str.append((enable ? "e" : ""));
168
169        return str.toString();
170    }
171
172    @Override
173    public void loadUrlPrameter(String parameter) {
174
175        String[] values = DOUBLE_COLON.split(parameter);
176
177        this.filteredCollection = new ORI(values[0]);
178        this.entityId = values[1];
179        String flags = values[2];
180        deletable = flags.contains("d");
181        enable = flags.contains("e");
182    }
183
184    @Override
185    public String getTitle(Query query) {
186
187        // Make collection URI absolute
188        ORI filteredCollection = getFilteredCollection().toAbsolute(query.getOn());
189        Collection collection = getResourceManager().load(Collection.class, filteredCollection);
190
191        String entityField = collection.getProperty("FIXED_ENTITY_FIELD");
192        String entityLabel = getEntityId();
193        if (entityField != null) {
194            IEntityTable entityTable = getCollectionManager().load(new SingleEntityQuery(filteredCollection, getEntityId()));
195
196            IEntity entity = new EntityIterator(entityTable, filteredCollection).next();
197
198            entityLabel = String.valueOf(entity.get(entityField));
199
200            if (entityLabel == null || entityLabel.isEmpty()) {
201                entityLabel = getEntityId();
202            }
203
204            entityTable.close();
205        }
206
207        String entityPattern = collection.getProperty("FIXED_ENTITY_PATTERN");
208        if (entityPattern != null) {
209            IEntityTable entityTable = getCollectionManager().load(new SingleEntityQuery(filteredCollection, getEntityId()));
210
211            IEntity entity = new EntityIterator(entityTable, filteredCollection).next();
212
213            if (entity != null) {
214                entityLabel = parseTemplate(entityPattern, entity);
215            }
216
217            entityTable.close();
218        }
219
220        return entityLabel;
221    }
222
223    private IResourceManager getResourceManager() {
224        if (resourceManager == null) {

226        }

229    }

232        if (collectionManager == null) {
233            WebsiteApplication.inject(this);
234        }
235
236        return collectionManager;
237    }
238
239    private String parseTemplate(String entityPattern, IEntity entity) {
240
241
242        for (Field field : entity.getCollection().getFields()) {
243            String fieldName = field.getId();
244            entityPattern = entityPattern.replaceAll(
245                    Pattern.quote("${" + fieldName + "}"),
246                    String.valueOf(entity.get(fieldName))
247            );
248        }
249
250        return entityPattern;
251    }

256        StringBuilder builder = new StringBuilder();
257        builder.append("FixedEntity [collectionId=");

261        builder.append("]");
262        return builder.toString();
263    }

265    @Override
266    public int hashCode() {
267        final int prime = 31;
268        int result = 1;
269        result = prime * result
270                + ((filteredCollection == null) ? 0 : filteredCollection.hashCode());
271        result = prime * result
272                + ((entityId == null) ? 0 : entityId.hashCode());
273        return result;
274    }

276    @Override
277    public boolean equals(Object obj) {
278        if (this == obj)
279            return true;
280        if (obj == null)
281            return false;
282        if (getClass() != obj.getClass())
283            return false;
284        FilterEntity other = (FilterEntity) obj;
285        if (filteredCollection == null) {
286            if (other.filteredCollection != null)
287                return false;
288        } else if (!filteredCollection.equals(other.filteredCollection))
289            return false;
290        if (entityId == null) {
291            if (other.entityId != null)
292                return false;
293        } else if (!entityId.equals(other.entityId))
294            return false;
295        return true;
296    }
