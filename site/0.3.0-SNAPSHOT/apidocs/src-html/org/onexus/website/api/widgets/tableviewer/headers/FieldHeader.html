018package org.onexus.website.api.widgets.tableviewer.headers;
019
020import org.onexus.collection.api.Collection;
021import org.onexus.collection.api.Field;
022import org.onexus.website.api.widgets.tableviewer.formaters.StringFormater;
023
024public class FieldHeader extends ElementHeader {
025
026    private final static int DEFAULT_MAX_LENGTH = 25;
027    public final static String SORT_PROPERTY_SEPARATOR = "::::";
028
029    private String defaultLabel;
030    private String defaultTitle;
031    private Collection collection;
032    private Field field;
033
034    public FieldHeader(String defaultLabel, String defaultTitle, Collection collection, Field field, IHeader parentHeader) {
035        super(field, parentHeader, new StringFormater(getMaxLength(field),
036                false));
037        this.defaultLabel = defaultLabel;
038        this.defaultTitle = defaultTitle;
039        this.field = field;
040        this.collection = collection;
041    }
042
043    @Override
044    public String getSortProperty() {
045        StringBuilder sortProperty = new StringBuilder();
046
047        sortProperty.append(collection.getORI()).append(SORT_PROPERTY_SEPARATOR);
048        sortProperty.append(field.getId());
049
050        return sortProperty.toString();
051    }
052
053    @Override
054    public String getLabel() {
055        if (defaultLabel != null) {
056            return defaultLabel;
057        }
058        String label = field.getLabel();
059        return (label == null ? field.getId() : label);
060    }
061
062    @Override
063    public String getTitle() {
064
065        if (defaultTitle != null) {
066            return defaultTitle;
067        }
068
069        String title = field.getTitle();
070        return (title == null ? field.getId() : title);
071    }
072
073    @Override
074    public boolean isSortable() {
075        String annotation = field.getProperty("SORTABLE");
076        return (annotation == null ? true : annotation.equalsIgnoreCase("TRUE"));
077    }
078
079    public static int getMaxLength(Field attribute) {
080        int maxLength;
081        String value = attribute.getProperty("MAX_LENGTH");
082
083        try {
084            maxLength = Integer.valueOf(value);
085        } catch (Exception e) {
086            maxLength = DEFAULT_MAX_LENGTH;
087        }
088
089        return maxLength;
090
091    }
092
093    public Collection getCollection() {
094        return collection;
095    }
096
097}
