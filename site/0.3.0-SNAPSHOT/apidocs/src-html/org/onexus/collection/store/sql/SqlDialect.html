064    static final String CREATE_TABLE_SQL = "CREATE TABLE IF NOT EXISTS system_properties (name varchar(128), version VARCHAR(128), primary key(name))";
065    static final String DELETE_OBJECT_SQL = "DELETE FROM system_properties WHERE name = ?";
066    static final String INSERT_OBJECT_SQL = "INSERT INTO system_properties (name, version) VALUES (?, ?)";
067    static final String SELECT_OBJECT_SQL = "SELECT version FROM system_properties WHERE name = ?";
068    static final String SELECT_PROPERTIES_KEYS = "SELECT name FROM system_properties";
069    static final String SELECT_PROPERTIES_MAP = "SELECT name, version FROM system_properties";

076    public SqlDialect() {
077        super();
078        this.sqlAdapters = registerAdapters();
079        this.builders = registerFilterBuilders();
080        this.columnTypes = registerColumnTypes();
081    }

083    protected Map<Class<?>, String> registerColumnTypes() {
084        Map<Class<?>, String> columnTypes = new HashMap<Class<?>, String>();
085        columnTypes.put(String.class, "VARCHAR(128)");
086        columnTypes.put(Text.class, "TEXT");
087        columnTypes.put(Boolean.class, "TINYINT(1)");
088        columnTypes.put(Date.class, "TIMESTAMP");
089        columnTypes.put(Integer.class, "INT(11)");
090        columnTypes.put(Long.class, "BIGINT");
091        columnTypes.put(Double.class, "DOUBLE");
092        return columnTypes;
093    }
094
095    protected Map<Class<?>, SqlAdapter> registerAdapters() {
096        Map<Class<?>, SqlAdapter> sqlAdapters = new HashMap<Class<?>, SqlAdapter>();
097        sqlAdapters.put(Double.class, new DoubleAdapter());
098        sqlAdapters.put(Integer.class, new IntegerAdapter());
099        sqlAdapters.put(Long.class, new LongAdapter());
100        sqlAdapters.put(String.class, new StringAdapter(this));
101        sqlAdapters.put(Boolean.class, new BooleanAdapter());
102        sqlAdapters.put(Text.class, new TextAdapter(this));
103        return sqlAdapters;
104    }
105
106    protected List<FilterBuilder> registerFilterBuilders() {
107        List<FilterBuilder> builders = new ArrayList<FilterBuilder>();
108        builders.add(new AtomicFilterBuilder(this));
109        builders.add(new EqualIdFilterBuilder(this));
110        builders.add(new BinaryFilterBuilder(this));
111        builders.add(new NotFilterBuilder(this));
112        builders.add(new InFilterBuilder(this));
113        builders.add(new IsNullFilterBuilder(this));
114        return builders;
115    }

122    public void createSystemPropertiesTable(Connection conn,
123                                            boolean closeConnection) throws SQLException {
124        Statement st = null;
125        try {
126            st = conn.createStatement();
127            st.execute(CREATE_TABLE_SQL);
128        } finally {
129            if (st != null)
130                st.close();
131            if (conn != null && closeConnection)
132                conn.close();
133        }
134    }

136    public List<String> loadPropertyKeys(Connection conn) throws Exception {
137        List<String> keys = new ArrayList<String>();
138        PreparedStatement pstmt = null;
139        ResultSet rs = null;
140        try {
141            pstmt = conn.prepareStatement(SELECT_PROPERTIES_KEYS);
142            rs = pstmt.executeQuery();
143
144            while (rs.next()) {
145                keys.add(rs.getString("name"));
146            }
147
148        } finally {
149            if (rs != null)
150                rs.close();
151            if (pstmt != null)
152                pstmt.close();
153
154        }
155        return keys;
156    }

181    public void saveProperty(Connection conn, String name, String version)
182            throws Exception {
183        PreparedStatement pstmt = null;
184        try {
185            pstmt = conn.prepareStatement(DELETE_OBJECT_SQL);
186            pstmt.setString(1, name);
187            pstmt.execute();
188            pstmt.close();
189
190            pstmt = conn.prepareStatement(INSERT_OBJECT_SQL);
191            pstmt.setString(1, name);
192            pstmt.setString(2, version);
193            pstmt.executeUpdate();
194        } finally {
195            if (pstmt != null)
196                pstmt.close();
197        }
198    }

214    public String loadProperty(Connection conn, String name)
215            throws Exception {
216        PreparedStatement pstmt = null;
217        ResultSet rs = null;
218        try {
219            pstmt = conn.prepareStatement(SELECT_OBJECT_SQL);
220
221            pstmt.setString(1, name);
222            rs = pstmt.executeQuery();
223
224            if (!rs.next()) {
225                return null;
226            }
227
228            String object = rs.getString(1);
229            return object;
230        } finally {
231            if (rs != null)
232                rs.close();
233            if (pstmt != null)
234                pstmt.close();
235        }
236    }

238    public void execute(Connection conn, String query)
239            throws SQLException {
240        Statement st = null;
241        try {
242            LOGGER.trace("Execute: " + query);
243            st = conn.createStatement();
244            st.execute(query);
245        } finally {
246            if (st != null)
247                st.close();
248        }
249    }

251    public void openInsert(StringBuilder sql, SqlCollectionDDL ddl) {
252
253        // Open insert
254        sql.append("INSERT INTO `").append(ddl.getTableName())
255                .append("` (`");
256
257        // Column names
258        Iterator<ColumnInfo> it = ddl.getColumnInfos().iterator();
259        while (it.hasNext()) {
260            sql.append(it.next().getColumnName());
261            if (it.hasNext()) {
262                sql.append("`, `");
263            }
264        }
265
266        sql.append("`) VALUES ");
267    }

269    public void addValues(StringBuilder sql, SqlCollectionDDL ddl,
270                          IEntity entity) {
271
272        sql.append("(");
273        Iterator<ColumnInfo> itColumns = ddl.getColumnInfos().iterator();
274        while (itColumns.hasNext()) {
275            ColumnInfo column = itColumns.next();
276            Object value = entity.get(column.getField().getId());
277            if (value == null) {
278
279                if (column.getField().isPrimaryKey() != null && column.getField().isPrimaryKey()) {
280                    if (column.getField().getType().equals(String.class)) {
281                        sql.append("\"NONE\"");
282                    } else {
283                        sql.append("0");
284                    }
285                } else {
286                    sql.append("NULL");
287                }
288            } else {
289                SqlAdapter adapter = column.getAdapter();
290
291                try {
292                    adapter.append(sql, value);
293                } catch (Exception e) {
294                    throw new RuntimeException(e);
295                }
296
297            }
298
299            if (itColumns.hasNext()) {
300                sql.append(",");
301            }
302        }
303        sql.append(")");
304    }

306    public SqlEntity loadEntity(Query query, SqlCollectionDDL ddl, ResultSet rs,
307                                SqlEntity dstObj, String columnPrefix) {
308
309        Set<String> columnWithError = new HashSet<String>();
310
311        String collectionAlias = QueryUtils.newCollectionAlias(query, ddl.getCollection().getORI());
312        List<String> queryFields = query.getSelect().get(collectionAlias);
313
314        for (ColumnInfo column : ddl.getColumnInfos()) {
315
316            if (queryFields != null && !queryFields.contains(column.getField().getId())) {
317                continue;
318            }
319
320            String columnName = column.getColumnName();
321            SqlAdapter adapter = column.getAdapter();
322            try {
323                String sqlColumnName = (columnPrefix == null ? columnName
324                        : columnPrefix + "_" + columnName);
325                dstObj.put(column.getField().getId(),
326                        adapter.extract(rs, sqlColumnName));
327            } catch (Exception e) {
328
329                if (!columnWithError.contains(columnName)) {
330                    LOGGER.error("Error loading column '" + columnName + "'");
331                    try {
332                        ResultSetMetaData m = rs.getMetaData();
333
334                        List<String> columnNames = new ArrayList<String>();
335                        for (int i = 1; i <= m.getColumnCount(); i++) {
336                            columnNames.add(m.getColumnName(i));
337                        }
338                        LOGGER.error("Valid column names: " + columnNames + "'");
339
340                    } catch (SQLException e1) {
341                        LOGGER.error("Error getting table metadata.", e1);
342                    }
343                    columnWithError.add(columnName);
344                }
345
346            }
347        }
348
349        return dstObj;
350    }

356    public String quoteString(Object value) {
357
358        if (value == null) {
359            return nullValue();
360        }
361
362        String s;
363
364        if (value instanceof String) {
365            s = (String) value;
366        } else {
367            s = String.valueOf(value);
368        }
369
370        int length = s.length();
371        StringBuilder buff = new StringBuilder(length + 2);
372        buff.append('\'');
373        for (int i = 0; i < length; i++) {
374            char c = s.charAt(i);
375            if (c == '\'' || c == '\\') {
376                buff.append(c);
377            }
378            buff.append(c);
379        }
380        buff.append('\'');
381        return buff.toString();
382    }

385    public FilterBuilder getFilterBuilder(Filter filter) {
386
387        for (FilterBuilder builder : builders) {
388            if (builder.canBuild(filter)) {
389                return builder;
390            }
391        }
392
393        return new UnknownFilterBuilder();
394
395    }
