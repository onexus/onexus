018package org.onexus.collection.store.sql;
019
020import org.onexus.collection.api.utils.FieldLink;
021import org.onexus.resource.api.ORI;
022
023import java.util.Collections;
024import java.util.Comparator;
025import java.util.HashSet;
026import java.util.List;
027import java.util.Map;
028import java.util.Set;
029
030public class LinksNetworkComparator implements Comparator<ORI> {
031
032    private ORI fromCollection;
033    private Map<ORI, List<FieldLink>> networkLinks;
034
035    public LinksNetworkComparator(Map<ORI, List<FieldLink>> networkLinks, ORI fromCollection) {
036        this.networkLinks = networkLinks;
037        this.fromCollection = fromCollection;
038    }
039
040    @Override
041    public int compare(ORI a, ORI b) {
042
043        int a_b = depend(a, b, 0, new HashSet<ORI>());
044        int b_a = depend(b, a, 0, new HashSet<ORI>());
045
046        if (a_b != -1 && b_a != -1) {
047            if (a_b == 0 && b_a == 0) {
048
049                List<FieldLink> linksA = networkLinks.get(a);
050                List<FieldLink> linksB = networkLinks.get(b);
051
052                if (linksA.size() > linksB.size()) {
053                    removeLinksToOrFromCollection(linksA, b);
054                } else {
055                    removeLinksToOrFromCollection(linksB, a);
056                }
057                return compare(a, b);
058            } else {
059                throw new RuntimeException("Mutually dependency on different depth level");
060            }
061        }
062
063        if (a_b != -1) {
064            return 1;
065        }
066
067        if (b_a != -1) {
068            return -1;
069        }
070
071        return 0;
072    }
073
074    private void removeLinksToOrFromCollection(List<FieldLink> links, ORI collection) {
075
076        for (int i = 0; i < links.size(); i++) {
077            FieldLink link = links.get(i);
078            if (link.getFromCollection().equals(collection) || link.getToCollection().equals(collection)) {
079                links.set(i, null);
080            }
081        }
082        links.removeAll(Collections.singletonList(null));
083
084    }
085
086    /*
087        Check if collection a depends on b
088
089        Returns -1 if not depends, 0 if there is a direct dependency. Otherwise the depth of the dependency.
090
091     */
092    private int depend(ORI a, ORI b, int depth, Set<ORI> visitedNodes) {
093
094        // End of the path, no dependency if we reach the fromCollection.
095        if (b.equals(fromCollection)) {
096            return -1;
097        }
098
099
100        // Break cycles
101        if (visitedNodes.contains(a)) {
102            return -1;
103        }
104
105        List<FieldLink> links = networkLinks.get(a);
106
107        if (links == null || links.isEmpty()) {
108            return -1;
109        }
110
111        // Check direct dependency
112        for (FieldLink link : links) {
113            if (link.getToCollection().equals(a) && link.getFromCollection().equals(b)) {
114                return depth;
115            }
116            if (link.getToCollection().equals(b) && link.getFromCollection().equals(a)) {
117                return depth;
118            }
119        }
120
121        visitedNodes.add(a);
122
123        // Check derived dependencies
124        int minDerived = -1;
125        for (FieldLink link : links) {
126            ORI nextCollection = (link.getToCollection().equals(a) ? link.getFromCollection() : link.getToCollection());
127            int derived = depend(nextCollection, b, depth + 1, visitedNodes);
128            if (derived != -1 && (minDerived == -1 || derived < minDerived)) {
129                minDerived = derived;
130            }
131        }
132
133        return minDerived;
134
135    }
136}
