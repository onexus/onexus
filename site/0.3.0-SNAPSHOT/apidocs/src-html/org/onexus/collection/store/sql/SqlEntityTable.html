018package org.onexus.collection.store.sql;
019
020import org.onexus.collection.api.Collection;
021import org.onexus.collection.api.IEntity;
022import org.onexus.collection.api.IEntityTable;
023import org.onexus.collection.api.query.Query;
024import org.onexus.collection.api.utils.QueryUtils;
025import org.onexus.resource.api.ORI;
026import org.onexus.resource.api.Progress;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030import java.sql.Connection;
031import java.sql.ResultSet;
032import java.sql.SQLException;
033import java.sql.Statement;
034
035public class SqlEntityTable implements IEntityTable {
036
037    private final static Logger LOGGER = LoggerFactory
038            .getLogger(SqlEntityTable.class);
039
040    private Query query;
041    private Progress progress;
042
043    private SqlQuery mysqlQuery;
044
045    private SqlCollectionStore store;
046
047    private Connection dataConn = null;
048    private Statement dataSt = null;
049    private ResultSet dataRS = null;
050    private Long size = null;
051
052    public SqlEntityTable(SqlCollectionStore store, Query query,
053                          Connection conn) {
054
055        this.store = store;
056        this.query = query;
057        this.mysqlQuery = store.newSqlQuery(query);
058        this.dataConn = conn;
059
060    }
061
062    private void initDataRS() {
063        String sql = null;
064        try {
065            dataSt = store.createReadStatement(dataConn);
066
067            sql = mysqlQuery.toSelectSQL();
068            LOGGER.debug(sql);
069            dataRS = dataSt.executeQuery(sql);
070        } catch (Exception e) {
071            throw new RuntimeException(e);
072        }
073    }
074
075    @Override
076    public boolean next() {
077
078        if (dataRS == null) {
079            initDataRS();
080        }
081
082        try {
083            boolean hasNext = (dataRS == null ? false : dataRS.next());
084
085            if (!hasNext) {
086                close();
087                return hasNext;
088            }
089
090            return hasNext;
091        } catch (SQLException e) {
092            return false;
093        }
094    }
095
096    @Override
097    public long size() {
098
099        if (size == null) {
100
101            String sql = null;
102            try {
103                // Count all
104                sql = mysqlQuery.toCountSQL();
105                LOGGER.debug(sql);
106                Statement dataSt = dataConn.createStatement();
107                ResultSet dataRS = dataSt.executeQuery(sql);
108                if (dataRS.next()) {
109                    size = dataRS.getLong("size");
110                } else {
111                    size = Long.valueOf(0);
112                }
113                dataRS.close();
114                dataSt.close();
115            } catch (SQLException e) {
116                throw new RuntimeException(e);
117            }
118
119        }
120
121        return size;
122    }
123
124    @Override
125    public void close() {
126        try {
127            if (dataRS != null && !dataRS.isClosed())
128                dataRS.close();
129            if (dataSt != null && !dataSt.isClosed())
130                dataSt.close();
131            if (dataConn != null && !dataConn.isClosed())
132                dataConn.close();
133        } catch (SQLException e) {
134            throw new RuntimeException(e);
135        }
136    }
137
138    @Override
139    public Query getQuery() {
140        return query;
141    }
142
143    @Override
144    public IEntity getEntity(ORI collectionURI) {
145
146        collectionURI = collectionURI.toAbsolute(query.getOn());
147
148        Collection collection = store.getCollection(collectionURI);
149        SqlEntity entity = new SqlEntity(collection);
150        store.getSqlDialect().loadEntity(query, store.getDDL(collectionURI), dataRS, entity,
151                QueryUtils.newCollectionAlias(query, collectionURI));
152
153        return entity;
154    }
155
156    @Override
157    public Progress getProgress() {
158        return progress;
159    }
160
161}
