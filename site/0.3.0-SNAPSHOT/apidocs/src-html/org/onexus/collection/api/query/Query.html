029public class Query implements Serializable {
030
031    private Map<String, ORI> define = new LinkedHashMap<String, ORI>();
032
033    private ORI on;

036

057    }

075    public Long getCount() {

079    public void addSelect(String key, List<String> fieldNames) {
080        if (select.containsKey(key) && select.get(key) != null) {
081            select.get(key).addAll(fieldNames);
082        } else {
083            select.put(key, (fieldNames == null ? null : new ArrayList<String>(fieldNames)));
084        }
085    }

129    @Override
130    public String toString() {
131        return toString(new StringBuilder(), true).toString();
132    }
133
134    public StringBuilder toString(StringBuilder oql, boolean prettyPrint) {
135
136
137        // Define section
138        if (define != null && !define.isEmpty()) {
139            oql.append("DEFINE");
140
141            Iterator<Map.Entry<String, ORI>> itDefine = define.entrySet().iterator();
142            while (itDefine.hasNext()) {
143                oql.append(prettyPrint ? "\n\t" : " ");
144
145                Map.Entry<String, ORI> entry = itDefine.next();
146                oql.append(entry.getKey()).append("=").append(escapeString(entry.getValue().toString()));
147
148                if (itDefine.hasNext()) {
149                    oql.append(',');
150                }
151            }
152        }
153
154        // On section
155        if (on != null) {
156            oql.append(prettyPrint ? "\n" : " ");
157            oql.append("ON");
158            oql.append(prettyPrint ? "\n\t" : " ");
159            oql.append(escapeString(on.toString()));
160        }
161
162
163        // Select section
164        if (select != null && !select.isEmpty()) {
165            oql.append(prettyPrint ? "\n" : " ");
166            oql.append("SELECT");
167            Iterator<Map.Entry<String, List<String>>> itSelect = select.entrySet().iterator();
168            while (itSelect.hasNext()) {
169                oql.append(prettyPrint ? "\n\t" : " ");
170                Map.Entry<String, List<String>> entry = itSelect.next();
171                oql.append(entry.getKey());
172                if (entry.getValue() != null) {
173                    Iterator<String> fields = entry.getValue().iterator();
174                    if (fields.hasNext()) {
175                        oql.append(" (");
176                        while (fields.hasNext()) {
177                            oql.append(fields.next());
178                            if (fields.hasNext()) {
179                                oql.append(", ");
180                            }
181                        }
182                        oql.append(")");
183                    }
184                }
185
186                if (itSelect.hasNext()) {
187                    oql.append(",");
188                }
189            }
190        }
191
192        // From section
193        if (from != null && !from.isEmpty()) {
194            oql.append(prettyPrint ? "\n" : " ");
195            oql.append("FROM");
196            oql.append(prettyPrint ? "\n\t" : " ");
197            oql.append(from);
198        }
199
200        // Where section
201        if (where != null) {
202            oql.append(prettyPrint ? "\n" : " ");
203            oql.append("WHERE");
204            oql.append(prettyPrint ? "\n\t" : " ");
205            where.toString(oql, prettyPrint);
206        }
207
208        // OrderBy section
209        if (orders != null && !orders.isEmpty()) {
210            oql.append(prettyPrint ? "\n" : " ");
211            oql.append("ORDER BY");
212
213            Iterator<OrderBy> itOrders = orders.iterator();
214            while (itOrders.hasNext()) {
215                oql.append(prettyPrint ? "\n\t" : " ");
216                OrderBy orderBy = itOrders.next();
217                orderBy.toString(oql, prettyPrint);
218                if (itOrders.hasNext()) {
219                    oql.append(',');
220                }
221            }
222        }
223
224        // Limit section
225        if (offset != null && count != null) {
226            oql.append(prettyPrint ? "\n" : " ");
227            oql.append("LIMIT");
228            oql.append(prettyPrint ? "\n\t" : " ");
229            oql.append("'").append(offset).append("', '").append(count).append("'");
230        }
231
232        return oql;
233    }
234
235    public static String unescapeString(String value) {
236
237        if (value == null) {
238            return null;
239        }
240
241        if (value.length() < 3) {
242            return "";
243        }
244
245        char quote = value.charAt(0);
246
247        value = value.substring(1, value.length() - 1);
248        if (quote == '"') {
249            value = value.replace("\\\"", "\"");
250        } else {
251            value = value.replace("\\'", "'");
252        }
253
254        return value;
255
256    }
257
258
259    public static String escapeString(String value) {
260        if (value == null) {
261            return null;
262        }
263        value = "'" + value.replace("'", "\\'") + "'";
264        return value;
265    }
