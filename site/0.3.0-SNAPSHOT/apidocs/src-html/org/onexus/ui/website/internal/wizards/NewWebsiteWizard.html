059public class NewWebsiteWizard extends AbstractNewResourceWizard<WebsiteConfig> {
060
061    private Boolean addTab = true;
062    private String tabName = "Tab-0";
063    private ORI mainTabCollection;
064    private List<ORI> linkCollections = new ArrayList<ORI>();
065    private TabConfig currentTab;
066
067    @PaxWicketBean(name = "resourceManager")
068    private IResourceManager resourceManager;
069
070    public NewWebsiteWizard(String id, IModel<? extends Resource> resourceModel) {
071        super(id, resourceModel);
072
073        AddTabCondition condition = new AddTabCondition();
074        WizardModel model = new WizardModel();
075        model.add(new ResourceName());
076        model.add(new AddTabs());
077        model.add(new TabName(), condition);
078        model.add(new MainTabCollection(), condition);
079        model.add(new LinkedCollections(), condition);
080        model.add(new StaticContentStep(Model.of("New website"), Model.of("Confirm website"), "Done", false));
081
082        init(model);
083    }
084
085    @Override
086    protected WebsiteConfig getDefaultResource() {
087
088        WebsiteConfig website = new WebsiteConfig();
089        website.setPages(new ArrayList<PageConfig>());
090        website.setLabel("website");
091        website.setTitle("Website title");
092
093        // Browser page
094        BrowserPageConfig browserPage = new BrowserPageConfig();
095        browserPage.setId("browser");
096        browserPage.setLabel("Browser");
097        browserPage.setTabs(new ArrayList<TabConfig>());
098        browserPage.setWidgets(new ArrayList<WidgetConfig>());
099
100        WidgetConfig downloadWidget = new DownloadWidgetConfig("download");
101        downloadWidget.setButton("download");
102        browserPage.getWidgets().add(downloadWidget);
103
104        WidgetConfig shareWidget = new ShareWidgetConfig("share");
105        shareWidget.setButton("share");
106        browserPage.getWidgets().add(shareWidget);
107
108        website.getPages().add(browserPage);
109
110        // Html page
111        HtmlPageConfig htmlPage = new HtmlPageConfig();
112        htmlPage.setId("about");
113        htmlPage.setLabel("About");
114        htmlPage.setContent("Website powered by Onexus");
115        website.getPages().add(htmlPage);
116
117
118        return website;
119    }

166    private final class MainTabCollection extends WizardStep {
167
168        public MainTabCollection() {
169            super("New website", "Main tab collection");
170
171            List<ORI> projectCollections = new ArrayList<ORI>();
172            addAllCollections(projectCollections, getParentUri());
173            add(new DropDownChoice<ORI>("mainTabCollection", projectCollections));
174
175        }
176
177        @Override
178        public void applyState() {
179
180            BrowserPageConfig browserPage = getBrowserPage();
181            ViewConfig view = new ViewConfig();
182            view.setTitle("default");
183
184            String tableWidgetId = currentTab.getId() + "-table";
185            String searchWidgetId = currentTab.getId() + "-search";
186            view.setMain(tableWidgetId);
187            view.setTop(searchWidgetId);
188            view.setTopRight("download, share");
189
190            // Create search widget
191            SearchWidgetConfig searchWidget = new SearchWidgetConfig(searchWidgetId);
192            List<SearchField> searchFields = new ArrayList<SearchField>();
193            searchWidget.setFields(searchFields);
194            browserPage.getWidgets().add(searchWidget);
195
196            // Create table widget
197            TableViewerConfig tableWidget = new TableViewerConfig(tableWidgetId, mainTabCollection);
198            tableWidget.setColumnSets(new ArrayList<ColumnSet>());
199
200            // Main collection columns
201            Collection collection = resourceManager.load(Collection.class, mainTabCollection.toAbsolute(getParentUri()));
202
203            List<String> keyFields = new ArrayList<String>();
204            List<String> fields = new ArrayList<String>();
205            List<String> search = new ArrayList<String>();
206
207            for (Field field : collection.getFields()) {
208
209                if (field.isPrimaryKey() != null && field.isPrimaryKey()) {
210                    keyFields.add(field.getId());
211                } else {
212                    fields.add(field.getId());
213                }
214
215                if (String.class.equals(field.getType())) {
216                    search.add(field.getId());
217                }
218            }
219
220            searchFields.add(new SearchField(mainTabCollection, StringUtils.join(search, ',')));
221
222            ColumnSet columnSet = new ColumnSet();
223            columnSet.setTitle("default");
224            columnSet.setColumns(new ArrayList<IColumnConfig>());
225
226            if (!keyFields.isEmpty()) {
227                columnSet.getColumns().add(new ColumnConfig(mainTabCollection, StringUtils.join(keyFields, ','), "LINK"));
228            }
229
230            if (!fields.isEmpty()) {
231                columnSet.getColumns().add(new ColumnConfig(mainTabCollection, StringUtils.join(fields, ',')));
232            }
233
234            tableWidget.getColumnSets().add(columnSet);
235            browserPage.getWidgets().add(tableWidget);
236
237            // Add view
238            currentTab.setViews(new ArrayList<ViewConfig>());
239            currentTab.getViews().add(view);
240
241            // Add current tab
242            browserPage.getTabs().add(currentTab);
243
244        }
245    }

247    private final class LinkedCollections extends WizardStep {
248
249        private List<ORI> linkedCollections;
250
251        public LinkedCollections() {
252            super("New website", "Linked collections");
253
254        }
255
256        @Override
257        protected void onBeforeRender() {
258
259            List<ORI> projectCollections = new ArrayList<ORI>();
260            addAllCollections(projectCollections, getParentUri());
261
262            linkedCollections = new ArrayList<ORI>();
263
264
265            Collection colA = resourceManager.load(Collection.class, getMainTabCollection().toAbsolute(getParentUri()));
266
267            for (ORI collection : projectCollections) {
268
269
270                Collection colB = resourceManager.load(Collection.class, collection.toAbsolute(getParentUri()));
271
272                if (colA.getORI().equals(colB.getORI())) {
273                    continue;
274                }
275
276                List<FieldLink> links = LinkUtils.getLinkFields(getParentUri(), colA, colB);
277
278                if (!links.isEmpty()) {
279                    linkedCollections.add(collection);
280                }
281            }
282
283            addOrReplace(new ListMultipleChoice<ORI>("linkCollections", linkedCollections));
284
285            super.onBeforeRender();
286        }
287
288        @Override
289        public void applyState() {
290
291            BrowserPageConfig browserPage = getBrowserPage();
292
293            if (!linkCollections.isEmpty()) {
294
295                String tableWidgetId = currentTab.getId() + "-table";
296                TableViewerConfig tableWidget = (TableViewerConfig) browserPage.getWidget(tableWidgetId);
297                ColumnSet columnSet = tableWidget.getColumnSets().get(0);
298
299                for (ORI linkCollection : linkCollections) {
300
301                    ORI resourceUri = linkCollection.toAbsolute(getParentUri());
302                    Collection collection = resourceManager.load(Collection.class, resourceUri);
303
304                    List<String> otherFields = new ArrayList<String>();
305
306                    for (Field field : collection.getFields()) {
307                        otherFields.add(field.getId());
308                    }
309
310                    if (!otherFields.isEmpty()) {
311                        columnSet.getColumns().add(new ColumnConfig(linkCollection, StringUtils.join(otherFields, ',')));
312                    }
313                }
314            }
315
316            // Prepare next tab iteration
317            tabName = "Tab-" + browserPage.getTabs().size();
318            mainTabCollection = null;
319            linkCollections.clear();
320            currentTab = null;
321            WizardModel model = (WizardModel) getWizardModel();
322            IWizardStep firstStep = model.stepIterator().next();
323            model.setActiveStep(firstStep);
324        }
325
326    }

360    private void addAllCollections(List<ORI> collectionUris, ORI parentUri) {
361
362        List<Collection> collections = resourceManager.loadChildren(Collection.class, parentUri);
363        for (Collection collection : collections) {
364            collectionUris.add(collection.getORI());
365        }
366
367        List<Folder> folders = resourceManager.loadChildren(Folder.class, parentUri);
368        for (Folder folder : folders) {
369            addAllCollections(collectionUris, folder.getORI());
370        }
371
372    }
