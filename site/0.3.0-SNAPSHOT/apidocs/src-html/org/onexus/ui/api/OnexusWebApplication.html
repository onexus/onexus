018package org.onexus.ui.api;
019
020import org.apache.wicket.Application;
021import org.apache.wicket.Page;
022import org.apache.wicket.RuntimeConfigurationType;
023import org.apache.wicket.authroles.authentication.AbstractAuthenticatedWebSession;
024import org.apache.wicket.authroles.authentication.AuthenticatedWebApplication;
025import org.apache.wicket.core.request.handler.PageProvider;
026import org.apache.wicket.core.request.handler.RenderPageRequestHandler;
027import org.apache.wicket.markup.html.WebPage;
028import org.apache.wicket.request.IRequestHandler;
029import org.apache.wicket.request.cycle.AbstractRequestCycleListener;
030import org.apache.wicket.request.cycle.RequestCycle;
031import org.onexus.resource.api.LoginContext;
032import org.onexus.ui.api.pages.SignOutPage;
033import org.onexus.ui.api.pages.error.ExceptionErrorPage;
034import org.onexus.ui.api.pages.resource.ResourcesPage;
035import org.onexus.ui.authentication.jaas.JaasSignInPage;
036import org.onexus.ui.authentication.persona.PersonaSignInPage;
037import org.ops4j.pax.wicket.api.InjectorHolder;
038
039import javax.servlet.http.HttpServletRequest;
040
041public class OnexusWebApplication extends AuthenticatedWebApplication {
042
043    private final static String SERVER_URL = System.getenv("ONEXUS_SERVER_URL");
044
045    public Class<? extends Page> getHomePage() {
046        return ResourcesPage.class;
047    }
048
049    @Override
050    public RuntimeConfigurationType getConfigurationType() {
051        return RuntimeConfigurationType.DEPLOYMENT;
052    }
053
054    @Override
055    protected void init() {
056        super.init();
057
058        // Authentication
059        getApplicationSettings().setAccessDeniedPage(getSignInPageClass());
060
061        /* In case of unhandled exception redirect it to a custom page */
062        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
063            @Override
064            public IRequestHandler onException(RequestCycle cycle, Exception e) {
065                return new RenderPageRequestHandler(new PageProvider(new ExceptionErrorPage(e)));
066            }
067        });
068
069        // Set the login context on each thread
070        getRequestCycleListeners().add(new AbstractRequestCycleListener() {
071            @Override
072            public void onBeginRequest(RequestCycle cycle) {
073                LoginContext.set(OnexusWebSession.get().getLoginContext());
074            }
075        });
076
077        // Mount pages
078        mountPage("/login", getSignInPageClass());
079        mountPage("/logout", SignOutPage.class);
080        mountPage("/admin", ResourcesPage.class);
081
082    }
083
084    public boolean usePersonSignIn() {
085        return Boolean.parseBoolean(System.getProperty("org.onexus.ui.authentication.persona", "false"));
086    }
087
088    @Override
089    protected Class<? extends WebPage> getSignInPageClass() {
090        if (usePersonSignIn()) {
091            return PersonaSignInPage.class;
092        }
093
094        return JaasSignInPage.class;
095    }
096
097    @Override
098    protected Class<? extends AbstractAuthenticatedWebSession> getWebSessionClass() {
099        return OnexusWebSession.class;
100    }
101
102    public static OnexusWebApplication get() {
103        return (OnexusWebApplication) Application.get();
104    }
105
106    public static String getRequestUrl() {
107
108        HttpServletRequest request = (HttpServletRequest) RequestCycle.get().getRequest().getContainerRequest();
109        String requestUrl = request.getRequestURL().toString();
110
111        if (SERVER_URL != null) {
112            String baseUrl = RequestCycle.get().getUrlRenderer().getBaseUrl().toString();
113            return SERVER_URL + '/' + baseUrl;
114        }
115
116        return requestUrl;
117    }
118
119    public static void inject(Object obj) {
120        if (obj != null) {
121            InjectorHolder.getInjector().inject(obj, obj.getClass());
122        }
123    }
124
125
126}
