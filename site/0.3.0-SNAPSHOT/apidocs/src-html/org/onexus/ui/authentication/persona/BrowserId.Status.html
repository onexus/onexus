001package org.onexus.ui.authentication.persona;
002
003
004import org.apache.wicket.util.string.Strings;
005
006import java.io.Serializable;
007import java.util.Date;
008
009/**
010 * An object representing the response from https://browserid.org
011 */
012public class BrowserId implements Serializable
013{
014
015    private static final long serialVersionUID = 1L;
016
017    public enum Status
018    {
019        OK, FAIL;
020
021        public static Status parse(final String t)
022        {
023            if ("okay".equals(t))
024            {
025                return OK;
026            }
027            else if ("failure".equals(t))
028            {
029                return FAIL;
030            }
031            else
032            {
033                throw new IllegalArgumentException("Unrecognized status: " + t);
034            }
035        }
036    };
037
038    private Status status;
039
040    private String email;
041
042    private String audience;
043
044    private Date validUntil;
045
046    private String issuer;
047
048    private String reason;
049
050    public String getReason()
051    {
052        return reason;
053    }
054
055    public void setReason(String reason)
056    {
057        this.reason = reason;
058    }
059
060    public Status getStatus()
061    {
062        return status;
063    }
064
065    public void setStatus(Status status)
066    {
067        this.status = status;
068    }
069
070    public String getEmail()
071    {
072        return email;
073    }
074
075    public void setEmail(String email)
076    {
077        this.email = email;
078    }
079
080    public String getAudience()
081    {
082        return audience;
083    }
084
085    public void setAudience(String audience)
086    {
087        this.audience = audience;
088    }
089
090    public Date getValidUntil()
091    {
092        return validUntil;
093    }
094
095    public void setValidUntil(Date validUntil)
096    {
097        this.validUntil = validUntil;
098    }
099
100    public String getIssuer()
101    {
102        return issuer;
103    }
104
105    public void setIssuer(String issuer)
106    {
107        this.issuer = issuer;
108    }
109
110    /**
111     * Create {@link BrowserId} object from success or failure JSON response
112     *
113     * @param json
114     *            the response returned by https://browserid.org
115     * @return a {@link BrowserId} instance with status OK if the authentication passed, or with
116     *         status FAIL if the authentication failed, or {@code null} if the passed JSON response
117     *         is invalid
118     */
119    public static BrowserId of(String json)
120    {
121        BrowserId browserId = null;
122
123        if (json != null && json.startsWith("{") && json.endsWith("}"))
124        {
125            json = json.substring(1);
126            json = json.substring(0, json.length() - 1);
127
128            String[] settings = Strings.split(json, ',');
129            if (settings.length > 0)
130            {
131                browserId = new BrowserId();
132                for (String setting : settings)
133                {
134                    String[] pair = Strings.split(setting, ':');
135                    pair[0] = Strings.replaceAll(pair[0], "\"", "").toString();
136                    pair[1] = Strings.replaceAll(pair[1], "\"", "").toString();
137
138                    if ("status".equals(pair[0]))
139                    {
140                        browserId.status = Status.parse(pair[1]);
141                    }
142                    else if ("email".equals(pair[0]))
143                    {
144                        browserId.email = pair[1];
145                    }
146                    else if ("audience".equals(pair[0]))
147                    {
148                        browserId.audience = pair[1];
149                    }
150                    else if ("issuer".equals(pair[0]))
151                    {
152                        browserId.issuer = pair[1];
153                    }
154                    else if ("expires".equals(pair[0]))
155                    {
156                        Long millis = Long.valueOf(pair[1]);
157                        browserId.validUntil = new Date(millis);
158                    }
159                    else if ("reason".equals(pair[0]))
160                    {
161                        browserId.reason = pair[1];
162                    }
163                    else
164                    {
165                        throw new IllegalArgumentException("Unknown setting: " + pair[0]);
166                    }
167                }
168            }
169        }
170
171        return browserId;
172    }
173}
