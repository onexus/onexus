001package org.onexus.ui.authentication.persona;
002
003import org.apache.wicket.Component;
004import org.apache.wicket.Session;
005import org.apache.wicket.ajax.AbstractDefaultAjaxBehavior;
006import org.apache.wicket.ajax.AjaxRequestTarget;
007import org.apache.wicket.markup.head.IHeaderResponse;
008import org.apache.wicket.markup.head.JavaScriptHeaderItem;
009import org.apache.wicket.markup.head.OnDomReadyHeaderItem;
010import org.apache.wicket.protocol.http.WebApplication;
011import org.apache.wicket.protocol.http.WebSession;
012import org.apache.wicket.request.cycle.RequestCycle;
013import org.apache.wicket.util.template.PackageTextTemplate;
014import org.apache.wicket.util.template.TextTemplate;
015
016import java.util.HashMap;
017import java.util.Map;
018
019
020public class SignOutBehavior extends AbstractDefaultAjaxBehavior {
021
022    @Override
023    protected void respond(AjaxRequestTarget target) {
024        WebSession.get().invalidate();
025        RequestCycle.get().setResponsePage(WebApplication.get().getHomePage());
026    }
027
028    @Override
029    public void renderHead(final Component component, final IHeaderResponse response)
030    {
031        component.setOutputMarkupId(true);
032        super.renderHead(component, response);
033
034        final Map<String, Object> variables = new HashMap<String, Object>();
035        variables.put("componentId", component.getMarkupId());
036        BrowserId personaId = SessionHelper.getBrowserId(Session.get());
037        String userName = (personaId == null ? "" : personaId.getEmail() );
038        variables.put("userName", userName);
039        variables.put("callbackUrl", getCallbackUrl());
040
041        final TextTemplate verifyTemplate = new PackageTextTemplate(VerifyBehavior.class, "signout.js.tmpl");
042        String asString = verifyTemplate.asString(variables);
043
044        response.render(JavaScriptHeaderItem.forUrl(GuestPanel.BROWSER_ID_JS));
045        response.render(OnDomReadyHeaderItem.forScript(asString));
046
047    }
048}
