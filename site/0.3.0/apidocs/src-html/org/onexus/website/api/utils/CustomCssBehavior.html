018package org.onexus.website.api.utils;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.behavior.Behavior;
022import org.apache.wicket.markup.head.CssHeaderItem;
023import org.apache.wicket.markup.head.IHeaderResponse;
024import org.apache.wicket.request.resource.CssResourceReference;
025import org.apache.wicket.util.string.Strings;
026import org.onexus.data.api.IDataManager;
027import org.onexus.resource.api.IResourceManager;
028import org.onexus.resource.api.ORI;
029import org.onexus.resource.api.Project;
030import org.onexus.website.api.WebsiteApplication;
031import org.ops4j.pax.wicket.api.PaxWicketBean;
032
033public class CustomCssBehavior extends Behavior {
034
035    private transient CssHeaderItem cssHeaderItem;
036
037    private ORI resourceUri;
038
039    @PaxWicketBean(name = "dataManager")
040    private IDataManager dataManager;
041
042    @PaxWicketBean(name = "resourceManager")
043    private IResourceManager resourceManager;
044
045    public CustomCssBehavior(ORI parentOri, String cssTag) {
046        if (!Strings.isEmpty(cssTag)) {
047            resourceUri = new ORI(cssTag);
048            if (!resourceUri.isAbsolute()) {
049                resourceUri = resourceUri.toAbsolute(parentOri);
050            }
051        }
052    }
053
054    private CustomCssBehavior(ORI resourceUri) {
055        this.resourceUri = resourceUri;
056    }
057
058    @Override
059    public void renderHead(Component component, IHeaderResponse response) {
060
061        if (cssHeaderItem == null) {
062
063            if (resourceUri != null) {
064
065                String dataService = getDataManager().getMount();
066                Project project = getResourceManager().getProject(resourceUri.getProjectUrl());
067
068                String url;
069                if (project != null) {
070                    url = WebsiteApplication.toAbsolutePath('/' + dataService + '/' + project.getName() + resourceUri.getPath());
071                } else {
072                    url = resourceUri.toString();
073                }
074                cssHeaderItem = CssHeaderItem.forUrl(url);
075
076            } else {
077                cssHeaderItem = CssHeaderItem.forReference(new CssResourceReference(component.getClass(), component.getClass().getSimpleName() + ".css"));
078            }
079        }
080
081        response.render(cssHeaderItem);
082    }
083
084    private IDataManager getDataManager() {
085
086        if (dataManager == null) {
087            WebsiteApplication.inject(this);
088        }
089
090        return dataManager;
091    }
092
093    private IResourceManager getResourceManager() {
094
095        if (resourceManager == null) {
096            WebsiteApplication.inject(this);
097        }
098
099        return resourceManager;
100    }
101
102}
