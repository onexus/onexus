31  import org.ops4j.pax.wicket.api.PaxWicketBean;
32  
33  import java.util.Arrays;
34  import java.util.List;
35  
36  public class ManageCollectionWizard extends AbstractWizard {
37  
38      // Commands
39      private static final String LOAD = "Load collection";
40      private static final String UNLOAD = "Unload collection";
41      private static final List<String> COMMANDS = Arrays.asList(new String[]{LOAD, UNLOAD});
42  
43      private String selected;
44      private ORI resourceURI;
45  
46      @PaxWicketBean(name = "collectionManager")
47      private ICollectionManager collectionManager;
48  
49      public ManageCollectionWizard(String id, IModel<? extends Resource> resourceModel) {
50          super(id);
51  
52          this.resourceURI = resourceModel.getObject().getORI();
53  
54          WizardModel model = new WizardModel();
55          model.add(new BasicOptions());
56  
57          init(model);
58      }
59  
60      @Override
61      public void onFinish() {
62  
63          super.onFinish();
64  
65          if (selected.equals(UNLOAD)) {
66              collectionManager.unload(resourceURI);
67              return;
68          }
69  
70          if (selected.equals(LOAD)) {
71              collectionManager.unload(resourceURI);
72              Query emptyQuery = new Query();
73              emptyQuery.addDefine("c", resourceURI);
74              emptyQuery.setFrom("c");
75              emptyQuery.setOffset(0);
76              emptyQuery.setCount(0);
77              ProgressBar.show(collectionManager.load(emptyQuery));
78              return;
79          }
80  
81      }
82  
83      public String getSelected() {
84          return selected;
85      }
86  
87      public void setSelected(String selected) {
88          this.selected = selected;
89      }
90  
91      private final class BasicOptions extends WizardStep {
92  
93  
94          public BasicOptions() {
95              super("Manage collection", "Basic collection tools");
96  
97              RadioChoice<String> commandOptions = new RadioChoice<String>("commands", new PropertyModel<String>(ManageCollectionWizard.this, "selected"), COMMANDS);
98              add(commandOptions);
99  
100         }
101 
102     }
