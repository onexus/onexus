069    @Override
070    public List<WidgetStatus> getActiveWidgetStatuses(ORI parentOri) {
071
072        BrowserPageConfig pageConfig = getConfig();
073        TabConfig tabConfig = pageConfig.getTab(currentTabId);
074        ViewConfig viewConfig = tabConfig.getView(currentView);
075
076        List<WidgetConfig> widgetConfigs = new ArrayList<WidgetConfig>();
077        Predicate visible = new VisiblePredicate(parentOri, getEntitySelections());
078        CollectionUtils.select(ViewConfig.getSelectedWidgetConfigs(
079                pageConfig,
080                viewConfig.getLeft(),
081                viewConfig.getTop(),
082                viewConfig.getTopRight(),
083                viewConfig.getMain()
084        ), visible, widgetConfigs);
085
086
087        List<WidgetStatus> statuses = new ArrayList<WidgetStatus>();
088        for (WidgetConfig config : widgetConfigs) {
089            WidgetStatus status = getWidgetStatus(config.getId());
090            if (status != null) {
091                statuses.add(status);
092            }
093        }
094
095        return statuses;
096    }

115    public void setCurrentTabId(String currentTabId) {
116        if (currentTabId != null && !currentTabId.equals(this.currentTabId)) {
117            BrowserPageStatus defaultStatus = getConfig().getDefaultStatus();
118            if (defaultStatus == null) {
119                defaultStatus = getConfig().createEmptyStatus();
120            }
121            this.currentView = defaultStatus.getCurrentView();
122            this.currentFilters = new ArrayList<FilterConfig>();
123        }
124        this.currentTabId = currentTabId;
125    }

155    @Override
156    public void beforeQueryBuild(Query query) {
157
158        if (!StringUtils.isEmpty(getBase())) {

160        }

164    @Override
165    public void afterQueryBuild(Query query) {
166
167        // General entity selections
168        if (getEntitySelections() != null) {
169            for (IEntitySelection fe : getEntitySelections()) {
170                if (getCollectionManager().isLinkable(query, fe.getSelectionCollection().toAbsolute(query.getOn()))) {
171                    Filter filter = fe.buildFilter(query);
172                    QueryUtils.and(query, filter);
173                    fe.setEnable(true);
174                } else {
175                    fe.setEnable(false);
176                }
177            }
178        }
179
180        // Current tab filters
181        MultipleEntitySelection filterCompiler = new MultipleEntitySelection();
182        for (FilterConfig filterConfig : getCurrentFilters()) {
183            filterCompiler.setFilterConfig(filterConfig);
184            Filter filter = filterCompiler.buildFilter(query);
185            QueryUtils.and(query, filter);
186        }
187
188    }

204    @Override
205    public void encodeParameters(PageParameters parameters, String keyPrefix, boolean global) {
206
207        BrowserPageStatus defaultStatus = getConfig().getDefaultStatus();
208        if (defaultStatus == null) {
209            defaultStatus = getConfig().createEmptyStatus();
210        }
211
212        if (global || !StringUtils.equals(currentTabId, defaultStatus.getCurrentTabId())) {
213            if (currentTabId == null) {
214                currentTabId = defaultStatus.getCurrentTabId();
215            }
216
217            if (currentTabId != null) {
218                parameters.add(keyPrefix + "tab", currentTabId);
219            }
220        }
221
222        if (global || !StringUtils.equals(currentView, defaultStatus.getCurrentView()) &&
223                getConfig().getTab(currentTabId).getViews().size() > 1) {
224
225            if (currentView == null) {
226                currentView = defaultStatus.getCurrentView();
227            }
228
229            if (currentView != null) {
230                parameters.add(keyPrefix + "view", currentView);
231            }
232
233        }
234
235        ORI parentOri = getORI();
236        for (IEntitySelection filter : getEntitySelections()) {
237            if (filter instanceof SingleEntitySelection) {
238                parameters.add(keyPrefix + "f", filter.toUrlParameter(global, parentOri));
239            } else {
240                parameters.add(keyPrefix + "fc", filter.toUrlParameter(global, parentOri));
241            }
242        }
243
244
245        super.encodeParameters(parameters, keyPrefix, global);
246    }
247
248    @Override
249    public void decodeParameters(PageParameters parameters, String keyPrefix) {
250
251        StringValue currentTabId = parameters.get(keyPrefix + "tab");
252        if (!currentTabId.isEmpty()) {
253            this.currentTabId = currentTabId.toString();
254
255            // Check that is a valid tabId
256            if (getConfig().getTab(this.currentTabId) == null) {
257
258                // Look for the more similar tab id
259                List<TabConfig> tabs = new ArrayList<TabConfig>(getConfig().getTabs());
260                Collections.sort(tabs, new Comparator<TabConfig>() {
261                    @Override
262                    public int compare(TabConfig o1, TabConfig o2) {
263                        Integer v1 = StringUtils.getLevenshteinDistance(BrowserPageStatus.this.currentTabId, o1.getId());
264                        Integer v2 = StringUtils.getLevenshteinDistance(BrowserPageStatus.this.currentTabId, o2.getId());
265
266                        return v1.compareTo(v2);
267                    }
268                });
269                this.currentTabId = tabs.get(0).getId();
270            }
271        }
272
273        StringValue currentView = parameters.get(keyPrefix + "view");
274        if (!currentView.isEmpty()) {
275            this.currentView = currentView.toString();
276
277            // Check that is a valid currentView
278            if (getConfig().getTab(this.currentTabId).getView(this.currentView) == null) {
279
280                // Look for the more similar view id
281                List<ViewConfig> views = new ArrayList<ViewConfig>(getConfig().getTab(this.currentTabId).getViews());
282
283                if (views.size() > 1) {
284                    Collections.sort(views, new Comparator<ViewConfig>() {
285                        @Override
286                        public int compare(ViewConfig o1, ViewConfig o2) {
287                            Integer v1 = StringUtils.getLevenshteinDistance(BrowserPageStatus.this.currentView, o1.getTitle());
288                            Integer v2 = StringUtils.getLevenshteinDistance(BrowserPageStatus.this.currentView, o2.getTitle());
289
290                            return v1.compareTo(v2);
291                        }
292                    });
293                }
294
295                this.currentView = views.get(0).getTitle();
296
297            }
298        }
299
300        selections = new ArrayList<IEntitySelection>();
301        List<StringValue> values = parameters.getValues(keyPrefix + "f");
302        if (!values.isEmpty()) {
303            for (StringValue value : values) {
304                SingleEntitySelection fe = new SingleEntitySelection();
305                fe.loadUrlPrameter(value.toString());
306                addEntitySelection(fe);
307            }
308        }
309
310        values = parameters.getValues(keyPrefix + "fc");
311        if (!values.isEmpty()) {
312            for (StringValue value : values) {
313                MultipleEntitySelection fe = new MultipleEntitySelection();
314                fe.loadUrlPrameter(value.toString());
315                addEntitySelection(fe);
316            }
317        }
318
319
320        super.decodeParameters(parameters, keyPrefix);    //To change body of overridden methods use File | Settings | File Templates.
321    }
