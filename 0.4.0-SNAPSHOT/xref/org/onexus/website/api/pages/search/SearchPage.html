312     private Iterator<IEntity> getAutocompleteChoices(String in) {
313 
314         int lastComma = in.lastIndexOf(',');
315         String input = lastComma > -1 ? in.substring(lastComma + 1).trim() : in.trim();
316 
317         Query query = new Query();
318         SearchType type = getStatus().getType();
319 
320         ORI collectionUri = getAbsoluteUri(type.getCollection());
321         String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
322         query.setFrom(collectionAlias);
323 
324         List<String> fieldList = type.getFieldsList();
325         query.addSelect(collectionAlias, fieldList);
326 
327         for (String field : fieldList) {
328             QueryUtils.or(query, new Contains(collectionAlias, field, input));
329         }
330         query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
331 
332         query.setCount(10);
333 
334         return new EntityIterator(collectionManager.load(query), collectionUri);
335     }

374     /**
375      * Generic IEntity renderer that show all the fields.
376      */
377     private class EntityRenderer implements IAutoCompleteRenderer<IEntity> {
378 
379         public final void render(final IEntity object, final Response response, final String criteria) {
380             String textValue = getTextValue(object, criteria);
381             if (textValue == null) {
382                 throw new IllegalStateException(
383                         "A call to textValue(Object) returned an illegal value: null for object: " +
384                                 object.toString());
385             }
386             textValue = textValue.replaceAll("\\\"", "&quot;");
387 
388             response.write("<li textvalue=\"" + textValue + "\"");
389             response.write(">");
390             renderChoice(object, response, criteria);
391             response.write("</li>");
392         }
393 
394         private String getTextValue(IEntity object, String in) {
395 
396             int lastComma = in.lastIndexOf(',');
397             String criteria = lastComma > -1 ? in.substring(lastComma + 1).trim() : in.trim();
398             String previous = lastComma > -1 ? in.substring(0, lastComma) + ", " : "";
399 
400 
401             SearchType type = getStatus().getType();
402             List<String> fields = type.getKeysList();
403 
404             for (String field : fields) {
405                 String value = String.valueOf(object.get(field));
406 
407                 if (StringUtils.containsIgnoreCase(value, criteria)) {
408                     return previous + value;
409                 }
410             }
411 
412             return previous + object.get(fields.get(0));
413         }
414 
415         public final void renderHeader(final Response response) {
416             response.write("<ul>");
417         }
418 
419         public final void renderFooter(final Response response, int count) {
420             response.write("</ul>");
421         }
422 
423         protected void renderChoice(IEntity object, Response response, String criteria) {
424 
425             SearchType type = getStatus().getType();
426             List<String> keys = type.getKeysList();
427             List<String> fields = type.getFieldsList();
428 
429 
430             boolean keyFieldAdded = false;
431             for (String field : fields) {
432                 String value = String.valueOf(object.get(field));
433 
434                 if (StringUtils.containsIgnoreCase(value, criteria)) {
435                     if (keys.contains(field)) {
436                         keyFieldAdded = true;
437                     }
438 
439                     renderValue(response, value, criteria, field);
440 
441                 }
442             }
443 
444             if (!keyFieldAdded) {
445                 String field = keys.get(0);
446                 String value = String.valueOf(object.get(field));
447                 renderValue(response, value, criteria, field);
448             }
449         }
450 
451         private void renderValue(Response response, String value, String criteria, String field) {
452 
453             String hlValue = value.replaceAll("(?i)(" + criteria + ")", "<strong>$1</strong>");
454             response.write("<span class='f'>" + field.toLowerCase() + ":</span>" + hlValue);
455             response.write("<br />");
456         }

462     private final class ExamplesModel extends AbstractReadOnlyModel<List<SearchExample>> {
463 
464         private IModel<SearchType> model;
465 
466         private ExamplesModel(IModel<SearchType> model) {
467             this.model = model;
468         }
469 
470         @Override
471         public List<SearchExample> getObject() {
472 
473             SearchType searchType = model.getObject();
474 
475             if (searchType == null || searchType.getExamples() == null) {
476                 return Collections.EMPTY_LIST;
477             }
478 
479             List<SearchExample> values = new ArrayList<SearchExample>();
480             for (String value : searchType.getExamples().split(",")) {
481 
482                 String label = value.trim();
483                 String query = label;
484 
485                 if (label.contains("|")) {
486                     String[] split = label.split("\\|");
487                     label = split[0].trim();
488                     query = split[1].replace(';', ',');
489                 }
490 
491                 values.add(new SearchExample(label, query));
492             }
493             return values;
494         }
495     }
